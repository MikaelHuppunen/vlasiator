135 Real getR(creal x,creal y,creal z, uint geometry, Real center[3]) {
136
137 Real r;
138
139 switch(geometry) {
140 case 0:
141 // infinity-norm, result is a diamond/square with diagonals aligned on the axes in 2D
142 r = fabs(x-center[0]) + fabs(y-center[1]) + fabs(z-center[2]);
143 break;
144 case 1:
145 // 1-norm, result is is a grid-aligned square in 2D
146 r = max(max(fabs(x-center[0]), fabs(y-center[1])), fabs(z-center[2]));
147 break;
148 case 2:
149 // 2-norm (Cartesian), result is a circle in 2D
150 r = sqrt((x-center[0])*(x-center[0]) + (y-center[1])*(y-center[1]) + (z-center[2])*(z-center[2]));
151 break;
152 case 3:
153 // 2-norm (Cartesian) cylinder aligned on y-axis
154 r = sqrt((x-center[0])*(x-center[0]) + (z-center[2])*(z-center[2]));
155 break;
156 default:
157 abort_mpi("copysphere.geometry has to be 0, 1 or 2.", 1);
158 }
159
160 return r;
161 }

163 void Copysphere::assignSysBoundary(dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
164 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid) {
165 const vector<CellID>& cells = getLocalCells();
166 for(uint i=0; i<cells.size(); i++) {
167 if(mpiGrid[cells[i]]->sysBoundaryFlag == sysboundarytype::DO_NOT_COMPUTE) {
168 continue;
169 }
170
171 creal* const cellParams = &(mpiGrid[cells[i]]->parameters[0]);
172 creal dx = cellParams[CellParams::DX];
173 creal dy = cellParams[CellParams::DY];
174 creal dz = cellParams[CellParams::DZ];
175 creal x = cellParams[CellParams::XCRD] + 0.5*dx;
176 creal y = cellParams[CellParams::YCRD] + 0.5*dy;
177 creal z = cellParams[CellParams::ZCRD] + 0.5*dz;
178
179 if(getR(x,y,z,this->geometry,this->center) < this->radius) {
180 mpiGrid[cells[i]]->sysBoundaryFlag = this->getIndex();
181 }
182 }

210 std::array<Real, 3> Copysphere::fieldSolverGetNormalDirection(
211 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
212 cint i,
213 cint j,
214 cint k
215 ) {
216 std::array<Real, 3> normalDirection{{ 0.0, 0.0, 0.0 }};
217
218 static creal DIAG2 = 1.0 / sqrt(2.0);
219 static creal DIAG3 = 1.0 / sqrt(3.0);
220
221 creal dx = technicalGrid.DX;
222 creal dy = technicalGrid.DY;
223 creal dz = technicalGrid.DZ;
224 const std::array<FsGridTools::FsIndex_t, 3> globalIndices = technicalGrid.getGlobalIndices(i,j,k);
225 creal x = P::xmin + (convert<Real>(globalIndices[0])+0.5)*dx;
226 creal y = P::ymin + (convert<Real>(globalIndices[1])+0.5)*dy;
227 creal z = P::zmin + (convert<Real>(globalIndices[2])+0.5)*dz;
228 creal xsign = divideIfNonZero(x, fabs(x));
229 creal ysign = divideIfNonZero(y, fabs(y));
230 creal zsign = divideIfNonZero(z, fabs(z));
231
232 Real length = 0.0;
233
234 if (Parameters::xcells_ini == 1) {
235 if (Parameters::ycells_ini == 1) {
236 if (Parameters::zcells_ini == 1) {
237 // X,Y,Z
238 abort_mpi("What do you expect to do with a single-cell simulation of copysphere boundary type? Stop kidding.", 1);
239 // end of X,Y,Z
240 } else {
241 // X,Y
242 normalDirection[2] = zsign;
243 // end of X,Y
244 }
245 } else if (Parameters::zcells_ini == 1) {
246 // X,Z
247 normalDirection[1] = ysign;
248 // end of X,Z
249 } else {
250 // X
251 switch(this->geometry) {
252 case 0:
253 normalDirection[1] = DIAG2*ysign;
254 normalDirection[2] = DIAG2*zsign;
255 break;
256 case 1:
257 if(fabs(y) == fabs(z)) {
258 normalDirection[1] = ysign*DIAG2;
259 normalDirection[2] = zsign*DIAG2;
260 break;
261 }
262 if(fabs(y) > (this->radius - dy)) {
263 normalDirection[1] = ysign;
264 break;
265 }
266 if(fabs(z) > (this->radius - dz)) {
267 normalDirection[2] = zsign;
268 break;
269 }
270 if(fabs(y) > (this->radius - 2.0*dy)) {
271 normalDirection[1] = ysign;
272 break;
273 }
274 if(fabs(z) > (this->radius - 2.0*dz)) {
275 normalDirection[2] = zsign;
276 break;
277 }
278 break;
279 case 2:
280 length = sqrt(y*y + z*z);
281 normalDirection[1] = y / length;
282 normalDirection[2] = z / length;
283 break;
284 default:
285 std::cerr << __FILE__ << ":" << __LINE__ << ":" << "copysphere.geometry has to be 0, 1 or 2 with this grid shape." << std::endl;
286 abort();
287 }
288 // end of X
289 }
290 } else if (Parameters::ycells_ini == 1) {
291 if (Parameters::zcells_ini == 1) {
292 // Y,Z
293 normalDirection[0] = xsign;
294 // end of Y,Z
295 } else {
296 // Y
297 switch(this->geometry) {
298 case 0:
299 normalDirection[0] = DIAG2*xsign;
300 normalDirection[2] = DIAG2*zsign;
301 break;
302 case 1:
303 if(fabs(x) == fabs(z)) {
304 normalDirection[0] = xsign*DIAG2;
305 normalDirection[2] = zsign*DIAG2;
306 break;
307 }
308 if(fabs(x) > (this->radius - dx)) {
309 normalDirection[0] = xsign;
310 break;
311 }
312 if(fabs(z) > (this->radius - dz)) {
313 normalDirection[2] = zsign;
314 break;
315 }
316 if(fabs(x) > (this->radius - 2.0*dx)) {
317 normalDirection[0] = xsign;
318 break;
319 }
320 if(fabs(z) > (this->radius - 2.0*dz)) {
321 normalDirection[2] = zsign;
322 break;
323 }
324 break;
325 case 2:
326 case 3:
327 length = sqrt(x*x + z*z);
328 normalDirection[0] = x / length;
329 normalDirection[2] = z / length;
330 break;
331 default:
332 std::cerr << __FILE__ << ":" << __LINE__ << ":" << "copysphere.geometry has to be 0, 1, 2 or 3 with this grid shape." << std::endl;
333 abort();
334 }
335 // end of Y
336 }
337 } else if (Parameters::zcells_ini == 1) {
338 // Z
339 switch(this->geometry) {
340 case 0:
341 normalDirection[0] = DIAG2*xsign;
342 normalDirection[1] = DIAG2*ysign;
343 break;
344 case 1:
345 if(fabs(x) == fabs(y)) {
346 normalDirection[0] = xsign*DIAG2;
347 normalDirection[1] = ysign*DIAG2;
348 break;
349 }
350 if(fabs(x) > (this->radius - dx)) {
351 normalDirection[0] = xsign;
352 break;
353 }
354 if(fabs(y) > (this->radius - dy)) {
355 normalDirection[1] = ysign;
356 break;
357 }
358 if(fabs(x) > (this->radius - 2.0*dx)) {
359 normalDirection[0] = xsign;
360 break;
361 }
362 if(fabs(y) > (this->radius - 2.0*dy)) {
363 normalDirection[1] = ysign;
364 break;
365 }
366 break;
367 case 2:
368 length = sqrt(x*x + y*y);
369 normalDirection[0] = x / length;
370 normalDirection[1] = y / length;
371 break;
372 default:
373 abort_mpi("copysphere.geometry has to be 0, 1 or 2 with this grid shape.", 1);
374 }
375 // end of Z
376 } else {
377 // 3D
378 switch(this->geometry) {
379 case 0:
380 normalDirection[0] = DIAG3*xsign;
381 normalDirection[1] = DIAG3*ysign;
382 normalDirection[2] = DIAG3*zsign;
383 break;
384 case 1:
385 if(fabs(x) == fabs(y) && fabs(x) == fabs(z) && fabs(x) > this->radius - dx) {
386 normalDirection[0] = xsign*DIAG3;
387 normalDirection[1] = ysign*DIAG3;
388 normalDirection[2] = zsign*DIAG3;
389 break;
390 }
391 if(fabs(x) == fabs(y) && fabs(x) == fabs(z) && fabs(x) > this->radius - 2.0*dx) {
392 normalDirection[0] = xsign*DIAG3;
393 normalDirection[1] = ysign*DIAG3;
394 normalDirection[2] = zsign*DIAG3;
395 break;
396 }
397 if(fabs(x) == fabs(y) && fabs(x) > this->radius - dx && fabs(z) < this->radius - dz) {
398 normalDirection[0] = xsign*DIAG2;
399 normalDirection[1] = ysign*DIAG2;
400 normalDirection[2] = 0.0;
401 break;
402 }
403 if(fabs(y) == fabs(z) && fabs(y) > this->radius - dy && fabs(x) < this->radius - dx) {
404 normalDirection[0] = 0.0;
405 normalDirection[1] = ysign*DIAG2;
406 normalDirection[2] = zsign*DIAG2;
407 break;
408 }
409 if(fabs(x) == fabs(z) && fabs(x) > this->radius - dx && fabs(y) < this->radius - dy) {
410 normalDirection[0] = xsign*DIAG2;
411 normalDirection[1] = 0.0;
412 normalDirection[2] = zsign*DIAG2;
413 break;
414 }
415 if(fabs(x) == fabs(y) && fabs(x) > this->radius - 2.0*dx && fabs(z) < this->radius - 2.0*dz) {
416 normalDirection[0] = xsign*DIAG2;
417 normalDirection[1] = ysign*DIAG2;
418 normalDirection[2] = 0.0;
419 break;
420 }
421 if(fabs(y) == fabs(z) && fabs(y) > this->radius - 2.0*dy && fabs(x) < this->radius - 2.0*dx) {
422 normalDirection[0] = 0.0;
423 normalDirection[1] = ysign*DIAG2;
424 normalDirection[2] = zsign*DIAG2;
425 break;
426 }
427 if(fabs(x) == fabs(z) && fabs(x) > this->radius - 2.0*dx && fabs(y) < this->radius - 2.0*dy) {
428 normalDirection[0] = xsign*DIAG2;
429 normalDirection[1] = 0.0;
430 normalDirection[2] = zsign*DIAG2;
431 break;
432 }
433 if(fabs(x) > (this->radius - dx)) {
434 normalDirection[0] = xsign;
435 break;
436 }
437 if(fabs(y) > (this->radius - dy)) {
438 normalDirection[1] = ysign;
439 break;
440 }
441 if(fabs(z) > (this->radius - dz)) {
442 normalDirection[2] = zsign;
443 break;
444 }
445 if(fabs(x) > (this->radius - 2.0*dx)) {
446 normalDirection[0] = xsign;
447 break;
448 }
449 if(fabs(y) > (this->radius - 2.0*dy)) {
450 normalDirection[1] = ysign;
451 break;
452 }
453 if(fabs(z) > (this->radius - 2.0*dz)) {
454 normalDirection[2] = zsign;
455 break;
456 }
457 break;
458 case 2:
459 length = sqrt(x*x + y*y + z*z);
460 normalDirection[0] = x / length;
461 normalDirection[1] = y / length;
462 normalDirection[2] = z / length;
463 break;
464 case 3:
465 length = sqrt(x*x + z*z);
466 normalDirection[0] = x / length;
467 normalDirection[2] = z / length;
468 break;
469 default:
470 abort_mpi("copysphere.geometry has to be 0, 1, 2 or 3 with this grid shape.", 1);
471 }
472 // end of 3D
473 }
474 return normalDirection;
475 }

483 Real Copysphere::fieldSolverBoundaryCondMagneticField(
484 FsGrid< std::array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & bGrid,
485 FsGrid< std::array<Real, fsgrids::bgbfield::N_BGB>, FS_STENCIL_WIDTH> & bgbGrid,
486 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
487 cint i,
488 cint j,
489 cint k,
490 creal dt,
491 cuint component
492 ) {
493 if(this->zeroPerB == true){
494 return bGrid.get(i,j,k)->at(fsgrids::bfield::PERBX+component);
495 } else {
496 if (technicalGrid.get(i,j,k)->sysBoundaryLayer == 1) {
497 switch(component) {
498 case 0:
499 if ( ((technicalGrid.get(i-1,j,k)->SOLVE & compute::BX) == compute::BX)
500 && ((technicalGrid.get(i+1,j,k)->SOLVE & compute::BX) == compute::BX)
501 ) {
502 return 0.5 * (bGrid.get(i-1,j,k)->at(fsgrids::bfield::PERBX) + bGrid.get(i+1,j,k)->at(fsgrids::bfield::PERBX));
503 } else if ((technicalGrid.get(i-1,j,k)->SOLVE & compute::BX) == compute::BX) {
504 return bGrid.get(i-1,j,k)->at(fsgrids::bfield::PERBX);
505 } else if ((technicalGrid.get(i+1,j,k)->SOLVE & compute::BX) == compute::BX) {
506 return bGrid.get(i+1,j,k)->at(fsgrids::bfield::PERBX);
507 } else {
508 Real retval = 0.0;
509 uint nCells = 0;
510 if ((technicalGrid.get(i,j-1,k)->SOLVE & compute::BX) == compute::BX) {
511 retval += bGrid.get(i,j-1,k)->at(fsgrids::bfield::PERBX);
512 nCells++;
513 }
514 if ((technicalGrid.get(i,j+1,k)->SOLVE & compute::BX) == compute::BX) {
515 retval += bGrid.get(i,j+1,k)->at(fsgrids::bfield::PERBX);
516 nCells++;
517 }
518 if ((technicalGrid.get(i,j,k-1)->SOLVE & compute::BX) == compute::BX) {
519 retval += bGrid.get(i,j,k-1)->at(fsgrids::bfield::PERBX);
520 nCells++;
521 }
522 if ((technicalGrid.get(i,j,k+1)->SOLVE & compute::BX) == compute::BX) {
523 retval += bGrid.get(i,j,k+1)->at(fsgrids::bfield::PERBX);
524 nCells++;
525 }
526 if (nCells == 0) {
527 for (int a=i-1; a<i+2; a++) {
528 for (int b=j-1; b<j+2; b++) {
529 for (int c=k-1; c<k+2; c++) {
530 if ((technicalGrid.get(a,b,c)->SOLVE & compute::BX) == compute::BX) {
531 retval += bGrid.get(a,b,c)->at(fsgrids::bfield::PERBX);
532 nCells++;
533 }
534 }
535 }
536 }
537 }
538 if (nCells == 0) {
539 cerr << __FILE__ << ":" << __LINE__ << ": ERROR: this should not have fallen through." << endl;
540 return 0.0;
541 }
542 return retval / nCells;
543 }
544 case 1:
545 if ( (technicalGrid.get(i,j-1,k)->SOLVE & compute::BY) == compute::BY
546 && (technicalGrid.get(i,j+1,k)->SOLVE & compute::BY) == compute::BY
547 ) {
548 return 0.5 * (bGrid.get(i,j-1,k)->at(fsgrids::bfield::PERBY) + bGrid.get(i,j+1,k)->at(fsgrids::bfield::PERBY));
549 } else if ((technicalGrid.get(i,j-1,k)->SOLVE & compute::BY) == compute::BY) {
550 return bGrid.get(i,j-1,k)->at(fsgrids::bfield::PERBY);
551 } else if ((technicalGrid.get(i,j+1,k)->SOLVE & compute::BY) == compute::BY) {
552 return bGrid.get(i,j+1,k)->at(fsgrids::bfield::PERBY);
553 } else {
554 Real retval = 0.0;
555 uint nCells = 0;
556 if ((technicalGrid.get(i-1,j,k)->SOLVE & compute::BY) == compute::BY) {
557 retval += bGrid.get(i-1,j,k)->at(fsgrids::bfield::PERBY);
558 nCells++;
559 }
560 if ((technicalGrid.get(i+1,j,k)->SOLVE & compute::BY) == compute::BY) {
561 retval += bGrid.get(i+1,j,k)->at(fsgrids::bfield::PERBY);
562 nCells++;
563 }
564 if ((technicalGrid.get(i,j,k-1)->SOLVE & compute::BY) == compute::BY) {
565 retval += bGrid.get(i,j,k-1)->at(fsgrids::bfield::PERBY);
566 nCells++;
567 }
568 if ((technicalGrid.get(i,j,k+1)->SOLVE & compute::BY) == compute::BY) {
569 retval += bGrid.get(i,j,k+1)->at(fsgrids::bfield::PERBY);
570 nCells++;
571 }
572 if (nCells == 0) {
573 for (int a=i-1; a<i+2; a++) {
574 for (int b=j-1; b<j+2; b++) {
575 for (int c=k-1; c<k+2; c++) {
576 if ((technicalGrid.get(a,b,c)->SOLVE & compute::BY) == compute::BY) {
577 retval += bGrid.get(a,b,c)->at(fsgrids::bfield::PERBY);
578 nCells++;
579 }
580 }
581 }
582 }

584 if (nCells == 0) {
585 cerr << __FILE__ << ":" << __LINE__ << ": ERROR: this should not have fallen through." << endl;
586 return 0.0;
587 }

589 }
590 case 2:
591 if ( (technicalGrid.get(i,j,k-1)->SOLVE & compute::BZ) == compute::BZ

597 } else if ((technicalGrid.get(i,j,k+1)->SOLVE & compute::BZ) == compute::BZ) {
598 return bGrid.get(i,j,k+1)->at(fsgrids::bfield::PERBZ);
599 } else {
600 Real retval = 0.0;
601 uint nCells = 0;
602 if ((technicalGrid.get(i-1,j,k)->SOLVE & compute::BZ) == compute::BZ) {
603 retval += bGrid.get(i-1,j,k)->at(fsgrids::bfield::PERBZ);
604 nCells++;
605 }
606 if ((technicalGrid.get(i+1,j,k)->SOLVE & compute::BZ) == compute::BZ) {
607 retval += bGrid.get(i+1,j,k)->at(fsgrids::bfield::PERBZ);
608 nCells++;
609 }
610 if ((technicalGrid.get(i,j-1,k)->SOLVE & compute::BZ) == compute::BZ) {
611 retval += bGrid.get(i,j-1,k)->at(fsgrids::bfield::PERBZ);
612 nCells++;
613 }
614 if ((technicalGrid.get(i,j+1,k)->SOLVE & compute::BZ) == compute::BZ) {
615 retval += bGrid.get(i,j+1,k)->at(fsgrids::bfield::PERBZ);
616 nCells++;
617 }
618 if (nCells == 0) {
619 for (int a=i-1; a<i+2; a++) {
620 for (int b=j-1; b<j+2; b++) {
621 for (int c=k-1; c<k+2; c++) {
622 if ((technicalGrid.get(a,b,c)->SOLVE & compute::BZ) == compute::BZ) {
623 retval += bGrid.get(a,b,c)->at(fsgrids::bfield::PERBZ);
624 nCells++;
625 }
626 }
627 }
628 }
629 }
630 if (nCells == 0) {
631 cerr << __FILE__ << ":" << __LINE__ << ": ERROR: this should not have fallen through." << endl;
632 return 0.0;
633 }
634 return retval / nCells;
635 }
636 default:
637 cerr << "ERROR: copysphere boundary tried to copy nonsensical magnetic field component " << component << endl;
638 return 0.0;
639 }
640 } else { // L2 cells
641 Real retval = 0.0;
642 uint nCells = 0;
643 for (int a=i-1; a<i+2; a++) {
644 for (int b=j-1; b<j+2; b++) {
645 for (int c=k-1; c<k+2; c++) {
646 if (technicalGrid.get(a,b,c)->sysBoundaryLayer == 1) {
647 retval += bGrid.get(a,b,c)->at(fsgrids::bfield::PERBX + component);
648 nCells++;
649 }
650 }
651 }
652 }
653 if (nCells == 0) {
654 cerr << __FILE__ << ":" << __LINE__ << ": ERROR: this should not have fallen through." << endl;
655 return 0.0;
656 }
657 return retval / nCells;
658 }

753 // WARNING not 0.0 here or the dipole() function fails miserably.
754 templateCell.sysBoundaryFlag = this->getIndex();
755 templateCell.sysBoundaryLayer = 1;
756 templateCell.parameters[CellParams::XCRD] = 1.0;
757 templateCell.parameters[CellParams::YCRD] = 1.0;

761 templateCell.parameters[CellParams::DZ] = 1;
762
763 Real initRho, initT, initV0X, initV0Y, initV0Z;
764 // Loop over particle species
765 for (uint popID=0; popID<getObjectWrapper().particleSpecies.size(); ++popID) {
766 templateCell.clear(popID,false); //clear, do not de-allocate memory
767 const CopysphereSpeciesParameters& sP = this->speciesParams[popID];
768 const Real mass = getObjectWrapper().particleSpecies[popID].mass;
769 initRho = sP.rho;
770 initT = sP.T;
771 initV0X = sP.V0[0];
772 initV0Y = sP.V0[1];
773 initV0Z = sP.V0[2];
774
775 // Find list of blocks to initialize.
776 const uint nRequested = SBC::findMaxwellianBlocksToInitialize(popID,templateCell, initRho, initT, initV0X, initV0Y, initV0Z);
777 // stores in vmesh->getGrid() (localToGlobalMap)
