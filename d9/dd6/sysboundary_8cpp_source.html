45bool precedenceSort(const SBC::SysBoundaryCondition* first, const SBC::SysBoundaryCondition* second) {
46 if (first->getPrecedence() < second->getPrecedence()) {
47 return true;
48 } else {
49 return false;
50 }

124void SysBoundary::addSysBoundary(SBC::SysBoundaryCondition* bc, Project& project, creal& t) {
125 // Initialize the boundary condition
126 stringstream timername;
127 timername<<"Initialize system boundary condition "<<bc->getName();
128 phiprof::Timer timer {timername.str()};
129 bc->initSysBoundary(t, project);
130 timer.stop();
131
132 sysBoundaries.push_back(bc);
133 if (sysBoundaries.size() > 1) {
134 sysBoundaries.sort(precedenceSort);
135 }
136
137 // This assumes that only one instance of each type is created.
138 indexToSysBoundary[bc->getIndex()] = bc;
139}

153void SysBoundary::initSysBoundaries(Project& project, creal& t) {
154 int myRank;
155 MPI_Comm_rank(MPI_COMM_WORLD, &myRank);
156 vector<string>::const_iterator it;
157
158 if (sysBoundaryCondList.size() == 0) {
159 if (!periodic[0] && !Readparameters::helpRequested) {
160 abort_mpi("Non-periodic in x but no boundary condtion loaded!");
161 }
162 if (!periodic[1] && !Readparameters::helpRequested) {
163 abort_mpi("Non-periodic in y but no boundary condtion loaded!");
164 }
165 if (!periodic[2] && !Readparameters::helpRequested) {
166 abort_mpi("Non-periodic in z but no boundary condtion loaded!");
167 }
168 }
169
170 for (it = sysBoundaryCondList.begin(); it != sysBoundaryCondList.end(); it++) {
171 if (*it == "Outflow" || *it == "outflow") {
172 this->addSysBoundary(::new SBC::Outflow, project, t);
173
174 anyDynamic = anyDynamic | this->getSysBoundary(sysboundarytype::OUTFLOW)->isDynamic();
175 bool faces[6];
176 this->getSysBoundary(sysboundarytype::OUTFLOW)->getFaces(&faces[0]);
177
178 if ((faces[0] || faces[1]) && periodic[0]) {
179 abort_mpi("Conflict: x boundaries set to periodic but found Outflow conditions!");
180 }
181
182 if ((faces[2] || faces[3]) && periodic[1]) {
183 abort_mpi("Conflict: y boundaries set to periodic but found Outflow conditions!");
184 }
185
186 if ((faces[4] || faces[5]) && periodic[2]) {
187 abort_mpi("Conflict: z boundaries set to periodic but found Outflow conditions!");
188 }
189 if ((faces[0] || faces[1]) && P::xcells_ini < 5) {
190 abort_mpi("Outflow condition loaded on x- or x+ face but not enough cells in x!");
191 }
192 if ((faces[2] || faces[3]) && P::ycells_ini < 5) {
193 abort_mpi("Outflow condition loaded on y- or y+ face but not enough cells in y!");
194 }
195
196 if ((faces[4] || faces[5]) && P::zcells_ini < 5) {
197 abort_mpi("Outflow condition loaded on z- or z+ face but not enough cells in z!");
198 }
199
200 } else if (*it == "Ionosphere" || *it == "ionosphere") {
201 this->addSysBoundary(::new SBC::Ionosphere, project, t);
202 this->addSysBoundary(::new SBC::DoNotCompute, project, t);
203 anyDynamic = anyDynamic | this->getSysBoundary(sysboundarytype::IONOSPHERE)->isDynamic();
204 } else if(*it == "Copysphere" || *it == "copysphere") {
205 this->addSysBoundary(::new SBC::Copysphere, project, t);
206 this->addSysBoundary(::new SBC::DoNotCompute, project, t);
207 anyDynamic = anyDynamic | this->getSysBoundary(sysboundarytype::COPYSPHERE)->isDynamic();
208 } else if (*it == "Maxwellian" || *it == "maxwellian") {
209 this->addSysBoundary(::new SBC::Maxwellian, project, t);
210 anyDynamic = anyDynamic | this->getSysBoundary(sysboundarytype::MAXWELLIAN)->isDynamic();
211 bool faces[6];
212 this->getSysBoundary(sysboundarytype::MAXWELLIAN)->getFaces(&faces[0]);
213 if ((faces[0] || faces[1]) && periodic[0]) {
214 abort_mpi("Conflict: x boundaries set to periodic but found Maxwellian also!");
215 }
216 if ((faces[2] || faces[3]) && periodic[1]) {
217 abort_mpi("Conflict: y boundaries set to periodic but found Maxwellian also!");
218 }
219 if ((faces[4] || faces[5]) && periodic[2]) {
220 abort_mpi("Conflict: z boundaries set to periodic but found Maxwellian also!");
221 }
222 if ((faces[0] || faces[1]) && P::xcells_ini < 5) {
223 abort_mpi("Maxwellian condition loaded on x- or x+ face but not enough cells in x!");
224 }
225 if ((faces[2] || faces[3]) && P::ycells_ini < 5) {
226 abort_mpi("Maxwellian condition loaded on y- or y+ face but not enough cells in y!");
227 }
228 if ((faces[4] || faces[5]) && P::zcells_ini < 5) {
229 abort_mpi("Maxwellian condition loaded on z- or z+ face but not enough cells in z!");
230 }
231 } else {
232 std::ostringstream msg;
233 msg << "Unknown type of boundary read: " << *it;
234 abort_mpi(msg.str());
235 }
236 }
237
238 for (auto& b : sysBoundaries) {
239 b->setPeriodicity(periodic);
240 }
241}

262void SysBoundary::checkRefinement(dccrg::Dccrg<spatial_cell::SpatialCell, dccrg::Cartesian_Geometry>& mpiGrid) {
263 // Verifies that all cells within Neighborhoods::FULL of L1 boundary cells are on the same refinement
264 // level (one group for inner boundary, another for outer boundary)
265
266 // Set is used to avoid storing duplicates - each cell only needs to be checked once
267 set<CellID> innerBoundaryCells;
268 set<CellID> outerBoundaryCells;
269
270 int innerBoundaryRefLvl = -1;
271 int outerBoundaryRefLvl = -1;
272
273 const vector<CellID>& local_cells = getLocalCells();
274 // Collect cells by sysboundarytype
275 for (auto cellId : local_cells) {
276 SpatialCell* cell = mpiGrid[cellId];
277 if (cell) {
278 if (cell->sysBoundaryFlag == sysboundarytype::IONOSPHERE || cell->sysBoundaryFlag == sysboundarytype::COPYSPHERE) {
279 innerBoundaryCells.insert(cellId);
280 innerBoundaryRefLvl = mpiGrid.get_refinement_level(cellId);
281 if (cell->sysBoundaryLayer == 1) {
282 // Add all stencil neighbors of layer 1 cells
283 auto* nbrPairVector = mpiGrid.get_neighbors_of(cellId, Neighborhoods::SYSBOUNDARIES);
284 for (auto nbrPair : *nbrPairVector) {
285 if (nbrPair.first != INVALID_CELLID) {
286 innerBoundaryCells.insert(nbrPair.first);
287 }
288 }
289 }
290 } else if (cell->sysBoundaryFlag != sysboundarytype::NOT_SYSBOUNDARY &&
291 cell->sysBoundaryFlag != sysboundarytype::DO_NOT_COMPUTE) {
292 outerBoundaryCells.insert(cellId);
293 outerBoundaryRefLvl = mpiGrid.get_refinement_level(cellId);
294 // Add all stencil neighbors of outer boundary cells
295 auto* nbrPairVector = mpiGrid.get_neighbors_of(cellId, Neighborhoods::SYSBOUNDARIES);
296 for (auto nbrPair : *nbrPairVector) {
297 if (nbrPair.first != INVALID_CELLID) {
298 outerBoundaryCells.insert(nbrPair.first);
299 }
300 }
301 }
302 }
303 }
304
305 for (auto cellId : innerBoundaryCells) {
306 if (cellId != INVALID_CELLID && mpiGrid.get_refinement_level(cellId) != innerBoundaryRefLvl) {
307 abort_mpi("ERROR: inner boundary cells must have identical refinement level!");
308 }
309 }
310
311 for (auto cellId : outerBoundaryCells) {
312 if (cellId != INVALID_CELLID && mpiGrid.get_refinement_level(cellId) != outerBoundaryRefLvl) {
313 abort_mpi("ERROR: outer boundary cells must have identical refinement level!");
314 }
315 }
316}

318bool belongsToLayer(const int layer, const int x, const int y, const int z,
319 FsGrid<fsgrids::technical, FS_STENCIL_WIDTH>& technicalGrid) {
320
321 bool belongs = false;
322
323 // loop through all neighbors (including diagonals)
324 for (int iz = -1; iz <= 1; ++iz) {
325 for (int iy = -1; iy <= 1; ++iy) {
326 for (int ix = -1; ix <= 1; ++ix) {
327
328 // not strictly necessary but logically we should not consider the cell itself
329 // among its neighbors.
330 if ((ix == 0 && iy == 0 && iz == 0) || !technicalGrid.get(x + ix, y + iy, z + iz)) {
331 continue;
332 }
333
334 if (layer == 1 && technicalGrid.get(x + ix, y + iy, z + iz)->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
335 // in the first layer, boundary cell belongs if it has a non-boundary neighbor
336 belongs = true;
337 return belongs;
338
339 } else if (layer > 1 && technicalGrid.get(x + ix, y + iy, z + iz)->sysBoundaryLayer == layer - 1) {
340 // in all other layers, boundary cell belongs if it has a neighbor in the previous layer
341 belongs = true;
342 return belongs;
343 }
344 }
345 }
346 }
347
348 return belongs;
349}

358void SysBoundary::classifyCells(dccrg::Dccrg<spatial_cell::SpatialCell, dccrg::Cartesian_Geometry>& mpiGrid,
359 FsGrid<fsgrids::technical, FS_STENCIL_WIDTH>& technicalGrid) {
360 const vector<CellID>& cells = getLocalCells();
361 auto localSize = technicalGrid.getLocalSize().data();
362
363 /*set all cells to default value, not_sysboundary */
364#pragma omp parallel for
365 for (uint i = 0; i < cells.size(); i++) {
366 mpiGrid[cells[i]]->sysBoundaryFlag = sysboundarytype::NOT_SYSBOUNDARY;
367 }
368#pragma omp parallel for collapse(2)
369 for (FsGridTools::FsIndex_t z = 0; z < localSize[2]; ++z) {
370 for (FsGridTools::FsIndex_t y = 0; y < localSize[1]; ++y) {
371 for (FsGridTools::FsIndex_t x = 0; x < localSize[0]; ++x) {
372 //technicalGrid.get(x, y, z)->sysBoundaryFlag = sysboundarytype::NOT_SYSBOUNDARY;
373 // Here for debugging since boundarytype should be fed from MPIGrid
374 technicalGrid.get(x, y, z)->sysBoundaryFlag = sysboundarytype::N_SYSBOUNDARY_CONDITIONS;
375 technicalGrid.get(x, y, z)->sysBoundaryLayer = 0;
376 // Function called on every refinement, we only want to reset dt on simulation start
377 if (P::tstep == P::tstep_min) {
378 technicalGrid.get(x, y, z)->maxFsDt = numeric_limits<Real>::max();
379 }
380 // Set the fsgrid rank in the technical grid
381 technicalGrid.get(x, y, z)->fsGridRank = technicalGrid.getRank();
382 }
383 }
384 }
385
386 /*
387 loop through sysboundaries and let all sysboundaries set in local
388 cells if they are part of which sysboundary (cell location needs to
389 be updated by now. No remote data needed/available, so assignement
390 has to be based individually on each cells location
391 */
392 for(auto& b : sysBoundaries) {
393 b->assignSysBoundary(mpiGrid, technicalGrid);
394 }
395
396 SpatialCell::set_mpi_transfer_type(Transfer::CELL_SYSBOUNDARYFLAG);
397 mpiGrid.update_copies_of_remote_neighbors(Neighborhoods::SYSBOUNDARIES);
398
399 feedBoundaryIntoFsGrid(mpiGrid, cells, technicalGrid);
400
401 // set distance 1 cells to boundary cells, that have neighbors which are normal cells
402 for (CellID cell : cells) {
403 mpiGrid[cell]->sysBoundaryLayer = 0; /*Initial value*/
404
405 std::array<double, 3> dx = mpiGrid.geometry.get_length(cell);
406 std::array<double, 3> x = mpiGrid.get_center(cell);
407 if (!isPeriodic(0) && (x[0] > Parameters::xmax - dx[0] || x[0] < Parameters::xmin + dx[0])) {
408 continue;
409 } else if (!isPeriodic(1) && (x[1] > Parameters::ymax - dx[1] || x[1] < Parameters::ymin + dx[1])) {
410 continue;
411 } else if (!isPeriodic(2) && (x[2] > Parameters::zmax - dx[2] || x[2] < Parameters::zmin + dx[2])) {
412 continue;
413 }
414
415 if (mpiGrid[cell]->sysBoundaryFlag != sysboundarytype::NOT_SYSBOUNDARY) {
416 // Cornerwise neighbor, i.e. cell must be in both neighbors_of and neighbors_to
417 for (auto i : *mpiGrid.get_neighbors_of(cell, Neighborhoods::SYSBOUNDARIES)) {
418 CellID neighbor = i.first;
419 if (neighbor && mpiGrid[neighbor]->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
420 for (auto j : *mpiGrid.get_neighbors_to(cell, Neighborhoods::SYSBOUNDARIES)) {
421 if (j.first == neighbor) {
422 mpiGrid[cell]->sysBoundaryLayer = 1;
423 }
424 }
425 }
426 }
427 }
428 }
429
430 /*communicate which cells have Layer 1 set above for local cells (sysBoundaryFlag
431 * and sysBoundaryLayer communicated)*/
432 SpatialCell::set_mpi_transfer_type(Transfer::CELL_SYSBOUNDARYFLAG);
433
434 mpiGrid.update_copies_of_remote_neighbors(Neighborhoods::SYSBOUNDARIES);
435
436 /*Compute distances*/
437 uint maxLayers = 3; // max(max(P::xcells_ini, P::ycells_ini), P::zcells_ini);
438 for (uint layer = 1; layer < maxLayers; layer++) {
439 for (CellID cell : cells) {
440 if (mpiGrid[cell]->sysBoundaryLayer == 0) {
441 // Note: this distance calculation will be non-plateau monotonic only assuming that
442 // SysBoundary::checkRefinement has been applied correctly and there are no refinement
443 // level changes within Neighborhoods::SYSBOUNDARIES.
444 for (auto i : *mpiGrid.get_neighbors_of(cell, Neighborhoods::SYSBOUNDARIES)) {
445 CellID neighbor = i.first;
446 if (neighbor && mpiGrid[neighbor]->sysBoundaryLayer == layer) {
447 for (auto j : *mpiGrid.get_neighbors_to(cell, Neighborhoods::SYSBOUNDARIES)) {
448 if (j.first == neighbor) {
449 mpiGrid[cell]->sysBoundaryLayer = layer + 1;
450 }
451 }
452 }
453 }
454 }
455 }
456
457 SpatialCell::set_mpi_transfer_type(Transfer::CELL_SYSBOUNDARYFLAG);
458 mpiGrid.update_copies_of_remote_neighbors(Neighborhoods::SYSBOUNDARIES);
459 }
460
461 /*set cells to DO_NOT_COMPUTE if they are on boundary, and are not
462 * in the first two layers of the boundary*/
463 for (uint i = 0; i < cells.size(); i++) {
464 if (mpiGrid[cells[i]]->sysBoundaryFlag != sysboundarytype::NOT_SYSBOUNDARY &&
465 mpiGrid[cells[i]]->sysBoundaryLayer != 1 && mpiGrid[cells[i]]->sysBoundaryLayer != 2) {
466 mpiGrid[cells[i]]->sysBoundaryFlag = sysboundarytype::DO_NOT_COMPUTE;
467 }
468 }
469
470 // The following is done so that everyone knows their neighbour's
471 // layer flags. This is needed for the correct use of the system
472 // boundary local communication patterns.
473 SpatialCell::set_mpi_transfer_type(Transfer::CELL_SYSBOUNDARYFLAG);
474 mpiGrid.update_copies_of_remote_neighbors(Neighborhoods::FULL);
475
476 SysBoundary& sysBoundaryContainer = getObjectWrapper().sysBoundaryContainer;
477 Real ionosphereDownmapRadius = 0;
478 if (sysBoundaryContainer.existSysBoundary("Ionosphere")) {
479 Readparameters::get("ionosphere.downmapRadius", ionosphereDownmapRadius);
480 }
481 if(ionosphereDownmapRadius < 1000) {
482 ionosphereDownmapRadius *= physicalconstants::R_E;
483 }
484
485 // Now the layers need to be set on fsgrid too
486 // In dccrg initialization the max number of boundary layers is set to 3.
487 const uint MAX_NUMBER_OF_BOUNDARY_LAYERS = 3 * pow(2, mpiGrid.get_maximum_refinement_level());
488
489 technicalGrid.updateGhostCells();
490
491 // loop through max number of layers
492 for (uint layer = 1; layer <= MAX_NUMBER_OF_BOUNDARY_LAYERS; ++layer) {
493
494// loop through all cells in grid
495#pragma omp parallel for collapse(2)
496 for (FsGridTools::FsIndex_t z = 0; z < localSize[2]; ++z) {
497 for (FsGridTools::FsIndex_t y = 0; y < localSize[1]; ++y) {
498 for (FsGridTools::FsIndex_t x = 0; x < localSize[0]; ++x) {
499
500 // for the first layer, consider all cells that belong to a boundary, for other layers
501 // consider all cells that have not yet been labeled.
502 if ((layer == 1 && technicalGrid.get(x, y, z)->sysBoundaryFlag != sysboundarytype::NOT_SYSBOUNDARY) ||
503 (layer > 1 && technicalGrid.get(x, y, z)->sysBoundaryLayer == 0)) {
504
505 if (belongsToLayer(layer, x, y, z, technicalGrid)) {
506
507 technicalGrid.get(x, y, z)->sysBoundaryLayer = layer;
508
509 if (layer > 2 && (technicalGrid.get(x,y,z)->sysBoundaryFlag == sysboundarytype::IONOSPHERE ||
510 technicalGrid.get(x,y,z)->sysBoundaryFlag == sysboundarytype::COPYSPHERE)) {
511 technicalGrid.get(x, y, z)->sysBoundaryFlag = sysboundarytype::DO_NOT_COMPUTE;
512 } else if (layer > 2 && technicalGrid.get(x, y, z)->sysBoundaryFlag != sysboundarytype::NOT_SYSBOUNDARY) {
513 technicalGrid.get(x, y, z)->sysBoundaryFlag = sysboundarytype::OUTER_BOUNDARY_PADDING;
514 }
515 }
516 }
517 }
518 }
519 }
520 // This needs an update every iteration as belongsToLayer() needs up to date data.
521 technicalGrid.updateGhostCells();
522 }
523
524// One more pass to make sure, in particular if the ionosphere is wide enough
525// there is remaining cells of IONOSPHERE type inside the max layers gone through previously.
526// This last pass now gets rid of them.
527#pragma omp parallel for collapse(2)
528 for (FsGridTools::FsIndex_t z = 0; z < localSize[2]; ++z) {
529 for (FsGridTools::FsIndex_t y = 0; y < localSize[1]; ++y) {
530 for (FsGridTools::FsIndex_t x = 0; x < localSize[0]; ++x) {
531 if (technicalGrid.get(x,y,z)->sysBoundaryLayer == 0 && (
532 technicalGrid.get(x,y,z)->sysBoundaryFlag == sysboundarytype::IONOSPHERE ||
533 technicalGrid.get(x,y,z)->sysBoundaryFlag == sysboundarytype::COPYSPHERE)) {
534 technicalGrid.get(x, y, z)->sysBoundaryFlag = sysboundarytype::DO_NOT_COMPUTE;
535 }
536 }
537 }
538 }
539
540 technicalGrid.updateGhostCells();
541
542 const array<FsGridTools::FsSize_t,3> fsGridDimensions = technicalGrid.getGlobalSize();
543
544 // One pass to setup the bit field to know which components the field solver should propagate.
545#pragma omp parallel for collapse(2)
546 for (FsGridTools::FsIndex_t z = 0; z < localSize[2]; ++z) {
547 for (FsGridTools::FsIndex_t y = 0; y < localSize[1]; ++y) {
548 for (FsGridTools::FsIndex_t x = 0; x < localSize[0]; ++x) {
549 technicalGrid.get(x, y, z)->SOLVE = 0;
550
551 array<FsGridTools::FsIndex_t, 3> globalIndices = technicalGrid.getGlobalIndices(x, y, z);
552
553 if (((globalIndices[0] == 0 || globalIndices[0] == (FsGridTools::FsIndex_t)fsGridDimensions[0] - 1) &&
554 !this->isPeriodic(0)) ||
555 ((globalIndices[1] == 0 || globalIndices[1] == (FsGridTools::FsIndex_t)fsGridDimensions[1] - 1) &&
556 !this->isPeriodic(1)) ||
557 ((globalIndices[2] == 0 || globalIndices[2] == (FsGridTools::FsIndex_t)fsGridDimensions[2] - 1) &&
558 !this->isPeriodic(2))) {
559 continue;
560 }
561 if (technicalGrid.get(x, y, z)->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
562 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::BX;
563 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::BY;
564 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::BZ;
565 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::EX;
566 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::EY;
567 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::EZ;
568 } else {
569 if (technicalGrid.get(x - 1, y, z)->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
570 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::BX;
571 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::EY;
572 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::EZ;
573 }
574 if (technicalGrid.get(x, y - 1, z)->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
575 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::BY;
576 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::EX;
577 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::EZ;
578 }
579 if (technicalGrid.get(x, y, z - 1)->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
580 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::BZ;
581 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::EX;
582 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::EY;
583 }
584 if (technicalGrid.get(x - 1, y - 1, z)->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
585 technicalGrid.get(x, y, z)->SOLVE = technicalGrid.get(x, y, z)->SOLVE | compute::EZ;
586 }
587 if (technicalGrid.get(x - 1, y, z - 1)->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {

674void SysBoundary::applySysBoundaryVlasovConditions(
675 dccrg::Dccrg<SpatialCell, dccrg::Cartesian_Geometry>& mpiGrid, creal& t,
676 const bool calculate_V_moments) {
677
678 if (sysBoundaries.size() == 0) {
679 return; // no system boundaries
680 }
681
682/*Transfer along boundaries*/
683// First the small stuff without overlapping in an extended neighbourhood:
684 SpatialCell::set_mpi_transfer_type(Transfer::CELL_PARAMETERS | Transfer::POP_METADATA | Transfer::CELL_SYSBOUNDARYFLAG, true);
685 mpiGrid.update_copies_of_remote_neighbors(Neighborhoods::SYSBOUNDARIES_EXTENDED);
686
687 // Loop over existing particle species
688 for (uint popID = 0; popID < getObjectWrapper().particleSpecies.size(); ++popID) {
689 SpatialCell::setCommunicatedSpecies(popID);
690 // update lists in neighborhood
691 updateRemoteVelocityBlockLists(mpiGrid, popID, Neighborhoods::SYSBOUNDARIES);
692
693 // Then the block data in the reduced neighbourhood:
694 phiprof::Timer commTimer {"Start comm of cell and block data", {"MPI"}};
695 SpatialCell::set_mpi_transfer_type(Transfer::VEL_BLOCK_DATA, true);
696 mpiGrid.start_remote_neighbor_copy_updates(Neighborhoods::SYSBOUNDARIES);
697 commTimer.stop();
698
699 phiprof::Timer computeInnerTimer {"Compute process inner cells"};
700 // Compute Vlasov boundary condition on system boundary/process inner cells
701 vector<CellID> localCells;
702 getBoundaryCellList(mpiGrid, mpiGrid.get_local_cells_not_on_process_boundary(Neighborhoods::SYSBOUNDARIES), localCells);
703
704#pragma omp parallel for
705 for (uint i = 0; i < localCells.size(); i++) {
706 cuint sysBoundaryType = mpiGrid[localCells[i]]->sysBoundaryFlag;
707 this->getSysBoundary(sysBoundaryType)->vlasovBoundaryCondition(mpiGrid, localCells[i], popID, calculate_V_moments);
708 }
709 if (popID==getObjectWrapper().particleSpecies.size()-1) {
710 // Only calculate moments when handling last population
711 if (calculate_V_moments) {
712 calculateMoments_V(mpiGrid, localCells, true);
713 } else {
714 calculateMoments_R(mpiGrid, localCells, true);
715 }
716 }
717 computeInnerTimer.stop();
718
719 phiprof::Timer waitimer {"Wait for receives", {"MPI", "Wait"}};
720 mpiGrid.wait_remote_neighbor_copy_updates(Neighborhoods::SYSBOUNDARIES);
721 waitimer.stop();
722
723 // Compute vlasov boundary on system boundary/process boundary cells
724 phiprof::Timer computeBoundaryTimer {"Compute process boundary cells"};
725 vector<CellID> boundaryCells;
726 getBoundaryCellList(mpiGrid, mpiGrid.get_local_cells_on_process_boundary(Neighborhoods::SYSBOUNDARIES),
727 boundaryCells);
728#pragma omp parallel for
729 for (uint i = 0; i < boundaryCells.size(); i++) {
730 cuint sysBoundaryType = mpiGrid[boundaryCells[i]]->sysBoundaryFlag;
731 this->getSysBoundary(sysBoundaryType)->vlasovBoundaryCondition(mpiGrid, boundaryCells[i], popID, calculate_V_moments);
732 }
733 if (popID==getObjectWrapper().particleSpecies.size()-1) {
734 // Only calculate moments when handling last population
735 if (calculate_V_moments) {
736 calculateMoments_V(mpiGrid, boundaryCells, true);
737 } else {
738 calculateMoments_R(mpiGrid, boundaryCells, true);
739 }
740 }
741 computeBoundaryTimer.stop();
742
743 // WARNING Blocks are changed but lists not updated now, if you need to use/communicate them before the next
744 // update is done, add an update here. reset lists in smaller default neighborhood
745 updateRemoteVelocityBlockLists(mpiGrid, popID);
746
747 } // for-loop over populations
748}
