796 std::array<Real, 3> otherB = getBVol(neighbor);

798

800 maxV = std::max(maxV,otherV);

816 if (maxB > EPS) {

823 alpha = std::max(alpha, dU * P::alphaDUWeight);
824 alpha = std::max(alpha, dPsq * P::alphaDPSqWeight);

873 for (const auto& pop : cell->get_populations()) {
