49template<typename ITEM> inline
50bool ItemStorage<ITEM>::get(const std::string& name,ITEM& item) const {
51 typename std::map<std::string,ITEM>::const_iterator it = warehouse.find(name);
52 if (it == warehouse.end()) return false;
53 item = it->second;
54 return true;
55}

62template<typename ITEM> inline
63bool ItemStorage<ITEM>::store(const std::string& name,ITEM item) {
64 // The insert returns a pair<iterator,bool>, where the boolean value is 'true'
65 // if the function was inserted to the map. Skip the pair creation and just return the boolean.
66 return warehouse.insert(make_pair(name,item)).second;
67}
