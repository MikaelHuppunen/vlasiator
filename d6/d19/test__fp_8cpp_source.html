95 //const speciesParameters& sP = this->speciesParams[popID];
96 // Fetch spatial cell center coordinates
97 const Real x = cell->parameters[CellParams::XCRD] + 0.5*cell->parameters[CellParams::DX];
98 const Real y = cell->parameters[CellParams::YCRD] + 0.5*cell->parameters[CellParams::DY];
99 const Real z = cell->parameters[CellParams::ZCRD] + 0.5*cell->parameters[CellParams::DZ];
100
101 const Real mass = getObjectWrapper().particleSpecies[popID].mass;
102 Real initRho = this->DENSITY;
103 Real initT = this->TEMPERATURE;
104
105 std::array<Real, 3> initV0 = this->getV0(x, y, z, popID)[0];
106 const Real initV0X = initV0[0];
107 const Real initV0Y = initV0[1];
108 const Real initV0Z = initV0[2];
109
110 #ifdef USE_GPU
111 vmesh::VelocityMesh *vmesh = cell->dev_get_velocity_mesh(popID);
112 vmesh::VelocityBlockContainer* VBC = cell->dev_get_velocity_blocks(popID);
113 #else
114 vmesh::VelocityMesh *vmesh = cell->get_velocity_mesh(popID);
115 vmesh::VelocityBlockContainer* VBC = cell->get_velocity_blocks(popID);
116 #endif
117 // Loop over blocks
118 Realf rhosum = 0;
119 arch::parallel_reduce<arch::null>(
120 {WID, WID, WID, nRequested},
121 ARCH_LOOP_LAMBDA (const uint i, const uint j, const uint k, const uint initIndex, Realf *lsum ) {
122 vmesh::GlobalID *GIDlist = vmesh->getGrid()->data();
123 Realf* bufferData = VBC->getData();
124 const vmesh::GlobalID blockGID = GIDlist[initIndex];
125 // Calculate parameters for new block
126 Real blockCoords[6];
127 vmesh->getBlockInfo(blockGID,&blockCoords[0]);
128 creal vxBlock = blockCoords[0];
129 creal vyBlock = blockCoords[1];
130 creal vzBlock = blockCoords[2];
131 creal dvxCell = blockCoords[3];
132 creal dvyCell = blockCoords[4];
133 creal dvzCell = blockCoords[5];
134 ARCH_INNER_BODY(i, j, k, initIndex, lsum) {
135 creal vx = vxBlock + (i+0.5)*dvxCell - initV0X;
136 creal vy = vyBlock + (j+0.5)*dvyCell - initV0Y;
137 creal vz = vzBlock + (k+0.5)*dvzCell - initV0Z;
138 const Realf value = MaxwellianPhaseSpaceDensity(vx,vy,vz,initT,initRho,mass);
139 bufferData[initIndex*WID3 + k*WID2 + j*WID + i] = value;
140 //lsum[0] += value;
141 };
142 }, rhosum);
143 return rhosum;
144 }

160 Real initRho = this->DENSITY;
161 Real initT = this->TEMPERATURE;
162
163 std::array<Real, 3> initV0 = this->getV0(x, y, z, popID)[0];
164 const Real initV0X = initV0[0];
165 const Real initV0Y = initV0[1];
166 const Real initV0Z = initV0[2];
167
168 creal vx = vx_in - initV0X;
169 creal vy = vy_in - initV0Y;
170 creal vz = vz_in - initV0Z;
171 const Realf value = MaxwellianPhaseSpaceDensity(vx,vy,vz,initT,initRho,mass);
172 return value;

175 void test_fp::setProjectBField(
176 FsGrid< std::array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & perBGrid,
177 FsGrid< std::array<Real, fsgrids::bgbfield::N_BGB>, FS_STENCIL_WIDTH> & BgBGrid,
178 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid
179 ) {
180 setBackgroundFieldToZero(BgBGrid);
181
182 if(!P::isRestart) {
183 auto localSize = perBGrid.getLocalSize().data();
184
185 creal dx = perBGrid.DX * 3.5;
186 creal dy = perBGrid.DY * 3.5;
187 creal dz = perBGrid.DZ * 3.5;
188
189 Real areaFactor = 1.0;
190
191 #pragma omp parallel for collapse(3)
192 for (FsGridTools::FsIndex_t i = 0; i < localSize[0]; ++i) {
193 for (FsGridTools::FsIndex_t j = 0; j < localSize[1]; ++j) {
194 for (FsGridTools::FsIndex_t k = 0; k < localSize[2]; ++k) {
195 const std::array<Real, 3> xyz = perBGrid.getPhysicalCoords(i, j, k);
196 std::array<Real, fsgrids::bfield::N_BFIELD>* cell = perBGrid.get(i, j, k);
197
198 creal x = xyz[0] + 0.5 * perBGrid.DX;
199 creal y = xyz[1] + 0.5 * perBGrid.DY;
200 creal z = xyz[2] + 0.5 * perBGrid.DZ;
201
202 switch (this->CASE) {
203 case BXCASE:
204 cell->at(fsgrids::bfield::PERBX) = 0.1 * this->B0 * areaFactor;
205 //areaFactor = (CellParams::DY * CellParams::DZ) / (dy * dz);
206 if (y >= -dy && y <= dy)
207 if (z >= -dz && z <= dz)
208 cell->at(fsgrids::bfield::PERBX) = this->B0 * areaFactor;
209 break;
210 case BYCASE:
211 cell->at(fsgrids::bfield::PERBY) = 0.1 * this->B0 * areaFactor;
212 //areaFactor = (CellParams::DX * CellParams::DZ) / (dx * dz);
213 if (x >= -dx && x <= dx)
214 if (z >= -dz && z <= dz)
215 cell->at(fsgrids::bfield::PERBY) = this->B0 * areaFactor;
216 break;
217 case BZCASE:
218 cell->at(fsgrids::bfield::PERBZ) = 0.1 * this->B0 * areaFactor;
219 //areaFactor = (CellParams::DX * CellParams::DY) / (dx * dy);
220 if (x >= -dx && x <= dx)
221 if (y >= -dy && y <= dy)
222 cell->at(fsgrids::bfield::PERBZ) = this->B0 * areaFactor;
223 break;
224 case BALLCASE:
225 cell->at(fsgrids::bfield::PERBX) = 0.1 * this->B0 * areaFactor;
226 cell->at(fsgrids::bfield::PERBY) = 0.1 * this->B0 * areaFactor;
227 cell->at(fsgrids::bfield::PERBZ) = 0.1 * this->B0 * areaFactor;
228
229 //areaFactor = (CellParams::DX * CellParams::DY) / (dx * dy);
230
231 if (y >= -dy && y <= dy)
232 if (z >= -dz && z <= dz)
233 cell->at(fsgrids::bfield::PERBX) = this->B0 * areaFactor;
234 if (x >= -dx && x <= dx)
235 if (z >= -dz && z <= dz)
236 cell->at(fsgrids::bfield::PERBY) = this->B0 * areaFactor;
237 if (x >= -dx && x <= dx)
238 if (y >= -dy && y <= dy)
239 cell->at(fsgrids::bfield::PERBZ) = this->B0 * areaFactor;
240 break;
241 }
242 }
243 }
244 }

252 vector<std::array<Real, 3>> test_fp::getV0(
253 creal x,
254 creal y,
255 creal z,
256 creal dx,
257 creal dy,
258 creal dz,
259 const uint popID
260 ) const {
261 vector<std::array<Real, 3>> centerPoints;
262
263 Real VX=0.0,VY=0.0,VZ=0.0;
264 if (this->shear == true)
265 {
266 //Real ksi;
267 Real eta;
268 switch (this->CASE) {
269 case BXCASE:
270 //ksi = ((y + 0.5 * dy) * cos(this->ALPHA) + (z + 0.5 * dz) * sin(this->ALPHA)) / (2.0 * sqrt(2.0));
271 eta = (-(y + 0.5 * dy) * sin(this->ALPHA) + (z + 0.5 * dz) * cos(this->ALPHA)) / (2.0 * sqrt(2.0));
272 VX = 0.0;
273 VY = sign(cos(this->ALPHA)) * 0.5 + 0.1*cos(this->ALPHA) * sin(2.0 * M_PI * eta);
274 VZ = sign(sin(this->ALPHA)) * 0.5 + 0.1*sin(this->ALPHA) * sin(2.0 * M_PI * eta);
275 break;
276 case BYCASE:
277 //ksi = ((z + 0.5 * dz) * cos(this->ALPHA) + (x + 0.5 * dx) * sin(this->ALPHA)) / (2.0 * sqrt(2.0));
278 eta = (-(z + 0.5 * dz) * sin(this->ALPHA) + (x + 0.5 * dx) * cos(this->ALPHA)) / (2.0 * sqrt(2.0));
279 VX = sign(sin(this->ALPHA)) * 0.5 + 0.1*sin(this->ALPHA) * sin(2.0 * M_PI * eta);
280 VY = 0.0;
281 VZ = sign(cos(this->ALPHA)) * 0.5 + 0.1*cos(this->ALPHA) * sin(2.0 * M_PI * eta);
282 break;
283 case BZCASE:
284 //ksi = ((x + 0.5 * dx) * cos(this->ALPHA) + (y + 0.5 * dy) * sin(this->ALPHA)) / (2.0 * sqrt(2.0));
285 eta = (-(x + 0.5 * dx) * sin(this->ALPHA) + (y + 0.5 * dy) * cos(this->ALPHA)) / (2.0 * sqrt(2.0));
286 VX = sign(cos(this->ALPHA)) * 0.5 + 0.1*cos(this->ALPHA) * sin(2.0 * M_PI * eta);
287 VY = sign(sin(this->ALPHA)) * 0.5 + 0.1*sin(this->ALPHA) * sin(2.0 * M_PI * eta);
288 VZ = 0.0;
289 break;
290 case BALLCASE:
291 std::cerr << "not implemented in " << __FILE__ << ":" << __LINE__ << std::endl;
292 exit(1);
293 break;
294 }
295 } else {
296 switch (this->CASE) {
297 case BXCASE:
298 VX = 0.0;
299 VY = cos(this->ALPHA) * 0.5;
300 VZ = sin(this->ALPHA) * 0.5;
301 break;
302 case BYCASE:
303 VX = sin(this->ALPHA) * 0.5;
304 VY = 0.0;
305 VZ = cos(this->ALPHA) * 0.5;
306 break;
307 case BZCASE:
308 VX = cos(this->ALPHA) * 0.5;
309 VY = sin(this->ALPHA) * 0.5;
310 VZ = 0.0;
311 break;
312 case BALLCASE:
313 VX = 0.5 / sqrt(3.0);
314 VY = 0.5 / sqrt(3.0);
315 VZ = 0.5 / sqrt(3.0);
316 break;
317 }
318 }
319
320 VX *= this->V0 * 2.0;
321 VY *= this->V0 * 2.0;
322 VZ *= this->V0 * 2.0;
323
324 std::array<Real, 3> point {{VX, VY, VZ}};
325 centerPoints.push_back(point);
326 return centerPoints;
327 }
