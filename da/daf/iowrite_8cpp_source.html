141bool writeVelocityDistributionData(const uint popID,Writer& vlsvWriter,
142 dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
143 const std::vector<CellID>& cells,MPI_Comm comm) {
144 // Write velocity blocks and related data.
145 // In restart we just write velocity grids for all cells.
146 // First write global Ids of those cells which write velocity blocks (here: all cells):
147 map<string,string> attribs;
148 const string popName = getObjectWrapper().particleSpecies[popID].name;
149 const string spatMeshName = "SpatialGrid";
150 attribs["name"] = popName;
151 bool success=true;
152
153 // Compute totalBlocks
154 uint64_t totalBlocks = 0;
155 vector<vmesh::LocalID> blocksPerCell;
156 for (size_t i=0; i<cells.size(); ++i){
157 totalBlocks+=mpiGrid[cells[i]]->get_number_of_velocity_blocks(popID);
158 blocksPerCell.push_back(mpiGrid[cells[i]]->get_number_of_velocity_blocks(popID));
159 }
160
161 // The name of the mesh is "SpatialGrid"
162 attribs["mesh"] = spatMeshName;
163
164 const unsigned int vectorSize = 1;
165 // Write the array:
166 if (vlsvWriter.writeArray("CELLSWITHBLOCKS",attribs,cells.size(),vectorSize,cells.data()) == false) success = false;
167 if (success == false) logFile << "(MAIN) writeGrid: ERROR failed to write CELLSWITHBLOCKS to file!" << endl << writeVerbose;
168 // Write blocks per cell, this has to be in the same order as cellswitblocks so that extracting works
169 if(vlsvWriter.writeArray("BLOCKSPERCELL",attribs,blocksPerCell.size(),vectorSize,blocksPerCell.data()) == false) success = false;
170 if (success == false) logFile << "(MAIN) writeGrid: ERROR failed to write CELLSWITHBLOCKS to file!" << endl << writeVerbose;
171
172 // Write (partial) velocity mesh data
173 // The mesh bounding box gives the outer extent of the available velocity space
174 // in blocks and cells. Note that this is not the physical extent of that
175 // space, but a purely numerical bounding box.
176 uint64_t bbox[6];
177 const size_t meshID = getObjectWrapper().particleSpecies[popID].velocityMesh;
178 bbox[0] = vmesh::getMeshWrapper()->velocityMeshes->at(meshID).gridLength[0];
179 bbox[1] = vmesh::getMeshWrapper()->velocityMeshes->at(meshID).gridLength[1];
180 bbox[2] = vmesh::getMeshWrapper()->velocityMeshes->at(meshID).gridLength[2];
181 bbox[3] = vmesh::getMeshWrapper()->velocityMeshes->at(meshID).blockLength[0];
182 bbox[4] = vmesh::getMeshWrapper()->velocityMeshes->at(meshID).blockLength[1];
183 bbox[5] = vmesh::getMeshWrapper()->velocityMeshes->at(meshID).blockLength[2];
184
185 attribs.clear();
186 attribs["mesh"] = getObjectWrapper().particleSpecies[popID].name;
187 attribs["type"] = vlsv::mesh::STRING_UCD_AMR;
188
189 // stringstream is necessary here to correctly convert refLevelMaxAllowed (hardcoded to zero now) into a string
190 stringstream ss;
191 //ss << static_cast<unsigned int>(vmesh::getMeshWrapper()->velocityMeshes->at(meshID).refLevelMaxAllowed);
192 ss << static_cast<unsigned int>(0);
193 attribs["max_velocity_ref_level"] = ss.str();
194
195 if (mpiGrid.get_rank() == MASTER_RANK) {
196 if (vlsvWriter.writeArray("MESH_BBOX",attribs,6,1,bbox) == false) success = false;
197
198 for (int crd=0; crd<3; ++crd) {
199 const size_t N_nodes = bbox[crd]*bbox[crd+3]+1;
200 Real* crds = new Real[N_nodes];
201 const Real dV = vmesh::getMeshWrapper()->velocityMeshes->at(meshID).cellSize[crd];
202
203 for (size_t i=0; i<N_nodes; ++i) {
204 crds[i] = vmesh::getMeshWrapper()->velocityMeshes->at(meshID).meshMinLimits[crd] + i*dV;
205 }
206
207 if (crd == 0) {
208 if (vlsvWriter.writeArray("MESH_NODE_CRDS_X",attribs,N_nodes,1,crds) == false) success = false;
209 }
210 if (crd == 1) {
211 if (vlsvWriter.writeArray("MESH_NODE_CRDS_Y",attribs,N_nodes,1,crds) == false) success = false;
212 }
213 if (crd == 2) {
214 if (vlsvWriter.writeArray("MESH_NODE_CRDS_Z",attribs,N_nodes,1,crds) == false) success = false;
215 }
216 delete [] crds; crds = NULL;
217 }
218 } else {
219 if (vlsvWriter.writeArray("MESH_BBOX",attribs,0,1,bbox) == false) success = false;
220 Real* crds = NULL;
221 if (vlsvWriter.writeArray("MESH_NODE_CRDS_X",attribs,0,1,crds) == false) success = false;
222 if (vlsvWriter.writeArray("MESH_NODE_CRDS_Y",attribs,0,1,crds) == false) success = false;
223 if (vlsvWriter.writeArray("MESH_NODE_CRDS_Z",attribs,0,1,crds) == false) success = false;
224 }
225
226 // Write velocity block IDs
227 vector<vmesh::GlobalID> velocityBlockIds(totalBlocks);
228 uint blockIndex = 0;
229 try {
230 // gather data for writing
231 for (size_t i=0; i<cells.size(); ++i) {
232 SpatialCell* SC = mpiGrid[cells[i]];
233 const vmesh::LocalID nBlocks = SC->get_number_of_velocity_blocks(popID);
234 #ifdef USE_GPU
235 const vmesh::GlobalID *GIDlist = SC->get_velocity_grid(popID);
236 CHK_ERR( gpuMemcpy(&velocityBlockIds[blockIndex], GIDlist, nBlocks*sizeof(vmesh::GlobalID), gpuMemcpyDeviceToHost));
237 #else
238 for (vmesh::LocalID block_i=0; block_i<nBlocks; ++block_i) {
239 const vmesh::GlobalID block = SC->get_velocity_block_global_id(block_i,popID);
240 velocityBlockIds[blockIndex + block_i] = block;
241 }
242 #endif
243 blockIndex += nBlocks;
244 }
245 } catch (...) {
246 cerr << "FAILED TO WRITE VELOCITY BLOCK IDS AT: " << __FILE__ << " " << __LINE__ << endl;
247 success=false;
248 }
249
250 if (globalSuccess(success,"(MAIN) writeGrid: ERROR: Failed to fill temporary array velocityBlockIds",MPI_COMM_WORLD) == false) {
251 vlsvWriter.close();
252 return false;
253 }
254
255 attribs.clear();
256 attribs["mesh"] = spatMeshName;
257 attribs["name"] = popName;
258 if (vlsvWriter.writeArray("BLOCKIDS", attribs, totalBlocks, vectorSize, velocityBlockIds.data()) == false) success = false;
259 if (success == false) logFile << "(MAIN) writeGrid: ERROR failed to write BLOCKIDS to file!" << endl << writeVerbose;
260 {
261 vector<vmesh::GlobalID>().swap(velocityBlockIds);
262 }
263
264 // Write the velocity space data
265 // set everything that is needed for writing in data such as the array name, size, datatype, etc..
266 attribs.clear();
267 attribs["mesh"] = spatMeshName; // Name of the spatial mesh
268 attribs["name"] = popName; // Name of the velocity space distribution is written avgs
269 const string datatype_avgs = "float";
270 const uint64_t arraySize_avgs = totalBlocks;
271 const uint64_t vectorSize_avgs = WID3; // There are 64 (WID=4) or 512 (WID=8) elements in every velocity block
272
273 // Get the data size needed for writing in data
274 uint64_t dataSize_avgs = sizeof(Realf);
275
276 // Start multi write
277 vlsvWriter.startMultiwrite(datatype_avgs,arraySize_avgs,vectorSize_avgs,dataSize_avgs);
278
279 #ifdef USE_GPU
280 // single pinned host buffer for facilitating IO from GPU memory
281 uint64_t bufferOffset = 0;
282 CHK_ERR( gpuMallocHost((void**)&IObuffer,totalBlocks*WID3*sizeof(Realf)) );
283 #endif
284 // Loop over cells
285 for (size_t i = 0; i<cells.size(); ++i) {
286 // Get the spatial cell
287 SpatialCell* SC = mpiGrid[cells[i]];
288
289 // Get the number of blocks in this cell
290 const uint64_t arrayElements = SC->get_number_of_velocity_blocks(popID);
291 // Add a subarray to write. Note: We told beforehands that the vectorsize = WID3
292 #ifdef USE_GPU
293 char* arrayToWrite = IObuffer+bufferOffset;
294 if (arrayElements > 0) {
295 CHK_ERR( gpuMemcpy(arrayToWrite, SC->get_data(popID), arrayElements*WID3*sizeof(Realf), gpuMemcpyDeviceToHost));
296 bufferOffset += arrayElements*WID3*sizeof(Realf);
297 }
298 #else
299 char* arrayToWrite = reinterpret_cast<char*>(SC->get_data(popID));
300 #endif
301 vlsvWriter.addMultiwriteUnit(arrayToWrite, arrayElements);
302
303 }
304 if (cells.size() == 0) {
305 vlsvWriter.addMultiwriteUnit(NULL, 0); //Dummy write to avoid hang in end multiwrite
306 }
307
308 // Write the subarrays
309 vlsvWriter.endMultiwrite("BLOCKVARIABLE", attribs);
310
311 if (globalSuccess(success,"(MAIN) writeGrid: ERROR: Failed to fill temporary velocityBlockData array",MPI_COMM_WORLD) == false) {
312 vlsvWriter.close();
313 return false;
314 }
315
316 if (success ==false) {
317 logFile << "(MAIN) writeGrid: ERROR occurred when writing BLOCKVARIABLE f" << endl << writeVerbose;
318 }
319
320 return success;
321}

332bool writeDataReducer(const dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
333 const std::vector<CellID>& cells,
334 FsGrid< std::array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & perBGrid,
335 FsGrid< std::array<Real, fsgrids::efield::N_EFIELD>, FS_STENCIL_WIDTH> & EGrid,
336 FsGrid< std::array<Real, fsgrids::ehall::N_EHALL>, FS_STENCIL_WIDTH> & EHallGrid,
337 FsGrid< std::array<Real, fsgrids::egradpe::N_EGRADPE>, FS_STENCIL_WIDTH> & EGradPeGrid,
338 FsGrid< std::array<Real, fsgrids::moments::N_MOMENTS>, FS_STENCIL_WIDTH> & momentsGrid,
339 FsGrid< std::array<Real, fsgrids::dperb::N_DPERB>, FS_STENCIL_WIDTH> & dPerBGrid,
340 FsGrid< std::array<Real, fsgrids::dmoments::N_DMOMENTS>, FS_STENCIL_WIDTH> & dMomentsGrid,
341 FsGrid< std::array<Real, fsgrids::bgbfield::N_BGB>, FS_STENCIL_WIDTH> & BgBGrid,
342 FsGrid< std::array<Real, fsgrids::volfields::N_VOL>, FS_STENCIL_WIDTH> & volGrid,
343 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
344 const bool writeAsFloat,
345 const bool writeFsGrid,
346 DataReducer& dataReducer,
347 cint dataReducerIndex,
348 Writer& vlsvWriter){
349 map<string,string> attribs;
350 string variableName,dataType,unitString,unitStringLaTeX, variableStringLaTeX, unitConversionFactor;
351 bool success=true;
352
353 if(!writeFsGrid) { // if we shouldn't write fsgrid DROs
354 variableName = dataReducer.getName(dataReducerIndex);
355 if(variableName.find("fg_", 0) == 0) { // and if the DRO's name includes the string "fg_"
356 return success; // we're good to go
357 }
358 }
359
360 const string meshName = "SpatialGrid";
361 variableName = dataReducer.getName(dataReducerIndex);
362 phiprof::Timer droTimer {"DRO_"+variableName};
363
364 //Get basic data on a variable:
365 uint dataSize,vectorSize;
366 attribs["mesh"] = meshName;
367 attribs["name"] = variableName;
368 if (dataReducer.getDataVectorInfo(dataReducerIndex,dataType,dataSize,vectorSize) == false) {
369 cerr << "ERROR when requesting info from DRO " << dataReducerIndex << endl;
370 return false;
371 }
372
373 // Request variable unit metadata: unit, latex-formatted unit, and conversion factor to SI
374 if (dataReducer.getMetadata(dataReducerIndex,unitString,unitStringLaTeX,variableStringLaTeX,unitConversionFactor) == false) {
375 cerr << "ERROR when requesting unit metadata from DRO " << dataReducerIndex << endl;
376 return false;
377 }
378 attribs["unit"]=unitString;
379 attribs["unitLaTeX"]=unitStringLaTeX;
380 attribs["unitConversion"]=unitConversionFactor;
381 attribs["variableLaTeX"]=variableStringLaTeX;
382
383 // If DRO has a vector size of 0 it means this DRO should not write out anything. This is used e.g. for DROs we want only for certain populations.
384 if (vectorSize == 0) {
385 return true;
386 }
387
388 const uint64_t varBufferArraySize = cells.size()*vectorSize*dataSize;
389
390 //Request DataReductionOperator to calculate the reduced data for all local cells:
391 char* varBuffer = NULL;
392 try {
393 varBuffer = new char[varBufferArraySize];
394 } catch( bad_alloc& ) {
395 cerr << "ERROR, FAILED TO ALLOCATE MEMORY AT: " << __FILE__ << " " << __LINE__ << endl;
396 logFile << "(MAIN) writeGrid: ERROR FAILED TO ALLOCATE MEMORY AT: " << __FILE__ << " " << __LINE__ << endl << writeVerbose;
397 return false;
398 }
399
400 for (size_t cell=0; cell<cells.size(); ++cell) {
401 //Reduce data ( return false if the operation fails )
402 if (dataReducer.reduceData(mpiGrid[cells[cell]],dataReducerIndex,varBuffer + cell*vectorSize*dataSize) == false){
403 success = false;
404 // Note that this is not an error (anymore), since fsgrid reducers will return false here.
405 }
406 }
407
408 if( dataReducer.getName(dataReducerIndex).find("fg_", 0) == 0 ) {
409 // Write fsgrid data
410 phiprof::Timer writeFsTimer {"writeFsGrid"};
411 success = dataReducer.writeFsGridData(perBGrid,EGrid,EHallGrid,EGradPeGrid,momentsGrid,dPerBGrid,dMomentsGrid,BgBGrid,volGrid, technicalGrid, "fsgrid", dataReducerIndex, vlsvWriter, writeAsFloat);
412 writeFsTimer.stop();
413
414 } else if( dataReducer.getName(dataReducerIndex).find("ig_", 0) == 0 ) {
415 // Or maybe it will be writing ionosphere data?
416 phiprof::Timer writeIonosphereTimer {"writeIonosphere"};
417 success |= dataReducer.writeIonosphereGridData(SBC::ionosphereGrid, "ionosphere", dataReducerIndex, vlsvWriter);
418 writeIonosphereTimer.stop();
419 } else {
420 // If the data reducer didn't want to write fg or ig data, maybe it will be happy writing dccrg data
421 if( (writeAsFloat == true && dataType.compare("float") == 0) && dataSize == sizeof(double) ) {
422 double * varBuffer_double = reinterpret_cast<double*>(varBuffer);
423 //Declare smaller varbuffer:
424 const uint64_t arraySize_smaller = cells.size();
425 const uint32_t vectorSize_smaller = vectorSize;
426 const uint32_t dataSize_smaller = sizeof(float);
427 const string dataType_smaller = dataType;
428 float * varBuffer_smaller = NULL;
429 try {
430 varBuffer_smaller = new float[arraySize_smaller * vectorSize_smaller];
431 } catch( bad_alloc& ) {
432 cerr << "ERROR, FAILED TO ALLOCATE MEMORY AT: " << __FILE__ << " " << __LINE__ << endl;
433 logFile << "(MAIN) writeGrid: ERROR FAILED TO ALLOCATE MEMORY AT: " << __FILE__ << " " << __LINE__ << endl << writeVerbose;
434 delete[] varBuffer;
435 varBuffer = NULL;
436 return false;
437 }
438 //Input varBuffer_double into varBuffer_smaller:
439 for( uint64_t i = 0; i < arraySize_smaller * vectorSize_smaller; ++i ) {
440 const double value = varBuffer_double[i];
441 varBuffer_smaller[i] = (float)(value);
442 }
443 //Cast the varBuffer to char:
444 char * varBuffer_smaller_char = reinterpret_cast<char*>(varBuffer_smaller);
445 //Write the array:
446 phiprof::Timer writeArrayTimer {"writeArray"};
447 if (vlsvWriter.writeArray("VARIABLE", attribs, dataType_smaller, arraySize_smaller, vectorSize_smaller, dataSize_smaller, varBuffer_smaller_char) == false) {
448 success = false;
449 logFile << "(MAIN) writeGrid: ERROR failed to write datareductionoperator data to file!" << endl << writeVerbose;
450 }
451 writeArrayTimer.stop();
452 delete[] varBuffer_smaller;
453 varBuffer_smaller = NULL;
454 } else {
455 // Write reduced data to file if DROP was successful:
456 phiprof::Timer writeArrayTimer {"writeArray"};
457 if (vlsvWriter.writeArray("VARIABLE",attribs, dataType, cells.size(), vectorSize, dataSize, varBuffer) == false) {
458 success = false;
459 logFile << "(MAIN) writeGrid: ERROR failed to write datareductionoperator data to file!" << endl << writeVerbose;
460 }
461 }
462 }
463
464 // Check if the DataReducer wants to write paramters to the output file
465 if (dataReducer.hasParameters(dataReducerIndex) == true) {
466 success = dataReducer.writeParameters(dataReducerIndex,vlsvWriter);
467 }
468
469 delete[] varBuffer;
470 varBuffer = NULL;
471 return success;
472}

545bool writeGhostZoneDomainAndLocalIdNumbers( dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
546 Writer & vlsvWriter,
547 const string & meshName,
548 const vector<uint64_t> & ghost_cells ) {
549 //Declare vectors for storing data
550 vector<uint64_t> ghostDomainIds;
551 ghostDomainIds.reserve( ghost_cells.size() );
552 vector<uint64_t> ghostLocalIds;
553 ghostLocalIds.reserve( ghost_cells.size() );
554
555 //Iterate through all ghost zones:
556 vector<uint64_t>::const_iterator it;
557 for( it = ghost_cells.begin(); it != ghost_cells.end(); ++it ) {
558 //Domain id is the MPI process rank owning the ghost zone
559
560 //get the local id of the zone in the process where THIS ghost zone is a local zone:
561 //In order to do this we need MPI (Done in createZone)
562 //Example:
563 //Marking zones with a letter A, B, C, D etc and local ids are numbers above the zones
564 //local id: 0 1 2 3 4 5
565 //Process 1. has: local zones ( A, B, C ), ghost zones ( D E F )
566
567 //local id: 0 1 2
568 //Process 2. has: local zones ( D, G ), ghost zones ( H )
569 //Now if we're in process 1. and our ghost zone is D, its domainId would be 2. because process 2. has D as local zone
570 //In process 2, the local id of D is 0, so that's the local id we want now
571 //The local id is being saved in createZone function
572
573 //Append to the vectors Note: Check updateLocalIds function
574 ghostDomainIds.push_back( mpiGrid.get_process( *it ) );
575 ghostLocalIds.push_back( mpiGrid[(*it)]->ioLocalCellId );
576 }

1139 mpiGrid[cells[i]]->parameters[CellParams::ISCELLSAVINGF] = 1.0;
1140 continue; // Avoid double entries in case the cell also matches following conditions.
1141 }
1142 // Cell lines selection
1143 // Determine cellID's 3D indices
1144
1145 // Loop over AMR levels
1146 uint startindex=1;
1147 uint endindex=1;
1148 for (int AMR = 0; AMR <= P::amrMaxSpatialRefLevel; AMR++) {
1149 int AMRm = 1u << AMR;
1150 uint cellsthislevel = (AMRm * P::xcells_ini) * (AMRm*P::ycells_ini) * (AMRm * P::zcells_ini);
1151 startindex = endindex;
1152 endindex = endindex + cellsthislevel;
1153
1154 // If cell belongs to this AMR level, find indices
1155 if ((cells[i]>=startindex)&&(cells[i]<endindex)) {
1156 lineX = (cells[i]-startindex) % (AMRm*P::xcells_ini);
1157 lineY = ((cells[i]-startindex) / (AMRm*P::xcells_ini)) % (AMRm*P::ycells_ini);
1158 lineZ = ((cells[i]-startindex) /((AMRm*P::xcells_ini) * (AMRm*P::ycells_ini))) % (AMRm*P::zcells_ini);
1159 // Check that indices are in correct intersection at least in one plane
1160 if ((P::systemWriteDistributionWriteXlineStride[index] > 0 &&
1161 P::systemWriteDistributionWriteYlineStride[index] > 0 &&
1162 lineX % P::systemWriteDistributionWriteXlineStride[index] == 0 &&
1163 lineY % P::systemWriteDistributionWriteYlineStride[index] == 0
1164 ) && (P::systemWriteDistributionWriteYlineStride[index] > 0 &&
1165 P::systemWriteDistributionWriteZlineStride[index] > 0 &&
1166 lineY % P::systemWriteDistributionWriteYlineStride[index] == 0 &&
1167 lineZ % P::systemWriteDistributionWriteZlineStride[index] == 0
1168 ) && (P::systemWriteDistributionWriteZlineStride[index] > 0 &&
1169 P::systemWriteDistributionWriteXlineStride[index] > 0 &&
1170 lineZ % P::systemWriteDistributionWriteZlineStride[index] == 0 &&
1171 lineX % P::systemWriteDistributionWriteXlineStride[index] == 0))
1172 {
1173 velSpaceCells.push_back(cells[i]);
1174 mpiGrid[cells[i]]->parameters[CellParams::ISCELLSAVINGF] = 1.0;
1175 break; // Avoid double entries in case the cell also matches following conditions.
1176 }
1177 }
1178 }
1179 // Avoid double entries in case the cell also matches following conditions.
1180 if (mpiGrid[cells[i]]->parameters[CellParams::ISCELLSAVINGF] > 0) continue;
1181
1182 // Loop over spherical shells at defined distances
1183 for (uint ishell = 0; ishell < P::systemWriteDistributionWriteShellRadius.size(); ishell++) {
1184 shellRadiusSquare = P::systemWriteDistributionWriteShellRadius[ishell] * P::systemWriteDistributionWriteShellRadius[ishell];
1185 cellX = mpiGrid[cells[i]]->parameters[CellParams::XCRD];
1186 cellY = mpiGrid[cells[i]]->parameters[CellParams::YCRD];
1187 cellZ = mpiGrid[cells[i]]->parameters[CellParams::ZCRD];
1188 DX = mpiGrid[cells[i]]->parameters[CellParams::DX];
1189 DY = mpiGrid[cells[i]]->parameters[CellParams::DY];
1190 DZ = mpiGrid[cells[i]]->parameters[CellParams::DZ];
1191
1192 dx_rm = cellX < 0 ? DX : 0;
1193 dx_rp = cellX < 0 ? 0 : DX;
1194 dy_rm = cellY < 0 ? DY : 0;
1195 dy_rp = cellY < 0 ? 0 : DY;
1196 dz_rm = cellZ < 0 ? DZ : 0;
1197 dz_rp = cellZ < 0 ? 0 : DZ;
1198 rsquare_minus = (cellX + dx_rm) * (cellX + dx_rm) + (cellY + dy_rm) * (cellY + dy_rm) + (cellZ + dz_rm) * (cellZ + dz_rm);
1199 rsquare_plus = (cellX + dx_rp) * (cellX + dx_rp) + (cellY + dy_rp) * (cellY + dy_rp) + (cellZ + dz_rp) * (cellZ + dz_rp);
1200 // Sometimes two face-neighboring cells can both intersect the sphere. In these cases, if the
1201 // stride applied in that region is in a different direction than the neighborhood, both cells will be saved.
1202 withinshell = (rsquare_minus <= shellRadiusSquare && rsquare_plus > shellRadiusSquare && P::systemWriteDistributionWriteShellStride[ishell] > 0);
1203 if (withinshell) {
1204 // sort centerpoints
1205 std::array<Real, 3> s = {abs(cellX+0.5*DX),abs(cellY+0.5*DY),abs(cellZ+0.5*DZ)};
1206 std::sort(s.begin(), s.end());
1207 Real shellR = P::systemWriteDistributionWriteShellRadius[ishell];
1208 int shellS = P::systemWriteDistributionWriteShellStride[ishell];
1209 // After this, assumes DX==DY==DZ
1210 // Dominant direction (+-x,+-y,+-z) is used for concentric rings
1211 Real D = s[2];
1212 // Tangential direction
1213 Real T;
1214 // Clock angle distance for stride steps
1215 Real clock;
1216 if ((P::xcells_ini==1) || (P::ycells_ini==1) || (P::zcells_ini==1)) {
1217 // 1D or 2D simulation
1218 T = s[1];
1219 s[0] = 0;
1220 clock = 0;
1221 } else { // 3D simulation
1222 T = sqrt(s[0]*s[0]+s[1]*s[1]);
1223 clock = T*atan(s[0]/s[1]);
1224 }
1225 // Distance along great circle away from dominant coordinate
1226 Real dist = shellR * atan(T/D);
1227 // Now find the closest point(s) which fulfills the stride requirement
1228 Real dist2 = DX * shellS * round(dist/DX/shellS);
1229 Real clock2 = DX * shellS * round(clock/DX/shellS);
1230
1231 // Find Cartesian coordinates of this stridepoint
1232 Real D2 = shellR * cos(dist2/shellR);
1233 Real T2 = shellR * sin(dist2/shellR);
1234
1235 stridecheck = false;
1236 // Now check if the stridepoint is exactly in this cell
1237 if ((P::xcells_ini==1) || (P::ycells_ini==1) || (P::zcells_ini==1)) {
1238 // 1D or 2D
1239 if ( (D2 >= D-0.5*DX) && (D2 < D+0.5*DX) && (T2 >= T-0.5*DX) && (T2 < T+0.5*DX) ) stridecheck=true;
1240 // Special case for corners:
1241 if ( (abs(D-T)<0.5*DX) && (dist2>dist) ) stridecheck=true;
1242
1243 // Only save 1 cell touching axes
1244 if ( (P::ycells_ini==1) && ( ( (cellX>-1.1*DX)&&(cellX<0) ) || ( (cellZ>-1.1*DZ)&&(cellZ<0) ) )) stridecheck=false;
1245 if ( (P::zcells_ini==1) && ( ( (cellX>-1.1*DX)&&(cellX<0) ) || ( (cellY>-1.1*DY)&&(cellY<0) ) )) stridecheck=false;
1246
1247 } else {
1248 // 3D simulation, account for clock angle
1249 Real T2A = T2 * cos(clock2/T2);
1250 Real T2B = T2 * sin(clock2/T2);
1251 // Rings at given stride from dominant direction
1252 bool ring = (D2 >= D-0.5*DX) && (D2 < D+0.5*DX) && (T2 >= T-0.5*DX) && (T2 < T+0.5*DX);
1253 // Special case for 45 degree ring:
1254 ring = ring || ((abs(D-T)<0.5*DX) && (dist2>dist));
1255 // Clock angle
1256 bool clockcheck = (T2A >= s[1]-0.5*DX) && (T2A < s[1]+0.5*DX) && (T2B >= s[0]-0.5*DX) && (T2B < s[0]+0.5*DX);
1257 // Special case for 45 degree clock angle
1258 clockcheck = clockcheck || ( (abs(s[1]-s[0])<0.5*DX) && (clock2>clock) );
1259 if ( ring && clockcheck ) stridecheck = true;
1260
1261 // Ensure cells touching Cartesian axes are included
1262 if ( (s[1]<DX) && (s[0]<DX) && (D2 >= D-0.5*DX) && (D2 < D+0.5*DX) ) stridecheck=true;
1263
1264 // Special corner-corner-case
1265 if ( (abs(s[2]-s[1])<DX) && (abs(s[1]-s[0])<DX) && (abs(s[2]-s[0])<DX) ) stridecheck=true;
1266
1267 // Only save 1 cell touching axes (assumes origin is at corner intersection of 8 cells)
1268 if ( ( (cellX>-1.1*DX)&&(cellX<0) ) || ( (cellY>-1.1*DY)&&(cellY<0) ) || ( (cellZ>-1.1*DZ)&&(cellZ<0) ) ) stridecheck=false;
1269 }
1270
1271 if (stridecheck) {
1272 velSpaceCells.push_back(cells[i]);
1273 mpiGrid[cells[i]]->parameters[CellParams::ISCELLSAVINGF] = 1.0;
1274 break; // Avoid double entries in case the cell also matches following conditions.
1275 }
1276 }
1277 }
1278 }
1279
1280 uint64_t numVelSpaceCells;
1281 uint64_t localNumVelSpaceCells;
1282 localNumVelSpaceCells=velSpaceCells.size();
1283 MPI_Allreduce(&localNumVelSpaceCells,&numVelSpaceCells,1,MPI_UINT64_T,MPI_SUM,MPI_COMM_WORLD);
1284 //write out velocity space data NOTE: There is mpi communication in writeVelocityDistributionData
1285 if (writeVelocityDistributionData(vlsvWriter, mpiGrid, velSpaceCells, MPI_COMM_WORLD ) == false ) {
1286 cerr << "ERROR, FAILED TO WRITE VELOCITY DISTRIBUTION DATA AT " << __FILE__ << " " << __LINE__ << endl;
1287 logFile << "(MAIN) writeGrid: ERROR FAILED TO WRITE VELOCITY DISTRIBUTION DATA AT: " << __FILE__ << " " << __LINE__ << endl << writeVerbose;
1288 }
1289 return true;
1290}
