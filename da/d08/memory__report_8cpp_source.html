49void report_cell_and_block_counts(dccrg::Dccrg<spatial_cell::SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid){
50 cint maxRefLevel = mpiGrid.get_maximum_refinement_level();
51 const std::vector<CellID> localCells = getLocalCells();
52 cint popCount = getObjectWrapper().particleSpecies.size();
53
54 // popCount+1 as we store the spatial cell counts and then the populations' v_cell counts.
55 // maxRefLevel+1 as e.g. there's 2 levels at maxRefLevel == 1
56 std::vector<int64_t> localCounts((popCount+1)*(maxRefLevel+1), 0), globalCounts((popCount+1)*(maxRefLevel+1), 0);
57
58 for (const auto cellid : localCells) {
59 cint level = mpiGrid.get_refinement_level(cellid);
60 localCounts[level]++;
61 for(int pop=0; pop<popCount; pop++) {
62 localCounts[maxRefLevel+1 + level*popCount + pop] += mpiGrid[cellid]->get_number_of_velocity_blocks(pop);
63 }
64 }
65
66 MPI_Reduce(localCounts.data(), globalCounts.data(), (popCount+1)*(maxRefLevel+1), MPI_INT64_T, MPI_SUM, MASTER_RANK, MPI_COMM_WORLD);
67
68 logFile << "(CELLS) tstep = " << P::tstep << " time = " << P::t << " spatial cells [ ";
69 for(int level = 0; level <= maxRefLevel; level++) {
70 logFile << globalCounts[level] << " ";
71 }
72 logFile << "] blocks ";
73 for(int pop=0; pop<popCount; pop++) {
74 logFile << getObjectWrapper().particleSpecies[pop].name << " [ ";
75 for(int level = 0; level <= maxRefLevel; level++) {
76 logFile << globalCounts[maxRefLevel+1 + level*popCount + pop] << " ";
77 }
78 logFile << "] ";
79 }
80 logFile << std::endl << std::flush;
81
82}

85uint64_t get_node_free_memory(){
86 uint64_t mem_proc_free = 0;
87 FILE * in_file = fopen("/proc/meminfo", "r");
88 char attribute_name[200];
89 uint64_t memory=0;
90 char memory_unit[10];
91 const char * memfree_attribute_name = "MemFree:";
92 if (in_file) {
93 while (!feof(in_file)) {
94 int retval = fscanf(in_file, "%199s %lu %9s", attribute_name, &memory, memory_unit);
95 if (retval >= 2) {
96 if (strcmp(attribute_name, memfree_attribute_name) == 0) {
97 // free memory in KB, transform to B
98 mem_proc_free = memory * 1024;
99 break;
100 }
101 } else {
102 // Skip to neww line
103 fscanf(in_file, "%*[^\n]\n");
104 }
105 }
106 fclose(in_file);
107 }
108
109 return mem_proc_free;
110}

117void report_memory_consumption(
118 dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
119 double extra_bytes/*=0*/
120){
121 /*Report memory consumption into logfile*/
122
123 char nodename[MPI_MAX_PROCESSOR_NAME];
124 int namelength, nodehash;
125 int rank, nProcs, nodeRank, interRank;
126 int nNodes;
127 const double GiB = pow(2,30);
128
129 std::hash<std::string> hasher;
130 MPI_Comm nodeComm;
131 MPI_Comm interComm;
132
133
134 MPI_Comm_size(MPI_COMM_WORLD, &nProcs);
135 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
136
137 //get name of this node
138 MPI_Get_processor_name(nodename,&namelength);
139 nodehash=(int)(hasher(std::string(nodename)) % std::numeric_limits<int>::max());
140
141 //intra-node communicator
142 MPI_Comm_split(MPI_COMM_WORLD, nodehash, rank, &nodeComm);
143 MPI_Comm_rank(nodeComm,&nodeRank);
144 //create communicator for inter-node communication
145 MPI_Comm_split(MPI_COMM_WORLD, nodeRank, rank, &interComm);
146 MPI_Comm_rank(interComm, &interRank);
147 MPI_Comm_size(interComm, &nNodes);
148
149 // Report /proc/meminfo memory consumption first so we then get resident and HWM from Papi below, easier to compare by eye in logfile in this order.
150 double mem_proc_free = (double)get_node_free_memory();
151 double total_mem_proc = 0;
152 double min_free,max_free;
153 const int numberOfParameters = 1;
154 MPI_Reduce( &mem_proc_free, &total_mem_proc, numberOfParameters, MPI_DOUBLE, MPI_SUM, 0, interComm );
155 MPI_Reduce( &mem_proc_free, &min_free, numberOfParameters, MPI_DOUBLE, MPI_MIN, MASTER_RANK, MPI_COMM_WORLD );
156 MPI_Reduce( &mem_proc_free, &max_free, numberOfParameters, MPI_DOUBLE, MPI_MAX, MASTER_RANK, MPI_COMM_WORLD );
157
158 char reportstring[512];
159 snprintf(reportstring,512, "(MEM) tstep %i t %.3g %-21s (GiB/node; avg, min, max, sum): %-8.3g %-8.3g %-8.3g %-8.3g on %i nodes\n",
160 P::tstep, P::t, "Free", total_mem_proc/nNodes/GiB, min_free/GiB, max_free/GiB, total_mem_proc/GiB, nNodes);
161 logFile << reportstring;
162
163#ifdef PAPI_MEM
164 /*If we have PAPI, we can report the resident usage of the process*/
165 if (PAPI_library_init(PAPI_VER_CURRENT) == PAPI_VER_CURRENT) {
166 PAPI_dmem_info_t dmem;
167 PAPI_get_dmem_info(&dmem);
168 double mem_papi[4] = {};
169 double node_mem_papi[4] = {};
170 double sum_mem_papi[4];
171 double min_mem_papi[4];
172 double max_mem_papi[4];
173 /*PAPI returns memory in KB units, transform to bytes*/
174 mem_papi[0] = dmem.high_water_mark * 1024;
175 mem_papi[1] = dmem.resident * 1024 + extra_bytes;
176 mem_papi[2] = dmem.resident * 1024;
177 mem_papi[3] = extra_bytes;
178 //sum node mem
179 MPI_Reduce(mem_papi, node_mem_papi, 4, MPI_DOUBLE, MPI_SUM, 0, nodeComm);
180
181 //rank 0 on all nodes do total reduces
182 if(nodeRank == 0) {
183 MPI_Reduce(node_mem_papi, sum_mem_papi, 4, MPI_DOUBLE, MPI_SUM, 0, interComm);
184 MPI_Reduce(node_mem_papi, min_mem_papi, 4, MPI_DOUBLE, MPI_MIN, 0, interComm);
185 MPI_Reduce(node_mem_papi, max_mem_papi, 4, MPI_DOUBLE, MPI_MAX, 0, interComm);
186 if (max_mem_papi[3] != 0.0) {
187 logFile << "(MEM) Estimating increased high water mark from refinement" << std::endl;
188 }
189 snprintf(reportstring,512, "(MEM) tstep %i t %.3g %-21s (GiB/node; avg, min, max, sum): %-8.3g %-8.3g %-8.3g %-8.3g on %i nodes\n",
190 P::tstep, P::t, "Resident", sum_mem_papi[2]/nNodes/GiB, min_mem_papi[2]/GiB, max_mem_papi[2]/GiB, sum_mem_papi[2]/GiB, nNodes);
191 logFile << reportstring;
192 snprintf(reportstring,512, "(MEM) tstep %i t %.3g %-21s (GiB/node; avg, min, max, sum): %-8.3g %-8.3g %-8.3g %-8.3g on %i nodes\n",
193 P::tstep, P::t, "High water mark \U0001F30A ", sum_mem_papi[0]/nNodes/GiB, min_mem_papi[0]/GiB, max_mem_papi[0]/GiB, sum_mem_papi[0]/GiB, nNodes);
194 logFile << reportstring;
195 if (max_mem_papi[3] != 0.0) {
196 snprintf(reportstring,512, "(MEM) tstep %i t %.3g %-21s (GiB/node; avg, min, max, sum): %-8.3g %-8.3g %-8.3g %-8.3g on %i nodes\n",
197 P::tstep, P::t, "Resident with refines", sum_mem_papi[1]/nNodes/GiB, min_mem_papi[1]/GiB, max_mem_papi[1]/GiB, sum_mem_papi[1]/GiB, nNodes);
198 logFile << reportstring;
199 }
200 }
201 if(rank == MASTER_RANK) {
202 bailout(max_mem_papi[0]/GiB > P::bailout_max_memory, "Memory high water mark per node exceeds bailout threshold", __FILE__, __LINE__);
203 bailout(max_mem_papi[1]/GiB > P::bailout_max_memory, "Estimated resident per node exceeds bailout threshold", __FILE__, __LINE__);
204 }
205 }
206#endif
207
208 /*now report memory consumption of mpiGrid specifically into logfile*/
209 const std::vector<CellID>& cells = getLocalCells();
210 const std::vector<CellID> remote_cells = mpiGrid.get_remote_cells_on_process_boundary();
211
212 /* Compute memory statistics of the memory consumption of the spatial cells.
213 * Internally we use double as MPI does
214 * not define proper uint64_t datatypes for MAXLOCNot Real, as we
215 * want double here not to loose accuracy.
216 */
217
218 /*report data for memory needed by blocks*/
219 double mem[6] = {0};
220 double sum_mem[6];
221
222 for(unsigned int i=0;i<cells.size();i++){
223 #ifdef USE_GPU
224 // GPU version keeps track of largest attained velocity mesh throughout acceleration cycles
225 mem[0] += mpiGrid[cells[i]]->largestvmesh * WID3 * sizeof(Realf);
226 #else
227 mem[0] += mpiGrid[cells[i]]->get_cell_memory_size();
228 #endif
229 mem[3] += mpiGrid[cells[i]]->get_cell_memory_capacity();
230 }
231
232 for(unsigned int i=0;i<remote_cells.size();i++){
233 if(mpiGrid[remote_cells[i]] != NULL) {
234 mem[1] += mpiGrid[remote_cells[i]]->get_cell_memory_size();
235 mem[4] += mpiGrid[remote_cells[i]]->get_cell_memory_capacity();
236 }
237 }
238
239 mem[2] = mem[0] + mem[1];//total memory according to size()
240 mem[5] = mem[3] + mem[4];//total memory according to capacity()
241
242
243 MPI_Reduce(mem, sum_mem, 6, MPI_DOUBLE, MPI_SUM, 0, MPI_COMM_WORLD);
244
245
246 struct {
247 double val;
248 int rank;
249 } max_mem[3],mem_usage_loc[3],min_mem[3];
250 for(uint i = 0; i<3; i++){
251 mem_usage_loc[i].val = mem[i + 3]; //report on capacity numbers (6: local cells, 7: remote cells, 8: all cells)
252 mem_usage_loc[i].rank = rank;
253 }
254
255 MPI_Reduce(mem_usage_loc, max_mem, 3, MPI_DOUBLE_INT, MPI_MAXLOC, 0, MPI_COMM_WORLD);
256 MPI_Reduce(mem_usage_loc, min_mem, 3, MPI_DOUBLE_INT, MPI_MINLOC, 0, MPI_COMM_WORLD);
257
258
259 snprintf(reportstring,512, "(MEM) tstep %i t %.3g %-21s (GiB/rank; avg, min, max, sum): %-8.3g %-8.3g %-8.3g %-8.3g min rank %i max rank %i\n",
260 P::tstep, P::t, "Local cells capacity", sum_mem[3]/nProcs/GiB, min_mem[0].val/GiB, max_mem[0].val/GiB, sum_mem[3]/GiB, min_mem[0].rank, max_mem[0].rank);
261 logFile << reportstring;
262 snprintf(reportstring,512, "(MEM) tstep %i t %.3g %-21s (GiB/rank; avg, min, max, sum): %-8.3g %-8.3g %-8.3g %-8.3g min rank %i max rank %i\n",
263 P::tstep, P::t, "Remote cells capacity", sum_mem[4]/nProcs/GiB, min_mem[1].val/GiB, max_mem[1].val/GiB, sum_mem[4]/GiB, min_mem[1].rank, max_mem[1].rank);
264 logFile << reportstring;
265 snprintf(reportstring,512, "(MEM) tstep %i t %.3g %-21s (GiB/rank; avg, min, max, sum): %-8.3g %-8.3g %-8.3g %-8.3g min rank %i max rank %i\n",
266 P::tstep, P::t, "Total cells capacity", sum_mem[5]/nProcs/GiB, min_mem[2].val/GiB, max_mem[2].val/GiB, sum_mem[5]/GiB, min_mem[2].rank, max_mem[2].rank);
267 logFile << reportstring;
268
269 snprintf(reportstring,512, "(MEM) tstep %i t %.3g Total size and capacity of SpatialCells (GiB): %-8.3g %-8.3g\n",
270 P::tstep, P::t, sum_mem[2]/GiB, sum_mem[5]/GiB);
271 logFile << reportstring;
272
273 logFile << writeVerbose;
274
275 MPI_Comm_free(&interComm);
276 MPI_Comm_free(&nodeComm);
277
278 #ifdef USE_GPU
279 // TODO: Clear duplicate output
280 // (local_cells_capacity, ghost_cells_capacity, local_cells_size, ghost_cells_size)
281 gpu_reportMemory(mem[3], mem[4], mem[0], mem[1]);
282
283 #endif
284}
