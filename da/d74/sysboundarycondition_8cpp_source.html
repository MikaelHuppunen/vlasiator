61 void SysBoundaryCondition::determineFace(
62 bool* isThisCellOnAFace,
63 creal x, creal y, creal z,
64 creal dx, creal dy, creal dz,
65 const bool excludeSlicesAndPeriodicDimensions //=false (default)
66 ) {
67 for(uint i=0; i<6; i++) {
68 isThisCellOnAFace[i] = false;
69 }
70 if(x > Parameters::xmax - dx * 2) {
71 isThisCellOnAFace[0] = true;
72 }
73 if(x < Parameters::xmin + dx * 2) {
74 isThisCellOnAFace[1] = true;
75 }
76 if(y > Parameters::ymax - dy * 2) {
77 isThisCellOnAFace[2] = true;
78 }
79 if(y < Parameters::ymin + dy * 2) {
80 isThisCellOnAFace[3] = true;
81 }
82 if(z > Parameters::zmax - dz * 2) {
83 isThisCellOnAFace[4] = true;
84 }
85 if(z < Parameters::zmin + dz * 2) {
86 isThisCellOnAFace[5] = true;
87 }
88 if(excludeSlicesAndPeriodicDimensions == true) {
89 if (Parameters::xcells_ini == 1 || this->periodic[0]) {
90 isThisCellOnAFace[0] = false;
91 isThisCellOnAFace[1] = false;
92 }
93 if (Parameters::ycells_ini == 1 || this->periodic[1]) {
94 isThisCellOnAFace[2] = false;
95 isThisCellOnAFace[3] = false;
96 }
97 if (Parameters::zcells_ini == 1 || this->periodic[2]) {
98 isThisCellOnAFace[4] = false;
99 isThisCellOnAFace[5] = false;
100 }
101 }

116 void SysBoundaryCondition::determineFace(
117 std::array<bool, 6> &isThisCellOnAFace,
118 const dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
119 CellID id,
120 const bool excludeSlicesAndPeriodicDimensions //=false (default)
121 ) {
122 isThisCellOnAFace.fill(false);
123
124 std::array<std::set<CellID>, 6> dirNeighbors;
125 auto* p = mpiGrid.get_neighbors_of(id, Neighborhoods::VLASOV_SOLVER);
126 if (!p) {
127 std::cerr << "No neighbors found for " << id << std::endl;
128 return;
129 }
130 int nbrs {0};
131 for (auto pair : *p) {
132 if (!pair.first) {
133 continue; // Error cells should obviously not be counted
134 } else if (pair.second[0] > 0) {
135 dirNeighbors[0].insert(pair.first);
136 } else if (pair.second[0] < 0) {
137 dirNeighbors[1].insert(pair.first);
138 } else if (pair.second[1] > 0) {
139 dirNeighbors[2].insert(pair.first);
140 } else if (pair.second[1] < 0) {
141 dirNeighbors[3].insert(pair.first);
142 } else if (pair.second[2] > 0) {
143 dirNeighbors[4].insert(pair.first);
144 } else if (pair.second[2] < 0) {
145 dirNeighbors[5].insert(pair.first);
146 }
147 ++nbrs;
148 //std::cerr << pair.second[0] << " " << pair.second[1] << " " << pair.second[2] << std::endl;
149 }
150
151 for (int i = 0; i < 6; ++i) {
152 if(dirNeighbors[i].size() < 2) {
153 isThisCellOnAFace[i] = true;
154 }
155 }
156
157 if(excludeSlicesAndPeriodicDimensions == true) {
158 if(Parameters::xcells_ini == 1 || this->periodic[0]) {
159 isThisCellOnAFace[0] = false;
160 isThisCellOnAFace[1] = false;
161 }
162 if(Parameters::ycells_ini == 1 || this->periodic[1]) {
163 isThisCellOnAFace[2] = false;
164 isThisCellOnAFace[3] = false;
165 }
166 if(Parameters::zcells_ini == 1 || this->periodic[2]) {
167 isThisCellOnAFace[4] = false;
168 isThisCellOnAFace[5] = false;
169 }
170 }
171 return;
172 }

382 void SysBoundaryCondition::copyCellData(
383 SpatialCell* from,
384 SpatialCell* to,
385 const bool copyMomentsOnly,
386 const uint popID,
387 const bool copy_V_moments
388 ) {
389 if (popID == 0) {
390 if (copy_V_moments) {
391 to->parameters[CellParams::RHOM_V] = from->parameters[CellParams::RHOM_V];
392 to->parameters[CellParams::VX_V] = from->parameters[CellParams::VX_V];
393 to->parameters[CellParams::VY_V] = from->parameters[CellParams::VY_V];
394 to->parameters[CellParams::VZ_V] = from->parameters[CellParams::VZ_V];
395 to->parameters[CellParams::RHOQ_V] = from->parameters[CellParams::RHOQ_V];
396 to->parameters[CellParams::P_11_V] = from->parameters[CellParams::P_11_V];
397 to->parameters[CellParams::P_22_V] = from->parameters[CellParams::P_22_V];
398 to->parameters[CellParams::P_33_V] = from->parameters[CellParams::P_33_V];
399 } else {
400 to->parameters[CellParams::RHOM_R] = from->parameters[CellParams::RHOM_R];
401 to->parameters[CellParams::VX_R] = from->parameters[CellParams::VX_R];
402 to->parameters[CellParams::VY_R] = from->parameters[CellParams::VY_R];
403 to->parameters[CellParams::VZ_R] = from->parameters[CellParams::VZ_R];
404 to->parameters[CellParams::RHOQ_R] = from->parameters[CellParams::RHOQ_R];
405 to->parameters[CellParams::P_11_R] = from->parameters[CellParams::P_11_R];
406 to->parameters[CellParams::P_22_R] = from->parameters[CellParams::P_22_R];
407 to->parameters[CellParams::P_33_R] = from->parameters[CellParams::P_33_R];
408 }
409 }
410
411 if(copyMomentsOnly) {
412 to->get_population(popID).RHO = from->get_population(popID).RHO;
413 if (copy_V_moments) {
414 to->get_population(popID).RHO_V = from->get_population(popID).RHO_V;
415 } else {
416 to->get_population(popID).RHO_R = from->get_population(popID).RHO_R;
417 }
418 for (uint i=0; i<3; i++) {
419 to->get_population(popID).V[i] = from->get_population(popID).V[i];
420 if (copy_V_moments) {
421 to->get_population(popID).V_V[i] = from->get_population(popID).V_V[i];
422 } else {
423 to->get_population(popID).V_R[i] = from->get_population(popID).V_R[i];
424 }
425 }
426 for (uint i=0; i<6; i++) {
427 to->get_population(popID).P[i] = from->get_population(popID).P[i];
428 if (copy_V_moments) {
429 to->get_population(popID).P_V[i] = from->get_population(popID).P_V[i];
430 } else {
431 to->get_population(popID).P_R[i] = from->get_population(popID).P_R[i];
432 }
433 }
434 } else {
435 to->set_population(from->get_population(popID), popID);
436 }
437 }

446 void averageCellData(
447 dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
448 const vector<CellID> cellList,
449 SpatialCell *to,
450 const uint popID,
451 creal fluffiness /* default =0.0*/
452 ) {
453 const size_t numberOfCells = cellList.size();
454 creal factor = fluffiness / convert<Real>(numberOfCells);
455
456 // Rescale own vspace
457 if (fluffiness != 0.0) {
458 to->scale_population(1.0 - fluffiness, popID);
459 }
460 // Add other cell vspaces
461 for (size_t i=0; i<numberOfCells; i++) {
462 const SpatialCell* from = mpiGrid[cellList[i]];
463 to->increment_population(from->get_population(popID), factor, popID);
464 }
465 }

472 bool SysBoundaryCondition::updateSysBoundaryConditionsAfterLoadBalance(
473 dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
474 const vector<CellID> & local_cells_on_boundary
475 ) {
476 // Loop over cellids
477 for(vector<CellID>::const_iterator it = local_cells_on_boundary.begin(); it != local_cells_on_boundary.end(); ++it) {
478 const CellID cellId = *it;
479 vector<CellID> & closestCells = allClosestNonsysboundaryCells[cellId];
480 closestCells.clear();
481 vector<CellID> & closeCells = allCloseNonsysboundaryCells[cellId];
482 closeCells.clear();
483 vector<CellID> & closestL1OutflowCells = allClosestL1OutflowCells[cellId];
484 closestL1OutflowCells.clear();
485 vector<CellID> & closeL1OutflowCells = allCloseL1OutflowCells[cellId];
486 closeL1OutflowCells.clear();
487 uint dist = numeric_limits<uint>::max();
488 uint d2 = numeric_limits<uint>::max();
489 uint distL1 = numeric_limits<uint>::max();
490 uint d2L1 = numeric_limits<uint>::max();
491
492 // Only closer neighborhood for layer 1
493 if(mpiGrid[cellId]->sysBoundaryLayer == 1) {
494 for (auto nbrPair : *mpiGrid.get_neighbors_of(cellId, Neighborhoods::SYSBOUNDARIES)) {
495 if(nbrPair.first != INVALID_CELLID) {
496 CellID neighbor = nbrPair.first;
497 if(mpiGrid[neighbor]->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
498 // Find distance and update closestCells
499 d2 = nbrPair.second[0]*nbrPair.second[0] + nbrPair.second[1]*nbrPair.second[1] + nbrPair.second[2]*nbrPair.second[2];
500 for (auto i : *mpiGrid.get_neighbors_to(cellId, Neighborhoods::SYSBOUNDARIES)) {
501 if (i.first == neighbor) {
502 closeCells.push_back(neighbor);
503 if(d2 == dist) {
504 closestCells.push_back(neighbor);
505 } else if (d2 < dist) {
506 closestCells.clear();
507 closestCells.push_back(neighbor);
508 dist = d2;
509 }
510 }
511 }
512 }
513 }
514 }
515 }
516 // search further for L2
517 if (mpiGrid[cellId]->sysBoundaryLayer == 2) {
518 for (auto nbrPair : *mpiGrid.get_neighbors_of(cellId, Neighborhoods::SYSBOUNDARIES_EXTENDED)) {
519 if(nbrPair.first != INVALID_CELLID) {
520 CellID neighbor = nbrPair.first;
521 if(mpiGrid[neighbor]->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
522 // Find distance and update closestCells
523 d2 = nbrPair.second[0]*nbrPair.second[0] + nbrPair.second[1]*nbrPair.second[1] + nbrPair.second[2]*nbrPair.second[2];
524 closeCells.push_back(neighbor);
525 if(d2 == dist) {
526 closestCells.push_back(neighbor);
527 } else if (d2 < dist) {
528 closestCells.clear();
529 closestCells.push_back(neighbor);
530 dist = d2;
531 }
532 }
533 if(mpiGrid[cellId]->sysBoundaryFlag == sysboundarytype::OUTFLOW && mpiGrid[neighbor]->sysBoundaryFlag == sysboundarytype::OUTFLOW && mpiGrid[neighbor]->sysBoundaryLayer == 1) {
534 // Find distance and update closestCells
535 d2L1 = nbrPair.second[0]*nbrPair.second[0] + nbrPair.second[1]*nbrPair.second[1] + nbrPair.second[2]*nbrPair.second[2];
536 closeL1OutflowCells.push_back(neighbor);
537 if(d2L1 == distL1) {
538 closestL1OutflowCells.push_back(neighbor);
539 } else if (d2L1 < distL1) {
540 closestL1OutflowCells.clear();
541 closestL1OutflowCells.push_back(neighbor);
542 distL1 = d2L1;
543 }
544 }
545 }
546 }
547 }
548
549 if(closestCells.size() == 0) {
550 closestCells.push_back(INVALID_CELLID);
551 }
552 if(closeCells.size() == 0) {
553 closeCells.push_back(INVALID_CELLID);
554 }
555 if(closestL1OutflowCells.size() == 0) {
556 closestL1OutflowCells.push_back(INVALID_CELLID);
557 }
558 if(closeL1OutflowCells.size() == 0) {
559 closeL1OutflowCells.push_back(INVALID_CELLID);
560 }
561 }
562 return true;
563 }

585 vector< array<int, 3> > SysBoundaryCondition::getAllClosestNonsysboundaryCells(
586 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
587 cint i,
588 cint j,
589 cint k
590 ) {
591 int distance = numeric_limits<int>::max();
592 vector< array<int,3> > closestCells;
593
594 for (int kk=-2; kk<3; kk++) {
595 for (int jj=-2; jj<3; jj++) {
596 for (int ii=-2; ii<3 ; ii++) {
597 if( technicalGrid.get(i+ii,j+jj,k+kk) && technicalGrid.get(i+ii,j+jj,k+kk)->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
598 distance = min(distance, ii*ii + jj*jj + kk*kk);
599 }
600 }
601 }
602 }
603
604 for (int kk=-2; kk<3; kk++) {
605 for (int jj=-2; jj<3; jj++) {
606 for (int ii=-2; ii<3 ; ii++) {
607 if( technicalGrid.get(i+ii,j+jj,k+kk) && technicalGrid.get(i+ii,j+jj,k+kk)->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
608 int d = ii*ii + jj*jj + kk*kk;
609 if( d == distance ) {
610 array<int, 3> cell = {i+ii, j+jj, k+kk};
611 closestCells.push_back(cell);
612 }
613 }
614 }
615 }
616 }
617
618 if(closestCells.size() == 0) {
619 array<int, 3> dummy = {numeric_limits<int>::min()};
620 closestCells.push_back(dummy);
621 }
622
623 return closestCells;
624 }

674 FsGrid< array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & bGrid,
675 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
676 cint i,
677 cint j,
678 cint k,
679 cuint component,
680 cuint mask
681 ) {
682
683 int distance = numeric_limits<int>::max();
684 vector< array<int,3> > closestCells;
685
686 for (int kk=-2; kk<3; kk++) {
687 for (int jj=-2; jj<3; jj++) {
688 for (int ii=-2; ii<3 ; ii++) {
689 if( technicalGrid.get(i+ii,j+jj,k+kk) // skip invalid cells returning NULL
690 && (technicalGrid.get(i+ii,j+jj,k+kk)->SOLVE & mask) == mask // Did that guy solve this component?
691 && technicalGrid.get(i+ii,j+jj,k+kk)->sysBoundaryFlag != sysboundarytype::DO_NOT_COMPUTE // Do not copy from there
692 && technicalGrid.get(i+ii,j+jj,k+kk)->sysBoundaryFlag != sysboundarytype::OUTER_BOUNDARY_PADDING // Do not copy from there either
693 ) {
694 distance = min(distance, ii*ii + jj*jj + kk*kk);
695 }
696 }
697 }
698 }
699
700 for (int kk=-2; kk<3; kk++) {
701 for (int jj=-2; jj<3; jj++) {
702 for (int ii=-2; ii<3 ; ii++) {
703 if( technicalGrid.get(i+ii,j+jj,k+kk) // skip invalid cells returning NULL
704 && (technicalGrid.get(i+ii,j+jj,k+kk)->SOLVE & mask) == mask // Did that guy solve this component?
705 && technicalGrid.get(i+ii,j+jj,k+kk)->sysBoundaryFlag != sysboundarytype::DO_NOT_COMPUTE // Do not copy from there
706 && technicalGrid.get(i+ii,j+jj,k+kk)->sysBoundaryFlag != sysboundarytype::OUTER_BOUNDARY_PADDING // Do not copy from there either
707 ) {
708 int d = ii*ii + jj*jj + kk*kk;
709 if( d == distance ) {
710 array<int, 3> cell = {i+ii, j+jj, k+kk};
711 closestCells.push_back(cell);
712 }
713 }
714 }
715 }
716 }
717
718 if (closestCells.size() == 0) {
719 abort_mpi("No closest cell found!", 1);
720 }
721
722 return bGrid.get(closestCells[0][0], closestCells[0][1], closestCells[0][2])->at(fsgrids::bfield::PERBX+component);

779 vmesh::LocalID findMaxwellianBlocksToInitialize(
780 const uint popID,
781 spatial_cell::SpatialCell& cell,
782 creal& rho,
783 creal& T,
784 creal& VX0,
785 creal& VY0,
786 creal& VZ0) {
787
788 bool search = true;
789 uint counter = 0;
790 const vmesh::LocalID* vblocks_ini = cell.get_velocity_grid_length(popID);
791 vmesh::VelocityMesh *vmesh = cell.get_velocity_mesh(popID);
792 const Real mass = getObjectWrapper().particleSpecies[popID].mass;
793
794 vmesh::GlobalID *GIDbuffer;
795 #ifdef USE_GPU
796 // Host-pinned memory buffer, max possible size
797 const uint blocksCount = vblocks_ini[0]*vblocks_ini[1]*vblocks_ini[2];
798 CHK_ERR( gpuMallocHost((void**)&GIDbuffer,blocksCount*sizeof(vmesh::GlobalID)) );
799 #endif
800 // Non-GPU: insert directly into vmesh
801
802 Real V_crds[3];
803 Real dV[3];
804 dV[0] = cell.get_velocity_grid_block_size(popID)[0];
805 dV[1] = cell.get_velocity_grid_block_size(popID)[1];
806 dV[2] = cell.get_velocity_grid_block_size(popID)[2];
807 creal minValue = cell.getVelocityBlockMinValue(popID);
808 // Single cell, not block
809 const Real dvx=cell.get_velocity_grid_cell_size(popID)[0];
810 const Real dvy=cell.get_velocity_grid_cell_size(popID)[1];
811 const Real dvz=cell.get_velocity_grid_cell_size(popID)[2];
812
813 while (search) {
814 if (0.1 * minValue > projects::MaxwellianPhaseSpaceDensity(counter*dV[0]+0.5*dvx, 0.5*dvy, 0.5*dvz, T, rho, mass) || counter > vblocks_ini[0]) {
815 search = false;
816 }
817 counter++;
818 }
819 counter+=2;
820
821 Real vRadiusSquared = (Real)counter * (Real)counter * dV[0] * dV[0];
822
823 #ifndef USE_GPU
824 // sphere volume is 4/3 pi r^3, approximate that 5*counterX*counterY*counterZ is enough.
825 vmesh::LocalID currentMaxSize = 5*counter*counter*counter;
826 vmesh->setNewSize(currentMaxSize);
827 GIDbuffer = vmesh->getGrid()->data();
828 #endif
829
830 vmesh::LocalID LID = 0;
831 for (uint kv=0; kv<vblocks_ini[2]; ++kv) {
832 for (uint jv=0; jv<vblocks_ini[1]; ++jv) {
833 for (uint iv=0; iv<vblocks_ini[0]; ++iv) {
834 const vmesh::GlobalID GID = vmesh->getGlobalID(iv,jv,kv);
835
836 cell.get_velocity_block_coordinates(popID,GID,V_crds);
837 V_crds[0] += 0.5*dV[0] - VX0;
838 V_crds[1] += 0.5*dV[1] - VY0;
839 V_crds[2] += 0.5*dV[2] - VZ0;
840 Real R2 = ((V_crds[0])*(V_crds[0])
841 + (V_crds[1])*(V_crds[1])
842 + (V_crds[2])*(V_crds[2]));
843
844 #ifndef USE_GPU
845 if (LID >= currentMaxSize) {
846 currentMaxSize = LID + counter*counter*counter;
847 vmesh->setNewSize(currentMaxSize);
848 GIDbuffer = vmesh->getGrid()->data();
849 }
850 #endif
851 if (R2 < vRadiusSquared) {
852 GIDbuffer[LID] = GID;
853 LID++;
854 }
855 }
856 }
857 }
858
859 // Set final size of vmesh
860 cell.get_population(popID).N_blocks = LID;
861
862 #ifdef USE_GPU
863 // Copy data into place
864 cell.dev_resize_vmesh(popID,LID);
865 vmesh::GlobalID *GIDtarget = vmesh->getGrid()->data();
866 gpuStream_t stream = gpu_getStream();
867 CHK_ERR( gpuMemcpyAsync(GIDtarget, GIDbuffer, LID*sizeof(vmesh::GlobalID), gpuMemcpyHostToDevice, stream));
868 CHK_ERR( gpuStreamSynchronize(stream) );
869 CHK_ERR( gpuFreeHost(GIDbuffer));
870 #else
871 vmesh->setNewSize(LID);
872 #endif
873
874 return LID;
875 }
