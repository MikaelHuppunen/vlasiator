52Real calculateCflSpeed(
53 const Real& v0,
54 const Real& v1,
55 const Real& vA,
56 const Real& vS,
57 const Real& vW
58) {
59 const Real v = sqrt(v0*v0 + v1*v1);
60 const Real vMS = sqrt(vA*vA + vS*vS);
61 return max(v + vMS, v + vW);
62}

160 // Effective wave speeds for advection and CFL calculation
161 // Note that these are calculated as if the plasma is purely made up of hydrogen, which
162 // is a reasonable approximation if it is proton-dominant.
163 // Simulations which predominantly contain heavier ion species will have to change this!
164 //
165 // See
166 // T E Stringer, Low-frequency waves in an unbounded plasma
167 // Journal of Nuclear Energy. Part C, Plasma Physics, Accelerators, Thermonuclear Research
168 // Volume 5, Number 2, page 89
169 // Section BC in Fig. 1 and Table 1, referenced later e.g. by Gary, for the current one,
170 // and the below for the heavy ions
171 // https://www.ann-geophys.net/26/1605/2008/ (Whistler waves)
172 // and
173 // http://iopscience.iop.org/article/10.1088/0253-6102/43/2/026/meta (Alfven waves)
174 // for details.
175 const Real vA2 = divideIfNonZero(Bmag2, pc::MU_0*rhom); // Alfven speed
176 const Real vS2 = divideIfNonZero(p11+p22+p33, 2.0*rhom); // sound speed, adiabatic coefficient 3/2, P=1/3*trace in sound speed
177// const Real vW = Parameters::ohmHallTerm > 0 ? divideIfNonZero(2.0*M_PI*vA2*pc::MASS_PROTON, perBGrid.DX*pc::CHARGE*sqrt(Bmag2)) : 0.0; // whistler speed
178 const Real vW = Parameters::ohmHallTerm > 0 ?
179 sqrt(vA2) * (1 + divideIfNonZero(2*M_PI*M_PI*pc::MASS_PROTON*pc::MASS_PROTON, perBGrid.DX*perBGrid.DX*rhom*pc::CHARGE*pc::CHARGE*pc::MU_0)
180 / sqrt(1 + divideIfNonZero( M_PI*M_PI*pc::MASS_PROTON*pc::MASS_PROTON, perBGrid.DX*perBGrid.DX*rhom*pc::CHARGE*pc::CHARGE*pc::MU_0)))

182
183 ret_vA = sqrt(vA2);
184 ret_vS = sqrt(vS2);
185 ret_vW = vW;
