156 calcCellParameters(cell,0.0);

171 // the moment no matter what when this function is called.
172 calculateCellMoments(cell,true,false,true);

175 /*
176 Brute force function, returns all possible velocity blocks. Much preferred to use
177 projectTriAxisSearch
178 */
179 uint Project::findBlocksToInitialize(spatial_cell::SpatialCell* cell,const uint popID) const {
180 const vmesh::LocalID* vblocks_ini = cell->get_velocity_grid_length(popID);
181 vmesh::VelocityMesh *vmesh = cell->get_velocity_mesh(popID);
182
183 const uint blocksCount = vblocks_ini[0]*vblocks_ini[1]*vblocks_ini[2];
184
185 #ifdef USE_GPU
186 // Host-pinned memory buffer
187 vmesh::GlobalID *GIDbuffer;
188 CHK_ERR( gpuMallocHost((void**)&GIDbuffer,blocksCount*sizeof(vmesh::GlobalID)) );
189 cell->dev_resize_vmesh(popID,blocksCount);
190 #else
191 vmesh->setNewSize(blocksCount);
192 vmesh::GlobalID *GIDbuffer = vmesh->getGrid()->data();
193 #endif
194
195 vmesh::LocalID LID = 0;
196 for (uint kv=0; kv<vblocks_ini[2]; ++kv) {
197 for (uint jv=0; jv<vblocks_ini[1]; ++jv) {
198 for (uint iv=0; iv<vblocks_ini[0]; ++iv) {
199 vmesh::LocalID blockIndices[3];
200 blockIndices[0] = iv;
201 blockIndices[1] = jv;
202 blockIndices[2] = kv;
203 const vmesh::GlobalID GID = cell->get_velocity_block(popID,blockIndices);
204 GIDbuffer[LID] = GID;
205 LID++;
206 }
207 }
208 }
209 cell->get_population(popID).N_blocks = LID;
210
211 #ifdef USE_GPU
212 vmesh::GlobalID *GIDtarget = vmesh->getGrid()->data();
213 gpuStream_t stream = gpu_getStream();
214 CHK_ERR( gpuMemcpyAsync(GIDtarget, GIDbuffer, blocksCount*sizeof(vmesh::GlobalID), gpuMemcpyHostToDevice, stream));
215 CHK_ERR( gpuStreamSynchronize(stream) );
216 CHK_ERR( gpuFreeHost(GIDbuffer));
217 #endif
218
219 return LID;
220 }

239 void Project::setVelocitySpace(const uint popID,SpatialCell* cell) const {
240 // Find list of blocks to initialize. The project.cpp version returns
241 // all possible blocks, projectTriAxisSearch provides a more educated guess.
242
243 const uint nRequested = this->findBlocksToInitialize(cell,popID);
244 // stores in vmesh->getGrid() (localToGlobalMap)
245 // with count in cell->get_population(popID).N_blocks
246
247 // Set and apply the reservation value
248 #ifdef USE_GPU
249 cell->setReservation(popID,nRequested,true); // Force to this value
250 cell->applyReservation(popID);
251 #endif
252
253 // Resize and populate mesh
254 cell->prepare_to_receive_blocks(popID);
255
256 // Call project-specific fill function, which loops over all requested blocks,
257 // fills v-space into target
258 const Realf nullsum = fillPhaseSpace(cell, popID, nRequested);
259 if (rescalesDensity(popID) == true) {
260 rescaleDensity(cell,popID);
261 }
262 return;
263 }

276 void Project::rescaleDensity(spatial_cell::SpatialCell* cell,const uint popID) const {
277 // Re-scale densities
278 Real sum = 0.0;
279 Realf* data = cell->get_data(popID);
280 const Real* blockParams = cell->get_block_parameters(popID);
281 for (vmesh::LocalID blockLID=0; blockLID<cell->get_number_of_velocity_blocks(popID); ++blockLID) {
282 Real tmp = 0.0;
283 for (unsigned int i=0; i<WID3; ++i) {
284 tmp += data[blockLID*WID3+i];
285 }
286 const Real DV3 = blockParams[BlockParams::DVX]*blockParams[BlockParams::DVY]*blockParams[BlockParams::DVZ];
287 sum += tmp*DV3;
288 blockParams += BlockParams::N_VELOCITY_BLOCK_PARAMS;
289 }
290
291 const Real correctSum = getCorrectNumberDensity(cell,popID);
292 const Real ratio = correctSum / sum;
293
294 for (size_t i=0; i<cell->get_number_of_velocity_blocks(popID)*WID3; ++i) {
295 data[i] *= ratio;
296 }
297 }

354 /*
355 Refine cells of mpiGrid. Each project that wants refinement should implement this function.
356 Base class function uses AMR box half width parameters
357 */
358 bool Project::refineSpatialCells( dccrg::Dccrg<spatial_cell::SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid ) const {
359 phiprof::Timer refineSCTimer {"Project: refine spatial cells"};
360 int myRank;
361 MPI_Comm_rank(MPI_COMM_WORLD,&myRank);
362
363 if(myRank == MASTER_RANK) {
364 std::cout << "Maximum refinement level is " << mpiGrid.mapping.get_maximum_refinement_level() << std::endl;
365 }
366
367 std::vector<bool> refineSuccess;
368
369 for (int level = 0; level < mpiGrid.mapping.get_maximum_refinement_level(); level++) {
370 int refineCount = 0;
371 for (int n = 0; n < P::amrBoxNumber; n++) {
372 if (level < P::amrBoxMaxLevel[n]) {
373 const int maxloop = pow(2, level+1);
374 #pragma omp parallel for schedule(guided) collapse(3)
375 for (int i = 0; i < maxloop * (int)P::amrBoxHalfWidthX[n]; ++i) {
376 for (int j = 0; j < maxloop * (int)P::amrBoxHalfWidthY[n]; ++j) {
377 for (int k = 0; k < maxloop * (int)P::amrBoxHalfWidthZ[n]; ++k) {
378
379 std::array<double,3> xyz;
380 xyz[0] = P::amrBoxCenterX[n] + (0.5 + i - pow(2, level)*P::amrBoxHalfWidthX[n]) * P::dx_ini / pow(2, level);
381 xyz[1] = P::amrBoxCenterY[n] + (0.5 + j - pow(2, level)*P::amrBoxHalfWidthY[n]) * P::dy_ini / pow(2, level);
382 xyz[2] = P::amrBoxCenterZ[n] + (0.5 + k - pow(2, level)*P::amrBoxHalfWidthZ[n]) * P::dz_ini / pow(2, level);
383
384 #pragma omp critical
385 {
386 if (mpiGrid.refine_completely_at(xyz)) {
387 refineCount++;
388 #ifdef DEBUG_REFINE
389 CellID myCell = mpiGrid.get_existing_cell(xyz);
390 std::cout << "Rank " << myRank << " is refining cell " << myCell << std::endl;
391 #endif
392 } // if
393 }
394 } // box z
395 } // box y
396 } // box x
397 } // if (P::amrBoxMaxLevel <= level)
398 } // box number
399 int totalRefineCount;
400 MPI_Allreduce(&refineCount, &totalRefineCount, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
401 if(totalRefineCount > 0) {
402 std::vector<CellID> refinedCells = mpiGrid.stop_refining();
403 #ifdef DEBUG_REFINE
404 if(refinedCells.size() > 0) {
405 std::cerr << "Refined cells produced by rank " << myRank << " for level " << level << " are: ";
406 for (auto cellid : refinedCells) {
407 std::cout << cellid << " ";
408 }
409 std::cout << endl;
410 }
411 #endif
412
413 // Don't do LB, as this function is called only before v-spaces have been created
414 // mpiGrid.balance_load();
415 }
416 if(myRank == MASTER_RANK) {
417 std::cout << "Finished level of refinement " << level+1 << endl;
418 }
419
420 } // refinement levels
421 return true;
422 }

429 bool Project::shouldRefineCell(dccrg::Dccrg<spatial_cell::SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid, CellID id, Real r_max2) const {
430 // Evaluate possible refinement for this cell
431
432 // Cells too far from the ionosphere should not be refined but
433 // induced refinement still possible just beyond this r_max2 limit.
434
435 std::array<double,3> xyz {mpiGrid.get_center(id)};
436 SpatialCell* cell {mpiGrid[id]};
437 int refLevel {mpiGrid.get_refinement_level(id)};
438 Real r2 {pow(xyz[0], 2) + pow(xyz[1], 2) + pow(xyz[2], 2)};
439
440 bool alpha1ShouldRefine = (P::useAlpha1 && cell->parameters[CellParams::AMR_ALPHA1] > P::alpha1RefineThreshold);
441 bool alpha2ShouldRefine = (P::useAlpha2 && cell->parameters[CellParams::AMR_ALPHA2] > P::alpha2RefineThreshold);
442 bool vorticityShouldRefine = (P::useVorticity && cell->parameters[CellParams::AMR_VORTICITY] > P::vorticityRefineThreshold);
443 bool anisotropyShouldRefine = (P::useAnisotropy && cell->parameters[CellParams::P_ANISOTROPY] >= 0 && cell->parameters[CellParams::P_ANISOTROPY] < P::anisotropyRefineThreshold && refLevel < P::anisotropyMaxReflevel);
444
445 bool shouldRefine {
446 (r2 < r_max2) && (
447 alpha1ShouldRefine ||
448 alpha2ShouldRefine ||
449 vorticityShouldRefine ||
450 anisotropyShouldRefine
451 )
452 };
453
454 if (P::refineBoxNumber > 0) {
455 bool outsideAllRefineBoxes = true;
456 for (int i = 0; i < P::refineBoxNumber; i++) {
457 if (
458 // If this cell is planned to be refined and is inside any of the refinement boxes, then go ahead with refinement.
459 // Induced refinement still possible just beyond that limit.
460 (xyz[0] >= P::refinementMinX.at(i)) && (xyz[0] <= P::refinementMaxX.at(i))
461 && (xyz[1] >= P::refinementMinY.at(i)) && (xyz[1] <= P::refinementMaxY.at(i))
462 && (xyz[2] >= P::refinementMinZ.at(i)) && (xyz[2] <= P::refinementMaxZ.at(i))) {
463 outsideAllRefineBoxes = false;
464 break;
465 }
466 }
467
468 shouldRefine = shouldRefine && !outsideAllRefineBoxes;
469 }
470
471 return shouldRefine;
472 }

474 bool Project::shouldUnrefineCell(dccrg::Dccrg<spatial_cell::SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid, CellID id, Real r_max2) const {
475 // Evaluate possible unrefinement for this cell
476
477 // Cells too far from the ionosphere should be unrefined but
478 // induced refinement still possible just beyond this r_max2 limit.
479
480 std::array<double,3> xyz {mpiGrid.get_center(id)};
481 SpatialCell* cell {mpiGrid[id]};
482 int refLevel {mpiGrid.get_refinement_level(id)};
483 Real r2 {pow(xyz[0], 2) + pow(xyz[1], 2) + pow(xyz[2], 2)};
484
485 bool alpha1ShouldUnrefine = (!P::useAlpha1 || cell->parameters[CellParams::AMR_ALPHA1] < P::alpha1CoarsenThreshold);
486 bool alpha2ShouldUnrefine = (!P::useAlpha2 || cell->parameters[CellParams::AMR_ALPHA2] < P::alpha2CoarsenThreshold);
487 bool vorticityShouldUnrefine = (!P::useVorticity || cell->parameters[CellParams::AMR_VORTICITY] < P::vorticityCoarsenThreshold);
488 bool anisotropyShouldUnrefine = (!(P::useAnisotropy && cell->parameters[CellParams::P_ANISOTROPY] >= 0) || cell->parameters[CellParams::P_ANISOTROPY] > P::anisotropyCoarsenThreshold || refLevel > P::anisotropyMaxReflevel);
489
490 bool shouldUnrefine {
491 (r2 > r_max2) || (
492 alpha1ShouldUnrefine &&
493 alpha2ShouldUnrefine &&
494 vorticityShouldUnrefine &&
495 anisotropyShouldUnrefine
496 )
497 };
498
499 if (P::refineBoxNumber > 0) {
500 bool outsideAllRefineBoxes = true;
501 for (int i = 0; i < P::refineBoxNumber; i++) {
502 if (
503 // If this cell is planned to remain at the current refinement level, but is outside all the allowed refinement regions,
504 // attempt to unrefine it instead. (If it is already at the lowest refinement level, DCCRG should not go belly-up.)
505 // Induced refinement still possible just beyond that limit.
506 (xyz[0] >= P::refinementMinX.at(i)) && (xyz[0] <= P::refinementMaxX.at(i))
507 && (xyz[1] >= P::refinementMinY.at(i)) && (xyz[1] <= P::refinementMaxY.at(i))
508 && (xyz[2] >= P::refinementMinZ.at(i)) && (xyz[2] <= P::refinementMaxZ.at(i))) {
509 outsideAllRefineBoxes = false;
510 break;
511 }
512 }
513
514 shouldUnrefine = shouldUnrefine || outsideAllRefineBoxes;
515 }
516
517 return shouldUnrefine;
518 }

520 uint64_t Project::adaptRefinement( dccrg::Dccrg<spatial_cell::SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid ) const {
521 phiprof::Timer refinesTimer {"Set refines"};
522 int myRank;
523 MPI_Comm_rank(MPI_COMM_WORLD,&myRank);
524
525 uint64_t refines {0};
526 if (!P::useAlpha1 && !P::useAlpha2 && !P::useAnisotropy && !P::useVorticity) {
527 if (myRank == MASTER_RANK) {
528 std::cout << "WARNING All refinement indices disabled" << std::endl;
529 }
530 return refines;
531 }
532
533 const std::vector<CellID> cells {getLocalCells()};
534 Real r_max2 {pow(P::refineRadius, 2)};
535
536 #pragma omp parallel for
537 for (uint cid = 0; cid < cells.size(); ++cid) {
538 CellID id = cells[cid];
539 int refLevel {mpiGrid.get_refinement_level(id)};
540
541 if (!canRefine(mpiGrid[id])) {
542 // Skip refining, touching boundaries during runtime breaks everything
543 #pragma omp critical
544 {
545 mpiGrid.dont_refine(id);
546 mpiGrid.dont_unrefine(id);
547 }
548 } else {
549 // Evaluate possible refinement or unrefinement for this cell
550 bool shouldRefine = shouldRefineCell(mpiGrid, id, r_max2);
551 bool shouldUnrefine = shouldUnrefineCell(mpiGrid, id, r_max2);
552
553 // Finally, check neighbors
554 int refined_neighbors {0};
555 int coarser_neighbors {0};
556 for (const auto& [neighbor, dir] : mpiGrid.get_face_neighbors_of(id)) {
557 // Evaluate all face neighbors of the current cell
558 bool shouldRefineNeighbor = shouldRefineCell(mpiGrid, neighbor, r_max2);
559 bool shouldUnrefineNeighbor = shouldUnrefineCell(mpiGrid, neighbor, r_max2);
560 int neighborRef {mpiGrid.get_refinement_level(neighbor)};
561
562 if (neighborRef > refLevel && !shouldUnrefineNeighbor) {
563 ++refined_neighbors;
564 } else if (neighborRef < refLevel && !shouldRefineNeighbor) {
565 ++coarser_neighbors;
566 } else if (shouldRefineNeighbor) {
567 // If neighbor refines, 4 of its children will be this cells refined neighbors
568 refined_neighbors += 4;
569 } else if (shouldUnrefineNeighbor) {
570 ++coarser_neighbors;
571 }
572 }
573 ; // CUDA compilation does not like the start of a new region with nothing, so a null semicolon is here.
574 #pragma omp critical
575 {
576 if ((shouldRefine || refined_neighbors > 12) && refLevel < P::amrMaxAllowedSpatialRefLevel) {
577 // Refine a cell if a majority of its neighbors are refined or about to be
578 // Increment count of refined cells only if we're actually refining
579 refines += mpiGrid.refine_completely(id) && refLevel < P::amrMaxSpatialRefLevel;
580 } else if (refLevel > 0 && shouldUnrefine && coarser_neighbors > 0) {
581 // Unrefine a cell only if any of its neighbors is unrefined or about to be
582 // refLevel check prevents dont_refine() being set
583 mpiGrid.unrefine_completely(id);
584 } else {
585 // Ensure no cells above both unrefine thresholds are unrefined
586 mpiGrid.dont_unrefine(id);
587 }
588 }
589 }
590 }
591
592 return refines;
593 }

605 bool Project::filterRefined( dccrg::Dccrg<spatial_cell::SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid ) const {
606 int myRank;
607 MPI_Comm_rank(MPI_COMM_WORLD,&myRank);
608
609 const vector<CellID>& cells = getLocalCells();
610 std::map<CellID, SpatialCell> cellsMap;
611 for (CellID id : cells) {
612 if (mpiGrid[id]->parameters[CellParams::RECENTLY_REFINED]) {
613 cellsMap.insert({id, *mpiGrid[id]});
614 }
615 }
616
617 for (auto cellPair : cellsMap) {
618 CellID id = cellPair.first;
619 // To preserve the mean, we must only consider refined cells
620 int refLevel = mpiGrid.get_refinement_level(id);
621 std::vector<CellID> refinedNeighbors;
622 for (auto& neighbor : *mpiGrid.get_neighbors_of(id, Neighborhoods::NEAREST)) {
623 if (mpiGrid[neighbor.first]->parameters[CellParams::RECENTLY_REFINED] && mpiGrid.get_refinement_level(neighbor.first) == refLevel) {
624 refinedNeighbors.push_back(neighbor.first);
625 }
626 }
627
628 if (refinedNeighbors.size() == 7) {
629 continue; // Simple heuristic, in these cases all neighbors are from the same parent cell, ergo are identical
630 }
631
632 // In boxcar filter, we take the average of each of the neighbors and the cell itself. For each missing neighbour, add the cell one more time
633 Real fluffiness = (Real) refinedNeighbors.size() / 27.0;
634 for (uint popID=0; popID<getObjectWrapper().particleSpecies.size(); ++popID) {
635 SBC::averageCellData(mpiGrid, refinedNeighbors, &cellPair.second, popID, fluffiness);
636 }
637
638 calculateCellMoments(&cellPair.second, true, false);
639 }
640
641 for (auto cellPair : cellsMap) {
642 *mpiGrid[cellPair.first] = cellPair.second;
643 mpiGrid[cellPair.first]->parameters[CellParams::RECENTLY_REFINED] = 0;
644 }
645
646 if (myRank == MASTER_RANK) {
647 std::cout << "Filtered refined cells!" << std::endl;
648 }
649
650 return true;
651 }
