74 void Outflow::getParameters() {
75 int myRank;
76 MPI_Comm_rank(MPI_COMM_WORLD,&myRank);
77 Readparameters::get("outflow.faceNoFields", this->faceNoFieldsList);
78 Readparameters::get("outflow.precedence", precedence);
79 uint reapply;
80 Readparameters::get("outflow.reapplyUponRestart", reapply);
81 this->applyUponRestart = false;
82 if(reapply == 1) {
83 this->applyUponRestart = true;
84 }
85
86 // Per-species parameters
87 for(uint i=0; i< getObjectWrapper().particleSpecies.size(); i++) {
88 const string& pop = getObjectWrapper().particleSpecies[i].name;
89 OutflowSpeciesParameters sP;
90
91 // Unless we find out otherwise, we assume that this species will not be treated at any boundary
92 for(int j=0; j<6; j++) {
93 sP.facesToSkipVlasov[j] = true;
94 }
95
96 vector<string> thisSpeciesFaceList;
97 Readparameters::get(pop + "_outflow.face", thisSpeciesFaceList);
98
99 for(auto& face : thisSpeciesFaceList) {
100 if(face == "x+") { facesToProcess[0] = true; sP.facesToSkipVlasov[0] = false; }
101 if(face == "x-") { facesToProcess[1] = true; sP.facesToSkipVlasov[1] = false; }
102 if(face == "y+") { facesToProcess[2] = true; sP.facesToSkipVlasov[2] = false; }
103 if(face == "y-") { facesToProcess[3] = true; sP.facesToSkipVlasov[3] = false; }
104 if(face == "z+") { facesToProcess[4] = true; sP.facesToSkipVlasov[4] = false; }
105 if(face == "z-") { facesToProcess[5] = true; sP.facesToSkipVlasov[5] = false; }
106 }
107
108 Readparameters::get(pop + "_outflow.reapplyFaceUponRestart", sP.faceToReapplyUponRestartList);
109 array<string, 6> vlasovSysBoundarySchemeName;
110 Readparameters::get(pop + "_outflow.vlasovScheme_face_x+", vlasovSysBoundarySchemeName[0]);
111 Readparameters::get(pop + "_outflow.vlasovScheme_face_x-", vlasovSysBoundarySchemeName[1]);
112 Readparameters::get(pop + "_outflow.vlasovScheme_face_y+", vlasovSysBoundarySchemeName[2]);
113
114 Readparameters::get(pop + "_outflow.vlasovScheme_face_y-", vlasovSysBoundarySchemeName[3]);
115 Readparameters::get(pop + "_outflow.vlasovScheme_face_z+", vlasovSysBoundarySchemeName[4]);
116 Readparameters::get(pop + "_outflow.vlasovScheme_face_z-", vlasovSysBoundarySchemeName[5]);
117 for(uint j=0; j<6 ; j++) {
118 if(vlasovSysBoundarySchemeName[j] == "None") {
119 sP.faceVlasovScheme[j] = vlasovscheme::NONE;
120 } else if (vlasovSysBoundarySchemeName[j] == "Copy") {
121 sP.faceVlasovScheme[j] = vlasovscheme::COPY;
122 } else {
123 abort_mpi("ERROR: " + vlasovSysBoundarySchemeName[j] + " is an invalid Outflow Vlasov scheme!");
124 }
125 }
126
127 Readparameters::get(pop + "_outflow.quench", sP.quenchFactor);
128
129 speciesParams.push_back(sP);
130 }

133 void Outflow::initSysBoundary(
134 creal& t,
135 Project &project
136 ) {
137 /* The array of bool describes which of the x+, x-, y+, y-, z+, z- faces are to have outflow system boundary conditions.
138 * A true indicates the corresponding face will have outflow.
139 * The 6 elements correspond to x+, x-, y+, y-, z+, z- respectively.
140 */
141 for(uint i=0; i<6; i++) {
142 facesToProcess[i] = false;
143 facesToSkipFields[i] = false;
144 facesToReapply[i] = false;
145 }
146
147 this->getParameters();
148
149 dynamic = false;
150
151 vector<string>::const_iterator it;
152 for (it = faceNoFieldsList.begin();
153 it != faceNoFieldsList.end();
154 it++) {
155 if(*it == "x+") facesToSkipFields[0] = true;
156 if(*it == "x-") facesToSkipFields[1] = true;
157 if(*it == "y+") facesToSkipFields[2] = true;
158 if(*it == "y-") facesToSkipFields[3] = true;
159 if(*it == "z+") facesToSkipFields[4] = true;
160 if(*it == "z-") facesToSkipFields[5] = true;
161 }
162
163 for(uint i=0; i< getObjectWrapper().particleSpecies.size(); i++) {
164 OutflowSpeciesParameters& sP = this->speciesParams[i];
165 for (it = sP.faceToReapplyUponRestartList.begin();
166 it != sP.faceToReapplyUponRestartList.end();
167 it++) {
168 if(*it == "x+") facesToReapply[0] = true;
169 if(*it == "x-") facesToReapply[1] = true;
170 if(*it == "y+") facesToReapply[2] = true;
171 if(*it == "y-") facesToReapply[3] = true;
172 if(*it == "z+") facesToReapply[4] = true;
173 if(*it == "z-") facesToReapply[5] = true;
174 }
175 }

179 void Outflow::assignSysBoundary(dccrg::Dccrg<SpatialCell, dccrg::Cartesian_Geometry>& mpiGrid,
180 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid) {
181
182 bool doAssign;
183 array<bool,6> isThisCellOnAFace;
184
185 // Assign boundary flags to local DCCRG cells
186 const vector<CellID>& cells = getLocalCells();
187 for(const auto& dccrgId : cells) {
188 if(mpiGrid[dccrgId]->sysBoundaryFlag == sysboundarytype::DO_NOT_COMPUTE) continue;
189 creal* const cellParams = &(mpiGrid[dccrgId]->parameters[0]);
190 creal dx = cellParams[CellParams::DX];
191 creal dy = cellParams[CellParams::DY];
192 creal dz = cellParams[CellParams::DZ];
193 creal x = cellParams[CellParams::XCRD] + 0.5*dx;
194 creal y = cellParams[CellParams::YCRD] + 0.5*dy;
195 creal z = cellParams[CellParams::ZCRD] + 0.5*dz;
196
197 isThisCellOnAFace.fill(false);
198 determineFace(isThisCellOnAFace.data(), x, y, z, dx, dy, dz);
199
200 // Comparison of the array defining which faces to use and the array telling on which faces this cell is
201 doAssign = false;
202 for(int j=0; j<6; j++) {
203 doAssign = doAssign || (facesToProcess[j] && isThisCellOnAFace[j]);
204 }
205 if(doAssign) {
206 mpiGrid[dccrgId]->sysBoundaryFlag = this->getIndex();
207 }
208 }
209
210 // Assign boundary flags to local fsgrid cells
211 const array<FsGridTools::FsIndex_t, 3> gridDims(technicalGrid.getLocalSize());
212 for (FsGridTools::FsIndex_t k=0; k<gridDims[2]; k++) {
213 for (FsGridTools::FsIndex_t j=0; j<gridDims[1]; j++) {
214 for (FsGridTools::FsIndex_t i=0; i<gridDims[0]; i++) {
215 const auto& coords = technicalGrid.getPhysicalCoords(i,j,k);
216
217 // Shift to the center of the fsgrid cell
218 auto cellCenterCoords = coords;
219 cellCenterCoords[0] += 0.5 * technicalGrid.DX;
220 cellCenterCoords[1] += 0.5 * technicalGrid.DY;
221 cellCenterCoords[2] += 0.5 * technicalGrid.DZ;
222 const auto refLvl = mpiGrid.get_refinement_level(mpiGrid.get_existing_cell(cellCenterCoords));
223
224 if (refLvl == -1) {
225 abort_mpi("ERROR: Could not get refinement level of remote DCCRG cell!", 1);
226 }
227
228 creal dx = P::dx_ini / pow(2, refLvl);
229 creal dy = P::dy_ini / pow(2, refLvl);
230 creal dz = P::dz_ini / pow(2, refLvl);
231
232 isThisCellOnAFace.fill(false);
233 doAssign = false;
234
235 determineFace(isThisCellOnAFace.data(), cellCenterCoords[0], cellCenterCoords[1], cellCenterCoords[2], dx, dy, dz);
236 for(int iface=0; iface<6; iface++) doAssign = doAssign || (facesToProcess[iface] && isThisCellOnAFace[iface]);
237 if(doAssign) {
238 technicalGrid.get(i,j,k)->sysBoundaryFlag = this->getIndex();
239 }
240 }
241 }
242 }

245 void Outflow::applyInitialState(
246 dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
247 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
248 FsGrid< array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & perBGrid,
249 FsGrid<std::array<Real, fsgrids::bgbfield::N_BGB>, FS_STENCIL_WIDTH>& BgBGrid,
250 Project &project
251 ) {
252 const vector<CellID>& cells = getLocalCells();
253 #pragma omp parallel for schedule(static)
254 for (uint i=0; i<cells.size(); ++i) {
255 CellID id = cells[i];
256 SpatialCell* cell = mpiGrid[id];
257 if (cell->sysBoundaryFlag != this->getIndex()) {
258 continue;
259 }
260
261 bool doApply = true;
262
263 if(Parameters::isRestart) {
264 std::array<bool, 6> isThisCellOnAFace;
265 determineFace(isThisCellOnAFace, mpiGrid, id);
266
267 doApply=false;
268 // Comparison of the array defining which faces to use and the array telling on which faces this cell is
269 for (uint j=0; j<6; j++) {
270 doApply = doApply || (facesToReapply[j] && isThisCellOnAFace[j]);
271 }
272 }
273
274 if (doApply) {
275 project.setCell(cell); // We set everything including VDF even in L2 cells to avoid a pile of spaghetti. Won't get communicated.
276 cell->parameters[CellParams::RHOM_DT2] = cell->parameters[CellParams::RHOM];
277 cell->parameters[CellParams::RHOQ_DT2] = cell->parameters[CellParams::RHOQ];
278 cell->parameters[CellParams::VX_DT2] = cell->parameters[CellParams::VX];
279 cell->parameters[CellParams::VY_DT2] = cell->parameters[CellParams::VY];
280 cell->parameters[CellParams::VZ_DT2] = cell->parameters[CellParams::VZ];
281 cell->parameters[CellParams::P_11_DT2] = cell->parameters[CellParams::P_11];
282 cell->parameters[CellParams::P_22_DT2] = cell->parameters[CellParams::P_22];
283 cell->parameters[CellParams::P_33_DT2] = cell->parameters[CellParams::P_33];
284 }

395 void Outflow::vlasovBoundaryCondition(
396 dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
397 const CellID& cellID,
398 const uint popID,
399 const bool calculate_V_moments
400 ) {
401
402 const OutflowSpeciesParameters& sP = this->speciesParams[popID];
403 if (mpiGrid[cellID]->sysBoundaryFlag != this->getIndex()) {
404 return;
405 }
406
407 std::array<bool, 6> isThisCellOnAFace;
408 determineFace(isThisCellOnAFace, mpiGrid, cellID, true);
409
410 for(uint i=0; i<6; i++) {
411 if(isThisCellOnAFace[i] && facesToProcess[i] && !sP.facesToSkipVlasov[i]) {
412 switch(sP.faceVlasovScheme[i]) {
413 case vlasovscheme::NONE:
414 break;
415 case vlasovscheme::COPY:
416 if(mpiGrid[cellID]->sysBoundaryLayer == 1) {
417 vlasovBoundaryCopyFromTheClosestNbr(mpiGrid,cellID,false,popID,calculate_V_moments); // copies VDF too
418 } else {
419 vlasovBoundaryCopyFromTheClosestNbr(mpiGrid,cellID,true,popID,calculate_V_moments); // no VDF copy
420 }
421 break;
422 default:
423 abort_mpi("ERROR: invalid Outflow Vlasov scheme", 1);
424 }
425 }
