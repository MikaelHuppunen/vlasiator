66Eigen::Transform<Real,3,Eigen::Affine> compute_acceleration_transformation(
67 SpatialCell* spatial_cell,
68 const uint popID,
69 const Real& dt) {
70 // total field
71 const Real Bx = spatial_cell->parameters[CellParams::BGBXVOL]+spatial_cell->parameters[CellParams::PERBXVOL];
72 const Real By = spatial_cell->parameters[CellParams::BGBYVOL]+spatial_cell->parameters[CellParams::PERBYVOL];
73 const Real Bz = spatial_cell->parameters[CellParams::BGBZVOL]+spatial_cell->parameters[CellParams::PERBZVOL];
74
75 // perturbed field
76 //const Real perBx = spatial_cell->parameters[CellParams::PERBXVOL];
77 //const Real perBy = spatial_cell->parameters[CellParams::PERBYVOL];
78 //const Real perBz = spatial_cell->parameters[CellParams::PERBZVOL];
79
80 // read in derivatives need for curl of B (only perturbed, curl of background field is always 0!)
81 const Real dBXdy = spatial_cell->derivativesBVOL[bvolderivatives::dPERBXVOLdy];
82 const Real dBXdz = spatial_cell->derivativesBVOL[bvolderivatives::dPERBXVOLdz];
83 const Real dBYdx = spatial_cell->derivativesBVOL[bvolderivatives::dPERBYVOLdx];
84
85 const Real dBYdz = spatial_cell->derivativesBVOL[bvolderivatives::dPERBYVOLdz];
86 const Real dBZdx = spatial_cell->derivativesBVOL[bvolderivatives::dPERBZVOLdx];
87 const Real dBZdy = spatial_cell->derivativesBVOL[bvolderivatives::dPERBZVOLdy];
88
89 const Eigen::Matrix<Real,3,1> B(Bx,By,Bz);
90 Eigen::Matrix<Real,3,1> unit_B(B.normalized());
91
92 // If B equals zero then gyro_period and unit_B are NAN.
93 // Guard against that by adding epsilons:
94 const Real B_mag = B.norm() + 1e-30;
95 if (B_mag < 1e-28) {
96 unit_B(0,0) = 0; unit_B(1,0) = 0; unit_B(2,0) = 1;
97 }
98
99 const Real gyro_period
100 = 2 * M_PI * getObjectWrapper().particleSpecies[popID].mass
101 / (getObjectWrapper().particleSpecies[popID].charge * B_mag);
102
103 // scale rho for hall term, if user requests
104 const Real EPSILON = 1e10 * numeric_limits<Real>::min();
105 const Real rhoq = spatial_cell->parameters[CellParams::RHOQ_V] + EPSILON;
106 const Real hallRhoq = (rhoq <= Parameters::hallMinimumRhoq ) ? Parameters::hallMinimumRhoq : rhoq ;
107 const Real hallPrefactor = 1.0 / (physicalconstants::MU_0 * hallRhoq );
108
109 Eigen::Matrix<Real,3,1> bulk_velocity(spatial_cell->parameters[CellParams::VX_V],
110 spatial_cell->parameters[CellParams::VY_V],
111 spatial_cell->parameters[CellParams::VZ_V]);
112
113 // compute total transformation
114 Transform<Real,3,Affine> total_transform(Matrix<Real, 4, 4>::Identity()); //CONTINUE
115
116 unsigned int bulk_velocity_substeps; // in this many substeps we iterate forward bulk velocity when the complete transformation is computed (0.1 deg per substep).
117 bulk_velocity_substeps = fabs(dt) / fabs(gyro_period*(0.1/360.0));
118 if (bulk_velocity_substeps < 1) {
119 bulk_velocity_substeps=1;
120 }
121
122 const Real substeps_radians = -(2.0*M_PI*dt/gyro_period)/bulk_velocity_substeps; // how many radians each substep is.
123 const Real substeps_dt=dt/bulk_velocity_substeps;
124 Eigen::Matrix<Real,3,1> EgradPe(
125 spatial_cell->parameters[CellParams::EXGRADPE],
126 spatial_cell->parameters[CellParams::EYGRADPE],
127 spatial_cell->parameters[CellParams::EZGRADPE]);
128
129 for (uint i=0; i<bulk_velocity_substeps; ++i) {
130 // rotation origin is the point through which we place our rotation axis (direction of which is unitB).
131 // first add bulk velocity (using the total transform computed this far.
132 Eigen::Matrix<Real,3,1> rotation_pivot(total_transform*bulk_velocity);
133
134 //include lorentzHallTerm (we should include, always)
135 rotation_pivot[0]-= hallPrefactor*(dBZdy - dBYdz);
136 rotation_pivot[1]-= hallPrefactor*(dBXdz - dBZdx);
137 rotation_pivot[2]-= hallPrefactor*(dBYdx - dBXdy);
138
139 // add to transform matrix the small rotation around pivot
140 // when added like this, and not using *= operator, the transformations
141 // are in the correct order
142 total_transform = Translation<Real,3>(-rotation_pivot)*total_transform;
143 total_transform = AngleAxis<Real>(substeps_radians,unit_B)*total_transform;
144 total_transform = Translation<Real,3>(rotation_pivot)*total_transform;
145
146 // Electron pressure gradient term
147 if(Parameters::ohmGradPeTerm > 0) {
148 total_transform=Translation<Real,3>( (fabs(getObjectWrapper().particleSpecies[popID].charge)/getObjectWrapper().particleSpecies[popID].mass) * EgradPe * substeps_dt) * total_transform;
149 }
150 }
151
152 return total_transform;
153}
