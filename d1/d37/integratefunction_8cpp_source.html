37double lineAverage(
38 const T3DFunction& f1,
39 coordinate line,
40 double accuracy,
41 const double r1[3],
42 double L
43) {
44 using namespace std::placeholders;
45 double value;
46 const double norm = 1/L;
47 const double acc = accuracy*L;
48 const double a = r1[line];
49 const double b = r1[line] + L;
50
51 switch (line) {
52 case X:
53 {
54 T1DFunction f=std::bind(f1,std::placeholders::_1,r1[1],r1[2]);
55 value= Romberg(f,a,b,acc)*norm;
56 }
57 break;
58 case Y:
59 {
60 T1DFunction f=std::bind(f1,r1[0],std::placeholders::_1,r1[2]);
61 value= Romberg(f,a,b,acc)*norm;
62 }
63 break;
64 case Z:
65 {
66 T1DFunction f=std::bind(f1,r1[0],r1[1],std::placeholders::_1);
67 value= Romberg(f,a,b,acc)*norm;
68 }
69 break;
70 default:
71 cerr << "*** lineAverage is bad\n";
72 value = 0.0;
73 break;
74 }
75 return value;
76}

79double surfaceAverage(
80 const T3DFunction& f1,
81 coordinate face, double accuracy,
82 const double r1[3],
83 double L1,
84 double L2
85) {
86 using namespace std::placeholders;
87 double value;
88 const double acc = accuracy*L1*L2;
89 const double norm = 1/(L1*L2);
90 switch (face) {
91 case X:
92 {
93 T2DFunction f = std::bind(f1,r1[0],std::placeholders::_1,std::placeholders::_2);
94 value = Romberg(f, r1[1],r1[1]+L1, r1[2],r1[2]+L2, acc)*norm;
95 }
96 break;
97 case Y:
98 {
99 T2DFunction f = std::bind(f1,std::placeholders::_1,r1[1],std::placeholders::_2);
100 value = Romberg(f, r1[0],r1[0]+L1, r1[2],r1[2]+L2, acc)*norm;
101 }
102 break;
103 case Z:
104 {
105 T2DFunction f = std::bind(f1,std::placeholders::_1,std::placeholders::_2,r1[2]);
106 value = Romberg(f, r1[0],r1[0]+L1, r1[1],r1[1]+L2, acc)*norm;
107 }
108 break;
109 default:
110 cerr << "*** SurfaceAverage is bad\n";
111 exit(1);
112 break;
113 }
114 return value;
115}

118double volumeAverage(
119 const T3DFunction& f1,
120 double accuracy,
121 const double r1[3],
122 const double r2[3]
123) {
124 double value;
125 const double acc = accuracy*(r2[0]-r1[0])*(r2[1]-r1[1])*(r2[2]-r1[2]);
126 const double norm = 1.0/((r2[0]-r1[0])*(r2[1]-r1[1])*(r2[2]-r1[2]));
127 value= Romberg(f1, r1[0],r2[0], r1[1],r2[1], r1[2],r2[2], acc)*norm;
128 return value;
129}
