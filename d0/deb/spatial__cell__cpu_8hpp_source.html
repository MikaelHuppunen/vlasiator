94 // Constructor, destructor
95 Population() {
96 vmesh = new vmesh::VelocityMesh();
97 blockContainer = new vmesh::VelocityBlockContainer();
98 // Set values to zero in case of zero-block populations
99 RHO = RHO_R = RHO_V = RHOLOSSADJUST = velocityBlockMinValue = ACCSUBCYCLES = N_blocks = 0;
100 for (uint i=0; i<2; ++i) {
101 max_dt[i] = 0;
102 }
103 for (uint i=0; i<3; ++i) {
104 V[i] = V_R[i] = V_V[i] = 0;
105 }
106 for (uint i=0; i<6; i++) {
107 P[i] = P_R[i] = P_V[i] = 0;
108 }
109 }
110 ~Population() {
111 delete vmesh;
112 delete blockContainer;
113 }
114 Population(const Population& other) {
115 vmesh = new vmesh::VelocityMesh(*(other.vmesh));
116 blockContainer = new vmesh::VelocityBlockContainer(*(other.blockContainer));
117
118 RHO = other.RHO;
119 RHO_R = other.RHO_R;
120 RHO_V = other.RHO_V;
121 RHOLOSSADJUST = other.RHOLOSSADJUST;
122 velocityBlockMinValue = other.velocityBlockMinValue;
123 ACCSUBCYCLES = other.ACCSUBCYCLES;
124 N_blocks = other.N_blocks;
125 for (uint i=0; i<2; ++i) {
126 max_dt[i] = other.max_dt[i];
127 }
128 for (uint i=0; i<3; ++i) {
129 V[i] = other.V[i];
130 V_R[i] = other.V_R[i];
131 V_V[i] = other.V_V[i];
132 }
133 for (uint i=0; i<6; i++) {
134 P[i] = other.P[i];
135 P_R[i] = other.P_R[i];
136 P_V[i] = other.P_V[i];
137 }
138 }

140 delete vmesh;

142 vmesh = new vmesh::VelocityMesh(*(other.vmesh));
143 blockContainer = new vmesh::VelocityBlockContainer(*(other.blockContainer));
144
145 RHO = other.RHO;

147 RHO_V = other.RHO_V;
148 RHOLOSSADJUST = other.RHOLOSSADJUST;
149 velocityBlockMinValue = other.velocityBlockMinValue;
150 ACCSUBCYCLES = other.ACCSUBCYCLES;
151 N_blocks = other.N_blocks;
152 for (uint i=0; i<2; ++i) {
153 max_dt[i] = other.max_dt[i];
154 }
155 for (uint i=0; i<3; ++i) {
156 V[i] = other.V[i];
157 V_R[i] = other.V_R[i];
158 V_V[i] = other.V_V[i];
159 }
160 for (uint i=0; i<6; i++) {
161 P[i] = other.P[i];
162 P_R[i] = other.P_R[i];
163 P_V[i] = other.P_V[i];
164 }
165 return *this;

167 void ResizeClear(const uint newSize) {
168 // Resizes the vmesh localToGlobalMap, clears the vmesh GlobalToLocalMap,
169 // and resizes the velocity block container.
170 // Contents of the localToGlobalMap or the VBC are not edited.
171 vmesh->setNewSize(newSize);
172 vmesh->clearMap(newSize);
173 blockContainer->setNewSize(newSize);
174 }
175 void Scale(creal factor) {
176 RHO *= factor;
177 RHO_R *= factor;
178 RHO_V *= factor;
179 for (uint i=0; i<3; ++i) {
180 P[i] *= factor;
181 P_R[i] *= factor;
182 P_V[i] *= factor;
183 }
184 // Now loop over whole velocity space and scale the values
185 for (vmesh::LocalID blockLID=0; blockLID < vmesh->size(); ++blockLID) {
186 // Pointer to target block data
187 Realf* toData = blockContainer->getData(blockLID);
188 // Scale data
189 for (uint i=0; i<WID3; ++i) {
190 toData[i] = toData[i] * factor;
191 }
192 }
193 }
194 void Increment(const Population& other, creal factor) {
195 // Note: moments will be invalidated.
196 // Loop over the whole velocity space, and add scaled values.
197 for (vmesh::LocalID incBlockLID=0; incBlockLID<(other.vmesh)->size(); ++incBlockLID) {
198 const Realf* fromData = (other.blockContainer)->getData(incBlockLID);
199
200 // Global ID of the block containing incoming data
201 vmesh::GlobalID incBlockGID = (other.vmesh)->getGlobalID(incBlockLID);
202
203 // Get local ID of the target block. If the block doesn't exist, create it.
204 vmesh::GlobalID toBlockLID = vmesh->getLocalID(incBlockGID);
205 if (toBlockLID == vmesh->invalidLocalID()) {
206 bool success = vmesh->push_back(incBlockGID);
207 toBlockLID = blockContainer->push_back_and_zero();
208 Real* parameters = blockContainer->getParameters(toBlockLID);
209 vmesh->getBlockInfo(incBlockGID, parameters+BlockParams::VXCRD);
210 }
211
212 // Pointer to target block data
213 Realf* toData = blockContainer->getData(toBlockLID);
214 // Add scaled values from source cells
215 for (uint i=0; i<WID3; ++i) {
216 toData[i] += fromData[i] * factor;
217 }
218 } // for-loop over velocity blocks
219 }
220 };

781 template <typename fileReal> void SpatialCell::add_velocity_blocks(const uint popID,const std::vector<vmesh::GlobalID>& blocks,fileReal* avgBuffer) {
782 debug_population_check(popID);
783 // Return if no blocks to add
784 const uint nBlocks = blocks.size();
785 if (nBlocks==0) {
786 return;
787 }
788 populations[popID].vmesh->setNewCapacity(nBlocks);
789 populations[popID].blockContainer->setNewCapacity(nBlocks);
790
791 // Add blocks to mesh
792 const uint8_t adds = populations[popID].vmesh->push_back(blocks);
793 if (adds == 0) {
794 std::cerr << "Failed to add blocks" << std::endl;
795 return;
796 }
797
798 // Add blocks to block container
799 vmesh::LocalID startLID = populations[popID].blockContainer->push_back(nBlocks);
800 Real* parameters = populations[popID].blockContainer->getParameters(startLID);
801
802 #ifdef DEBUG_SPATIAL_CELL
803 if (populations[popID].vmesh->size() != populations[popID].blockContainer->size()) {
804 std::cerr << "size mismatch in " << __FILE__ << ' ' << __LINE__ << std::endl; exit(1);
805 }
806 #endif
807
808 // Set block parameters
809 for (size_t b=0; b<nBlocks; ++b) {
810 vmesh::GlobalID VBC_GID = blocks.at(b);
811 populations[popID].vmesh->getBlockInfo(VBC_GID, parameters);
812 parameters += BlockParams::N_VELOCITY_BLOCK_PARAMS;
813 }
814
815 //copy avgs data, here a conversion may happen between float and double
816 Realf *cellBlockData = populations[popID].blockContainer->getData(startLID);
817 for(uint64_t i = 0; i< WID3 * nBlocks ; i++){
818 cellBlockData[i] = avgBuffer[i];
819 }
820 }

826 inline void SpatialCell::remove_velocity_block(const vmesh::GlobalID& block,const uint popID) {
827 debug_population_check(popID);
828 if (block == invalid_global_id()) {
829 //std::cerr << "not removing, block " << block << " is invalid" << std::endl;
830 return;
831 }
832
833 const vmesh::LocalID removedLID = populations[popID].vmesh->getLocalID(block);
834 if (removedLID == invalid_local_id()) {
835 //std::cerr << "not removing since block " << block << " does not exist" << std::endl;
836 return;
837 }
838
839 // Get local ID of the last block:
840 const vmesh::LocalID lastLID = populations[popID].vmesh->size()-1;
841 // If block to remove is already last:
842 if (lastLID == removedLID) {
843 // Just remove the block
844 populations[popID].vmesh->pop();
845 populations[popID].blockContainer->pop();
846 } else {
847 // Move the last block to the removed position
848 populations[popID].vmesh->move(lastLID,removedLID);
849 populations[popID].blockContainer->move(lastLID,removedLID);
850 }
851 }
