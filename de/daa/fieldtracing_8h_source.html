40// Get the (integer valued) global fsgrid cell index (i,j,k) for the magnetic-field traced mapping point that node n is
41// associated with
42template<class T> std::array<FsGridTools::FsSize_t, 3> getGlobalFsGridCellIndexForCoord(T& grid,const std::array<Real, 3>& x) {
43 std::array<FsGridTools::FsSize_t, 3> retval;
44 retval[0] = floor((x[0] - grid.physicalGlobalStart[0]) / grid.DX);
45 retval[1] = floor((x[1] - grid.physicalGlobalStart[1]) / grid.DY);
46 retval[2] = floor((x[2] - grid.physicalGlobalStart[2]) / grid.DZ);
47 return retval;
48}
49// Get the (integer valued) local fsgrid cell index (i,j,k) for the magnetic-field traced mapping point that node n is
50// associated with If the cell is not in our local domain, will return {-1,-1,-1}
51template<class T> std::array<FsGridTools::FsIndex_t, 3> getLocalFsGridCellIndexForCoord(T& grid, const std::array<Real, 3>& x) {
52 std::array<FsGridTools::FsSize_t, 3> globalInd = getGlobalFsGridCellIndexForCoord(grid,x);
53 std::array<FsGridTools::FsIndex_t, 3> retval = grid.globalToLocal(globalInd[0], globalInd[1], globalInd[2]);
54 return retval;
55}
56// Get the (integer valued) local fsgrid cell index (i,j,k) for the magnetic-field traced mapping point that node n is associated with
57// This includes indices beyond local size (positive and negative) as we need to access ghost cells
58template<class T> std::array<FsGridTools::FsIndex_t, 3> getLocalFsGridCellIndexWithGhostsForCoord(T& grid, const std::array<Real, 3>& x) {
59 std::array<FsGridTools::FsSize_t, 3> globalInd = getGlobalFsGridCellIndexForCoord(grid,x);
60 const std::array<FsGridTools::FsIndex_t, 3> localStart = grid.getLocalStart();
61 std::array<FsGridTools::FsIndex_t,3> retval = {(FsGridTools::FsIndex_t)globalInd[0]-localStart[0], (FsGridTools::FsIndex_t)globalInd[1]-localStart[1], (FsGridTools::FsIndex_t)globalInd[2]-localStart[2]};
62 return retval;
63}
64// Get the fraction fsgrid cell index for the magnetic-field traced mapping point that node n is associated with.
65// Note that these are floating point values between 0 and 1
66template<class T> std::array<Real, 3> getFractionalFsGridCellForCoord(T& grid, const std::array<Real, 3>& x) {
67 std::array<Real, 3> retval;
68 std::array<FsGridTools::FsSize_t, 3> fsgridCell = getGlobalFsGridCellIndexForCoord(grid,x);
69 retval[0] = (x[0] - grid.physicalGlobalStart[0]) / grid.DX - fsgridCell[0];
70 retval[1] = (x[1] - grid.physicalGlobalStart[1]) / grid.DY - fsgridCell[1];
71 retval[2] = (x[2] - grid.physicalGlobalStart[2]) / grid.DZ - fsgridCell[2];
72 return retval;
73}

165 template<typename REAL> bool traceFullFieldFunction(
166 FsGrid< std::array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & perBGrid,
167 FsGrid< std::array<Real, fsgrids::dperb::N_DPERB>, FS_STENCIL_WIDTH> & dPerBGrid,
168 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
169 std::array<REAL,3>& r,
170 const bool alongB,
171 std::array<REAL,3>& b
172 ) {
173 if( r[0] > P::xmax - 2*P::dx_ini
174 || r[0] < P::xmin + 2*P::dx_ini
175 || r[1] > P::ymax - 2*P::dy_ini
176 || r[1] < P::ymin + 2*P::dy_ini
177 || r[2] > P::zmax - 2*P::dz_ini
178 || r[2] < P::zmin + 2*P::dz_ini
179 ) {
180 cerr << (string)("(fieldtracing) Error: fsgrid coupling trying to step outside of the global domain?\n");
181 return false;
182 }
183
184 // Get field direction
185 b[0] = SBC::ionosphereGrid.dipoleField(r[0],r[1],r[2],X,0,X) + SBC::ionosphereGrid.BGB[0];
186 b[1] = SBC::ionosphereGrid.dipoleField(r[0],r[1],r[2],Y,0,Y) + SBC::ionosphereGrid.BGB[1];
187 b[2] = SBC::ionosphereGrid.dipoleField(r[0],r[1],r[2],Z,0,Z) + SBC::ionosphereGrid.BGB[2];
188
189 std::array<FsGridTools::FsSize_t, 3> fsgridCellu = getGlobalFsGridCellIndexForCoord(technicalGrid,{(TReal)r[0], (TReal)r[1], (TReal)r[2]});
190 std::array<FsGridTools::FsIndex_t,3> fsgridCell = {(FsGridTools::FsIndex_t)fsgridCellu[0],(FsGridTools::FsIndex_t)fsgridCellu[1],(FsGridTools::FsIndex_t)fsgridCellu[2]};
191 const std::array<FsGridTools::FsIndex_t, 3> localStart = technicalGrid.getLocalStart();
192 const std::array<FsGridTools::FsIndex_t, 3> localSize = technicalGrid.getLocalSize();
193 // Make the global index a local one, bypass the fsgrid function that yields (-1,-1,-1) also for ghost cells.
194 fsgridCell[0] -= localStart[0];
195 fsgridCell[1] -= localStart[1];
196 fsgridCell[2] -= localStart[2];
197
198 if(fsgridCell[0] > localSize[0] || fsgridCell[1] > localSize[1] || fsgridCell[2] > localSize[2]
199 || fsgridCell[0] < -1 || fsgridCell[1] < -1 || fsgridCell[2] < -1) {
200 cerr << (string)("(fieldtracing) Error: fsgrid coupling trying to access local ID " + to_string(fsgridCell[0]) + " " + to_string(fsgridCell[1]) + " " + to_string(fsgridCell[2])
201 + " for local domain size " + to_string(localSize[0]) + " " + to_string(localSize[1]) + " " + to_string(localSize[2])
202 + " at position " + to_string(r[0]) + " " + to_string(r[1]) + " " + to_string(r[2]) + " radius " + to_string(sqrt(r[0]*r[0]+r[1]*r[1]+r[2]*r[2]))
203 + "\n");
204 abort();
205 return false;
206 } else {
207 if(technicalGrid.get(fsgridCell[0],fsgridCell[1],fsgridCell[2])->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
208 const std::array<Real, 3> perB = interpolatePerturbedB(
209 perBGrid,
210 dPerBGrid,
211 technicalGrid,
212 fieldTracingParameters.reconstructionCoefficientsCache,
213 fsgridCell[0],fsgridCell[1],fsgridCell[2],
214 {(Real)r[0], (Real)r[1], (Real)r[2]}
215 );
216 b[0] += perB[0];
217 b[1] += perB[1];
218 b[2] += perB[2];
219 }
220 }
221
222 // Normalize
223 REAL norm = 1. / sqrt(b[0]*b[0] + b[1]*b[1] + b[2]*b[2]);
224 for(int c=0; c<3; c++) {
225 b[c] = b[c] * norm;
226 }
227
228 // Make sure motion is outwards. Flip b if dot(r,b) < 0
229 if(!(std::isfinite(b[0]) && std::isfinite(b[1]) && std::isfinite(b[2]))) {
230 cerr << "(fieldtracing) Error: magnetic field is nan or inf in getRadialBfieldDirection at location "
231 << r[0] << ", " << r[1] << ", " << r[2] << ", with B = " << b[0] << ", " << b[1] << ", " << b[2] << endl;
232 b[0] = 0;
233 b[1] = 0;
234 b[2] = 0;
235 }
236 if(!alongB) { // In this function, outwards indicates whether we trace along (true) or against (false) the field direction
237 b[0] *= -1;
238 b[1] *= -1;
239 b[2] *= -1;
240 }
241 return true;
242 }

245 /*Modified Midpoint Method used by the Bulirsch Stoer integrations
246 * stepsize: initial step size
247 * r: initial position
248 * r1: new position
249 * n: number of substeps
250 * stepsize: big stepsize to use
251 * z0,zmid,z2: intermediate approximations
252 * */
253 template <typename REAL> void modifiedMidpointMethod(
254 std::array<REAL,3> r,
255 std::array<REAL,3>& r1,
256 int n,
257 REAL stepSize,
258 TracingFieldFunction<REAL>& BFieldFunction,
259 const bool outwards=true
260 ){
261 //Allocate some memory.
262 std::array<REAL,3> bunit,crd,z0,zmid,z1;
263 //Divide by number of sub steps
264 REAL h= stepSize/(REAL)n;
265
266 //First step
267 BFieldFunction(r,outwards,bunit);
268 z0=r;
269 z1={ r[0]+h*bunit[0], r[1]+h*bunit[1], r[2]+h*bunit[2] };
270 BFieldFunction(z1,outwards,bunit);
271
272 crd = { r[0]+h*bunit[0], r[1]+h*bunit[1], r[2]+h*bunit[2] };
273
274 for (int m =0; m<=n; m++){
275 zmid= { z0[0]+2*h*bunit[0], z0[1]+2*h*bunit[1], z0[2]+2*h*bunit[2] };
276 z0=z1;
277 z1=zmid;
278 crd = { crd[0]+h*bunit[0], crd[1]+h*bunit[1], crd[2]+h*bunit[2] };
279 BFieldFunction(crd,outwards,bunit);
280 }
281
282 //These are now are new position
283 for (int c=0; c<3; c++){
284 r1[c] = 0.5*(z0[c]+z1[c]+h*bunit[c]);
285 }
286 }; // Modified Midpoint Method used by BS step

289 template <typename REAL> void richardsonExtrapolation(
290 int i,
291 std::vector<REAL>& table,
292 REAL& maxError,
293 std::array<int,3>dims
294 ) {
295 int k;
296 maxError = 0;
297 for (int dim=0; dim<3; dim++){
298 for(k =1; k<i+1; k++){
299
300 table.at(ijk2Index(i,k,dim,dims)) = table.at(ijk2Index(i,k-1,dim,dims)) +(table.at(ijk2Index(i,k-1,dim,dims)) -table.at(ijk2Index(i-1,k-1,dim,dims)))/(std::pow(4,i) -1);
301 }
302
303 REAL thisError = fabs(table.at(ijk2Index(k-1,k-1,dim,dims)) - table.at(ijk2Index(k-2,k-2,dim,dims)));
304 if(thisError > maxError) {
305 maxError = thisError;
306 }
307 }
308
309 }; //Richardson extrapolation method used by BS step

311 template <typename REAL> bool bulirschStoerStep(
312 std::array<REAL, 3>& r,
313 std::array<REAL, 3>& b,
314 REAL& stepSize,
315 const REAL minStepSize,
316 const REAL maxStepSize,
317 TracingFieldFunction<REAL>& BFieldFunction,
318 const bool outwards=true
319 ) {
320 //Factors by which the stepsize is multiplied
321 REAL shrink = 0.95;
322 REAL grow = 1.2;
323 //Max substeps for midpoint method
324 int kMax = 8;
325 //Optimal row to converge at
326 int kOpt = 6;
327
328 const int ndim = kMax*kMax*3;
329 std::array<int,3> dims={kMax,kMax,3};
330 std::vector<REAL>table(ndim);
331 std::array<REAL,3> rold,rnew,r1;
332 REAL error;
333
334 //Get B field unit vector in case we don't converge yet
335 BFieldFunction(r,outwards,b);
336
337 //Let's start things up with 2 substeps
338 int n =2;
339 //Save old state
340 rold = r;
341 //Take a first Step
342 modifiedMidpointMethod(r,r1,n,stepSize,BFieldFunction,outwards);
343
344 //Save values in table
345 for (int c =0; c<3; ++c) {
346 table[ijk2Index(0,0,c,dims)] = r1[c];
347 }
348
349 for (int i=1; i<kMax; ++i) {
350
351 //Increment n by 2 at every iteration.
352 n+=2;
353 modifiedMidpointMethod(r,rnew,n,stepSize,BFieldFunction,outwards);
354
355 //Save values in table
356 for (int c =0; c<3; ++c) {
357 table[ijk2Index(i,0,c,dims)] = rnew[c];
358 }
359
360 //Now let's perform a Richardson extrapolatation
361 richardsonExtrapolation(i,table,error,dims);
362
363 //Normalize error
364 error/=fieldTracingParameters.max_allowed_error;
365
366 //If we are below eps good, let's return but also let's modify the stepSize accordingly
367 if (error<1. || stepSize == minStepSize) {
368 if (i> kOpt) {
369 stepSize*=shrink;
370 }
371 if (i< kOpt) {
372 stepSize*=grow;
373 }
374 //Make sure stepsize does not exceed maxStepsize
375 stepSize = stepSize > maxStepSize ? maxStepSize : stepSize;
376 stepSize = stepSize < minStepSize ? minStepSize : stepSize;
377 //Keep our new position
378 r=rnew;
379 //Evaluate B here
380 BFieldFunction(r,outwards,b);
381 return true;
382 }
383 }
384
385 //If we end up here it means our tracer did not converge so we need to reduce the stepSize all along and try again
386 stepSize*=shrink;
387 stepSize = stepSize < minStepSize ? minStepSize : stepSize;
388 return false;
389 }; //Bulirsch Stoer step

392 template<typename REAL> bool dormandPrinceStep(
393 std::array<REAL, 3>& r,
394 std::array<REAL, 3>& b,
395 REAL& stepSize,
396 const REAL minStepSize,
397 const REAL maxStepSize,
398 TracingFieldFunction<REAL>& BFieldFunction,
399 const bool outwards=true
400 ) {
401 // BFieldFunction can return false if it steps out of the "comfort zone"
402 bool proceed = true;
403 std::array<REAL,7> kx,ky,kz;
404 std::array<REAL,3> b_unit;
405 std::array<REAL,3> _r{0,0,0};
406
407 //K1 slope
408 proceed = BFieldFunction(r,outwards,b_unit);
409 kx[0]=stepSize*b_unit[0];
410 ky[0]=stepSize*b_unit[1];
411 kz[0]=stepSize*b_unit[2];
412
413 //K2 slope
414 _r[0]=r[0]+(1./5.)*kx[0];
415 _r[1]=r[1]+(1./5.)*ky[0];
416 _r[2]=r[2]+(1./5.)*kz[0];
417 if (proceed) {
418 proceed = BFieldFunction(_r,outwards,b_unit);
419 }
420 kx[1]=stepSize*b_unit[0];
421 ky[1]=stepSize*b_unit[1];
422 kz[1]=stepSize*b_unit[2];
423
424 //K3 slope
425 _r[0]=r[0]+ (3./10.)*kx[1];
426 _r[1]=r[1]+ (3./10.)*ky[1];
427 _r[2]=r[2]+ (3./10.)*kz[1];
428 if (proceed) {
429 proceed = BFieldFunction(_r,outwards,b_unit);
430 }
431 kx[2]=stepSize*b_unit[0];
432 ky[2]=stepSize*b_unit[1];
433 kz[2]=stepSize*b_unit[2];
434
435 //K4 slope
436 _r[0]=r[0]+(4./5.)*kx[2];
437 _r[1]=r[1]+(4./5.)*ky[2];
438 _r[2]=r[2]+(4./5.)*kz[2];
439 if (proceed) {
440 proceed = BFieldFunction(_r,outwards,b_unit);
441 }
442 kx[3]=stepSize*b_unit[0];
443 ky[3]=stepSize*b_unit[1];
444 kz[3]=stepSize*b_unit[2];
445
446 //K5 slope
447 _r[0]=r[0]+(8./9.)*kx[3];
448 _r[1]=r[1]+(8./9.)*ky[3];
449 _r[2]=r[2]+(8./9.)*kz[3];
450 if (proceed) {
451 proceed = BFieldFunction(_r,outwards,b_unit);
452 }
453 kx[4]=stepSize*b_unit[0];
454 ky[4]=stepSize*b_unit[1];
455 kz[4]=stepSize*b_unit[2];
456
457 //K6 slope
458 _r[0]=r[0]+kx[4];
459 _r[1]=r[1]+ky[4];
460 _r[2]=r[2]+kz[4];
461 if (proceed) {
462 proceed = BFieldFunction(_r,outwards,b_unit);
463 }
464 kx[5]=stepSize*b_unit[0];
465 ky[5]=stepSize*b_unit[1];
466 kz[5]=stepSize*b_unit[2];
467
468 //K7 slope
469 _r[0]=r[0]+ kx[5];
470 _r[1]=r[1]+ ky[5];
471 _r[2]=r[2]+ kz[5];
472 if (proceed) {
473 proceed = BFieldFunction(_r,outwards,b_unit);
474 }
475 kx[6]=stepSize*b_unit[0];
476 ky[6]=stepSize*b_unit[1];
477 kz[6]=stepSize*b_unit[2];
478
479 REAL err=0;
480 std::array<REAL,3>rf;
481 if (proceed) {
482 //Error calculation
483 std::array<REAL,3>error_xyz;
484 rf[0]=r[0] +(35./384.)*kx[0] + (500./1113.)*kx[2] + (125./192.)*kx[3] - (2187./6784.)*kx[4] +(11./84.)*kx[5];
485 rf[1]=r[1] +(35./384.)*ky[0] + (500./1113.)*ky[2] + (125./192.)*ky[3] - (2187./6784.)*ky[4] +(11./84.)*ky[5];
486 rf[2]=r[2] +(35./384.)*kz[0] + (500./1113.)*kz[2] + (125./192.)*kz[3] - (2187./6784.)*kz[4] +(11./84.)*kz[5];
487
488 error_xyz[0]=abs((71./57600.)*kx[0] -(71./16695.)*kx[2] + (71./1920.)*kx[3] -(17253./339200.)*kx[4]+(22./525.)*kx[5] -(1./40.)*kx[6] );
489 error_xyz[1]=abs((71./57600.)*ky[0] -(71./16695.)*ky[2] + (71./1920.)*ky[3] -(17253./339200.)*ky[4]+(22./525.)*ky[5] -(1./40.)*ky[6] );
490 error_xyz[2]=abs((71./57600.)*kz[0] -(71./16695.)*kz[2] + (71./1920.)*kz[3] -(17253./339200.)*kz[4]+(22./525.)*kz[5] -(1./40.)*kz[6] );
491
492 //Estimate proper stepsize
493 err=std::max( std::max(error_xyz[0], error_xyz[1]), error_xyz[2]);
494 REAL s=pow((fieldTracingParameters.max_allowed_error/(2*err)),1./5.);
495 stepSize=stepSize*s;
496 } else { // proceed is false, we probably stepped too far
497 stepSize /= 2;
498 }
499
500 stepSize = stepSize > maxStepSize ? maxStepSize : stepSize;
501 stepSize = stepSize < minStepSize ? minStepSize : stepSize;
502 if ((err>fieldTracingParameters.max_allowed_error && stepSize > minStepSize) || !proceed) {
503 return false;
504 }
505 // No explicit else so the compiler sees a return at the end of the non-void function.
506 // With the return in if it's the same.
507 // Evaluate B at the final stepping point to get the b vector of the fieldline
508 BFieldFunction(r,outwards,b);
509 r=rf;
510 return true;
511
512 }; //Dormand Prince step

515 template<typename REAL> bool adaptiveEulerStep(
516 std::array<REAL, 3>& r,
517 std::array<REAL, 3>& b,
518 REAL& stepSize,
519 const REAL minStepSize,
520 const REAL maxStepSize,
521 TracingFieldFunction<REAL>& BFieldFunction,
522 const bool outwards=true
523 ) {
524 //First evaluation
525 std::array<REAL, 3> r1;
526 BFieldFunction(r,outwards,b);
527
528 for(int c=0; c<3; c++) {
529 r1[c] = r[c]+ stepSize * b[c];
530 }
531
532 //Second more accurate evaluation
533 std::array<REAL, 3> r2,b2;
534 for(int c=0; c<3; c++) {
535 r2[c] = r[c]+ 0.5*stepSize * b[c];
536 }
537
538 BFieldFunction(r2,outwards,b2);
539 for(int c=0; c<3; c++) {
540 r2[c] = r2[c]+ 0.5*stepSize * b2[c];
541 }
542
543 //Local error estimate
544 std::array<REAL,3> error_xyz{
545 fabs(r2[0]-r1[0]),
546 fabs(r2[1]-r1[1]),
547 fabs(r2[2]-r1[2])
548 };
549
550 //Max Error and step adjustment
551 const REAL err=std::max( std::max(error_xyz[0], error_xyz[1]), error_xyz[2]);
552 stepSize=stepSize*sqrt(fieldTracingParameters.max_allowed_error/err);
553 stepSize = stepSize > maxStepSize ? maxStepSize : stepSize;
554 stepSize = stepSize < minStepSize ? minStepSize : stepSize;
555
556 if (err<=fieldTracingParameters.max_allowed_error || stepSize == minStepSize) {
557 // Note: B at r has been evaluated above so no need to do it here and we're not returning it anyway as it's not needed.
558 r=r2;
559 return true;
560 } else {

584 template<typename REAL> void stepFieldLine(
585 std::array<REAL, 3>& x,
586 std::array<REAL, 3>& v,
587 REAL& stepsize,
588 const REAL minStepSize,
589 const REAL maxStepSize,
590 TracingMethod method,
591 TracingFieldFunction<REAL>& BFieldFunction,
592 const bool outwards
593 ) {
594 bool reTrace;
595 uint32_t attempts=0;
596 switch(method) {
597 case Euler:
598 eulerStep(x, v,stepsize, BFieldFunction, outwards);
599 break;
600 case ADPT_Euler:
601 do {
602 reTrace=!adaptiveEulerStep(x, v, stepsize, minStepSize, maxStepSize, BFieldFunction, outwards);
603 attempts++;
604 } while (reTrace && attempts<= fieldTracingParameters.max_field_tracer_attempts);
605 if (reTrace) {
606 logFile << "(fieldtracing) Warning: Adaptive Euler field line tracer exhausted all available attempts and still did not converge." << std::endl;
607 }
608 break;
609 case BS:
610 do{
611 reTrace=!bulirschStoerStep(x, v, stepsize, minStepSize, maxStepSize, BFieldFunction, outwards);
612 attempts++;
613 } while (reTrace && attempts<= fieldTracingParameters.max_field_tracer_attempts);
614 break;
615 case DPrince:
616 do {
617 reTrace=!dormandPrinceStep(x, v, stepsize, minStepSize, maxStepSize, BFieldFunction, outwards);
618 attempts++;
619 } while (reTrace && attempts<= fieldTracingParameters.max_field_tracer_attempts);
620 if (reTrace) {
621 logFile << "(fieldtracing) Warning: Dormand Prince field line tracer exhausted all available attempts and still did not converge..." << std::endl;
622 }
623 break;
624 default:
625 std::cerr << "(fieldtracing) Error: No field line tracing method defined."<<std::endl;
626 abort();
627 break;
628 }
629 }//stepFieldLine
