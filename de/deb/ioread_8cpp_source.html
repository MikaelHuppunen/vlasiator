166bool readCellIds(vlsv::ParallelReader & file, vector<CellID>& fileCells, const int masterRank,MPI_Comm comm)
167{
168 // Get info on array containing cell Ids:
169 uint64_t arraySize = 0;
170 uint64_t vectorSize;
171 vlsv::datatype::type dataType;
172 uint64_t byteSize;
173 list<pair<string,string> > attribs;
174 bool success=true;
175 int rank;
176 MPI_Comm_rank(comm,&rank);
177 if (rank==masterRank) {
178 const short int readFromFirstIndex = 0;
179 //let's let master read cellId's, we anyway have at max ~1e6 cells
180 attribs.push_back(make_pair("name","CellID"));
181 attribs.push_back(make_pair("mesh","SpatialGrid"));
182 if (file.getArrayInfoMaster("VARIABLE",attribs,arraySize,vectorSize,dataType,byteSize) == false) {
183 logFile << "(RESTART) ERROR: Failed to read cell ID array info!" << endl << write;
184 return false;
185 }
186
187 //Make a routine error check:
188 if( vectorSize != 1 ) {
189 logFile << "(RESTART) ERROR: Bad vectorsize at " << __FILE__ << " " << __LINE__ << endl << write;
190 return false;
191 }
192
193 // Read cell Ids:
194 char* IDbuffer = new char[arraySize*vectorSize*byteSize];
195 if (file.readArrayMaster("VARIABLE",attribs,readFromFirstIndex,arraySize,IDbuffer) == false) {
196 logFile << "(RESTART) ERROR: Failed to read cell Ids!" << endl << write;
197 success = false;
198 }
199
200 // Convert global Ids into our local DCCRG 64 bit uints
201 const uint64_t& numberOfCells = arraySize;
202 fileCells.resize(numberOfCells);
203 if (dataType == vlsv::datatype::type::UINT && byteSize == 4) {
204 uint32_t* ptr = reinterpret_cast<uint32_t*>(IDbuffer);
205 //Input cell ids
206 for (uint64_t i=0; i<numberOfCells; ++i) {
207 const CellID cellID = ptr[i];
208 fileCells[i] = cellID;
209 }
210 } else if (dataType == vlsv::datatype::type::UINT && byteSize == 8) {
211 uint64_t* ptr = reinterpret_cast<uint64_t*>(IDbuffer);
212 for (uint64_t i=0; i<numberOfCells; ++i) {
213 const CellID cellID = ptr[i];
214 fileCells[i] = cellID;
215 }
216 } else {
217 logFile << "(RESTART) ERROR: ParallelReader returned an unsupported datatype for cell Ids!" << endl << write;
218 success = false;
219 }
220 delete[] IDbuffer;
221 }
222
223 //broadcast cellId's to everybody
224 MPI_Bcast(&arraySize,1,MPI_UINT64_T,masterRank,comm);
225 fileCells.resize(arraySize);
226 MPI_Bcast(&(fileCells[0]),arraySize,MPI_UINT64_T,masterRank,comm);
227
228 return success;
229}

231/* Read the total number of velocity blocks per spatial cell in the spatial mesh.
232 * The returned value for each cell is a sum of the velocity blocks associated in each particle species.
233 * The value is used to calculate an initial load balance after restart.
234 * @param file Some vlsv reader with a file open (can be old or new vlsv reader)
235 * @param nBlocks Vector for holding information on cells and the number of blocks in them -- this function saves data here
236 * @param masterRank The master rank of this process (Vlasiator uses masterRank = 0 and so it should be the default)
237 * @param comm MPI comm
238 * @return Returns true if the operation was successful
239 @ @see exec_readGrid
240*/
241bool readNBlocks(vlsv::ParallelReader& file,const std::string& meshName,
242 std::vector<size_t>& nBlocks,int masterRank,MPI_Comm comm) {
243 bool success = true;
244
245 // Get info on array containing cell IDs:
246 uint64_t arraySize;
247 uint64_t vectorSize;
248 vlsv::datatype::type dataType;
249 uint64_t byteSize;
250
251 // Read mesh bounding box to all processes, the info in bbox contains
252 // the number of spatial cells in the mesh.
253 // (This is *not* the physical coordinate bounding box.)
254 list<pair<string,string> > attribsIn;
255 map<string,string> attribsOut;
256 attribsIn.push_back(make_pair("mesh",meshName));
257
258 // Read number of domains and domain sizes
259 uint64_t N_domains;
260 file.getArrayAttributes("MESH_DOMAIN_SIZES",attribsIn,attribsOut);
261 auto it = attribsOut.find("arraysize");
262 if (it == attribsOut.end()) {
263 cerr << "VLSV\t\t ERROR: Array 'MESH_DOMAIN_SIZES' XML tag does not have attribute 'arraysize'" << endl;
264 return false;
265 } else {
266 N_domains = atoi(it->second.c_str());
267 }
268
269 uint64_t N_spatialCells = 0;
270 int64_t* domainInfo = NULL;
271 if (file.read("MESH_DOMAIN_SIZES",attribsIn,0,N_domains,domainInfo) == false) return false;
272
273 for (uint i_domain = 0; i_domain < N_domains; ++i_domain) {
274 N_spatialCells += domainInfo[2*i_domain];
275 }
276 nBlocks.resize(N_spatialCells);
277
278 #pragma omp parallel for
279 for (size_t i=0; i<nBlocks.size(); ++i) nBlocks[i] = 0;
280
281 // Note: the input file contains N particle species, which are also
282 // defined in the configuration file. We need to read the BLOCKSPERCELL
283 // array for each species, and sum the values for each spatial cell.
284 set<string> speciesNames;
285 if (file.getUniqueAttributeValues("BLOCKSPERCELL","name",speciesNames) == false) return false;
286
287 // Iterate over all particle species and read in BLOCKSPERCELL array
288 // to all processes, and add the values to nBlocks
289 uint64_t* buffer = new uint64_t[N_spatialCells];
290 for (set<string>::const_iterator s=speciesNames.begin(); s!=speciesNames.end(); ++s) {
291 attribsIn.clear();
292 attribsIn.push_back(make_pair("mesh",meshName));
293 attribsIn.push_back(make_pair("name",*s));
294 if (file.getArrayInfo("BLOCKSPERCELL",attribsIn,arraySize,vectorSize,dataType,byteSize) == false) return false;
295
296 if (file.read("BLOCKSPERCELL",attribsIn,0,arraySize,buffer) == false) {
297 delete [] buffer; buffer = NULL;
298 return false;
299 }
300
301 #pragma omp parallel for
302 for (size_t i=0; i<N_spatialCells; ++i) {
303 nBlocks[i] += buffer[i];
304 }
305 }
306 delete [] buffer; buffer = NULL;
307 return success;
308}

370template <typename fileReal>
371bool _readBlockData(
372 vlsv::ParallelReader & file,
373 const std::string& spatMeshName,
374 const std::vector<uint64_t>& fileCells,
375 const uint64_t localCellStartOffset,
376 const uint64_t localCells,
377 const vmesh::LocalID* blocksPerCell,
378 const std::vector<uint64_t> &blockSumOffsets,
379 const uint64_t localBlockStartOffset,
380 const uint64_t localBlocks,
381 dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
382 std::function<vmesh::GlobalID(vmesh::GlobalID)> blockIDremapper,
383 const uint popID
384) {
385 uint64_t arraySize;
386 uint64_t avgVectorSize;
387 vlsv::datatype::type dataType;
388 uint64_t byteSize;
389 list<pair<string,string> > avgAttribs;
390 bool success=true;
391 const string popName = getObjectWrapper().particleSpecies[popID].name;
392 const string tagName = "BLOCKIDS";
393
394 avgAttribs.push_back(make_pair("mesh",spatMeshName));
395 avgAttribs.push_back(make_pair("name",popName));
396
397 //Get block id array info and store them into blockIdAttribs, lockIdByteSize, blockIdDataType, blockIdVectorSize
398 list<pair<string,string> > blockIdAttribs;
399 uint64_t blockIdVectorSize, blockIdByteSize;
400 vlsv::datatype::type blockIdDataType;
401 blockIdAttribs.push_back( make_pair("mesh", spatMeshName));
402 blockIdAttribs.push_back( make_pair("name", popName));
403 if (file.getArrayInfo("BLOCKIDS",blockIdAttribs,arraySize,blockIdVectorSize,blockIdDataType,blockIdByteSize) == false ){
404 logFile << "(RESTART) ERROR: Failed to read BLOCKCOORDINATES array info " << endl << write;
405 return false;
406 }
407 if(file.getArrayInfo("BLOCKVARIABLE",avgAttribs,arraySize,avgVectorSize,dataType,byteSize) == false ){
408 logFile << "(RESTART) ERROR: Failed to read BLOCKVARIABLE array info " << endl << write;
409 return false;
410 }
411
412 //Some routine error checks:
413 if( avgVectorSize!=WID3 ){
414 logFile << "(RESTART) ERROR: Blocksize does not match in restart file " << endl << write;
415 return false;
416 }
417 if( byteSize != sizeof(fileReal) ) {
418 logFile << "(RESTART) ERROR: Bad avgs bytesize at " << __FILE__ << " " << __LINE__ << endl << write;
419 return false;
420 }
421
422 if( blockIdByteSize != sizeof(vmesh::GlobalID)) {
423 logFile << "(RESTART) ERROR: BlockID data size does not match " << __FILE__ << " " << __LINE__ << endl << write;
424 return false;
425 }
426
427 vmesh::GlobalID* blockIdBuffer;
428 fileReal* avgBuffer;
429 #ifdef USE_GPU
430 CHK_ERR( gpuMallocHost((void**)&avgBuffer, avgVectorSize*localBlocks*sizeof(Realf)) ); // Pinned memory
431 blockIdBuffer = ::new vmesh::GlobalID[blockIdVectorSize * localBlocks]; //blockids of all cells
432 #else
433 avgBuffer = new fileReal[avgVectorSize * localBlocks]; //avgs data for all cells
434 blockIdBuffer = new vmesh::GlobalID[blockIdVectorSize * localBlocks]; //blockids of all cells
435 #endif
436
437 //Read block ids and data
438 if (file.readArray("BLOCKIDS", blockIdAttribs, localBlockStartOffset, localBlocks, (char*)blockIdBuffer ) == false) {
439 cerr << "ERROR, failed to read BLOCKIDS in " << __FILE__ << ":" << __LINE__ << endl;
440 success = false;
441 }
442 if (file.readArray("BLOCKVARIABLE", avgAttribs, localBlockStartOffset, localBlocks, (char*)avgBuffer) == false) {
443 cerr << "ERROR, failed to read BLOCKVARIABLE in " << __FILE__ << ":" << __LINE__ << endl;
444 success = false;
445 }
446
447 //Go through all spatial cells
448 #pragma omp parallel for schedule(dynamic,1)
449 for(uint64_t i=0; i<localCells; i++) {
450 CellID cell = fileCells[localCellStartOffset + i]; //spatial cell id
451 uint64_t blockBufferOffset = blockSumOffsets[i];
452 vmesh::LocalID nBlocksInCell = blocksPerCell[i];
453 //copy blocks in this cell to vector blockIdsInCell, size of read in data has been checked earlier
454 vector<vmesh::GlobalID> blockIdsInCell; //blockIds in a particular cell, temporary usage
455 blockIdsInCell.reserve(nBlocksInCell);
456 blockIdsInCell.assign(blockIdBuffer + blockBufferOffset, blockIdBuffer + blockBufferOffset + nBlocksInCell);
457 for(auto& id : blockIdsInCell) {
458 id = blockIDremapper(id);
459 }
460 // allocate space for all blocks and create them and fill them
461 // a conversion may happen between float and double
462 mpiGrid[cell]->add_velocity_blocks(popID,blockIdsInCell,&avgBuffer[blockBufferOffset*WID3]);
463 #if defined(USE_GPU) && defined(DEBUG_VLASIATOR)
464 mpiGrid[cell]->checkMesh(popID);
465 #endif
466 }
467 #ifdef USE_GPU
468 if (avgBuffer) {
469 CHK_ERR( gpuFreeHost(avgBuffer) );
470 avgBuffer = 0;
471 }
472 #else
473 delete[] avgBuffer;
474 #endif
475 delete[] blockIdBuffer;
476 return success;
477}

488bool readBlockData(
489 vlsv::ParallelReader& file,
490 const string& meshName,
491 const vector<CellID>& fileCells,
492 const uint64_t localCellStartOffset,
493 const uint64_t localCells,
494 dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid
495 ) {
496 bool success = true;
497
498 const uint64_t bytesReadStart = file.getBytesRead();
499 int N_processes;
500 MPI_Comm_size(MPI_COMM_WORLD,&N_processes);
501
502 uint64_t arraySize;
503 uint64_t vectorSize;
504 vlsv::datatype::type dataType;
505 uint64_t byteSize;
506 uint64_t* offsetArray = new uint64_t[N_processes];
507
508 for (uint popID=0; popID<getObjectWrapper().particleSpecies.size(); ++popID) {
509 const string& popName = getObjectWrapper().particleSpecies[popID].name;
510
511 // Create a cellID remapping lambda that can renumber our velocity space, should it's size have changed.
512 // By default, this is a no-op that keeps the blockIDs untouched.
513 std::function<vmesh::GlobalID(vmesh::GlobalID)> blockIDremapper = [](vmesh::GlobalID oldID) -> vmesh::GlobalID {return oldID;};
514
515 // Check that velocity space extents and DV matches the grids we have created
516 list<pair<string,string> > attribs;
517 attribs.push_back(make_pair("mesh",popName));
518 std::array<unsigned int, 6> fileMeshBBox;
519 unsigned int* bufferpointer = &fileMeshBBox[0];
520 if (file.read("MESH_BBOX",attribs,0,6,bufferpointer,false) == false) {
521 logFile << "(RESTART) ERROR: Failed to read MESH_BBOX at " << __FILE__ << ":" << __LINE__ << endl << write;
522 success = false;
523 }
524
525 const size_t meshID = getObjectWrapper().particleSpecies[popID].velocityMesh;
526 const vmesh::MeshParameters& ourMeshParams = vmesh::getMeshWrapper()->velocityMeshes->at(meshID);
527 if(fileMeshBBox[0] != ourMeshParams.gridLength[0] ||
528 fileMeshBBox[1] != ourMeshParams.gridLength[1] ||
529 fileMeshBBox[2] != ourMeshParams.gridLength[2]) {
530
531 logFile << "(RESTART) INFO: velocity mesh sizes don't match:" << endl
532 << " restart file has " << fileMeshBBox[0] << " x " << fileMeshBBox[1] << " x " << fileMeshBBox[2] << "," << endl
533 << " config specifies " << ourMeshParams.gridLength[0] << " x " << ourMeshParams.gridLength[1] << " x " << ourMeshParams.gridLength[2] << endl << write;
534
535 if(ourMeshParams.gridLength[0] < fileMeshBBox[0] ||
536 ourMeshParams.gridLength[1] < fileMeshBBox[1] ||
537 ourMeshParams.gridLength[2] < fileMeshBBox[2]) {
538 logFile << "(RESTART) ERROR: trying to shrink velocity space." << endl << write;
539 abort();
540 }
541
542 // If we are mismatched, we have to iterate through the velocity coords to see if we have a
543 // chance at renumbering.
544 std::vector<Real> fileVelCoordsX(fileMeshBBox[0]*fileMeshBBox[3]+1);
545 std::vector<Real> fileVelCoordsY(fileMeshBBox[1]*fileMeshBBox[4]+1);
546 std::vector<Real> fileVelCoordsZ(fileMeshBBox[2]*fileMeshBBox[5]+1);
547
548 Real* tempPointer = fileVelCoordsX.data();
549 if (file.read("MESH_NODE_CRDS_X",attribs,0,fileMeshBBox[0]*fileMeshBBox[3]+1,tempPointer,false) == false) {
550 logFile << "(RESTART) ERROR: Failed to read MESH_NODE_CRDS_X at " << __FILE__ << ":" << __LINE__ << endl << write;
551 success = false;
552 }
553 tempPointer = fileVelCoordsY.data();
554 if (file.read("MESH_NODE_CRDS_Y",attribs,0,fileMeshBBox[1]*fileMeshBBox[4]+1,tempPointer,false) == false) {
555 logFile << "(RESTART) ERROR: Failed to read MESH_NODE_CRDS_X at " << __FILE__ << ":" << __LINE__ << endl << write;
556 success = false;
557 }
558 tempPointer = fileVelCoordsZ.data();
559 if (file.read("MESH_NODE_CRDS_Z",attribs,0,fileMeshBBox[2]*fileMeshBBox[5]+1,tempPointer,false) == false) {
560 logFile << "(RESTART) ERROR: Failed to read MESH_NODE_CRDS_X at " << __FILE__ << ":" << __LINE__ << endl << write;
561 success = false;
562 }
563
564 const Real dVx = vmesh::getMeshWrapper()->velocityMeshes->at(meshID).cellSize[0];
565 for(const auto& c : fileVelCoordsX) {
566 Real cellindex = (c - vmesh::getMeshWrapper()->velocityMeshes->at(meshID).meshMinLimits[0]) / dVx;
567 if(fabs(nearbyint(cellindex) - cellindex) > 1./10000.) {
568 logFile << "(RESTART) ERROR: Can't resize velocity space as cell coordinates don't match." << endl
569 << " (X coordinate " << c << " = " << cellindex <<" * " << dVx << " + " << vmesh::getMeshWrapper()->velocityMeshes->at(meshID).meshMinLimits[0] << endl
570 << " coordinate = cellindex * dV + meshMinLimits)" << endl << write;
571 abort();
572 }
573 }
574
575 const Real dVy = vmesh::getMeshWrapper()->velocityMeshes->at(meshID).cellSize[1];
576 for(const auto& c : fileVelCoordsY) {
577 Real cellindex = (c - vmesh::getMeshWrapper()->velocityMeshes->at(meshID).meshMinLimits[1]) / dVy;
578 if(fabs(nearbyint(cellindex) - cellindex) > 1./10000.) {
579 logFile << "(RESTART) ERROR: Can't resize velocity space as cell coordinates don't match." << endl
580 << " (Y coordinate " << c << " = " << cellindex <<" * " << dVy << " + " << vmesh::getMeshWrapper()->velocityMeshes->at(meshID).meshMinLimits[1] << endl
581 << " coordinate = cellindex * dV + meshMinLimits)" << endl << write;
582 abort();
583 }
584 }
585
586 const Real dVz = vmesh::getMeshWrapper()->velocityMeshes->at(meshID).cellSize[2];
587 for(const auto& c : fileVelCoordsY) {
588 Real cellindex = (c - vmesh::getMeshWrapper()->velocityMeshes->at(meshID).meshMinLimits[2]) / dVz;
589 if(fabs(nearbyint(cellindex) - cellindex) > 1./10000.) {
590 logFile << "(RESTART) ERROR: Can't resize velocity space as cell coordinates don't match." << endl
591 << " (Z coordinate " << c << " = " << cellindex <<" * " << dVz << " + " << vmesh::getMeshWrapper()->velocityMeshes->at(meshID).meshMinLimits[2] << endl
592 << " coordinate = cellindex * dV + meshMinLimits)" << endl << write;
593 abort();
594 }
595 }
596
597 // If we haven't aborted above, we can apparently renumber our
598 // cellIDs. Build an approprita blockIDremapper lambda for this purpose.
599 std::array<int, 3> velGridOffset;
600 velGridOffset[0] = (fileVelCoordsX[0] - vmesh::getMeshWrapper()->velocityMeshes->at(meshID).meshMinLimits[0]) / dVx;
601 velGridOffset[1] = (fileVelCoordsY[0] - vmesh::getMeshWrapper()->velocityMeshes->at(meshID).meshMinLimits[1]) / dVy;
602 velGridOffset[2] = (fileVelCoordsZ[0] - vmesh::getMeshWrapper()->velocityMeshes->at(meshID).meshMinLimits[2]) / dVz;
603
604 if((velGridOffset[0] % ourMeshParams.blockLength[0] != 0) ||
605 (velGridOffset[1] % ourMeshParams.blockLength[1] != 0) ||
606 (velGridOffset[2] % ourMeshParams.blockLength[2] != 0)) {
607 logFile << "(RESTART) ERROR: resizing velocity space on restart must end up with the old velocity space" << endl
608 << " at a block boundary of the new space!" << endl
609 << " (It now starts at cell [" << velGridOffset[0] << ", " << velGridOffset[1] << "," << velGridOffset[2] << "])" << endl << write;
610 abort();
611 }
612
613 velGridOffset[0] /= ourMeshParams.blockLength[0];
614 velGridOffset[1] /= ourMeshParams.blockLength[1];
615 velGridOffset[2] /= ourMeshParams.blockLength[2];
616
617 blockIDremapper = [fileMeshBBox,velGridOffset,ourMeshParams](vmesh::GlobalID oldID) -> vmesh::GlobalID {
618 unsigned int x,y,z;
619 x = oldID % fileMeshBBox[0];
620 y = (oldID / fileMeshBBox[0]) % fileMeshBBox[1];
621 z = oldID / (fileMeshBBox[0] * fileMeshBBox[1]);
622
623 x += velGridOffset[0];
624 y += velGridOffset[1];
625 z += velGridOffset[2];
626
627 //logFile << " Remapping " << oldID << "(" << x << "," << y << "," << z << ") to " << x + y * ourMeshParams.gridLength[0] + z* ourMeshParams.gridLength[0] * ourMeshParams.gridLength[1] << endl << write;
628 return x + y * ourMeshParams.gridLength[0] + z* ourMeshParams.gridLength[0] * ourMeshParams.gridLength[1];
629 };
630
631 logFile << " => Resizing velocity space by renumbering GlobalIDs." << endl << endl << write;
632 }
633
634 // In restart files each spatial cell has an entry in CELLSWITHBLOCKS.
635 // Each process calculates how many velocity blocks it has for this species.
636 attribs.clear();
637 attribs.push_back(make_pair("mesh",meshName));
638 attribs.push_back(make_pair("name",popName));
639 vmesh::LocalID* blocksPerCell = NULL;
640
641 if (file.read("BLOCKSPERCELL",attribs,localCellStartOffset,localCells,blocksPerCell,true) == false) {
642 logFile << "(RESTART) ERROR: Failed to read BLOCKSPERCELL at " << __FILE__ << ":" << __LINE__ << endl << write;
643 success = false;
644 }
645
646 // Count how many velocity blocks this process gets
647 uint64_t blockSum = 0;
648 std::vector<uint64_t> blockSumOffsets(localCells);
649 for (uint64_t i=0; i<localCells; ++i){
650 blockSumOffsets[i] = blockSum;
651 blockSum += blocksPerCell[i];
652 }
653
654 // Gather all block sums to master process who will them broadcast
655 // the values to everyone
656 MPI_Allgather(&blockSum,1,MPI_Type<uint64_t>(),offsetArray,1,MPI_Type<uint64_t>(),MPI_COMM_WORLD);
657
658 // Calculate the offset from which this process starts reading block data
659 uint64_t myOffset = 0;
660 for (int64_t i=0; i<mpiGrid.get_rank(); ++i) myOffset += offsetArray[i];
661
662 if (file.getArrayInfo("BLOCKVARIABLE",attribs,arraySize,vectorSize,dataType,byteSize) == false) {
663 logFile << "(RESTART) ERROR: Failed to read BLOCKVARIABLE INFO" << endl << write;
664 return false;
665 }
666
667 // Call _readBlockData
668 if (dataType == vlsv::datatype::type::FLOAT) {
669 switch (byteSize) {
670 case sizeof(double):
671 if (_readBlockData<double>(file,meshName,fileCells,localCellStartOffset,localCells,blocksPerCell,blockSumOffsets,
672 myOffset,blockSum,mpiGrid,blockIDremapper,popID) == false) success = false;
673 break;
674 case sizeof(float):
675 if (_readBlockData<float>(file,meshName,fileCells,localCellStartOffset,localCells,blocksPerCell,blockSumOffsets,
676 myOffset,blockSum,mpiGrid,blockIDremapper,popID) == false) success = false;
677 break;
678 }
679 } else if (dataType == vlsv::datatype::type::UINT) {
680 switch (byteSize) {
681 case sizeof(uint32_t):
682 if (_readBlockData<uint32_t>(file,meshName,fileCells,localCellStartOffset,localCells,blocksPerCell,blockSumOffsets,
683 myOffset,blockSum,mpiGrid,blockIDremapper,popID) == false) success = false;
684 break;
685 case sizeof(uint64_t):
686 if (_readBlockData<uint64_t>(file,meshName,fileCells,localCellStartOffset,localCells,blocksPerCell,blockSumOffsets,
687 myOffset,blockSum,mpiGrid,blockIDremapper,popID) == false) success = false;
688 break;
689 }
690 } else if (dataType == vlsv::datatype::type::INT) {
691 switch (byteSize) {
692 case sizeof(int32_t):
693 if (_readBlockData<int32_t>(file,meshName,fileCells,localCellStartOffset,localCells,blocksPerCell,blockSumOffsets,
694 myOffset,blockSum,mpiGrid,blockIDremapper,popID) == false) success = false;
695 break;
696 case sizeof(int64_t):
697 if (_readBlockData<int64_t>(file,meshName,fileCells,localCellStartOffset,localCells,blocksPerCell,blockSumOffsets,
698 myOffset,blockSum,mpiGrid,blockIDremapper,popID) == false) success = false;
699 break;
700 }
701 } else {
702 logFile << "(RESTART) ERROR: Failed to read data type at readCellParamsVariable" << endl << write;
703 success = false;
704 }
705 delete [] blocksPerCell; blocksPerCell = NULL;
706 } // for-loop over particle species
707
708 delete [] offsetArray; offsetArray = NULL;
709
710 const uint64_t bytesReadEnd = file.getBytesRead() - bytesReadStart;
711 logFile << "Velocity meshes and data read, approximate data rate is ";
712 logFile << vlsv::printDataRate(bytesReadEnd,file.getReadTime()) << endl << write;
713
714 return success;
715}

852template<unsigned long int N> bool readFsGridVariable(
853 vlsv::ParallelReader& file, const string& variableName, int numWritingRanks, FsGrid<std::array<Real, N>,FS_STENCIL_WIDTH> & targetGrid) {
854
855 phiprof::Timer preparations {"preparations"};
856
857 uint64_t arraySize;
858 uint64_t vectorSize;
859 vlsv::datatype::type dataType;
860 uint64_t byteSize;
861 list<pair<string,string> > attribs;
862 bool convertFloatType = false;
863
864 attribs.push_back(make_pair("name",variableName));
865 attribs.push_back(make_pair("mesh","fsgrid"));
866
867 phiprof::Timer getArrayInfo {"getArrayInfo"};
868 if (file.getArrayInfo("VARIABLE",attribs,arraySize,vectorSize,dataType,byteSize) == false) {
869 logFile << "(RESTART) ERROR: Failed to read FsGrid ArrayInfo " << endl << write;
870 return false;
871 }
872 if(! (dataType == vlsv::datatype::type::FLOAT && byteSize == sizeof(Real))) {
873 logFile << "(RESTART) Converting floating point format of fsgrid variable " << variableName << " from " << byteSize * 8 << " bits to " << sizeof(Real) * 8 << " bits." << endl << write;
874 convertFloatType = true;
875 // Note: this implicitly assumes that Real is of type double, and we either read a double in directly, or read a float and convert it to double.
876 }
877 getArrayInfo.stop();
878
879 // Are we restarting from the same number of tasks, or a different number?
880 int size, myRank;
881 MPI_Comm_size(MPI_COMM_WORLD, &size);
882 MPI_Comm_rank(MPI_COMM_WORLD, &myRank);
883
884 std::array<FsGridTools::FsIndex_t,3>& localSize = targetGrid.getLocalSize();
885 std::array<FsGridTools::FsIndex_t,3>& localStart = targetGrid.getLocalStart();
886 std::array<FsGridTools::FsSize_t,3>& globalSize = targetGrid.getGlobalSize();
887
888 // Determine our tasks storage size
889 size_t storageSize = localSize[0]*localSize[1]*localSize[2];
890
891 preparations.stop();
892 std::array<FsGridTools::Task_t,3> fileDecomposition={0,0,0};
893 // No override given (zero array)
894 if(P::overrideReadFsGridDecomposition == fileDecomposition){
895 // Try and read the decomposition from file
896 if (readFsgridDecomposition(file, fileDecomposition) == false) {
897 exitOnError(false, "(RESTART) Failed to read Fsgrid decomposition", MPI_COMM_WORLD);
898 }
899 } else {
900 // Override
901 logFile << "(RESTART) Using manual override for FsGrid MESH_DECOMPOSITION." << endl << write;
902 fileDecomposition = P::overrideReadFsGridDecomposition;
903 int fsgridInputRanks=0;
904 // Read numWritingRanks from file, that should exist and be sane
905 if(readScalarParameter(file,"numWritingRanks",fsgridInputRanks, MASTER_RANK, MPI_COMM_WORLD) == false) {
906 exitOnError(false, "(RESTART) FSGrid writing rank number not found in restart file", MPI_COMM_WORLD);
907 }
908 // Check that the override is sane wrt. numWritingRanks
909 if(fileDecomposition[0]*fileDecomposition[1]*fileDecomposition[2] != fsgridInputRanks){
910 exitOnError(false, "(RESTART) Trying to use a manual FsGrid decomposition for a file with a differing number of input ranks.", MPI_COMM_WORLD);
911 }
912 }
913
914 std::array<FsGridTools::Task_t,3> decomposition = targetGrid.getDecomposition();
915 // targetGrid.computeDomainDecomposition(globalSize, size, decomposition);
916
917 if(decomposition == fileDecomposition) {
918 // Easy case: same decomposition => slurp it in.
919 //
920
921 // Determine offset in file by summing up all the previous tasks' sizes.
922 size_t localStartOffset = 0;
923 for(FsGridTools::Task_t task = 0; task < myRank; task++) {
924 std::array<FsGridTools::FsIndex_t, 3> thatTasksSize;
925 thatTasksSize[0] = targetGrid.calcLocalSize(globalSize[0], decomposition[0], task/decomposition[2]/decomposition[1]);
926 thatTasksSize[1] = targetGrid.calcLocalSize(globalSize[1], decomposition[1], (task/decomposition[2])%decomposition[1]);
927 thatTasksSize[2] = targetGrid.calcLocalSize(globalSize[2], decomposition[2], task%decomposition[2]);
928 localStartOffset += thatTasksSize[0] * thatTasksSize[1] * thatTasksSize[2];
929 }
930
931 // Read into buffer
932 std::vector<Real> buffer(storageSize*N);
933
934 if(file.readArray("VARIABLE",attribs, localStartOffset, storageSize, buffer.data()) == false) {
935 logFile << "(RESTART) ERROR: Failed to read fsgrid variable " << variableName << endl << write;
936 return false;
937 }
938
939 // Assign buffer into fsgrid
940 int index=0;
941 for(FsGridTools::FsIndex_t z=0; z<localSize[2]; z++) {
942 for(FsGridTools::FsIndex_t y=0; y<localSize[1]; y++) {
943 for(FsGridTools::FsIndex_t x=0; x<localSize[0]; x++) {
944 memcpy(targetGrid.get(x,y,z), &buffer[index], N*sizeof(Real));
945 index += N;
946 }
947 }
948 }
949
950 } else {
951
952 // More difficult case: different number of tasks.
953 // In this case, our own fsgrid domain overlaps (potentially many) domains in the file.
954 // We read the whole source rank into a temporary buffer, and transfer the overlapping
955 // part.
956 //
957 // +------------+----------------+
958 // | | |
959 // | . . . . . . . . . . . . |
960 // | .<----->|<----------->. |
961 // | .<----->|<----------->. |
962 // | .<----->|<----------->. |
963 // +----+-------+-------------+--|
964 // | .<----->|<----------->. |
965 // | .<----->|<----------->. |
966 // | .<----->|<----------->. |
967 // | . . . . . . . . . . . . |
968 // | | |
969 // +------------+----------------+
970
971 // Iterate through tasks and find their overlap with our domain.
972 size_t fileOffset = 0;
973 for(int task = 0; task < numWritingRanks; task++) {
974
975 phiprof::Timer taskArithmetics1 {"task overlap arithmetics 1"};
976
977 std::array<FsGridTools::FsIndex_t,3> thatTasksSize;
978 std::array<FsGridTools::FsIndex_t,3> thatTasksStart;
979 thatTasksSize[0] = targetGrid.calcLocalSize(globalSize[0], fileDecomposition[0], task/fileDecomposition[2]/fileDecomposition[1]);
980 thatTasksSize[1] = targetGrid.calcLocalSize(globalSize[1], fileDecomposition[1], (task/fileDecomposition[2])%fileDecomposition[1]);
981 thatTasksSize[2] = targetGrid.calcLocalSize(globalSize[2], fileDecomposition[2], task%fileDecomposition[2]);
982
983 thatTasksStart[0] = targetGrid.calcLocalStart(globalSize[0], fileDecomposition[0], task/fileDecomposition[2]/fileDecomposition[1]);
984 thatTasksStart[1] = targetGrid.calcLocalStart(globalSize[1], fileDecomposition[1], (task/fileDecomposition[2])%fileDecomposition[1]);
985 thatTasksStart[2] = targetGrid.calcLocalStart(globalSize[2], fileDecomposition[2], task%fileDecomposition[2]);
986
987 // Iterate through overlap area
988 std::array<FsGridTools::FsIndex_t,3> overlapStart,overlapEnd,overlapSize;
989 overlapStart[0] = max(localStart[0],thatTasksStart[0]);
990 overlapStart[1] = max(localStart[1],thatTasksStart[1]);
991 overlapStart[2] = max(localStart[2],thatTasksStart[2]);
992
993 overlapEnd[0] = min(localStart[0]+localSize[0], thatTasksStart[0]+thatTasksSize[0]);
994 overlapEnd[1] = min(localStart[1]+localSize[1], thatTasksStart[1]+thatTasksSize[1]);
995 overlapEnd[2] = min(localStart[2]+localSize[2], thatTasksStart[2]+thatTasksSize[2]);
996
997 overlapSize[0] = max(overlapEnd[0]-overlapStart[0],(FsGridTools::FsIndex_t)0);
998 overlapSize[1] = max(overlapEnd[1]-overlapStart[1],(FsGridTools::FsIndex_t)0);
999 overlapSize[2] = max(overlapEnd[2]-overlapStart[2],(FsGridTools::FsIndex_t)0);
1000
1001 taskArithmetics1.stop();
1002
1003 // Read into buffer
1004 std::vector<Real> buffer(thatTasksSize[0]*thatTasksSize[1]*thatTasksSize[2]*N);
1005
1006 phiprof::Timer multiRead {"multiRead"};
1007 file.startMultiread("VARIABLE", attribs);
1008 // Read every source rank that we have an overlap with.
1009 if(overlapSize[0]*overlapSize[1]*overlapSize[2] > 0) {
1010
1011
1012 if(!convertFloatType) {
1013 if(file.addMultireadUnit((char*)buffer.data(), thatTasksSize[0]*thatTasksSize[1]*thatTasksSize[2])==false) {
1014 logFile << "(RESTART) ERROR: Failed to read fsgrid variable " << variableName << endl << write;
1015 return false;
1016 }
1017 file.endMultiread(fileOffset);
1018 } else {
1019 std::vector<float> readBuffer(thatTasksSize[0]*thatTasksSize[1]*thatTasksSize[2]*N);
1020 if(file.addMultireadUnit((char*)readBuffer.data(), thatTasksSize[0]*thatTasksSize[1]*thatTasksSize[2])==false) {
1021 logFile << "(RESTART) ERROR: Failed to read fsgrid variable " << variableName << endl << write;
1022 return false;
1023 }
1024 file.endMultiread(fileOffset);
1025
1026 for(uint64_t i=0; i< thatTasksSize[0]*thatTasksSize[1]*thatTasksSize[2]*N; i++) {
1027 buffer[i]=readBuffer[i];
1028 }
1029 }
1030
1031 // Copy continuous stripes in x direction.
1032 for(FsGridTools::FsIndex_t z=overlapStart[2]; z<overlapEnd[2]; z++) {
1033 for(FsGridTools::FsIndex_t y=overlapStart[1]; y<overlapEnd[1]; y++) {
1034 for(FsGridTools::FsIndex_t x=overlapStart[0]; x<overlapEnd[0]; x++) {
1035 FsGridTools::FsIndex_t index = (z - thatTasksStart[2]) * thatTasksSize[0]*thatTasksSize[1]
1036 + (y - thatTasksStart[1]) * thatTasksSize[0]
1037 + (x - thatTasksStart[0]);
1038
1039 memcpy(targetGrid.get(x - localStart[0], y - localStart[1], z - localStart[2]), &buffer[index*N], N*sizeof(Real));
1040 }
1041 }
1042 }
1043 } else {
1044 // If we don't overlap, just perform a dummy read.
1045 file.endMultiread(fileOffset);
1046 }
1047 fileOffset += thatTasksSize[0] * thatTasksSize[1] * thatTasksSize[2];
1048 multiRead.stop();
1049 }
1050 }
1051 phiprof::Timer updateGhostsTimer {"updateGhostCells"};
1052 targetGrid.updateGhostCells();
1053 updateGhostsTimer.stop();
1054 return true;
1055}
