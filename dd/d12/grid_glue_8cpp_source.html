33/*
34Find closest in-domain cell and return the given moment from that cell
35*/
36Real copyMomentFromClosestSimCell(FsGrid< std::array<Real, fsgrids::moments::N_MOMENTS>, FS_STENCIL_WIDTH> & momentsGrid,
37 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
38 cint i,
39 cint j,
40 cint k,
41 cuint moment) {
42 int distance = numeric_limits<int>::max();
43 vector< array<int,3> > closestCells;
44
45 for (int kk=-2; kk<3; kk++) {
46 for (int jj=-2; jj<3; jj++) {
47 for (int ii=-2; ii<3 ; ii++) {
48 if( technicalGrid.get(i+ii,j+jj,k+kk) // skip invalid cells returning NULL
49 && technicalGrid.get(i+ii,j+jj,k+kk)->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY // Copy only from sim domain
50 ) {
51 distance = min(distance, ii*ii + jj*jj + kk*kk);
52 }
53 }
54 }
55 }
56
57 for (int kk=-2; kk<3; kk++) {
58 for (int jj=-2; jj<3; jj++) {
59 for (int ii=-2; ii<3 ; ii++) {
60 if( technicalGrid.get(i+ii,j+jj,k+kk) // skip invalid cells returning NULL
61 && technicalGrid.get(i+ii,j+jj,k+kk)->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY // Copy only from sim domain
62 ) {
63 int d = ii*ii + jj*jj + kk*kk;
64 if( d == distance ) {
65 array<int, 3> cell = {i+ii, j+jj, k+kk};
66 closestCells.push_back(cell);
67 }
68 }
69 }
70 }
71 }
72
73 if (closestCells.size() == 0) {
74 abort_mpi("No closest cell found!", 1);
75 }
76
77 return momentsGrid.get(closestCells[0][0], closestCells[0][1], closestCells[0][2])->at(fsgrids::moments::RHOM+moment);
78}

159 const FsGridTools::FsIndex_t* mntDims = &momentsGrid.getLocalSize()[0];
160 FsGrid<std::array<Real, fsgrids::moments::N_MOMENTS>, FS_STENCIL_WIDTH> swapGrid = momentsGrid; //swap array

163 for (int blurPass = 0; blurPass < Parameters::maxFilteringPasses; blurPass++){

165 // Blurring Pass
166 #pragma omp parallel for collapse(2)
167 for (FsGridTools::FsIndex_t k = 0; k < mntDims[2]; k++){
168 for (FsGridTools::FsIndex_t j = 0; j < mntDims[1]; j++){

173
174 // Skip pass and copy value
175 if (blurPass >= P::numPasses.at(refLevel) || technicalGrid.get(i, j, k)->sysBoundaryFlag != sysboundarytype::NOT_SYSBOUNDARY) {
176 *swap = *momentsGrid.get(i, j, k);
177 continue;
178 }
179

181 for (int e = 0; e < fsgrids::moments::N_MOMENTS; ++e) {
182 swap->at(e) = 0.0;
183 }
184
185 // Perform the blur
186 for (int c=-kernelOffset; c<=kernelOffset; c++){
187 for (int b=-kernelOffset; b<=kernelOffset; b++){
188 for (int a=-kernelOffset; a<=kernelOffset; a++){

205
206 // If outflow boundaries exist, filtered moments must be recopied there
207 copyMomentsToOutflow(momentsGrid, technicalGrid);
208 momentsGrid.updateGhostCells();
209 }

540Map from dccrg cell id to fsgrid global cell ids when they aren't identical (ie. when dccrg has refinement).

543std::vector<CellID> mapDccrgIdToFsGridGlobalID(dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
544 CellID dccrgID) {
545 const auto maxRefLvl = mpiGrid.get_maximum_refinement_level();
546 const auto refLvl = mpiGrid.get_refinement_level(dccrgID);
547 const auto cellLength = pow(2,maxRefLvl-refLvl);
548 const auto topLeftIndices = mpiGrid.mapping.get_indices(dccrgID);
549 std::array<int,3> fsgridDims;
550
551 fsgridDims[0] = P::xcells_ini * pow(2,mpiGrid.get_maximum_refinement_level());
552 fsgridDims[1] = P::ycells_ini * pow(2,mpiGrid.get_maximum_refinement_level());
553 fsgridDims[2] = P::zcells_ini * pow(2,mpiGrid.get_maximum_refinement_level());
554
555 std::vector<CellID> fsgridIDs(cellLength * cellLength * cellLength);
556 for (uint k = 0; k < cellLength; ++k) {
557 for (uint j = 0; j < cellLength; ++j) {
558 for (uint i = 0; i < cellLength; ++i) {
559 const std::array<uint64_t,3> indices = {{topLeftIndices[0] + i,topLeftIndices[1] + j,topLeftIndices[2] + k}};
560 fsgridIDs[k*cellLength*cellLength + j*cellLength + i] = indices[0] + indices[1] * fsgridDims[0] + indices[2] * fsgridDims[1] * fsgridDims[0];
561 }
562 }
563 }
564 return fsgridIDs;
565}
