63 void calculateIonosphereFsgridCoupling(
64 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
65 FsGrid< std::array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & perBGrid,
66 FsGrid< std::array<Real, fsgrids::dperb::N_DPERB>, FS_STENCIL_WIDTH> & dPerBGrid,
67 std::vector<SBC::SphericalTriGrid::Node> & nodes,
68 creal couplingRadius
69 ) {
70
71 // we don't need to do anything if we have no nodes
72 if(nodes.size() == 0) {
73 return;
74 }
75
76 phiprof::Timer timer {"fieldtracing-ionosphere-fsgridCoupling"};
77 // Pick an initial stepsize
78 creal stepSize = min(100e3, technicalGrid.DX / 2.);
79 std::vector<Real> nodeTracingStepSize(nodes.size(), stepSize); // In-flight storage of step size, needed when crossing into next MPI domain
80 std::vector<Real> reducedNodeTracingStepSize(nodes.size());
81
82 std::vector<Real> nodeDistance(nodes.size(), std::numeric_limits<Real>::max()); // For reduction of node coordinate in case of multiple hits
83 std::vector<int> nodeNeedsContinuedTracing(nodes.size(), 1); // Flag, whether tracing needs to continue on another task
84 std::vector<std::array<Real, 3>> nodeTracingCoordinates(nodes.size()); // In-flight node upmapping coordinates (for global reduction)
85 for(uint n=0; n<nodes.size(); n++) {
86 nodeTracingCoordinates.at(n) = nodes.at(n).x;
87 nodes.at(n).haveCouplingData = 0;
88 for (uint c=0; c<3; c++) {
89 nodes.at(n).xMapped.at(c) = 0;
90 nodes.at(n).parameters.at(ionosphereParameters::UPMAPPED_BX+c) = 0;
91 }

99 int itCount = 0;
100 do {
101 itCount++;
102 anyNodeNeedsTracing = false;
103
104 #pragma omp parallel
105 {
106 // Trace node coordinates outwards until a non-sysboundary cell is encountered or the local fsgrid domain has been left.
107 #pragma omp for schedule(dynamic)
108 for(uint n=0; n<nodes.size(); n++) {
109
110 if(!nodeNeedsContinuedTracing[n]) {
111 // This node has already found its target, no need for us to do anything about it.
112 continue;
113 }
114 SBC::SphericalTriGrid::Node& no = nodes[n];
115
116 std::array<Real, 3> x = nodeTracingCoordinates[n];
117 std::array<Real, 3> v({0,0,0});
118
119 while( true ) {
120
121 // Check if the current coordinates (pre-step) are in our own domain.
122 std::array<FsGridTools::FsIndex_t, 3> fsgridCell = getLocalFsGridCellIndexForCoord(technicalGrid,x);
123 // If it is not in our domain, somebody else takes care of it.

126 nodeTracingCoordinates[n] = {0,0,0};
127 nodeTracingStepSize[n]=0;
128 break;
129 }
130
131 // Make one step along the fieldline

137 nodeTracingCoordinates.at(n) = {0,0,0};
138 break;
139 }
140

144
145 // Take a step back and find the downmapRadius crossing point
146 stepFieldLine(x,v, nodeTracingStepSize[n],fieldTracingParameters.min_tracer_dx_full_box,technicalGrid.DX/2,fieldTracingParameters.tracingMethod,tracingFullField,!(no.x[2] < 0));

164 // Interpolate and record upmapped B at final xMapped ccordinates
165 const std::array<Real, 3> perB = interpolatePerturbedB(
166 perBGrid,
167 dPerBGrid,
168 technicalGrid,
169 fieldTracingParameters.reconstructionCoefficientsCache,
170 fsgridCell[0], fsgridCell[1], fsgridCell[2],
171 no.xMapped
172 );
173 no.parameters[ionosphereParameters::UPMAPPED_BX] = SBC::ionosphereGrid.dipoleField(x[0],x[1],x[2],X,0,X) + SBC::ionosphereGrid.BGB[0] + perB[0];

177 no.haveCouplingData = 1;
178 nodeDistance[n] = sqrt((no.xMapped[0]-no.x[0])*(no.xMapped[0]-no.x[0])+(no.xMapped[1]-no.x[1])*(no.xMapped[1]-no.x[1])+(no.xMapped[2]-no.x[2])*(no.xMapped[2]-no.x[2]));
179 nodeNeedsContinuedTracing[n] = 0;
180 nodeTracingCoordinates[n] = {0,0,0};
181 break;
182 }
183
184 // Look up the fsgrid cell belonging to these coordinates, again only local (no ghosts)
185 fsgridCell = getLocalFsGridCellIndexForCoord(technicalGrid,x);
186 // Now, after stepping, if it is no longer in our domain, another MPI rank will pick up later.
187 if(fsgridCell[0] == -1) {
188 nodeNeedsContinuedTracing[n] = 1;
189 nodeTracingCoordinates[n] = x;
190 break;
191 }
192 } // while(true)
193 } // for
194 } // pragma omp parallel

196 // Globally reduce whether any node still needs to be picked up and traced onwards
197 std::vector<int> sumNodeNeedsContinuedTracing(nodes.size());
198 std::vector<std::array<Real, 3>> sumNodeTracingCoordinates(nodes.size());
199 MPI_Allreduce(nodeNeedsContinuedTracing.data(), sumNodeNeedsContinuedTracing.data(), nodes.size(), MPI_INT, MPI_SUM, MPI_COMM_WORLD);

201 MPI_Allreduce(nodeTracingCoordinates.data(), sumNodeTracingCoordinates.data(), 3*nodes.size(), MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
202 MPI_Allreduce(nodeTracingStepSize.data(), reducedNodeTracingStepSize.data(), nodes.size(), MPI_DOUBLE, MPI_MAX, MPI_COMM_WORLD);
203 } else {
204 MPI_Allreduce(nodeTracingCoordinates.data(), sumNodeTracingCoordinates.data(), 3*nodes.size(), MPI_FLOAT, MPI_SUM, MPI_COMM_WORLD);
205 MPI_Allreduce(nodeTracingStepSize.data(), reducedNodeTracingStepSize.data(), nodes.size(), MPI_FLOAT, MPI_MAX, MPI_COMM_WORLD);
206 }
207 for(uint n=0; n<nodes.size(); n++) {
208 if(sumNodeNeedsContinuedTracing[n] > 0) {
209 anyNodeNeedsTracing=true;
210 nodeNeedsContinuedTracing[n] = 1;
211
212 // Update that nodes' tracing coordinates
213 nodeTracingCoordinates[n][0] = sumNodeTracingCoordinates[n][0] / sumNodeNeedsContinuedTracing[n];
214 nodeTracingCoordinates[n][1] = sumNodeTracingCoordinates[n][1] / sumNodeNeedsContinuedTracing[n];
215 nodeTracingCoordinates[n][2] = sumNodeTracingCoordinates[n][2] / sumNodeNeedsContinuedTracing[n];
216 }
217 nodeTracingStepSize[n] = reducedNodeTracingStepSize[n];
218 }
219
220 } while(anyNodeNeedsTracing);
221
222 logFile << "(fieldtracing) fsgrid coupling traced in " << itCount << " iterations of the tracing loop." << endl;
223
224 std::vector<Real> reducedNodeDistance(nodes.size());
225 if(sizeof(Real) == sizeof(double)) {
226 MPI_Allreduce(nodeDistance.data(), reducedNodeDistance.data(), nodes.size(), MPI_DOUBLE, MPI_MIN, MPI_COMM_WORLD);
227 } else {
228 MPI_Allreduce(nodeDistance.data(), reducedNodeDistance.data(), nodes.size(), MPI_FLOAT, MPI_MIN, MPI_COMM_WORLD);
229 }
230
231 // Reduce upmapped magnetic field to be consistent on all nodes
232 std::vector<Real> sendUpmappedB(3 * nodes.size());
233 std::vector<Real> reducedUpmappedB(3 * nodes.size());
234 // Likewise, reduce upmapped coordinates
235 std::vector<Real> sendxMapped(3 * nodes.size());
236 std::vector<Real> reducedxMapped(3 * nodes.size());
237 // And coupling rank number
238 std::vector<int> sendCouplingNum(nodes.size());
239 std::vector<int> reducedCouplingNum(nodes.size());
240
241 for(uint n=0; n<nodes.size(); n++) {
242 SBC::SphericalTriGrid::Node& no = nodes[n];
243 // Discard false hits from cells that are further out from the node
244 if(nodeDistance[n] > reducedNodeDistance[n]) {
245 no.haveCouplingData = 0;
246 for(int c=0; c<3; c++) {
247 no.parameters[ionosphereParameters::UPMAPPED_BX+c] = 0;
248 no.xMapped[c] = 0;
249 }
250 } else {
251 // Cell found, add association.
252 SBC::ionosphereGrid.isCouplingInwards = true;
253 }
254
255
256 sendUpmappedB[3*n] = no.parameters[ionosphereParameters::UPMAPPED_BX];
257 sendUpmappedB[3*n+1] = no.parameters[ionosphereParameters::UPMAPPED_BY];
258 sendUpmappedB[3*n+2] = no.parameters[ionosphereParameters::UPMAPPED_BZ];
259 sendxMapped[3*n] = no.xMapped[0];
260 sendxMapped[3*n+1] = no.xMapped[1];
261 sendxMapped[3*n+2] = no.xMapped[2];
262 sendCouplingNum[n] = no.haveCouplingData;
263 }
264 if(sizeof(Real) == sizeof(double)) {
265 MPI_Allreduce(sendUpmappedB.data(), reducedUpmappedB.data(), 3*nodes.size(), MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
266 MPI_Allreduce(sendxMapped.data(), reducedxMapped.data(), 3*nodes.size(), MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
267 } else {
268 MPI_Allreduce(sendUpmappedB.data(), reducedUpmappedB.data(), 3*nodes.size(), MPI_FLOAT, MPI_SUM, MPI_COMM_WORLD);
269 MPI_Allreduce(sendxMapped.data(), reducedxMapped.data(), 3*nodes.size(), MPI_FLOAT, MPI_SUM, MPI_COMM_WORLD);
270 }
271 MPI_Allreduce(sendCouplingNum.data(), reducedCouplingNum.data(), nodes.size(), MPI_INT, MPI_SUM, MPI_COMM_WORLD);
272
273 for(uint n=0; n<nodes.size(); n++) {
274 SBC::SphericalTriGrid::Node& no = nodes[n];
275
276 // We don't even care about nodes that couple nowhere.
277 if(reducedCouplingNum[n] == 0) {
278 continue;
279 }
280 no.parameters[ionosphereParameters::UPMAPPED_BX] = reducedUpmappedB[3*n] / reducedCouplingNum[n];
281 no.parameters[ionosphereParameters::UPMAPPED_BY] = reducedUpmappedB[3*n+1] / reducedCouplingNum[n];
282 no.parameters[ionosphereParameters::UPMAPPED_BZ] = reducedUpmappedB[3*n+2] / reducedCouplingNum[n];
283 no.xMapped[0] = reducedxMapped[3*n] / reducedCouplingNum[n];
284 no.xMapped[1] = reducedxMapped[3*n+1] / reducedCouplingNum[n];
285 no.xMapped[2] = reducedxMapped[3*n+2] / reducedCouplingNum[n];
286 }

297 std::array<std::pair<int, Real>, 3> calculateIonosphereVlasovGridCoupling(
298 std::array<Real,3> x,
299 std::vector<SBC::SphericalTriGrid::Node> & nodes,
300 creal couplingRadius
301 ) {
302
303 std::array<std::pair<int, Real>, 3> coupling;
304
305 Real stepSize = 100e3;
306 std::array<Real,3> v;
307
308 // For tracing towards the vlasov boundary, we only require the dipole field.
309 TracingFieldFunction<Real> dipoleFieldOnly = [](std::array<Real,3>& r, const bool outwards, std::array<Real,3>& b)->bool {
310
311 // Get field direction
312 b[0] = SBC::ionosphereGrid.dipoleField(r[0],r[1],r[2],X,0,X) + SBC::ionosphereGrid.BGB[0];
313 b[1] = SBC::ionosphereGrid.dipoleField(r[0],r[1],r[2],Y,0,Y) + SBC::ionosphereGrid.BGB[1];
314 b[2] = SBC::ionosphereGrid.dipoleField(r[0],r[1],r[2],Z,0,Z) + SBC::ionosphereGrid.BGB[2];
315
316 // Normalize
317 Real norm = 1. / sqrt(b[0]*b[0] + b[1]*b[1] + b[2]*b[2]);
318 for(int c=0; c<3; c++) {
319 b[c] = b[c] * norm;
320 }
321
322 // Make sure motion is outwards. Flip b if dot(r,b) < 0
323 if(outwards) {
324 if(b[0]*r[0] + b[1]*r[1] + b[2]*r[2] < 0) {
325 b[0]*=-1;
326 b[1]*=-1;
327 b[2]*=-1;
328 }
329 } else {
330 if(b[0]*r[0] + b[1]*r[1] + b[2]*r[2] > 0) {
331 b[0]*=-1;
332 b[1]*=-1;
333 b[2]*=-1;
334 }
335 }
336 return true;
337 };
338
339 while(x[0]*x[0]+x[1]*x[1]+x[2]*x[2] > SBC::Ionosphere::innerRadius*SBC::Ionosphere::innerRadius) {
340
341 // Make one step along the fieldline
342 stepFieldLine(x,v, stepSize,fieldTracingParameters.min_tracer_dx_ionospere_coupling,fieldTracingParameters.max_tracer_dx_ionospere_coupling,fieldTracingParameters.tracingMethod,dipoleFieldOnly,false);
343
344 // If the field lines is moving even further outwards, abort.
345 // (this shouldn't happen under normal magnetospheric conditions, but who
346 // knows what crazy driving this will be run with)
347 if(x[0]*x[0]+x[1]*x[1]+x[2]*x[2] > 1.5*1.5*couplingRadius*couplingRadius) {
348 cerr << "(fieldtracing) Warning: coupling of Vlasov grid cell failed due to weird magnetic field topology." << endl;
349
350 // Return a coupling that has 0 value and results in zero potential
351 return coupling;
352 }
353 }
354
355 const std::array<Real,3> x_in = x;
356 Real r_in = sqrt(x[0]*x[0] + x[1]*x[1] + x[2]*x[2]);
357 // Take a step back and find the innerRadius crossing point
358 stepFieldLine(x,v, stepSize,fieldTracingParameters.min_tracer_dx_ionospere_coupling,fieldTracingParameters.max_tracer_dx_ionospere_coupling,fieldTracingParameters.tracingMethod,dipoleFieldOnly,true);
359 Real r_out = sqrt(x[0]*x[0] + x[1]*x[1] + x[2]*x[2]);
360 Real alpha = (SBC::Ionosphere::innerRadius - r_in)/(r_out - r_in);
361 alpha = std::fmax(std::fmin(alpha,1.0),0.0);
362 if (fabs(r_out-r_in) < 0.01*fieldTracingParameters.min_tracer_dx_ionospere_coupling) {
363 alpha = 0.5;
364 }
365 Real xi = x[0]-x_in[0];
366 Real yi = x[1]-x_in[1];
367 Real zi = x[2]-x_in[2];
368 x[0] = x_in[0] + xi*alpha;
369 x[1] = x_in[1] + yi*alpha;
370 x[2] = x_in[2] + zi*alpha;
371
372 // Determine the nearest ionosphere node to this point.
373 uint32_t nearestNode = SBC::ionosphereGrid.findNodeAtCoordinates(x);
374 int32_t elementIndex = nodes[nearestNode].touchingElements[0];
375 int32_t oldElementIndex;
376
377 std::unordered_set<int32_t> elementHistory;
378 bool override=false;
379
380 for (uint toto=0; toto<15; toto++) {
381 const SBC::SphericalTriGrid::Element& el = SBC::ionosphereGrid.elements[elementIndex];
382 oldElementIndex = elementIndex;
383
384 if(elementHistory.find(elementIndex) == elementHistory.end()) {
385 elementHistory.insert(elementIndex);
386 } else {
387 // This element was already seen, entering a loop, let's get out
388 // It happens when the projection rx is left seen from the right triangle and right seen from the left triangle.
389 cerr << "Entered a loop, taking the current element " << elementIndex << "." << endl;
390 override=true;
391 }
392
393 // Calculate barycentric coordinates for x in this element.
394 Vec3d r1(nodes[el.corners[0]].x.data());
395 Vec3d r2(nodes[el.corners[1]].x.data());
396 Vec3d r3(nodes[el.corners[2]].x.data());
397
398 Vec3d rx(x[0],x[1],x[2]);
399
400 cint handedness = sign(dot_product(cross_product(r2-r1, r3-r1), r1));
401
402 creal kappa1 = handedness*sign(dot_product(cross_product(r1, r2-r1), rx-r1));
403 creal kappa2 = handedness*sign(dot_product(cross_product(r2, r3-r2), rx-r2));
404 creal kappa3 = handedness*sign(dot_product(cross_product(r3, r1-r3), rx-r3));
405
406 if(override || (kappa1 > 0 && kappa2 > 0 && kappa3 > 0)) {
407 // Total area
408 Real A = vector_length(cross_product(r2-r1,r3-r1));
409
410 // Project x into the plane of this triangle
411 Vec3d normal = normalize_vector(cross_product(r2-r1, r3-r1));
412 rx -= normal*dot_product(rx-r1, normal);
413
414 // Area of the sub-triangles
415 Real lambda1 = vector_length(cross_product(r2-rx, r3-rx)) / A;
416 Real lambda2 = vector_length(cross_product(r1-rx, r3-rx)) / A;
417 Real lambda3 = vector_length(cross_product(r1-rx, r2-rx)) / A;
418
419 coupling[0] = {el.corners[0], lambda1};
420 coupling[1] = {el.corners[1], lambda2};
421 coupling[2] = {el.corners[2], lambda3};
422 return coupling;
423 } else if (kappa1 > 0 && kappa2 > 0 && kappa3 < 0) {
424 elementIndex = SBC::ionosphereGrid.findElementNeighbour(elementIndex,0,2);
425 } else if (kappa2 > 0 && kappa3 > 0 && kappa1 < 0) {
426 elementIndex = SBC::ionosphereGrid.findElementNeighbour(elementIndex,0,1);
427 } else if (kappa3 > 0 && kappa1 > 0 && kappa2 < 0) {
428 elementIndex = SBC::ionosphereGrid.findElementNeighbour(elementIndex,1,2);
429 } else if (kappa1 < 0 && kappa2 < 0 && kappa3 > 0) {
430 if (handedness > 0) {
431 elementIndex = SBC::ionosphereGrid.findElementNeighbour(elementIndex,0,1);
432 } else {
433 elementIndex = SBC::ionosphereGrid.findElementNeighbour(elementIndex,1,2);
434 }
435 } else if (kappa1 < 0 && kappa2 > 0 && kappa3 < 0) {
436 if (handedness > 0) {
437 elementIndex = SBC::ionosphereGrid.findElementNeighbour(elementIndex,0,2);
438 } else {
439 elementIndex = SBC::ionosphereGrid.findElementNeighbour(elementIndex,0,1);
440 }
441 } else if (kappa1 > 0 && kappa2 < 0 && kappa3 < 0) {
442 if (handedness > 0) {
443 elementIndex = SBC::ionosphereGrid.findElementNeighbour(elementIndex,1,2);
444 } else {
445 elementIndex = SBC::ionosphereGrid.findElementNeighbour(elementIndex,0,2);
446 }
447 } else {
448 cerr << "This fell through, strange."
449 << " kappas " << kappa1 << " " << kappa2 << " " << kappa3
450 << " handedness " << handedness
451 << " r1 " << r1[0] << " " << r1[1] << " " << r1[2]
452 << " r2 " << r2[0] << " " << r2[1] << " " << r2[2]
453 << " r3 " << r3[0] << " " << r3[1] << " " << r3[2]
454 << " rx " << rx[0] << " " << rx[1] << " " << rx[2]
455 << endl;
456 }
457 if(elementIndex == -1) {
458 cerr << __FILE__ << ":" << __LINE__ << ": invalid elementIndex returned for coordinate "
459 << x[0] << " " << x[1] << " " << x[2] << " projected to rx " << rx[0] << " " << rx[1] << " " << rx[2]
460 << ". Last valid elementIndex: " << oldElementIndex << "." << endl;
461 return coupling;
462 }
463 }
464
465 // If we arrived here, we did not find an element to couple to (why?)
466 // Return an empty coupling instead
467 cerr << "(fieldtracing) Failed to find an ionosphere element to couple to for coordinate " <<
468 x[0] << " " << x[1] << " " << x[2] << endl;
469 return coupling;
470 }

474 void traceOpenClosedConnection(
475 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
476 FsGrid< std::array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & perBGrid,
477 FsGrid< std::array<Real, fsgrids::dperb::N_DPERB>, FS_STENCIL_WIDTH> & dPerBGrid,
478 std::vector<SBC::SphericalTriGrid::Node> & nodes
479 ) {
480
481 // we don't need to do anything if we have no nodes

489 std::vector<TReal> nodeTracingStepSize(nodes.size(), stepSize); // In-flight storage of step size, needed when crossing into next MPI domain
490 std::vector<TReal> reducedNodeTracingStepSize(nodes.size());

495 std::vector<uint64_t> nodeStepCounter(nodes.size());

505 nodeTracingCoordinates.at(n)[2] = nodes.at(n).x[2];

513 int itCount=0;
514 bool warnMaxStepsExceeded = false;

517 itCount++;

519 #pragma omp parallel
520 {
521 // Trace node coordinates outwards until a non-sysboundary cell is encountered or the local fsgrid domain has been left.
522 #pragma omp for schedule(dynamic)

560 nodeTracingStepCount[n]++;

608 MPI_Allreduce(nodeTracingStepSize.data(), reducedNodeTracingStepSize.data(), nodes.size(), MPI_FLOAT, MPI_MAX, MPI_COMM_WORLD);
609 }
610 for(uint n=0; n<nodes.size(); n++) {

616 nodeTracingCoordinates[n][0] = sumNodeTracingCoordinates[n][0] / sumNodeNeedsContinuedTracing[n];
617 nodeTracingCoordinates[n][1] = sumNodeTracingCoordinates[n][1] / sumNodeNeedsContinuedTracing[n];
618 nodeTracingCoordinates[n][2] = sumNodeTracingCoordinates[n][2] / sumNodeNeedsContinuedTracing[n];

620 nodeStepCounter[n] = maxNodeStepCounter[n];
621 }
622 nodeTracingStepSize[n] = reducedNodeTracingStepSize[n];
623 }
624 } while(anyNodeNeedsTracing);

630 int rank;
631 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
632 if(redWarning && rank == MASTER_RANK) {
633 logFile << "(fieldtracing) Warning: reached the maximum number of tracing steps " << maxTracingSteps << " allowed for open-closed ionosphere tracing." << endl;

643 }

651 void stepCellAcrossTaskDomain(
652 cint n,
653 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
654 TracingFieldFunction<TReal> & tracingFullField,
655 const std::vector<std::array<TReal,3>> & cellInitialCoordinates,
656 const std::vector<TReal> & cellCurvatureRadius,
657 std::vector<std::array<TReal, 3>> & cellTracingCoordinates,
658 std::vector<TReal> & cellTracingStepSize,
659 std::vector<TReal> & cellRunningDistance,
660 std::vector<TReal> & cellMaxExtension,
661 std::vector<signed char> & cellConnection,
662 bool & warnMaxDistanceExceeded,
663 const TReal maxTracingDistance,
664 cuint DIRECTION
665 ) {
666 std::array<TReal, 3> x = cellTracingCoordinates[n];
667 std::array<TReal, 3> v({0,0,0});
668 while( true ) {
669 // Check if the current coordinates (pre-step) are in our own domain.
670 std::array<FsGridTools::FsIndex_t, 3> fsgridCell = getLocalFsGridCellIndexForCoord(technicalGrid,{(Real)x[0], (Real)x[1], (Real)x[2]});
671 // If it is not in our domain, somebody else takes care of it.
672 if(fsgridCell[0] == -1) {
673 cellTracingCoordinates[n] = {0,0,0};
674 cellTracingStepSize[n]=0;
675 break;
676 }
677
678 // Make one step along the fieldline
679 // Forward tracing means true for last argument
680 stepFieldLine(x,v, cellTracingStepSize[n],(TReal)100e3,(TReal)technicalGrid.DX/2,fieldTracingParameters.tracingMethod,tracingFullField,(DIRECTION == Direction::FORWARD));
681 cellRunningDistance[n] += cellTracingStepSize[n];
682
683 // Look up the fsgrid cell belonging to these coordinates
684 fsgridCell = getLocalFsGridCellIndexForCoord(technicalGrid,{(Real)x[0], (Real)x[1], (Real)x[2]});
685
686 // If we map into the ionosphere, discard this field line.
687 if(x.at(0)*x.at(0) + x.at(1)*x.at(1) + x.at(2)*x.at(2) < fieldTracingParameters.innerBoundaryRadius*fieldTracingParameters.innerBoundaryRadius) {
688 cellTracingCoordinates[n] = x;
689 cellConnection[n] += TracingLineEndType::CLOSED;
690
691 // Take a step back and find the innerRadius crossing point
692 stepFieldLine(x,v, cellTracingStepSize[n],(TReal)fieldTracingParameters.min_tracer_dx_full_box,(TReal)technicalGrid.DX/2,fieldTracingParameters.tracingMethod,tracingFullField,!(DIRECTION == Direction::FORWARD));
693 Real r_in = sqrt(cellTracingCoordinates[n][0]*cellTracingCoordinates[n][0] + cellTracingCoordinates[n][1]*cellTracingCoordinates[n][1] + cellTracingCoordinates[n][2]*cellTracingCoordinates[n][2]);
694 Real r_out = sqrt(x[0]*x[0] + x[1]*x[1] + x[2]*x[2]);
695 Real alpha = (fieldTracingParameters.innerBoundaryRadius-r_in)/(r_out - r_in);
696 alpha = std::fmax(std::fmin(alpha,1.0),0.0);
697 if (fabs(r_out-r_in) < 0.01*fieldTracingParameters.min_tracer_dx_full_box) {
698 alpha = 0.5;
699 }
700 TReal xi = x[0]-cellTracingCoordinates[n][0];
701 TReal yi = x[1]-cellTracingCoordinates[n][1];
702 TReal zi = x[2]-cellTracingCoordinates[n][2];
703 cellTracingCoordinates[n][0] += xi*alpha;
704 cellTracingCoordinates[n][1] += yi*alpha;
705 cellTracingCoordinates[n][2] += zi*alpha;
706 cellRunningDistance[n] -= cellTracingStepSize[n]*alpha;
707 break;
708 }
709
710 // If we map out of the box, discard this field line.
711 if(
712 x[0] > fieldTracingParameters.x_max
713 || x[0] < fieldTracingParameters.x_min
714 || x[1] > fieldTracingParameters.y_max
715 || x[1] < fieldTracingParameters.y_min
716 || x[2] > fieldTracingParameters.z_max
717 || x[2] < fieldTracingParameters.z_min
718 ) {
719 cellTracingCoordinates[n] = x;
720 cellConnection[n] += TracingLineEndType::OPEN;
721 break;
722 }
723
724 // If we exceed the max tracing distance we're probably looping
725 if(cellRunningDistance[n] > maxTracingDistance) {
726 cellTracingCoordinates[n] = x;
727 cellConnection[n] += TracingLineEndType::DANGLING;
728 #pragma omp critical
729 {
730 warnMaxDistanceExceeded = true;
731 }
732 break;
733 }
734
735 // See the longer comment for the function traceFullBoxConnectionAndFluxRopes for details.
736 // If we are still in the race for flux rope...
737 if(cellConnection[n] < TracingLineEndType::N_TYPES) {
738 const TReal extension = sqrt(
739 (x[0]-(cellInitialCoordinates[n])[0])*(x[0]-(cellInitialCoordinates[n])[0])
740 + (x[1]-(cellInitialCoordinates[n])[1])*(x[1]-(cellInitialCoordinates[n])[1])
741 + (x[2]-(cellInitialCoordinates[n])[2])*(x[2]-(cellInitialCoordinates[n])[2])
742 );
743 cellMaxExtension[n] = max(cellMaxExtension[n], extension);
744 // ...and if we traced too far from the seed, this is not a flux rope candidate and we do a single +=
745 if(extension > fieldTracingParameters.fluxrope_max_curvature_radii_extent*cellCurvatureRadius[n]) {
746 cellConnection[n] += TracingLineEndType::N_TYPES;
747 } else if(cellRunningDistance[n] > fieldTracingParameters.fluxrope_max_curvature_radii_to_trace*cellCurvatureRadius[n]) {
748 // If we're still in the game and reach this limit we have a hit and we do a double +=
749 cellConnection[n] += 2*TracingLineEndType::N_TYPES;
750 }
751 }
752
753 // Now, after stepping, if it is no longer in our domain, another MPI rank will pick up later.
754 if(fsgridCell[0] == -1) {
755 cellTracingCoordinates[n] = x;
756 break;
757 }
758 } // while true
759 }
