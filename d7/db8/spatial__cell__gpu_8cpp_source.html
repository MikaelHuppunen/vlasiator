410 void SpatialCell::adjust_velocity_blocks(const uint popID, bool doDeleteEmptyBlocks) {
411 debug_population_check(popID);
412
413 const uint cpuThreadID = gpu_getThread();
414 const gpuStream_t stream = gpu_getStream();
415
416 vmesh::VelocityMesh* host_vmesh = populations[popID].vmesh;
417 vmesh::VelocityMesh* dev_vmesh = populations[popID].dev_vmesh;
418 vmesh::GlobalID* _withContentData = velocity_block_with_content_list->data();
419
420 // Evaluate velocity halo for local content blocks
421 if (velocity_block_with_content_list_size>0) {
422 const int addWidthV = getObjectWrapper().particleSpecies[popID].sparseBlockAddWidthV;
423 if (addWidthV!=1) {
424 std::cerr<<"Warning! "<<__FILE__<<":"<<__LINE__<<" Halo extent is not 1, unsupported size."<<std::endl;
425 }
426 // Halo of 1 in each direction adds up to 26 velocity neighbours.
427 // For NVIDIA/CUDA, we dan do 26 neighbours and 32 threads per warp in a single block.
428 // For AMD/HIP, we dan do 13 neighbours and 64 threads per warp in a single block, meaning two loops per cell.
429 // In either case, we launch blocks equal to velocity_block_with_content_list_size
430 update_velocity_halo_kernel<<<velocity_block_with_content_list_size, 26*32, 0, stream>>> (
431 dev_vmesh,
432 velocity_block_with_content_list_size,
433 _withContentData,
434 dev_velocity_block_with_content_map,
435 dev_velocity_block_with_no_content_map
436 );
437 CHK_ERR( gpuPeekAtLastError() );
438 //CHK_ERR( gpuStreamSynchronize(stream) );
439 }
440
445
446 /****
447 // Gather pointers and counts from neighbours
448 uint neighbours_count = neighbor_ptrs.size();
449 uint neighbours_blocks_count = 0;
450 std::vector<vmesh::GlobalID*> neigh_vbwcls;
451 std::vector<vmesh::LocalID> neigh_Nvbwcls;
452
453 if (neighbours_count > 0) {
454 for (std::vector<SpatialCell*>::const_iterator neighbor=neighbor_ptrs.begin();
455 neighbor != neighbor_ptrs.end(); ++neighbor) {
456 if ((*neighbor)->velocity_block_with_content_list_size > 0) {
457 neigh_Nvbwcls.push_back((*neighbor)->velocity_block_with_content_list_size);
458 neigh_vbwcls.push_back((*neighbor)->velocity_block_with_content_list->data());
459 neighbours_blocks_count += (*neighbor)->velocity_block_with_content_list_size;
460 }
461 }
462 neighbours_count = neigh_Nvbwcls.size(); // Only include neighbours with content.
463 }
464
465 if (neighbours_count > 0) {
466 // Upload pointers and counters for neighbours
467 vmesh::GlobalID** dev_neigh_vbwcls;
468 vmesh::GlobalID* dev_neigh_Nvbwcls;
469 CHK_ERR( gpuMallocAsync((void**)&dev_neigh_vbwcls, neighbours_count*sizeof(vmesh::GlobalID*), stream) ); // where would these be cleared?
470 CHK_ERR( gpuMallocAsync((void**)&dev_neigh_Nvbwcls, neighbours_count*sizeof(vmesh::LocalID), stream) );
471 CHK_ERR( gpuMemcpyAsync(dev_neigh_vbwcls, neigh_vbwcls.data(), neighbours_count*sizeof(vmesh::GlobalID*), gpuMemcpyHostToDevice, stream) );
472 CHK_ERR( gpuMemcpyAsync(dev_neigh_Nvbwcls, neigh_Nvbwcls.data(), neighbours_count*sizeof(vmesh::LocalID), gpuMemcpyHostToDevice, stream) );
473 // For NVIDIA/CUDA, we dan do 32 neighbour GIDs and 32 threads per warp in a single block.
474 // For AMD/HIP, we dan do 16 neighbour GIDs and 64 threads per warp in a single block
475 // This is handled in-kernel.
476 // ceil int division
477 uint launchBlocks = 1 + ((neighbours_blocks_count - 1) / WARPSPERBLOCK);
