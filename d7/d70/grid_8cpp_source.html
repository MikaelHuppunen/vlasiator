92void initializeGrids(
93 int argn,
94 char **argc,
95 dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
96 FsGrid< std::array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & perBGrid,
97 FsGrid< std::array<Real, fsgrids::bgbfield::N_BGB>, FS_STENCIL_WIDTH> & BgBGrid,
98 FsGrid< std::array<Real, fsgrids::moments::N_MOMENTS>, FS_STENCIL_WIDTH> & momentsGrid,
99 FsGrid< std::array<Real, fsgrids::moments::N_MOMENTS>, FS_STENCIL_WIDTH> & momentsDt2Grid,
100 FsGrid< std::array<Real, fsgrids::dmoments::N_DMOMENTS>, FS_STENCIL_WIDTH> & dMomentsGrid,
101 FsGrid< std::array<Real, fsgrids::efield::N_EFIELD>, FS_STENCIL_WIDTH> & EGrid,
102 FsGrid< std::array<Real, fsgrids::egradpe::N_EGRADPE>, FS_STENCIL_WIDTH> & EGradPeGrid,
103 FsGrid< std::array<Real, fsgrids::volfields::N_VOL>, FS_STENCIL_WIDTH> & volGrid,
104 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
105 SysBoundary& sysBoundaries,
106 Project& project
107) {
108 int myRank;
109 MPI_Comm_rank(MPI_COMM_WORLD,&myRank);
110
111 // Init Zoltan:
112 float zoltanVersion;
113 if (Zoltan_Initialize(argn,argc,&zoltanVersion) != ZOLTAN_OK) {
114 if(myRank == MASTER_RANK) cerr << "\t ERROR: Zoltan initialization failed." << endl;
115 exit(1);
116 } else {
117 logFile << "\t Zoltan " << zoltanVersion << " initialized successfully" << std::endl << writeVerbose;
118 }
119
120 MPI_Comm comm = MPI_COMM_WORLD;
121 int neighborhood_size = VLASOV_STENCIL_WIDTH;
122 if (P::vlasovSolverGhostTranslate) {
123 // One extra layer for translation of ghost cells
124 neighborhood_size++;
125 }
126
127 const std::array<uint64_t, 3> grid_length = {{P::xcells_ini, P::ycells_ini, P::zcells_ini}};
128 dccrg::Cartesian_Geometry::Parameters geom_params;
129 geom_params.start[0] = P::xmin;
130 geom_params.start[1] = P::ymin;
131 geom_params.start[2] = P::zmin;
132 geom_params.level_0_cell_length[0] = P::dx_ini;
133 geom_params.level_0_cell_length[1] = P::dy_ini;
134 geom_params.level_0_cell_length[2] = P::dz_ini;
135
136 phiprof::Timer dccrgTimer {"Initialize DCCRG grid"};
137 mpiGrid.set_initial_length(grid_length)
138 .set_load_balancing_method(&P::loadBalanceAlgorithm[0])
139 .set_neighborhood_length(neighborhood_size)
140 .set_maximum_refinement_level(P::amrMaxSpatialRefLevel)
141 .set_periodic(sysBoundaries.isPeriodic(0),
142 sysBoundaries.isPeriodic(1),
143 sysBoundaries.isPeriodic(2))
144 .initialize(comm)
145 .set_geometry(geom_params);
146 dccrgTimer.stop();
147
148 phiprof::Timer refineTimer {"Refine spatial cells"};
149 // We need this first as well
150 recalculateLocalCellsCache(mpiGrid);
151 if (!P::isRestart) {
152 // Note call to project.refineSpatialCells below
153 if (P::amrMaxSpatialRefLevel > 0 && project.refineSpatialCells(mpiGrid)) {
154 mpiGrid.balance_load();
155 recalculateLocalCellsCache(mpiGrid);
156 mapRefinement(mpiGrid, technicalGrid);
157 }
158 } else {
159 if (myRank == MASTER_RANK) logFile << "(INIT): Reading grid structure from " << P::restartFileName << endl << writeVerbose;
160 bool restartSuccess = readFileCells(mpiGrid, P::restartFileName);
161 if (myRank == MASTER_RANK) logFile << " ...done." << endl << writeVerbose;
162 if (restartSuccess) {
163 mpiGrid.balance_load();
164 recalculateLocalCellsCache(mpiGrid);
165 mapRefinement(mpiGrid, technicalGrid);
166 }
167 }
168 refineTimer.stop();
169 initializeStencils(mpiGrid);
170
171 for (const auto& [key, value] : P::loadBalanceOptions) {
172 mpiGrid.set_partitioning_option(key, value);
173 }
174 phiprof::Timer initialLBTimer {"Initial load-balancing"};
175 if (myRank == MASTER_RANK) logFile << "(INIT): Starting initial load balance." << endl << writeVerbose;
176 mpiGrid.balance_load(); // Direct DCCRG call, recalculate cache afterwards
177 recalculateLocalCellsCache(mpiGrid);
178
179 SpatialCell::set_mpi_transfer_type(Transfer::VEL_BLOCK_DATA);
180 mpiGrid.update_copies_of_remote_neighbors(Neighborhoods::NEAREST);
181
182 if (!P::vlasovSolverGhostTranslate) {
183 setFaceNeighborRanks( mpiGrid ); // Only needed for remote contribution in translation
184 }
185 const vector<CellID>& cells = getLocalCells();
186 initialLBTimer.stop();
187
188 if (myRank == MASTER_RANK) {
189 logFile << "(INIT): Set initial state." << endl << writeVerbose;
190 }
191
192 phiprof::Timer initialStateTimer {"Set initial state"};
193
194 phiprof::Timer setCoordsTimer {"Set spatial cell coordinates"};
195 initSpatialCellCoordinates(mpiGrid);
196 setCoordsTimer.stop();
197
198 phiprof::Timer initBoundaryTimer {"Initialize system boundary conditions"};
199 sysBoundaries.initSysBoundaries(project, P::t_min);
200 initBoundaryTimer.stop();
201
202 SpatialCell::set_mpi_transfer_type(Transfer::CELL_DIMENSIONS);
203 mpiGrid.update_copies_of_remote_neighbors(Neighborhoods::SYSBOUNDARIES);
204
205 computeCoupling(mpiGrid, cells, technicalGrid);
206
207 // We want this before restart refinement
208 phiprof::Timer classifyTimer {"Classify cells (sys boundary conditions)"};
209 sysBoundaries.classifyCells(mpiGrid,technicalGrid);
210 classifyTimer.stop();
211
212 if (P::isRestart) {
213 logFile << "Restart from "<< P::restartFileName << std::endl << writeVerbose;
214 phiprof::Timer restartReadTimer {"Read restart"};
215 if (readGrid(mpiGrid,perBGrid,EGrid,technicalGrid,P::restartFileName) == false) {
216 logFile << "(MAIN) ERROR: restarting failed" << endl;
217 exit(1);
218 }
219 restartReadTimer.stop();
220
221 if (P::forceRefinement) {
222 // Adapt refinement to match new static refinement parameters
223 phiprof::Timer timer {"Restart refinement"};
224 for (int i = 0; i < P::amrMaxSpatialRefLevel; ++i) {
225 // (un)Refinement is done one level at a time so we don't blow up memory
226 if (!adaptRefinement(mpiGrid, technicalGrid, sysBoundaries, project, i)) {
227 cerr << "(MAIN) ERROR: Forcing refinement takes too much memory" << endl;
228 exit(1);
229 }
230 balanceLoad(mpiGrid, sysBoundaries, technicalGrid);
231 }
232 } else if (P::refineOnRestart) {
233 // Considered deprecated
234 phiprof::Timer timer {"Restart refinement"};
235 // Get good load balancing for refinement
236 balanceLoad(mpiGrid, sysBoundaries, technicalGrid);
237 adaptRefinement(mpiGrid, technicalGrid, sysBoundaries, project);
238 balanceLoad(mpiGrid, sysBoundaries, technicalGrid);
239 }
240 }
241
242 // Check refined cells do not touch boundary cells
243 phiprof::Timer boundaryCheckTimer {"Check boundary refinement"};
244 sysBoundaries.checkRefinement(mpiGrid);
245 boundaryCheckTimer.stop();
246
247 if (P::isRestart) {
248 //initial state for sys-boundary cells, will skip those not set to be reapplied at restart
249 sysBoundaries.applyInitialState(mpiGrid, technicalGrid, perBGrid, BgBGrid, project);
250 }
251
252 // Update technicalGrid (e.g. sysboundary flags)
253 technicalGrid.updateGhostCells();
254
255 if (!P::isRestart && !P::writeFullBGB) {
256 // If we are starting a new regular simulation, we need to prepare all cells with their initial state.
257 // If we're only after writing out the full BGB we don't need all this shebang EXCEPT the weights!
258
259 //Initial state based on project, background field in all cells
260 //and other initial values in non-sysboundary cells
261 phiprof::Timer applyInitialTimer {"Apply initial state"};
262 // Go through every cell on this node and initialize the
263 // -Background field on all cells
264 // -Perturbed fields and ion distribution function in non-sysboundary cells
265 // Each initialization has to be independent to avoid threading problems
266
267 // Allow the project to set up data structures for it's setCell calls
268 project.setupBeforeSetCell(cells);
269
270 phiprof::Timer setCellTimer {"setCell"};
271 #pragma omp parallel for schedule(dynamic)
272 for (size_t i=0; i<cells.size(); ++i) {
273 SpatialCell* cell = mpiGrid[cells[i]];
274 if (cell->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY) {
275 project.setCell(cell);
276 }
277 }
278 setCellTimer.stop();
279
280 // Initial state for sys-boundary cells
281 sysBoundaries.applyInitialState(mpiGrid, technicalGrid, perBGrid, BgBGrid, project);
282
283 #pragma omp parallel for schedule(static)
284 for (size_t i=0; i<cells.size(); ++i) {
285 mpiGrid[cells[i]]->parameters[CellParams::LBWEIGHTCOUNTER] = 0;
286 }
287
288 for (uint popID=0; popID<getObjectWrapper().particleSpecies.size(); ++popID) {
289 adjustVelocityBlocks(mpiGrid,cells,true,popID);
290 // set initial LB metric based on number of blocks
291 #pragma omp parallel for schedule(static)
292 for (size_t i=0; i<cells.size(); ++i) {
293 SpatialCell* SC = mpiGrid[cells[i]];
294 if (SC->sysBoundaryFlag == sysboundarytype::DO_NOT_COMPUTE) {
295 SC->parameters[CellParams::LBWEIGHTCOUNTER] = 0;
296 } else if (SC->sysBoundaryFlag != sysboundarytype::NOT_SYSBOUNDARY) {
297 // Set sysb cells to a small weight
298 SC->parameters[CellParams::LBWEIGHTCOUNTER] += 0.5 * SC->get_number_of_velocity_blocks(popID);
299 } else {
300 SC->parameters[CellParams::LBWEIGHTCOUNTER] += 3.0 * SC->get_number_of_velocity_blocks(popID);
301 }
302 }
303 }
304
305 shrink_to_fit_grid_data(mpiGrid); //get rid of excess data already here
306
307 /*
308 // Apply boundary conditions so that we get correct initial moments
309 sysBoundaries.applySysBoundaryVlasovConditions(mpiGrid,Parameters::t);
310 //compute moments, and set them in RHO* and RHO_*_DT2. If restart, they are already read in
311 phiprof::Timer initMomentsTimer {"Init moments"};
312 calculateInitialVelocityMoments(mpiGrid);
313 initMomentsTimer.stop();
314 */
315
316 } else if (P::writeFullBGB) {
317 // If, instead of starting a regular simulation, we are only writing out the background field, it is enough to set a dummy load balance value of 1 here.
318 for (size_t i=0; i<cells.size(); ++i) {
319 mpiGrid[cells[i]]->parameters[CellParams::LBWEIGHTCOUNTER] = 1;
320 }
321 }
322
323
324 // Balance load before we transfer all data below
325 balanceLoad(mpiGrid, sysBoundaries, technicalGrid, false);
326 // Function includes re-calculation of local cells cache, but
327 // setting third parameter to false skips preparation of
328 // translation cell lists and building of pencils.
329
330 phiprof::Timer fetchNeighbourTimer {"Fetch Neighbour data", {"MPI"}};
331 // update complete cell spatial data for full stencil
332 SpatialCell::set_mpi_transfer_type(Transfer::ALL_SPATIAL_DATA);
333 mpiGrid.update_copies_of_remote_neighbors(Neighborhoods::FULL);
334 fetchNeighbourTimer.stop();
335
336 phiprof::Timer setBTimer {"project.setProjectBField"};
337 project.setProjectBField(perBGrid, BgBGrid, technicalGrid);
338 setBTimer.stop();
339 if (P::isRestart) {
340 // There are projects that have non-uniform and non-zero perturbed B, e.g. Magnetosphere with dipole type 4.
341 // If restarting with reapplyUponRestart active, we need to set PerB again
342 // in boundary cells after setProjectBField has populated the BGBXVDCORR etc. terms
343 sysBoundaries.applyInitialState(mpiGrid, technicalGrid, perBGrid, BgBGrid, project);
344 }
345 phiprof::Timer fsGridGhostTimer {"fsgrid-ghost-updates"};
346 perBGrid.updateGhostCells();
347 BgBGrid.updateGhostCells();
348 EGrid.updateGhostCells();
349
350 // This will only have the BGB set up properly at this stage but we need the BGBvol for the Vlasov boundaries below.
351 volGrid.updateGhostCells();
352 fsGridGhostTimer.stop();
353 phiprof::Timer getFieldsTimer {"getFieldsFromFsGrid"};
354 getFieldsFromFsGrid(volGrid, BgBGrid, EGradPeGrid, dMomentsGrid, technicalGrid, mpiGrid, cells);
355 getFieldsTimer.stop();
356
357 setBTimer.stop();
358
359 // If we only want the full BGB for writeout, we have it now and we can return early.
360 if(P::writeFullBGB == true) {
361 return;
362 }
363
364 if (P::isRestart == false) {
365 // Apply boundary conditions so that we get correct initial moments
366 sysBoundaries.applySysBoundaryVlasovConditions(mpiGrid,Parameters::t, true); // It doesn't matter here whether we put _R or _V moments
367
368 //compute moments, and set them in RHO* and RHO_*_DT2. If restart, they are already read in
369 phiprof::Timer timer {"Init moments"};
370 calculateInitialVelocityMoments(mpiGrid);
371 } else {
372 phiprof::Timer timer {"Init moments"};
373 #pragma omp parallel for schedule(guided,1)
374 for (size_t i=0; i<cells.size(); ++i) {
375 // easier to skip here than adding one more bool flag to calculateCellMoments - handles L2 outflow cells without VDF
376 if(mpiGrid[cells[i]]->sysBoundaryFlag == sysboundarytype::OUTFLOW && mpiGrid[cells[i]]->sysBoundaryLayer != 1) {
377 continue;
378 }
379 calculateCellMoments(mpiGrid[cells[i]], true, true);
380 }
381 }
382
383
384 phiprof::Timer finishFSGridTimer {"Finish fsgrid setup"};
385 feedMomentsIntoFsGrid(mpiGrid, cells, momentsGrid, technicalGrid, false);
386 if(!P::isRestart) {
387 // WARNING this means moments and dt2 moments are the same here at t=0, which is a feature so far.
388 feedMomentsIntoFsGrid(mpiGrid, cells, momentsDt2Grid, technicalGrid, false);
389 } else {
390 feedMomentsIntoFsGrid(mpiGrid, cells, momentsDt2Grid, technicalGrid, true);
391 }
392 momentsGrid.updateGhostCells();
393 momentsDt2Grid.updateGhostCells();
394 finishFSGridTimer.stop();
395
396 // Set this so CFL doesn't break
397 if(P::refineOnRestart) {
398 // Half-step acceleration
399 if( P::propagateVlasovAcceleration ) {
400 calculateAcceleration(mpiGrid, -0.5*P::dt + 0.5*P::bailout_min_dt);
401 } else {
402 calculateAcceleration(mpiGrid, 0.0);
403 }
404 P::dt = P::bailout_min_dt;
405 }
406
407 // With all cell data in place, make preparations for translation
408 prepareAMRLists(mpiGrid);
409 initialStateTimer.stop();
410}

414 #pragma omp parallel for

427 mpiGrid[cells[i]]->parameters[CellParams::REFINEMENT_LEVEL] = mpiGrid.get_refinement_level(cells[i]);
428 }

434void setFaceNeighborRanks( dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid ) {
435
436 const vector<CellID>& cells = getLocalCells();
437 // TODO: Try a #pragma omp parallel for
438 for (const auto& cellid : cells) {
439
440 if (cellid == INVALID_CELLID) continue;
441
442 SpatialCell* cell = mpiGrid[cellid];
443
444 if (!cell) continue;
445
446 cell->face_neighbor_ranks.clear();
447
448 for (const auto& [neighbor, dir] : mpiGrid.get_face_neighbors_of(cellid)) {
449
450 int neighborhood;
451
452 // We store rank numbers into a map that has neighborhood ids as its key values.
453
454 switch (dir) {
455 case -3:
456 neighborhood = Neighborhoods::SHIFT_M_Z;
457 break;
458 case -2:
459 neighborhood = Neighborhoods::SHIFT_M_Y;
460 break;
461 case -1:
462 neighborhood = Neighborhoods::SHIFT_M_X;
463 break;
464 case +1:
465 neighborhood = Neighborhoods::SHIFT_P_X;
466 break;
467 case +2:
468 neighborhood = Neighborhoods::SHIFT_P_Y;
469 break;
470 case +3:
471 neighborhood = Neighborhoods::SHIFT_P_Z;
472 break;
473 default:
474 cerr << "Invalid face neighbor dimension: " << dir << " in " << __FILE__ << ":" << __LINE__ << std::endl;
475 abort();
476 }
477
478 cell->face_neighbor_ranks[neighborhood].insert(mpiGrid.get_process(neighbor));
479
480 }
481 }

484void balanceLoad(dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid, SysBoundary& sysBoundaries, FsGrid<fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid, bool doTranslationLists){
485 // Invalidate cached cell lists
486 Parameters::meshRepartitioned = true;
487
488 // tell other processes which velocity blocks exist in remote spatial cells
489 phiprof::Timer balanceLoadTimer {"Balancing load", {"Load balance"}};
490
491 phiprof::Timer deallocTimer {"deallocate boundary data"};
492 //deallocate blocks in remote cells to decrease memory load
493 deallocateRemoteCellBlocks(mpiGrid);
494 deallocTimer.stop();
495
496 //set weights based on each cells LB weight counter
497 const vector<CellID>& cells = getLocalCells();
498 for (size_t i=0; i<cells.size(); ++i){
499 // Set cell weight. We could use different counters or number of blocks if different solvers are active.
500 // if (P::propagateVlasovAcceleration)
501 // When using the FS-SPLIT functionality, Jaro Hokkanen reported issues with using the regular
502 // CellParams::LBWEIGHTCOUNTER, so use of blockscounts + 1 might be required.
503 mpiGrid.set_cell_weight(cells[i], (Real)1 + mpiGrid[cells[i]]->parameters[CellParams::LBWEIGHTCOUNTER]);
504 }
505
506 phiprof::Timer initLBTimer {"dccrg.initialize_balance_load"};
507 mpiGrid.initialize_balance_load(true);
508 initLBTimer.stop();
509
510 const std::unordered_set<CellID>& incoming_cells = mpiGrid.get_cells_added_by_balance_load();
511 std::vector<CellID> incoming_cells_list (incoming_cells.begin(),incoming_cells.end());
512
513 const std::unordered_set<CellID>& outgoing_cells = mpiGrid.get_cells_removed_by_balance_load();
514 std::vector<CellID> outgoing_cells_list (outgoing_cells.begin(),outgoing_cells.end());
515
516 /*transfer cells in parts to preserve memory*/
517 phiprof::Timer transfersTimer {"Data transfers"};
518
519 // Idea: do as many cell sending passes hereafter so that there's not more than transfer_block_fraction_limit
520 // blocks of this task's total block count that gets sent. Helps in reducing memory peaks during load balancing.
521 creal transfer_block_fraction_limit = 0.1;
522 uint64_t num_part_transfers_local = 1, num_part_transfers, outgoing_block_count = 0, total_block_count = 0;
523 bool count_determined = false;
524 Real outgoing_block_fraction;
525
526 // count blocks
527 for (unsigned int i=0; i<outgoing_cells_list.size(); i++) {
528 CellID cell_id=outgoing_cells_list[i];
529 SpatialCell* cell = mpiGrid[cell_id];
530 outgoing_block_count += cell->get_number_of_all_velocity_blocks();
531 }
532 for (unsigned int i=0; i<cells.size(); i++) {
533 CellID cell_id=cells[i];
534 SpatialCell* cell = mpiGrid[cell_id];
535 total_block_count += cell->get_number_of_all_velocity_blocks();
536 }
537 outgoing_block_fraction = (Real)outgoing_block_count / ((Real)total_block_count + 1);
538 // if we're not exceeding transfer_block_fraction_limit we're good
539 if(outgoing_block_fraction < transfer_block_fraction_limit) {
540 count_determined = true;
541 }
542 // otherwise we increase the number of chunks until all chunks are below transfer_block_fraction_limit
543 while(!count_determined) {
544 uint64_t transfer_part; // we use this in the logic after the for
545 for (transfer_part=0; transfer_part<num_part_transfers_local; transfer_part++) {
546 uint64_t transfer_part_block_count=0;
547 for (unsigned int i=0;i<outgoing_cells_list.size();i++){
548 CellID cell_id=outgoing_cells_list[i];
549 if (cell_id%num_part_transfers_local==transfer_part) {
550 transfer_part_block_count += mpiGrid[cell_id]->get_number_of_all_velocity_blocks();
551 }
552 }
553 outgoing_block_fraction = (Real)transfer_part_block_count / ((Real)total_block_count + 1);
554 if(outgoing_block_fraction > transfer_block_fraction_limit) {
555 num_part_transfers_local *= 2;
556 break; // out of for
557 }
558 }
559 if((transfer_part == num_part_transfers_local // either the loop ended or we hit that number with the *= 2
560 && outgoing_block_fraction <= transfer_block_fraction_limit) // so cross-check with this
561 || num_part_transfers_local >= cells.size()
562 ) {
563 count_determined = true; // we got a break out if any chunk was still too big
564 }
565 }
566 // ...and finally we reduce this across all tasks of course.
567 MPI_Allreduce(&num_part_transfers_local, &num_part_transfers, 1, MPI_UINT64_T, MPI_MAX, MPI_COMM_WORLD);
568
569 for (uint64_t transfer_part=0; transfer_part<num_part_transfers; transfer_part++) {
570 //Set transfers on/off for the incoming cells in this transfer set and prepare for receive
571 for (unsigned int i=0;i<incoming_cells_list.size();i++){
572 CellID cell_id=incoming_cells_list[i];
573 SpatialCell* cell = mpiGrid[cell_id];
574 if (cell_id%num_part_transfers!=transfer_part) {
575 cell->set_mpi_transfer_enabled(false);
576 } else {
577 cell->set_mpi_transfer_enabled(true);
578 }
579 }
580
581 //Set transfers on/off for the outgoing cells in this transfer set
582 for (unsigned int i=0; i<outgoing_cells_list.size(); i++) {
583 CellID cell_id=outgoing_cells_list[i];
584 SpatialCell* cell = mpiGrid[cell_id];
585 if (cell_id%num_part_transfers!=transfer_part) {
586 cell->set_mpi_transfer_enabled(false);
587 } else {
588 cell->set_mpi_transfer_enabled(true);
589 }
590 }
591
592 for (size_t popID=0; popID<getObjectWrapper().particleSpecies.size(); ++popID) {
593 // Set active population
594 SpatialCell::setCommunicatedSpecies(popID);
595
596 // Transfer velocity block lists. On-device GPU mesh preparation tasks require
597 // device synchronization between transfer phases.
598 SpatialCell::set_mpi_transfer_type(Transfer::VEL_BLOCK_LIST_STAGE1);
599 mpiGrid.continue_balance_load();
600 #ifdef USE_GPU
601 CHK_ERR( gpuDeviceSynchronize() );
602 #endif
603 SpatialCell::set_mpi_transfer_type(Transfer::VEL_BLOCK_LIST_STAGE2);
604 mpiGrid.continue_balance_load();
605 #ifdef USE_GPU
606 CHK_ERR( gpuDeviceSynchronize() );
607 #endif
608
609 int prepareReceives {phiprof::initializeTimer("Preparing receives")};
610 int receives = 0;
611 #pragma omp parallel for schedule(guided)
612 for (unsigned int i=0; i<incoming_cells_list.size(); i++) {
613 CellID cell_id=incoming_cells_list[i];
614 SpatialCell* cell = mpiGrid[cell_id];
615 if (cell_id % num_part_transfers == transfer_part) {
616 receives++;
617 // reserve space for velocity block data in arriving remote cells
618 phiprof::Timer timer {prepareReceives};
619 cell->prepare_to_receive_blocks(popID);
620 timer.stop(1, "Spatial cells");
621 }
622 }
623 if(receives == 0) {
624 //empty phiprof timer, to avoid unneccessary divergence in unique
625 //profiles (keep order same)
626 phiprof::Timer timer {prepareReceives};
627 timer.stop(0, "Spatial cells");
628 }
629
630 //do the actual transfer of data for the set of cells to be transferred
631 phiprof::Timer transferTimer {"transfer_all_data"};
632 SpatialCell::set_mpi_transfer_type(Transfer::ALL_DATA);
633 mpiGrid.continue_balance_load();
634 transferTimer.stop();
635
636 // Free memory for cells that have been sent (the block data)
637 for (unsigned int i=0;i<outgoing_cells_list.size();i++){
638 CellID cell_id=outgoing_cells_list[i];
639 SpatialCell* cell = mpiGrid[cell_id];
640
641 // Free memory of this cell as it has already been transferred,
642 // it will not be used anymore. NOTE: Only clears memory allocated
643 // to the active population.
644 if (cell_id % num_part_transfers == transfer_part) cell->clear(popID,true);
645 }
646
647 memory_purge(); // Purge jemalloc allocator to actually release memory
648 } // for-loop over populations
649 } // for-loop over transfer parts
650 transfersTimer.stop();
651
652 //finish up load balancing
653 phiprof::Timer finishLBTimer {"dccrg.finish_balance_load"};
654 mpiGrid.finish_balance_load();
655 finishLBTimer.stop();
656
657 //Make sure transfers are enabled for all cells
658 recalculateLocalCellsCache(mpiGrid);
659 #pragma omp parallel for
660 for (uint i=0; i<cells.size(); ++i) {
661 mpiGrid[cells[i]]->set_mpi_transfer_enabled(true);
662 }
663
664 // recompute coupling of grids after load balance
665 computeCoupling(mpiGrid, cells, technicalGrid);
666
667 // Communicate all spatial data for FULL neighborhood, which
668 // includes all data with the exception of dist function data
669 SpatialCell::set_mpi_transfer_type(Transfer::ALL_SPATIAL_DATA);
670 mpiGrid.update_copies_of_remote_neighbors(Neighborhoods::FULL);
671
672 phiprof::Timer updateBlocksTimer {"update block lists"};
673 //new partition, re/initialize blocklists of remote cells.
674 for (uint popID=0; popID<getObjectWrapper().particleSpecies.size(); ++popID) {
675 if (P::vlasovSolverGhostTranslate) {
676 updateRemoteVelocityBlockLists(mpiGrid,popID,Neighborhoods::VLASOV_SOLVER_GHOST);
677 } else {
678 updateRemoteVelocityBlockLists(mpiGrid,popID);
679 }
680 }
681 updateBlocksTimer.stop();
682
683 phiprof::Timer updateBoundariesTimer {"update sysboundaries"};
684 sysBoundaries.updateSysBoundariesAfterLoadBalance( mpiGrid );
685 updateBoundariesTimer.stop();
686
687 // Prepare ghost translation cell lists and build pencils for translation.
688 if (doTranslationLists) {
689 prepareAMRLists(mpiGrid);
690 }
691
692 // Record ranks of face neighbors for translation remote neighbor contribution
693 if (!P::vlasovSolverGhostTranslate) {
694 phiprof::Timer timer {"set face neighbor ranks"};
695 setFaceNeighborRanks( mpiGrid );
696 }
697
698#ifdef USE_GPU
699 phiprof::Timer gpuReservationsTimer("GPU LB set cell reservations");
700 uint gpuMaxBlockCount = 0;
701 vmesh::LocalID gpuBlockCount = 0;
702 // Not parallelized
703 const vector<CellID>& newCells = getLocalCells();
704 const uint newCellsSize = newCells.size();
705 const std::vector<CellID>& remote_cells = mpiGrid.get_remote_cells_on_process_boundary(Neighborhoods::FULL);
706 for (uint i=0; i<newCellsSize+remote_cells.size(); ++i) {
707 SpatialCell* SC;
708 if (i < newCells.size()) {
709 SC = mpiGrid[newCells[i]];
710 } else {
711 SC = mpiGrid[remote_cells[i - newCells.size()]];
712 }
713 for (size_t popID=0; popID<getObjectWrapper().particleSpecies.size(); ++popID) {
714 const vmesh::VelocityMesh* vmesh = SC->get_velocity_mesh(popID);
715 vmesh::VelocityBlockContainer* blockContainer = SC->get_velocity_blocks(popID);
716 gpuBlockCount = vmesh->size();
717 if (gpuBlockCount > gpuMaxBlockCount) {
718 gpuMaxBlockCount = gpuBlockCount;
719 }
720 // Ensure cell has sufficient reservation, then apply it
721 SC->setReservation(popID,gpuBlockCount);
722 SC->applyReservation(popID);
723 SC->dev_upload_population(popID);
724 }
725 }
726 CHK_ERR( gpuDeviceSynchronize() );
727 gpuReservationsTimer.stop();
728 // Call GPU routines for memory allocation for Vlasov solvers
729 // deallocates first if necessary
730 phiprof::Timer gpuAllocationsTimer("GPU LB set buffer allocations");
731 gpu_vlasov_allocate(gpuMaxBlockCount);
732 gpu_calculateProbeAllocation(gpuMaxBlockCount);
733 gpu_acc_allocate(gpuMaxBlockCount);
734 gpuAllocationsTimer.stop();
735#endif // end USE_GPU
736
737}

982

984 std::vector<neigh_t> neighborhood;
985 for (int z = -1; z <= 1; z++) {
986 for (int y = -1; y <= 1; y++) {
987 for (int x = -1; x <= 1; x++) {
988 if (x == 0 && y == 0 && z == 0) {
989 continue;
990 }
991 neigh_t offsets = {{x, y, z}};
992 neighborhood.push_back(offsets);
993 }
994 }
995 }

998 abort();
999 }
1000 if (!mpiGrid.add_neighborhood(Neighborhoods::SYSBOUNDARIES, neighborhood)){
1001 std::cerr << "Failed to add neighborhood Neighborhoods::SYSBOUNDARIES \n";
1002 abort();
1003 }
1004
1005 neighborhood.clear();
1006 for (int z = -2; z <= 2; z++) {
1007 for (int y = -2; y <= 2; y++) {
1008 for (int x = -2; x <= 2; x++) {
1009 if (x == 0 && y == 0 && z == 0) {
1010 continue;
1011 }
1012 neigh_t offsets = {{x, y, z}};
1013 neighborhood.push_back(offsets);
1014 }
1015 }
1016 }
1017 if (!mpiGrid.add_neighborhood(Neighborhoods::SYSBOUNDARIES_EXTENDED, neighborhood)){
1018 std::cerr << "Failed to add neighborhood Neighborhoods::SYSBOUNDARIES_EXTENDED \n";
1019 abort();
1020 }
1021

1023 if (P::vlasovSolverGhostTranslate) {
1024 // One extra layer for translation of ghost cells
1025 full_neighborhood_size++;
1026 }
1027 neighborhood.clear();
1028 for (int z = -full_neighborhood_size; z <= full_neighborhood_size; z++) {
1029 for (int y = -full_neighborhood_size; y <= full_neighborhood_size; y++) {
1030 for (int x = -full_neighborhood_size; x <= full_neighborhood_size; x++) {
1031 if (x == 0 && y == 0 && z == 0) {
1032 continue;
1033 }
1034 neigh_t offsets = {{x, y, z}};
1035 neighborhood.push_back(offsets);
1036 }
1037 }
1038 }
1039 /*all possible communication pairs*/
1040 if( !mpiGrid.add_neighborhood(Neighborhoods::FULL, neighborhood)){
1041 std::cerr << "Failed to add neighborhood Neighborhoods::FULL \n";
1042 abort();
1043 }
1044
1045 /*stencils for semilagrangian propagators*/
1046 neighborhood.clear();
1047 for (int d = -VLASOV_STENCIL_WIDTH; d <= VLASOV_STENCIL_WIDTH; d++) {
1048 if (d != 0) {
1049 neighborhood.push_back({{d, 0, 0}});
1050 neighborhood.push_back({{0, d, 0}});
1051 neighborhood.push_back({{0, 0, d}});
1052 }

1064 if (x == 0 && y == 0 ) continue;

1066 if (y == 0 && z == 0 ) continue;

1068 neighborhood.push_back(offsets);
1069 }
1070 }
1071 }
1072 if( !mpiGrid.add_neighborhood(Neighborhoods::DIST_FUNC, neighborhood)){
1073 std::cerr << "Failed to add neighborhood Neighborhoods::DIST_FUNC \n";
1074 abort();
1075 }

1078 for (int d = -VLASOV_STENCIL_WIDTH; d <= VLASOV_STENCIL_WIDTH; d++) {
1079 if (d != 0) {
1080 neighborhood.push_back({{d, 0, 0}});
1081 }
1082 }

1084 std::cerr << "Failed to add neighborhood Neighborhoods::VLASOV_SOLVER_X \n";
1085 abort();
1086 }
1087
1088 neighborhood.clear();
1089 for (int d = -VLASOV_STENCIL_WIDTH; d <= VLASOV_STENCIL_WIDTH; d++) {
1090 if (d != 0) {
1091 neighborhood.push_back({{0, d, 0}});
1092 }
1093 }

1095 std::cerr << "Failed to add neighborhood Neighborhoods::VLASOV_SOLVER_Y \n";
1096 abort();
1097 }
1098
1099 neighborhood.clear();
1100 for (int d = -VLASOV_STENCIL_WIDTH; d <= VLASOV_STENCIL_WIDTH; d++) {
1101 if (d != 0) {
1102 neighborhood.push_back({{0, 0, d}});
1103 }
1104 }

1106 std::cerr << "Failed to add neighborhood Neighborhoods::VLASOV_SOLVER_Z \n";
1107 abort();
1108 }

1110 if (P::vlasovSolverGhostTranslate) {
1111 neighborhood.clear();
1112 for (int d = -VLASOV_STENCIL_WIDTH-1; d <= VLASOV_STENCIL_WIDTH+1; d++) {
1113 if (d != 0) {
1114 neighborhood.push_back({{d, 0, 0}});

1118 std::cerr << "Failed to add neighborhood Neighborhoods::VLASOV_SOLVER_X_GHOST \n";
1119 abort();
1120 }
1121
1122 neighborhood.clear();
1123 for (int d = -VLASOV_STENCIL_WIDTH-1; d <= VLASOV_STENCIL_WIDTH+1; d++) {
1124 if (d != 0) {
1125 neighborhood.push_back({{0, d, 0}});
1126 }
1127 }

1177 for (int dy = -(int)P::vlasovSolverGhostTranslateExtent; dy <= (int)P::vlasovSolverGhostTranslateExtent; dy++){

1182 }

1184 }
1185 }
1186 }

1189 abort();
