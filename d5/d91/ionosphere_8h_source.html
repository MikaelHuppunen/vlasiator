235 // Returns the surface area of one element on the sphere
236 Real elementArea(uint32_t elementIndex) {
237 const std::array<Real, 3>& a = nodes[elements[elementIndex].corners[0]].x;
238 const std::array<Real, 3>& b = nodes[elements[elementIndex].corners[1]].x;
239 const std::array<Real, 3>& c = nodes[elements[elementIndex].corners[2]].x;
240
241 // Two edges e1 = b-c, e2 = c-a
242 std::array<Real, 3> e1{b[0]-c[0], b[1]-c[1],b[2]-c[2]};
243 std::array<Real, 3> e2{c[0]-a[0], c[1]-a[1],c[2]-a[2]};
244 // Area vector A = cross(e1 e2)
245 std::array<Real, 3> area{ e1[1]*e2[2] - e1[2]*e2[1],
246 e1[2]*e2[0] - e1[0]*e2[2],
247 e1[0]*e2[1] - e1[1]*e2[0]};
248
249 return 0.5 * sqrt( area[0]*area[0] + area[1]*area[1] + area[2]*area[2] );
250 }
251
252 // Returns the projected surface area of one element, mapped up along the magnetic field to
253 // the simulation boundary. If one of the nodes maps nowhere, returns 0.
254 // Returns an oriented vector, which can be dotted with B
255 std::array<Real, 3> mappedElementArea(uint32_t elementIndex) {
256 const std::array<Real, 3>& a = nodes[elements[elementIndex].corners[0]].xMapped;
257 const std::array<Real, 3>& b = nodes[elements[elementIndex].corners[1]].xMapped;
258 const std::array<Real, 3>& c = nodes[elements[elementIndex].corners[2]].xMapped;
259
260 // Check if any node maps to zero
261 if( sqrt( a[0]*a[0] + a[1]*a[1] + a[2]*a[2] ) == 0 ||
262 sqrt( b[0]*b[0] + b[1]*b[1] + b[2]*b[2] ) == 0 ||
263 sqrt( c[0]*c[0] + c[1]*c[1] + c[2]*c[2] ) == 0) {
264
265 return {0,0,0};
266 }
267
268 // Two edges e1 = b-c, e2 = c-a
269 std::array<Real, 3> e1{b[0]-c[0], b[1]-c[1],b[2]-c[2]};
270 std::array<Real, 3> e2{c[0]-a[0], c[1]-a[1],c[2]-a[2]};
271 // Area vector A = cross(e1 e2)
272 const Real HALF = 0.5;
273 const Real THIRD = 1./3.;
274 std::array<Real, 3> area{ HALF * (e1[1]*e2[2] - e1[2]*e2[1]),
275 HALF * (e1[2]*e2[0] - e1[0]*e2[2]),
276 HALF * (e1[0]*e2[1] - e1[1]*e2[0])};
277
278 // By definition, the area is oriented outwards, so if dot(r,A) < 0, flip it.
279 std::array<Real, 3> r{
280 (a[0]+b[0]+c[0]) * THIRD,
281 (a[1]+b[1]+c[1]) * THIRD,
282 (a[2]+b[2]+c[2]) * THIRD};
283 if(area[0]*r[0] + area[1]*r[1] + area[2] *r[2] < 0) {
284 area[0]*=-1.;
285 area[1]*=-1.;
286 area[2]*=-1.;
287 }
288 return area;
289 }

291 Real nodeNeighbourArea(uint32_t nodeIndex) { // Summed area of all touching elements
292
293 Node& n = nodes[nodeIndex];
294 Real area=0;
295
296 for(uint i=0; i<n.numTouchingElements; i++) {
297 area += elementArea(n.touchingElements[i]);
298 }
299 return area;
300 }
