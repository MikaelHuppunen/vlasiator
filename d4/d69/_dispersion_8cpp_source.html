99 void Dispersion::hook(
100 cuint& stage,
101 const dccrg::Dccrg<spatial_cell::SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
102 FsGrid< std::array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & perBGrid
103 ) const {
104 if(hook::END_OF_TIME_STEP == stage) {
105 int myRank;
106 MPI_Comm_rank(MPI_COMM_WORLD,&myRank);
107
108 vector<Real> localRhom(P::xcells_ini, 0.0), outputRhom(P::xcells_ini, 0.0);
109
110 const vector<CellID>& cells = getLocalCells();
111
112 for(uint i=0; i<cells.size(); i++) {
113 if(cells[i] <= P::xcells_ini) {
114 localRhom[cells[i] - 1] = mpiGrid[cells[i]]->parameters[CellParams::RHOM];
115 }
116 }
117
118 MPI_Reduce(&(localRhom[0]), &(outputRhom[0]), P::xcells_ini, MPI_DOUBLE, MPI_SUM, MASTER_RANK, MPI_COMM_WORLD);
119
120 vector<Real> localPerBx(P::xcells_ini, 0.0);
121 vector<Real> localPerBy(P::xcells_ini, 0.0);
122 vector<Real> localPerBz(P::xcells_ini, 0.0);
123 vector<Real> outputPerBx(P::xcells_ini, 0.0);
124 vector<Real> outputPerBy(P::xcells_ini, 0.0);
125 vector<Real> outputPerBz(P::xcells_ini, 0.0);
126
127 const std::array<FsGridTools::FsIndex_t, 3> localSize = perBGrid.getLocalSize();
128 const std::array<FsGridTools::FsIndex_t, 3> localStart = perBGrid.getLocalStart();
129 for (FsGridTools::FsIndex_t x = 0; x < localSize[0]; ++x) {
130 localPerBx[x + localStart[0]] = perBGrid.get(x, 0, 0)->at(fsgrids::bfield::PERBX);
131 localPerBy[x + localStart[0]] = perBGrid.get(x, 0, 0)->at(fsgrids::bfield::PERBY);
132 localPerBz[x + localStart[0]] = perBGrid.get(x, 0, 0)->at(fsgrids::bfield::PERBZ);
133 }
134
135 MPI_Reduce(&(localPerBx[0]), &(outputPerBx[0]), P::xcells_ini, MPI_DOUBLE, MPI_SUM, MASTER_RANK, MPI_COMM_WORLD);
136 MPI_Reduce(&(localPerBy[0]), &(outputPerBy[0]), P::xcells_ini, MPI_DOUBLE, MPI_SUM, MASTER_RANK, MPI_COMM_WORLD);
137 MPI_Reduce(&(localPerBz[0]), &(outputPerBz[0]), P::xcells_ini, MPI_DOUBLE, MPI_SUM, MASTER_RANK, MPI_COMM_WORLD);
138
139 if(myRank == MASTER_RANK) {
140 FILE* outputFile = fopen("perBxt.bin", "ab");
141 fwrite(&(outputPerBx[0]), sizeof(outputPerBx[0]), P::xcells_ini, outputFile);
142 fclose(outputFile);
143 outputFile = fopen("perByt.bin", "ab");
144 fwrite(&(outputPerBy[0]), sizeof(outputPerBy[0]), P::xcells_ini, outputFile);
145 fclose(outputFile);
146 outputFile = fopen("perBzt.bin", "ab");
147 fwrite(&(outputPerBz[0]), sizeof(outputPerBz[0]), P::xcells_ini, outputFile);
148 fclose(outputFile);
149 outputFile = fopen("rhomt.bin", "ab");
150 fwrite(&(outputRhom[0]), sizeof(outputRhom[0]), P::xcells_ini, outputFile);
151 fclose(outputFile);
152 }
153 }
154 }

156 Realf Dispersion::fillPhaseSpace(spatial_cell::SpatialCell *cell,
157 const uint popID,
158 const uint nRequested
159 ) const {
160 const DispersionSpeciesParameters& sP = speciesParams[popID];
161 const Real mass = getObjectWrapper().particleSpecies[popID].mass;
162 Real initT = sP.TEMPERATURE;
163 Real initRho = sP.DENSITY * (1.0 + sP.densityPertRelAmp * (0.5 - this->rndRho));
164 const Real initV0X = sP.VX0 + sP.velocityPertAbsAmp * (0.5 - this->rndVel[0]);
165 const Real initV0Y = sP.VY0 + sP.velocityPertAbsAmp * (0.5 - this->rndVel[1]);
166 const Real initV0Z = sP.VZ0 + sP.velocityPertAbsAmp * (0.5 - this->rndVel[2]);
167
168 #ifdef USE_GPU
169 vmesh::VelocityMesh *vmesh = cell->dev_get_velocity_mesh(popID);
170 vmesh::VelocityBlockContainer* VBC = cell->dev_get_velocity_blocks(popID);
171 #else
172 vmesh::VelocityMesh *vmesh = cell->get_velocity_mesh(popID);
173 vmesh::VelocityBlockContainer* VBC = cell->get_velocity_blocks(popID);
174 #endif
175 // Loop over blocks
176 Realf rhosum = 0;
177 arch::parallel_reduce<arch::null>(
178 {WID, WID, WID, nRequested},
179 ARCH_LOOP_LAMBDA (const uint i, const uint j, const uint k, const uint initIndex, Realf *lsum ) {
180 vmesh::GlobalID *GIDlist = vmesh->getGrid()->data();
181 Realf* bufferData = VBC->getData();
182 const vmesh::GlobalID blockGID = GIDlist[initIndex];
183 // Calculate parameters for new block
184 Real blockCoords[6];
185 vmesh->getBlockInfo(blockGID,&blockCoords[0]);
186 creal vxBlock = blockCoords[0];
187 creal vyBlock = blockCoords[1];
188 creal vzBlock = blockCoords[2];
189 creal dvxCell = blockCoords[3];
190 creal dvyCell = blockCoords[4];
191 creal dvzCell = blockCoords[5];
192 ARCH_INNER_BODY(i, j, k, initIndex, lsum) {
193 creal vx = vxBlock + (i+0.5)*dvxCell - initV0X;
194 creal vy = vyBlock + (j+0.5)*dvyCell - initV0Y;
195 creal vz = vzBlock + (k+0.5)*dvzCell - initV0Z;
196 Realf value = MaxwellianPhaseSpaceDensity(vx,vy,vz,initT,initRho,mass);
197 bufferData[initIndex*WID3 + k*WID2 + j*WID + i] = value;
198 //lsum[0] += value;
199 };
200 }, rhosum);
201 return rhosum;
202 }

215 void Dispersion::setProjectBField(
216 FsGrid< std::array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & perBGrid,
217 FsGrid< std::array<Real, fsgrids::bgbfield::N_BGB>, FS_STENCIL_WIDTH> & BgBGrid,
218 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid
219 ) {
220 ConstantField bgField;
221 bgField.initialize(this->B0 * cos(this->angleXY) * cos(this->angleXZ),
222 this->B0 * sin(this->angleXY) * cos(this->angleXZ),
223 this->B0 * sin(this->angleXZ));
224
225 setBackgroundField(bgField, BgBGrid);
226
227 if(!P::isRestart) {
228 const auto localSize = BgBGrid.getLocalSize().data();
229
230#pragma omp parallel for collapse(3)
231 for (FsGridTools::FsIndex_t x = 0; x < localSize[0]; ++x) {
232 for (FsGridTools::FsIndex_t y = 0; y < localSize[1]; ++y) {
233 for (FsGridTools::FsIndex_t z = 0; z < localSize[2]; ++z) {
234 std::array<Real, fsgrids::bfield::N_BFIELD>* cell = perBGrid.get(x, y, z);
235 const int64_t cellid = perBGrid.GlobalIDForCoords(x, y, z);
236
237 std::default_random_engine rndState;
238 setRandomSeed(cellid,rndState);
239
240 Real rndBuffer[3];
241 rndBuffer[0]=getRandomNumber(rndState);
242 rndBuffer[1]=getRandomNumber(rndState);
243 rndBuffer[2]=getRandomNumber(rndState);
244
245 cell->at(fsgrids::bfield::PERBX) = this->magXPertAbsAmp * (0.5 - rndBuffer[0]);
246 cell->at(fsgrids::bfield::PERBY) = this->magYPertAbsAmp * (0.5 - rndBuffer[1]);
247 cell->at(fsgrids::bfield::PERBZ) = this->magZPertAbsAmp * (0.5 - rndBuffer[2]);
248 }
249 }
250 }
251 }
