37void assignConductivityTensor(std::vector<SphericalTriGrid::Node>& nodes, Real sigmaP, Real sigmaH) {
38 static const char epsilon[3][3][3] = {
39 {{0,0,0},{0,0,1},{0,-1,0}},
40 {{0,0,-1},{0,0,0},{1,0,0}},
41 {{0,1,0},{-1,0,0},{0,0,0}}
42 };
43
44 for(uint n=0; n<nodes.size(); n++) {
45 std::array<Real, 3> b = {nodes[n].x[0] / Ionosphere::innerRadius, nodes[n].x[1] / Ionosphere::innerRadius, nodes[n].x[2] / Ionosphere::innerRadius};
46 if(nodes[n].x[2] >= 0) {
47 b[0] *= -1;
48 b[1] *= -1;
49 b[2] *= -1;
50 }
51 for(int i=0; i<3; i++) {
52 for(int j=0; j<3; j++) {
53 nodes[n].parameters[ionosphereParameters::SIGMA + i*3 + j] = sigmaP * (((i==j)? 1. : 0.) - b[i]*b[j]);
54 for(int k=0; k<3; k++) {
55 nodes[n].parameters[ionosphereParameters::SIGMA + i*3 + j] -= sigmaH * epsilon[i][j][k]*b[k];
56 }
57 }
58 }
