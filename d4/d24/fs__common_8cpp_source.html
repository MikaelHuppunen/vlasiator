55void reconstructionCoefficients(

57 FsGrid< std::array<Real, fsgrids::dperb::N_DPERB>, FS_STENCIL_WIDTH> & dPerBGrid,

71 FsGrid< std::array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> * params = & perBGrid;
72
73 cep_i1j1k1 = params->get(i,j,k);
74 dummyCellParams = cep_i1j1k1;
75 cep_i2j1k1 = dummyCellParams;
76 cep_i1j2k1 = dummyCellParams;
77 cep_i1j1k2 = dummyCellParams;
78 if (params->get(i+1,j,k) != NULL) cep_i2j1k1 = params->get(i+1,j,k);
79 if (params->get(i,j+1,k) != NULL) cep_i1j2k1 = params->get(i,j+1,k);

84 // Create a dummy array for containing zero values for derivatives on non-existing cells:

86 for (int ii=0; ii<fsgrids::dperb::N_DPERB; ii++) {
87 dummyDerivatives.at(ii) = 0.0;

96 if (dPerBGrid.get(i,j+1,k) != NULL) der_i1j2k1 = dPerBGrid.get(i,j+1,k);

102 perturbedResult[Rec::a_zz] = 0.0;
103 perturbedResult[Rec::a_yz] = 0.0;

112 perturbedResult[Rec::b_zz] = 0.0;

114 perturbedResult[Rec::b_xyy] = 0.0;

130 perturbedResult[Rec::a_zz] = HALF * (der_i2j1k1->at(fsgrids::dperb::dPERBxdzz) + der_i1j1k1->at(fsgrids::dperb::dPERBxdzz));
131 perturbedResult[Rec::a_yz] = HALF * (der_i2j1k1->at(fsgrids::dperb::dPERBxdyz) + der_i1j1k1->at(fsgrids::dperb::dPERBxdyz));

133 perturbedResult[Rec::a_xyz] = (der_i2j1k1->at(fsgrids::dperb::dPERBxdyz) - der_i1j1k1->at(fsgrids::dperb::dPERBxdyz));
134 perturbedResult[Rec::a_xzz] = (der_i2j1k1->at(fsgrids::dperb::dPERBxdzz) - der_i1j1k1->at(fsgrids::dperb::dPERBxdzz));

137 perturbedResult[Rec::b_xz] = HALF * (der_i1j2k1->at(fsgrids::dperb::dPERBydxz) + der_i1j1k1->at(fsgrids::dperb::dPERBydxz));
138 perturbedResult[Rec::b_zz] = HALF * (der_i1j2k1->at(fsgrids::dperb::dPERBydzz) + der_i1j1k1->at(fsgrids::dperb::dPERBydzz));
139 perturbedResult[Rec::b_xxy] = (der_i1j2k1->at(fsgrids::dperb::dPERBydxx) - der_i1j1k1->at(fsgrids::dperb::dPERBydxx));
140 perturbedResult[Rec::b_xyz] = (der_i1j2k1->at(fsgrids::dperb::dPERBydxz) - der_i1j1k1->at(fsgrids::dperb::dPERBydxz));
141 perturbedResult[Rec::b_yzz] = (der_i1j2k1->at(fsgrids::dperb::dPERBydzz) - der_i1j1k1->at(fsgrids::dperb::dPERBydzz));

144 perturbedResult[Rec::c_xy] = HALF * (der_i1j1k2->at(fsgrids::dperb::dPERBzdxy) + der_i1j1k1->at(fsgrids::dperb::dPERBzdxy));
145 perturbedResult[Rec::c_yy] = HALF * (der_i1j1k2->at(fsgrids::dperb::dPERBzdyy) + der_i1j1k1->at(fsgrids::dperb::dPERBzdyy));

147 perturbedResult[Rec::c_xyz] = (der_i1j1k2->at(fsgrids::dperb::dPERBzdxy) - der_i1j1k1->at(fsgrids::dperb::dPERBzdxy));
148 perturbedResult[Rec::c_yyz] = (der_i1j1k2->at(fsgrids::dperb::dPERBzdyy) - der_i1j1k1->at(fsgrids::dperb::dPERBzdyy));
149
150 perturbedResult[Rec::a_xxx] = -THIRD*(perturbedResult[Rec::b_xxy] + perturbedResult[Rec::c_xxz]);
151 perturbedResult[Rec::a_xxy] = -FOURTH*perturbedResult[Rec::c_xyz];
152 perturbedResult[Rec::a_xxz] = -FOURTH*perturbedResult[Rec::b_xyz];
153
154 perturbedResult[Rec::b_xyy] = -FOURTH*perturbedResult[Rec::c_xyz];

161 } else {

163 abort();
164 }

166 // Calculate 2nd order reconstruction coefficients:
167 perturbedResult[Rec::a_xy] = der_i2j1k1->at(fsgrids::dperb::dPERBxdy) - der_i1j1k1->at(fsgrids::dperb::dPERBxdy);
168 perturbedResult[Rec::a_xz] = der_i2j1k1->at(fsgrids::dperb::dPERBxdz) - der_i1j1k1->at(fsgrids::dperb::dPERBxdz);

170 perturbedResult[Rec::a_z ] = HALF*(der_i2j1k1->at(fsgrids::dperb::dPERBxdz) + der_i1j1k1->at(fsgrids::dperb::dPERBxdz)) - SIXTH*perturbedResult[Rec::a_xxz];
171
172 perturbedResult[Rec::b_xy] = der_i1j2k1->at(fsgrids::dperb::dPERBydx) - der_i1j1k1->at(fsgrids::dperb::dPERBydx);
173 perturbedResult[Rec::b_yz] = der_i1j2k1->at(fsgrids::dperb::dPERBydz) - der_i1j1k1->at(fsgrids::dperb::dPERBydz);
174 perturbedResult[Rec::b_x ] = HALF*(der_i1j2k1->at(fsgrids::dperb::dPERBydx) + der_i1j1k1->at(fsgrids::dperb::dPERBydx)) - SIXTH*perturbedResult[Rec::b_xyy];
175 perturbedResult[Rec::b_z ] = HALF*(der_i1j2k1->at(fsgrids::dperb::dPERBydz) + der_i1j1k1->at(fsgrids::dperb::dPERBydz)) - SIXTH*perturbedResult[Rec::b_yyz];
176
177 perturbedResult[Rec::c_xz] = der_i1j1k2->at(fsgrids::dperb::dPERBzdx) - der_i1j1k1->at(fsgrids::dperb::dPERBzdx);
178 perturbedResult[Rec::c_yz] = der_i1j1k2->at(fsgrids::dperb::dPERBzdy) - der_i1j1k1->at(fsgrids::dperb::dPERBzdy);
179 perturbedResult[Rec::c_x ] = HALF*(der_i1j1k2->at(fsgrids::dperb::dPERBzdx) + der_i1j1k1->at(fsgrids::dperb::dPERBzdx)) - SIXTH*perturbedResult[Rec::c_xzz];
180 perturbedResult[Rec::c_y ] = HALF*(der_i1j1k2->at(fsgrids::dperb::dPERBzdy) + der_i1j1k1->at(fsgrids::dperb::dPERBzdy)) - SIXTH*perturbedResult[Rec::c_yzz];
181
182 perturbedResult[Rec::a_xx] = -HALF*(perturbedResult[Rec::b_xy] + perturbedResult[Rec::c_xz]);
183 perturbedResult[Rec::b_yy] = -HALF*(perturbedResult[Rec::a_xy] + perturbedResult[Rec::c_yz]);

187 perturbedResult[Rec::b_y ] = cep_i1j2k1->at(fsgrids::bfield::PERBY) - cep_i1j1k1->at(fsgrids::bfield::PERBY) - TENTH*perturbedResult[Rec::b_yyy];
188 perturbedResult[Rec::c_z ] = cep_i1j1k2->at(fsgrids::bfield::PERBZ) - cep_i1j1k1->at(fsgrids::bfield::PERBZ) - TENTH*perturbedResult[Rec::c_zzz];
189
190 #else
191 for (int i=0; i<Rec::N_REC_COEFFICIENTS; ++i) {
192 perturbedResult[i] = 0.0;
193 }
194 #endif

197 perturbedResult[Rec::a_0 ] = HALF*(cep_i2j1k1->at(fsgrids::bfield::PERBX) + cep_i1j1k1->at(fsgrids::bfield::PERBX)) - SIXTH*perturbedResult[Rec::a_xx];

216std::array<Real, 3> interpolatePerturbedB(
217 FsGrid< std::array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & perBGrid,
218 FsGrid< std::array<Real, fsgrids::dperb::N_DPERB>, FS_STENCIL_WIDTH> & dPerBGrid,
219 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
220 std::map< std::array<int, 3>, std::array<Real, Rec::N_REC_COEFFICIENTS> > & reconstructionCoefficientsCache,
221 cint i,
222 cint j,
223 cint k,
224 const std::array<Real, 3> x
225) {
226 cuint cellSysBoundaryFlag = technicalGrid.get(i,j,k)->sysBoundaryFlag;
227 if (cellSysBoundaryFlag != sysboundarytype::NOT_SYSBOUNDARY) {
228 std::array<Real, 3> zero = {0,0,0};
229 return zero;
230 }
231
232 // Balsara reconstruction formulas: x,y,z are in [-1/2, 1/2] local coordinates
233 std::array<Real, 3> xLocal = getFractionalFsGridCellForCoord(technicalGrid, x);
234 xLocal[0] -= 0.5;
235 xLocal[1] -= 0.5;
236 xLocal[2] -= 0.5;
237
238 if (fabs(xLocal[0]) > 0.5 || fabs(xLocal[1]) > 0.5 || fabs(xLocal[2]) > 0.5) {
239 cerr << __FILE__ << ":" << __LINE__ << ": Coordinate (" << xLocal[0] << "," << xLocal[1] << "," << xLocal[2] << ") outside of this cell!" << endl;
240 abort();
241 }
242
243 std::array<int, 3> cellIds = {i,j,k};
244 std::array<Real, Rec::N_REC_COEFFICIENTS> rc;
245
246 if(FieldTracing::fieldTracingParameters.useCache) {
247 #pragma omp critical
248 {
249 if (reconstructionCoefficientsCache.find(cellIds) == reconstructionCoefficientsCache.end()) {
250 reconstructionCoefficients(
251 perBGrid,
252 dPerBGrid,
253 rc,
254 i,
255 j,
256 k,
257 3 // Reconstruction order of the fields after Balsara 2009, 2 used for general B, but 3 used here to allow for cache reuse, see interpolatePerturbedJ below
258 );
259 reconstructionCoefficientsCache.insert({cellIds, rc});
260 } else {
261 rc = reconstructionCoefficientsCache.at(cellIds);
262 }
263 }
264 } else {
265 reconstructionCoefficients(
266 perBGrid,
267 dPerBGrid,
268 rc,
269 i,
270 j,
271 k,
272 3 // // Reconstruction order of the fields after Balsara 2009, 3 used to obtain 2nd order curl(B) and allows for cache reuse, see interpolatePerturbedB above
273 );
274 }
275
276 std::array<Real, 3> interpolatedB;
277 // Eq. (7) Balsara 2009
278 interpolatedB[0] = rc[Rec::a_0] + rc[Rec::a_x]*xLocal[0] + rc[Rec::a_y]*xLocal[1] + rc[Rec::a_z]*xLocal[2]
279 + rc[Rec::a_xx] * (xLocal[0]*xLocal[0] - TWELWTH) + rc[Rec::a_xy]*xLocal[0]*xLocal[1] + rc[Rec::a_xz]*xLocal[0]*xLocal[2];
280 // Eq. (8) Balsara 2009
281 interpolatedB[1] = rc[Rec::b_0] + rc[Rec::b_x]*xLocal[0] + rc[Rec::b_y]*xLocal[1] + rc[Rec::b_z]*xLocal[2]
282 + rc[Rec::b_yy] * (xLocal[1]*xLocal[1] - TWELWTH) + rc[Rec::b_xy]*xLocal[0]*xLocal[1] + rc[Rec::b_yz]*xLocal[1]*xLocal[2];
283 // Eq. (9) Balsara 2009
284 interpolatedB[2] = rc[Rec::c_0] + rc[Rec::c_x]*xLocal[0] + rc[Rec::c_y]*xLocal[1] + rc[Rec::c_z]*xLocal[2]
285 + rc[Rec::c_zz] * (xLocal[2]*xLocal[2] - TWELWTH) + rc[Rec::c_xz]*xLocal[0]*xLocal[2] + rc[Rec::c_yz]*xLocal[1]*xLocal[2];
286 return interpolatedB;
287}
