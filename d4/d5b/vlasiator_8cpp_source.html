129void computeNewTimeStep(dccrg::Dccrg<SpatialCell,dccrg::Cartesian_Geometry>& mpiGrid,
130 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid, Real &newDt, bool &isChanged) {
131
132 phiprof::Timer computeTimestepTimer {"compute-timestep"};
133 // Compute maximum time step. This cannot be done at the first step as the solvers compute the limits for each cell.
134
135 isChanged = false;
136
137 const vector<CellID>& cells = getLocalCells();
138 /* Arrays for storing local (per process) and global max dt
139 0th position stores ordinary space propagation dt
140 1st position stores velocity space propagation dt
141 2nd position stores field propagation dt
142 */
143 Real dtMaxLocal[3];
144 Real dtMaxGlobal[3];
145
146 dtMaxLocal[0] = numeric_limits<Real>::max();
147 dtMaxLocal[1] = numeric_limits<Real>::max();
148 dtMaxLocal[2] = numeric_limits<Real>::max();
149
150 // Compute max dt for Vlasov solver
151 reduce_vlasov_dt(mpiGrid, cells, dtMaxLocal);
152
153 // compute max dt for fieldsolver
154 const std::array<FsGridTools::FsIndex_t, 3> gridDims(technicalGrid.getLocalSize());
155 for (FsGridTools::FsIndex_t k = 0; k < gridDims[2]; k++) {
156 for (FsGridTools::FsIndex_t j = 0; j < gridDims[1]; j++) {
157 for (FsGridTools::FsIndex_t i = 0; i < gridDims[0]; i++) {
158 fsgrids::technical* cell = technicalGrid.get(i, j, k);
159 if (cell->sysBoundaryFlag == sysboundarytype::NOT_SYSBOUNDARY ||
160 (cell->sysBoundaryLayer == 1 && cell->sysBoundaryFlag != sysboundarytype::NOT_SYSBOUNDARY)) {
161 dtMaxLocal[2] = min(dtMaxLocal[2], cell->maxFsDt);
162 }
163 }
164 }
165 }
166
167 MPI_Allreduce(&(dtMaxLocal[0]), &(dtMaxGlobal[0]), 3, MPI_Type<Real>(), MPI_MIN, MPI_COMM_WORLD);
168
169 // If any of the solvers are disabled there should be no limits in timespace from it
170 if (!P::propagateVlasovTranslation)
171 dtMaxGlobal[0] = numeric_limits<Real>::max();
172 if (!P::propagateVlasovAcceleration)
173 dtMaxGlobal[1] = numeric_limits<Real>::max();
174 if (!P::propagateField)
175 dtMaxGlobal[2] = numeric_limits<Real>::max();
176
177 creal meanVlasovCFL = 0.5 * (P::vlasovSolverMaxCFL + P::vlasovSolverMinCFL);
178 creal meanFieldsCFL = 0.5 * (P::fieldSolverMaxCFL + P::fieldSolverMinCFL);
179 Real subcycleDt;
180
181 // reduce/increase dt if it is too high for any of the three propagators or too low for all propagators
182 if ((P::dt > dtMaxGlobal[0] * P::vlasovSolverMaxCFL ||
183 P::dt > dtMaxGlobal[1] * P::vlasovSolverMaxCFL * P::maxSlAccelerationSubcycles ||
184 P::dt > dtMaxGlobal[2] * P::fieldSolverMaxCFL * P::maxFieldSolverSubcycles) ||
185 (P::dt < dtMaxGlobal[0] * P::vlasovSolverMinCFL &&
186 P::dt < dtMaxGlobal[1] * P::vlasovSolverMinCFL * P::maxSlAccelerationSubcycles &&
187 P::dt < dtMaxGlobal[2] * P::fieldSolverMinCFL * P::maxFieldSolverSubcycles)) {
188
189 // new dt computed
190 isChanged = true;
191
192 // set new timestep to the lowest one of all interval-midpoints
193 newDt = meanVlasovCFL * dtMaxGlobal[0];
194 newDt = min(newDt, meanVlasovCFL * dtMaxGlobal[1] * P::maxSlAccelerationSubcycles);
195 newDt = min(newDt, meanFieldsCFL * dtMaxGlobal[2] * P::maxFieldSolverSubcycles);
196
197 logFile << "(TIMESTEP) New dt = " << newDt << " computed on step " << P::tstep << " at " << P::t
198 << "s Maximum possible dt (not including vlasovsolver CFL " << P::vlasovSolverMinCFL << "-"
199 << P::vlasovSolverMaxCFL << " or fieldsolver CFL " << P::fieldSolverMinCFL << "-" << P::fieldSolverMaxCFL
200 << ") in {r, v, BE} was " << dtMaxGlobal[0] << " " << dtMaxGlobal[1] << " " << dtMaxGlobal[2] << " "
201 << " Including subcycling { v, BE} was " << dtMaxGlobal[1] * P::maxSlAccelerationSubcycles << " "
202 << dtMaxGlobal[2] * P::maxFieldSolverSubcycles << " " << endl
203 << writeVerbose;
204
205 if (P::dynamicTimestep) {
206 // Check if the calculated value was and continues to be above the ceiling
207 if (P::dt_ceil > 0.0 && newDt >= P::dt_ceil && P::dt == P::dt_ceil) {
208 isChanged = false;
209 newDt = P::dt_ceil;
210 return;
211 }
212 // Check if we at this time exceeded the ceiling
213 if (P::dt_ceil > 0.0 && newDt > P::dt_ceil) {
214 newDt = P::dt_ceil;
215 logFile << "(TIMESTEP) However, ceiling timestep in config overrides larger dynamic and dt = " << P::dt_ceil << endl << writeVerbose;
216 }
217 subcycleDt = newDt;
218 } else {
219 logFile << "(TIMESTEP) However, fixed timestep in config overrides and dt = " << P::dt << endl << writeVerbose;
220 subcycleDt = P::dt;
221 }
222 } else {
223 subcycleDt = P::dt;
224 }
225
226 // Subcycle if field solver dt < global dt (including CFL) (new or old dt hence the hassle with subcycleDt
227 if (meanFieldsCFL * dtMaxGlobal[2] < subcycleDt && P::propagateField) {
228 P::fieldSolverSubcycles =
229 min(convert<uint>(ceil(subcycleDt / (meanFieldsCFL * dtMaxGlobal[2]))), P::maxFieldSolverSubcycles);
230 } else {
231 P::fieldSolverSubcycles = 1;
232 }
233}
