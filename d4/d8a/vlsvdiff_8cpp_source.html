82static uint64_t convUInt(const char* ptr, const vlsv::datatype::type& dataType, const uint64_t& dataSize) {
83 if (dataType != vlsv::datatype::type::UINT) {
84 cerr << "Erroneous datatype given to convUInt" << endl;
85 exit(1);
86 }
87
88 switch (dataSize) {
89 case 1:
90 return *reinterpret_cast<const unsigned char*> (ptr);
91 break;
92 case 2:
93 return *reinterpret_cast<const unsigned short int*> (ptr);
94 break;
95 case 4:
96 return *reinterpret_cast<const unsigned int*> (ptr);
97 break;
98 case 8:
99 return *reinterpret_cast<const unsigned long int*> (ptr);
100 break;
101 }
102 return 0;
103}

112bool copyArray(vlsv::Reader& input,vlsv::Writer& output,
113 const std::string& tagName,
114 const list<pair<string,string> >& inputAttribs,
115 bool optional=false) {
116 bool success = true;
117
118 // Read input array attributes
119 map<string,string> outputAttribs;
120 if (input.getArrayAttributes(tagName,inputAttribs,outputAttribs) == false) {
121
122 if(!optional) {
123 cerr << "ERROR: Failed to read array '" << tagName << "' attributes in " << __FILE__ << ":" << __LINE__ << endl;
124 cerr << "Input attributes are:" << endl;
125 for (list<pair<string,string> >::const_iterator it=inputAttribs.begin(); it!=inputAttribs.end(); ++it) {
126 cerr << "\t '" << it->first << "' = '" << it->second << "'" << endl;
127 }
128 return false;
129 } else {
130 // This was an optional parameter, so whatever.
131 return true;
132 }
133 }
134
135 // Figure out arraysize, vectorsize, datasize, and datatype of the copied array
136 map<string,string>::const_iterator it;
137 map<string,string>::iterator it2;
138 it = outputAttribs.find("arraysize"); if (it == outputAttribs.end()) return false;
139 uint64_t arraysize = atol(it->second.c_str());
140 it = outputAttribs.find("vectorsize"); if (it == outputAttribs.end()) return false;
141 const uint64_t vectorsize = atol(it->second.c_str());
142 it = outputAttribs.find("datasize"); if (it == outputAttribs.end()) return false;
143 const uint64_t datasize = atol(it->second.c_str());
144 it = outputAttribs.find("datatype"); if (it == outputAttribs.end()) return false;
145 const string datatype = it->second;
146
147 const size_t bytes = arraysize*vectorsize*datasize;
148
149 // Read values from input file
150 char* ptr = new char[bytes];
151 if (input.readArray(tagName,inputAttribs,0,arraysize,ptr) == false) {
152 cerr << "ERROR: Failed to clone array '" << tagName << "' in " << __FILE__ << ":" << __LINE__ << endl;
153 delete [] ptr; return false;
154 }
155
156
157 // Write array to output file
158 if (output.writeArray(tagName,outputAttribs,datatype,arraysize,vectorsize,datasize,ptr) == false) {
159 cerr << "ERROR: Failed to write array '" << tagName << "' in " << __FILE__ << ":" << __LINE__ << endl;
160 success = false;
161 }
162
163 delete [] ptr; ptr = NULL;
164 return success;
165}

169bool HandleFsGrid(const string& inputFileName,

172{
173
174
175 //Open input file
176 vlsv::Reader input;

183 std::string tagName="MESH";
184 list<pair<string,string> > inputAttribs;
185 inputAttribs.push_back(make_pair("name","fsgrid"));
186 map<string,string> outputAttribs;
187
188 if (input.getArrayAttributes(tagName,inputAttribs,outputAttribs) == false) {
189 cerr << "ERROR: Failed to read array '" << tagName << "' attributes in " << __FILE__ << ":" << __LINE__ << endl;
190 cerr << "Input attributes are:" << endl;
191 for (list<pair<string,string> >::const_iterator it=inputAttribs.begin(); it!=inputAttribs.end(); ++it) {
192 cerr << "\t '" << it->first << "' = '" << it->second << "'" << endl;
193 }
194 return false;
195 }
196
197 //Collect needed attributes to a map named patch
198 map<string,string>::const_iterator it;
199 it = outputAttribs.find("arraysize"); if (it == outputAttribs.end()) return false;
200 uint64_t arraysize = atol(it->second.c_str());
201 it = outputAttribs.find("vectorsize"); if (it == outputAttribs.end()) return false;
202 const uint64_t vectorsize = atol(it->second.c_str());
203 it = outputAttribs.find("datasize"); if (it == outputAttribs.end()) return false;
204 const uint64_t datasize = atol(it->second.c_str());
205 it = outputAttribs.find("datatype"); if (it == outputAttribs.end()) return false;
206 const string datatype = it->second;
207 it = outputAttribs.find("xperiodic"); if (it == outputAttribs.end()) return false;
208 const string xperiodic = it->second;
209 it = outputAttribs.find("yperiodic"); if (it == outputAttribs.end()) return false;
210 const string yperiodic = it->second;
211 it = outputAttribs.find("zperiodic"); if (it == outputAttribs.end()) return false;
212 const string zperiodic = it->second;
213 it = outputAttribs.find("type"); if (it == outputAttribs.end()) return false;
214 const string type = it->second;
215
216 map<string,string>patch;
217 patch["arraysize"]=std::to_string(arraysize);
218 patch["datasize"]=std::to_string(datasize);
219 patch["datatype"]=datatype;
220 patch["name"]="fsgrid";
221 patch["type"]=type;
222 patch["vectorsize"]=std::to_string(vectorsize);
223 patch["xperiodic"]=xperiodic;
224 patch["yperiodic"]=yperiodic;
225 patch["zperiodic"]=zperiodic;
226
227
228 //Get the global IDs in a vector
229 std::vector<uint64_t> globalIds;
230 for (const auto iter : orderedData){
231 globalIds.push_back( iter.first );
232 }
233
234 //Write to file
235 output.writeArray("MESH",patch,arraysize,1,&globalIds[0]);
236
237 std::array<int,1> numWritingRanks = {1};
238 output.writeParameter("numWritingRanks", &numWritingRanks[0]);
239
240 // Save the FSgrid decomposition
241 std::map<std::string, std::string> xmlAttributes;
242 const std::string meshName="fsgrid";
243 xmlAttributes["mesh"] = meshName;
244 std::array<FsGridTools::Task_t, 3> decom = {1,1,1};
245 output.writeArray("MESH_DECOMPOSITION", outputAttribs, 3u, 1u, &decom[0]);
246
247 //Now for MESH_DOMAIN_SIZES
248 inputAttribs.clear();
249 inputAttribs.push_back(make_pair("mesh","fsgrid"));
250 tagName="MESH_DOMAIN_SIZES";
251
252 if (input.getArrayAttributes(tagName,inputAttribs,outputAttribs) == false) {
253 cerr << "ERROR: Failed to read array '" << tagName << "' attributes in " << __FILE__ << ":" << __LINE__ << endl;
254 cerr << "Input attributes are:" << endl;
255 for (list<pair<string,string> >::const_iterator it=inputAttribs.begin(); it!=inputAttribs.end(); ++it) {
256 cerr << "\t '" << it->first << "' = '" << it->second << "'" << endl;
257 }
258 return false;
259 }
260
261
262 //Read some attributes we need and parse to our map
263 it = outputAttribs.find("datasize"); if (it == outputAttribs.end()) return false;
264 const uint64_t datasize2 = atol(it->second.c_str());
265 it = outputAttribs.find("datatype"); if (it == outputAttribs.end()) return false;
266 const string datatype2 = it->second;
267 it = outputAttribs.find("vectorsize"); if (it == outputAttribs.end()) return false;
268 const uint64_t vectorsize2 = atol(it->second.c_str());
269
270 patch.clear();
271 patch["arraysize"]="1";
272 patch["datasize"]=to_string(datasize2);
273 patch["datatype"]=datatype2;
274 patch["mesh"]="fsgrid";
275 patch["vectorsize"]=to_string(vectorsize2);
276
277 //Override MESH_DOMAIN_SIZES
278 std::array<uint64_t,2> meshDomainSize({globalIds.size(), 0});
279 output.writeArray("MESH_DOMAIN_SIZES",patch ,1,vectorsize2, &meshDomainSize[0]);
280
281
282 //Close the file
283 input.close();
284
285
286 return true;

292 uint64_t arraySize;
293 uint64_t vectorSize;
294 vlsv::datatype::type dataType;
295 uint64_t byteSize;
296
297 list<pair<string,string> > attribs;
298 attribs.push_back(make_pair("mesh","fsgrid"));
299
300
301 std::array<FsGridTools::Task_t,3> fsGridDecomposition={0,0,0};
302 int* ptr = &fsGridDecomposition[0];
303
304 // Check if array exists:
305 bool success = file.getArrayInfo("MESH_DECOMPOSITION",attribs,arraySize,vectorSize,dataType,byteSize);
306 if (success == false) {
307 // std::cout << "Could not read MESH_DECOMPOSITION" << endl;

310

312 if(file.readParameter("numWritingRanks",fsgridInputRanks) == false) {
313 std::cerr << "FSGrid writing rank number not found in restart file" << endl;
314 exit(1);
315 }
316 std::array<FsGridTools::FsSize_t,3> gridSize;

326 std::cerr << "Could not read MESH_DOMAIN_SIZES from file" << endl;
327 exit(1);
328 }

330 for (int i = 0; i < 2*fsgridInputRanks; i+=2){
331 mesh_domain_sizes.push_back(domainInfo[i]);
332 }
333 list<pair<string,string> > mesh_attribs;
334 mesh_attribs.push_back(make_pair("name","fsgrid"));
335 std::vector<FsGridTools::FsSize_t> rank_first_ids(fsgridInputRanks);
336 FsGridTools::FsSize_t* ids_ptr = &rank_first_ids[0];
337
338 std::set<FsGridTools::FsIndex_t> x_corners, y_corners, z_corners;
339
340 int64_t begin_rank = 0;
341 int i = 0;
342 for(auto rank_size : mesh_domain_sizes){
343 if(file.read("MESH", mesh_attribs, begin_rank, 1, ids_ptr, false) == false){
344 std::cerr << "Reading MESH failed.\n";
345 exit(1);
346 }
347 std::array<FsGridTools::FsIndex_t,3> inds = FsGridTools::globalIDtoCellCoord(*ids_ptr, gridSize);
348 x_corners.insert(inds[0]);
349 y_corners.insert(inds[1]);
350 z_corners.insert(inds[2]);
351 ++ids_ptr;
352 begin_rank += rank_size;
353 }
354
355 decomposition[0] = x_corners.size();
356 decomposition[1] = y_corners.size();
357 decomposition[2] = z_corners.size();
358 std::cout << "Fsgrid decomposition computed from MESH to be " << decomposition[0] << " " << decomposition[1] << " " <<decomposition[2] << endl;
359
360 return true;
361 } else {
362 // data exists, now read it
363 success = file.read("MESH_DECOMPOSITION",attribs, 0, 3, ptr, false);
364 decomposition[0] = fsGridDecomposition[0];
365 decomposition[1] = fsGridDecomposition[1];
366 decomposition[2] = fsGridDecomposition[2];
367 std::cout << "Fsgrid decomposition read as " << decomposition[0] << " " << decomposition[1] << " " <<decomposition[2] << endl;
368 return true;
369 }

371 return false;
372}

379bool cloneMesh(const string& inputFileName,vlsv::Writer& output,const string& meshName, std::map<uint, Real> orderedData) {
380 bool success = true;
381
382 vlsv::Reader input;
383 if (input.open(inputFileName) == false) {
384 cerr << "ERROR failed to open input file '" << inputFileName << "' in " << __FILE__ << ":" << __LINE__ << endl;
385 return false;
386 }
387
388 list<pair<string,string> > inputAttribs;
389 inputAttribs.push_back(make_pair("name",meshName));
390 inputAttribs.clear();
391 inputAttribs.push_back(make_pair("mesh",meshName));
392 if (copyArray(input,output,"MESH_BBOX",inputAttribs) == false) success = false;
393
394 // Mesh have either individual coordinate arrays (for cartesian geometries)...
395 if (copyArray(input,output,"MESH_NODE_CRDS_X",inputAttribs, meshName == "ionosphere") == false) success = false;
396 if (copyArray(input,output,"MESH_NODE_CRDS_Y",inputAttribs, meshName == "ionosphere") == false) success = false;
397 if (copyArray(input,output,"MESH_NODE_CRDS_Z",inputAttribs, meshName == "ionosphere") == false) success = false;
398
399 // Or they have per-node coordinate arrays (for unstructured meshes)
400 if (copyArray(input,output,"MESH_NODE_CRDS",inputAttribs, meshName != "ionosphere") == false) success = false;
401 if (copyArray(input,output,"MESH_OFFSETS",inputAttribs, meshName != "ionosphere") == false) success = false;
402
403 if (copyArray(input,output,"MESH_GHOST_LOCALIDS",inputAttribs, meshName == "ionosphere") == false) success = false;
404 if (copyArray(input,output,"MESH_GHOST_DOMAINS",inputAttribs, meshName == "ionosphere") == false) success = false;
405
406 //Only do this if we diff SpatialGrid data
407 if (gridName==gridType::SpatialGrid || gridName==gridType::ionosphere){
408 if (copyArray(input,output,"MESH_DOMAIN_SIZES",inputAttribs) == false) success = false;
409
410 inputAttribs.clear();
411 inputAttribs.push_back(make_pair("name",meshName));
412 if (copyArray(input,output,"MESH",inputAttribs) == false) success = false;
413 }else{
414 HandleFsGrid(inputFileName,output,orderedData);
415 }
416
417 input.close();
418 return success;
419}

428bool convertMesh(vlsvinterface::Reader& vlsvReader,
429 const string& meshName,
430 const char * varToExtract,
431 const uint compToExtract,
432 map<uint, Real> * orderedData,
433 unordered_map<size_t,size_t>& cellOrder,
434 const bool& storeCellOrder) {
435
436 //Check for null pointer:
437 if( !varToExtract || !orderedData ) {
438 cerr << "ERROR, PASSED A NULL POINTER AT " << __FILE__ << " " << __LINE__ << endl;
439 return false;
440 }
441 bool meshSuccess = true;
442 bool variableSuccess = true;
443
444 datatype::type meshDataType;
445 datatype::type variableDataType;
446 uint64_t meshArraySize, meshVectorSize, meshDataSize;
447 uint64_t variableArraySize, variableVectorSize, variableDataSize;
448
449 list<pair<string, string> > variableAttributes;
450 const string _varToExtract( varToExtract );
451 variableAttributes.push_back( make_pair("mesh", meshName) );
452 variableAttributes.push_back( make_pair("name", _varToExtract) );
453 //Read in array size, vector size, data type and data size of the array "VARIABLE" in the vlsv file (Needed in reading the array)
454 if (vlsvReader.getArrayInfo("VARIABLE", variableAttributes, variableArraySize, variableVectorSize, variableDataType, variableDataSize) == false) {
455 cerr << "ERROR, failed to get array info for '" << _varToExtract << "' on mesh '" << meshName << "' at " << __FILE__ << " " << __LINE__ << endl;
456 return false;
457 }
458
459 switch(gridName) {
460 case gridType::SpatialGrid:
461 {
462 std::vector<char> variableBuffer(variableVectorSize * variableDataSize);
463 float *variablePtrFloat = reinterpret_cast<float *>(variableBuffer.data());
464 double *variablePtrDouble = reinterpret_cast<double *>(variableBuffer.data());
465 uint *variablePtrUint = reinterpret_cast<uint *>(variableBuffer.data());
466 int *variablePtrInt = reinterpret_cast<int *>(variableBuffer.data());
467
468 // Read the mesh array one node (of a spatial cell) at a time
469 // and create a map which contains each cell's CellID and variable to be extracted
470 //Get local cell ids:
471 vector<uint64_t> local_cells;
472 if ( vlsvReader.getCellIds( local_cells, meshName) == false ) {
473 cerr << "Failed to read cell ids at " << __FILE__ << " " << __LINE__ << endl;
474 return false;
475 }
476
477 //Check for correct output:
478 if (local_cells.size() != variableArraySize) {
479 cerr << "ERROR array size mismatch: " << local_cells.size() << " " << variableArraySize << endl;
480 }
481 if (compToExtract + 1 > variableVectorSize) {
482 cerr << "ERROR invalid component, this variable has size " << variableVectorSize << endl;
483 abort();
484 }
485
486 if (storeCellOrder == true) {
487 cellOrder.clear();
488 }
489
490 orderedData->clear();
491
492 for (uint64_t i=0; i<local_cells.size(); ++i) {
493 const short int amountToReadIn = 1;
494 const uint64_t & startingReadIndex = i;
495 if (vlsvReader.readArray("VARIABLE", variableAttributes, startingReadIndex, amountToReadIn, variableBuffer.data()) == false) {
496 cerr << "ERROR, failed to read variable '" << _varToExtract << "' at " << __FILE__ << " " << __LINE__ << endl;
497 variableSuccess = false;
498 break;
499 }
500 // Get the CellID
501 uint64_t & CellID = local_cells[i];
502
503 // Get the variable value
504 Real extract = NAN;
505
506 switch (variableDataType) {
507 case datatype::type::FLOAT:
508 if(variableDataSize == sizeof(float)) extract = (Real)(variablePtrFloat[compToExtract]);
509 if(variableDataSize == sizeof(double)) extract = (Real)(variablePtrDouble[compToExtract]);
510 break;
511 case datatype::type::UINT:
512 extract = (Real)(variablePtrUint[compToExtract]);
513 break;
514 case datatype::type::INT:
515 extract = (Real)(variablePtrInt[compToExtract]);
516 break;
517 case datatype::type::UNKNOWN:
518 cerr << "ERROR, BAD DATATYPE AT " << __FILE__ << " " << __LINE__ << endl;
519 break;
520 }
521 // Put those into the map
522 orderedData->insert(pair<uint64_t, Real>(CellID, extract));
523 if (storeCellOrder == true) {
524 cellOrder[CellID] = i;
525 }
526 }
527 }
528 break;
529
530 case gridType::fsgrid:
531
532 {
533 // Get Spatial Grid's max refinement Level
534 int maxRefLevel=0;
535 list<pair<string, string>> meshAttributesIn;
536 meshAttributesIn.push_back(make_pair("name", "SpatialGrid"));
537 map<string,string> meshAttributesOut;
538 if (vlsvReader.getArrayAttributes("MESH", meshAttributesIn,meshAttributesOut) == false)
539 {
540 cerr << "ERROR, failed to get array info for '" << _varToExtract << "' at " << __FILE__ << " " << __LINE__ << endl;
541 return false;
542 }
543
544 std::map<string, string>::iterator attributesOutIt;
545 attributesOutIt = meshAttributesOut.find("max_refinement_level");
546 if (attributesOutIt != meshAttributesOut.end())
547 {
548 maxRefLevel = stoi(attributesOutIt->second);
549 }
550 int numtasks;
551 int xcells,ycells,zcells;
552 vlsvReader.readParameter("numWritingRanks",numtasks);
553 vlsvReader.readParameter("xcells_ini",xcells);
554 vlsvReader.readParameter("ycells_ini",ycells);
555 vlsvReader.readParameter("zcells_ini",zcells);
556 xcells*=pow(2,maxRefLevel);
557 ycells*=pow(2,maxRefLevel);
558 zcells*=pow(2,maxRefLevel);
559 std::array<int,3> GlobalBox={xcells,ycells,zcells};
560 std::array<int,3> thisDomainDecomp;
561
562 //Compute Domain Decomposition Scheme for this vlsv file
563 //FsGridTools::computeDomainDecomposition(GlobalBox,numtasks,thisDomainDecomp);
564 getFsgridDecomposition(vlsvReader, thisDomainDecomp);
565
566
567 std::array<int32_t,3> taskSize,taskStart;
568 std::array<int32_t,3> taskEnd;
569 int readOffset=0;
570 int index,my_x,my_y,my_z;
571 orderedData->clear();
572
573 for (int task=0; task<numtasks; task++){
574
575 my_x=task/thisDomainDecomp[2]/thisDomainDecomp[1];
576 my_y=(task/thisDomainDecomp[2])%thisDomainDecomp[1];
577 my_z=task%thisDomainDecomp[2];
578
579
580 taskStart[0] = FsGridTools::calcLocalStart(GlobalBox[0], thisDomainDecomp[0], my_x);
581 taskStart[1] = FsGridTools::calcLocalStart(GlobalBox[1], thisDomainDecomp[1], my_y);
582 taskStart[2] = FsGridTools::calcLocalStart(GlobalBox[2], thisDomainDecomp[2], my_z);
583
584 taskSize[0] = FsGridTools::calcLocalSize(GlobalBox[0], thisDomainDecomp[0], my_x);
585 taskSize[1] = FsGridTools::calcLocalSize(GlobalBox[1], thisDomainDecomp[1], my_y);
586 taskSize[2] = FsGridTools::calcLocalSize(GlobalBox[2], thisDomainDecomp[2], my_z);
587
588 taskEnd[0]= taskStart[0]+taskSize[0];
589 taskEnd[1]= taskStart[1]+taskSize[1];
590 taskEnd[2]= taskStart[2]+taskSize[2];
591
592 int64_t readSize= taskSize[0] * taskSize[1] * taskSize[2] ;
593 //Allocate vector for reading
594 std::vector<Real> buffer(readSize*variableVectorSize);
595
596 if ( variableDataSize==sizeof(Real)){
597 if (vlsvReader.readArray("VARIABLE", variableAttributes, readOffset, readSize, (char*)buffer.data()) == false) {
598 cerr << "ERROR, failed to read variable '" << _varToExtract << "' at " << __FILE__ << " " << __LINE__ << endl;
599 variableSuccess = false;
600 break;
601 }
602 }else{
603 std::vector<float> tmpbuffer(readSize * variableVectorSize);
604 if (vlsvReader.readArray("VARIABLE", variableAttributes, readOffset, readSize, (char *)tmpbuffer.data()) == false){
605 cerr << "ERROR, failed to read variable '" << _varToExtract << "' at " << __FILE__ << " " << __LINE__ << endl;
606 variableSuccess = false;
607 break;
608 }
609 for (unsigned int i = 0; i < readSize * variableVectorSize; i++){
610 buffer[i] = tmpbuffer[i];
611 }
612 }
613
614 uint64_t globalindex,counter=0;;
615 for (int z=taskStart[2]; z<taskEnd[2]; z++){
616 for (int y=taskStart[1]; y< taskEnd[1]; y++){
617 for (int x=taskStart[0]; x<taskEnd[0]; x++){
618 globalindex= x + y*xcells + z*xcells*ycells;
619 Real data;
620 switch (variableDataType){
621 case datatype::type::FLOAT:
622 if (variableDataSize == sizeof(float))
623 memcpy(&data, &buffer[counter + compToExtract], sizeof(float));

626 break;
627 case datatype::type::UINT:

645 }

658 case datatype::type::FLOAT:

669 for(unsigned int i=0; i<variableArraySize; i++) {
670 orderedData->insert(pair<uint64_t, Real>(i, buffer[i*variableVectorSize + compToExtract]));

672 } else if(variableDataSize == sizeof(float)) {
673 std::vector<double> buffer(variableVectorSize * variableArraySize);
674 // The mesh is simply one big blob that can be read in one go.
675 if(vlsvReader.readArray("VARIABLE", variableAttributes, 0, variableArraySize, (char*)buffer.data()) == false) {
676 cerr << "ERROR, failed to read variable '" << _varToExtract << "' at " << __FILE__ << " " << __LINE__ << endl;
677 variableSuccess = false;
678 break;
679 }
680
681 for(unsigned int i=0; i<variableArraySize; i++) {
682 orderedData->insert(pair<uint64_t, Real>(i, buffer[i*variableVectorSize + compToExtract]));
683 }
684 }
685 }
686 break;
687 default:
688 cerr << "Error: No support for ionosphere parameters that are not float-valued implemented, at " << __FILE__ << " " << __LINE__ << endl;
689 break;
690 }

703 }

834 data2 = &shiftedData2;
835 }
836
837 // Reset old values
838 *absolute = 0.0;
839 *relative = 0.0;
840
841 vector<Real> array(orderedData1.size());
842 for (size_t i=0; i<array.size(); ++i) array[i] = -1.0;

866 value = abs(it1->second - it2->second);
867 *absolute += value;
868 length += abs(it1->second);
869
870 }
871 if (gridName==gridType::SpatialGrid){
872 array[cellOrder.at(it1->first)] = value;
873 }else if (gridName==gridType::fsgrid || gridName==gridType::ionosphere) {
874 array[it1->first]=value;

891 }
892 }

894 length = pow(length, 1.0 / p);
895 }
896
897 if (length != 0.0) *relative = *absolute / length;
898 else {
899 cout << "WARNING (pDistance) : length of reference is 0.0, cannot divide to give relative distance." << endl;
900 *relative = -1;
901 }
902
903 // Write out the difference (if requested):
904 if (attributes.find("--diff") != attributes.end()) {
905 map<string,string> attributes;

907 attributes["name"] = varName;

909 attributes["centering"] = "node";
910 }
911
912

914 cerr << "ERROR failed to write variable '" << varName << "' to output file in " << __FILE__ << ":" << __LINE__ << endl;
915 return 1;
916 }
917 }
918
919 return 0;
920}

1381template <class T, class U>
1382bool compareAvgs( const string fileName1,
1383 const string fileName2,
1384 const bool verboseOutput,
1385 vector<uint64_t> & cellIds1,
1386 vector<uint64_t> & cellIds2
1387 ) {
1388 if( cellIds1.empty() == true || cellIds2.empty() == true ) {
1389 cerr << "ERROR, CELL IDS EMPTY IN COMPARE AVGS" << endl;
1390 return false;
1391 }
1392 // Declare map for locating velocity spaces within cell ids
1393 // Note: Key = cell id, value->first = blockOffset, value->second = numberOfBlocksToRead
1394 unordered_map<uint64_t, pair<uint64_t, uint32_t>> cellsWithBlocksLocations1;
1395 unordered_map<uint64_t, pair<uint64_t, uint32_t>> cellsWithBlocksLocations2;
1396 // Open the files for reading:
1397 T vlsvReader1;
1398 if( vlsvReader1.open(fileName1) == false ) {
1399 cerr << "Error opening file name " << fileName1 << " at " << __FILE__ << " " << __LINE__ << endl;
1400 return false;
1401 }
1402
1403 U vlsvReader2;
1404 if( vlsvReader2.open(fileName2) == false ) {
1405 cerr << "Error opening file name " << fileName2 << " at " << __FILE__ << " " << __LINE__ << endl;
1406 return false;
1407 }
1408
1409 if( getCellsWithBlocksLocations( vlsvReader1, cellsWithBlocksLocations1 ) == false ) {
1410 cerr << "ERROR AT " << __FILE__ << " " << __LINE__ << endl;
1411 return false;
1412 }
1413
1414 if( getCellsWithBlocksLocations( vlsvReader2, cellsWithBlocksLocations2 ) == false ) {
1415 cerr << "ERROR AT " << __FILE__ << " " << __LINE__ << endl;
1416 return false;
1417 }
1418 // Consistency check:
1419 if( cellsWithBlocksLocations2.size() != cellsWithBlocksLocations1.size() ) {
1420 cerr << "BAD CELLS WITH BLOCKS SIZE AT " << __FILE__ << " " << __LINE__ << endl;
1421 return false;
1422 }
1423
1424 // Create a few variables for the cell id loop:
1425 vector< double > avgsDiffs;
1426 double totalAbsAvgs = 0;
1427 double totalAbsDiff = 0;
1428 double totalAbsLog10Diff = 0;
1429 double threshold=1e-16;
1430 uint64_t numOfRelevantCells = 0;
1431 uint64_t numOfIdenticalBlocks = 0;
1432 uint64_t numOfNonIdenticalBlocks = 0;
1433 if( cellIds1[0] == 0 || cellIds2[0] == 0 ) {
1434 // User input 0 as the cell id -- compare all cell ids
1435 cellIds1.clear();
1436 cellIds2.clear();
1437 for( unordered_map<uint64_t, pair<uint64_t, uint32_t>>::const_iterator it = cellsWithBlocksLocations1.begin(); it != cellsWithBlocksLocations1.end(); ++it ) {
1438 cellIds1.push_back(it->first);
1439 cellIds2.push_back(it->first);
1440 }
1441 }
1442
1443 if( cellIds1.size() != cellIds2.size() ) {
1444 cerr << "ERROR, BAD CELL ID SIZES AT " << __FILE__ << " " << __LINE__ << endl;
1445 return false;
1446 }
1447 // Go through cell ids:
1448 for( uint cellIndex = 0; cellIndex < cellIds2.size(); cellIndex++ ) {
1449 const uint64_t & cellId1 = cellIds1[cellIndex];
1450 const uint64_t & cellId2 = cellIds2[cellIndex];
1451 // Get the avgs in a hash map (The velocity block id is the key and avgs is the value):
1452 uint64_t vectorSize1 = 0, vectorSize2;
1453 unordered_map<uint32_t, vector<double> > avgs1;
1454 unordered_map<uint32_t, vector<double> > avgs2;
1455 // Store the avgs in avgs1 and 2:
1456 if( readAvgs( vlsvReader1, "proton", cellsWithBlocksLocations1, cellId1, avgs1, vectorSize1 ) == false ) {
1457 if( readAvgs( vlsvReader1, "avgs", cellsWithBlocksLocations1, cellId1, avgs1, vectorSize1 ) == false ) {
1458 cerr << "ERROR, FAILED TO READ AVGS AT " << __FILE__ << " " << __LINE__ << endl;
1459 return false;
1460 }
1461 }
1462
1463 if( readAvgs( vlsvReader2, "proton", cellsWithBlocksLocations2, cellId2, avgs2, vectorSize2 ) == false ) {
1464 if( readAvgs( vlsvReader2, "avgs", cellsWithBlocksLocations2, cellId2, avgs2, vectorSize2 ) == false ) {
1465 cerr << "ERROR, FAILED TO READ AVGS AT " << __FILE__ << " " << __LINE__ << endl;
1466 return false;
1467 }
1468 }
1469
1470 if (vectorSize1 != vectorSize2) {
1471 cerr << "ERROR, VECTORSIZES DON'T MATCH " << vectorSize1 << " VS " << vectorSize2 << " AT " << __FILE__ << " " << __LINE__ << endl;
1472 return false;
1473 }
1474 const uint64_t velocityCellsPerBlock = vectorSize1;
1475
1476 //Compare the avgs values:
1477 // First make a check on how many of the block ids are identical:
1478 const size_t sizeOfAvgs1 = avgs1.size();
1479 const size_t sizeOfAvgs2 = avgs2.size();
1480 // Vector of block ids that are the same
1481 vector<uint32_t> blockIds1;
1482 vector<uint32_t> blockIds2;
1483 blockIds1.reserve(sizeOfAvgs1);
1484 blockIds2.reserve(sizeOfAvgs2);
1485 // Input block ids:
1486 for( unordered_map<uint32_t, vector<double> >::const_iterator it = avgs1.begin(); it != avgs1.end(); ++it ) {
1487 blockIds1.push_back(it->first);
1488 }
1489 for( unordered_map<uint32_t, vector<double> >::const_iterator it = avgs2.begin(); it != avgs2.end(); ++it ) {
1490 blockIds2.push_back(it->first);
1491 }
1492 // Compare block ids:
1493 // Sort
1494 sort( blockIds1.begin(), blockIds1.end() );
1495 sort( blockIds2.begin(), blockIds2.end() );
1496 // Create iterators
1497 vector<uint32_t>::const_iterator it1 = blockIds1.begin();
1498 vector<uint32_t>::const_iterator it2 = blockIds2.begin();
1499 // Separate block ids into two categories -- the ones that blockids1 and blockids2 share and ones that only one of them shares
1500 vector<uint32_t> identicalBlockIds;
1501 vector<uint32_t> nonIdenticalBlockIds;
1502
1503 while( true ) {
1504 if( it1 == blockIds1.end() || it2 == blockIds2.end() ) {
1505 // Reach end of block ids
1506 break;
1507 }
1508 if( *it1 == *it2 ) {
1509 // Identical block id
1510 identicalBlockIds.push_back(*it1);
1511 it1++; it2++;
1512 } else if( *it1 < *it2 ) {
1513 // Non identical block id
1514 // The block ids are sorted so to get identical block ids one must increment the lower value
1515 nonIdenticalBlockIds.push_back(*it1);
1516 it1++;
1517 } else if( *it2 < *it1 ) {
1518 // Non identical block id
1519 // The block ids are sorted so to get identical block ids one must increment the lower value
1520 nonIdenticalBlockIds.push_back(*it2);
1521 it2++;
1522 }
1523 }
1524 // Get the rest of the non identical block ids (If there are any)
1525 // Note: This is only needed if for example it1 hit the end of the iteration and it2 still isn't at the end
1526 for( ; it1 != blockIds1.end(); ++it1 ) {
1527 nonIdenticalBlockIds.push_back(*it1);
1528 }
1529 for( ; it2 != blockIds2.end(); ++it2 ) {
1530 nonIdenticalBlockIds.push_back(*it2);
1531 }
1532 // Compare block ids:
1533 const uint64_t totalNumberOfBlocks = identicalBlockIds.size() + nonIdenticalBlockIds.size();
1534 const double percentageOfIdenticalBlocks = (double)(totalNumberOfBlocks) / (double)(identicalBlockIds.size());
1535 // Compare the avgs values of the identical blocks:
1536 avgsDiffs.reserve(avgsDiffs.size() + identicalBlockIds.size() * velocityCellsPerBlock);
1537 for( vector<uint32_t>::const_iterator it = identicalBlockIds.begin(); it != identicalBlockIds.end(); ++it ) {
1538 // Get the block id
1539 const uint32_t blockId = *it;
1540 // Get avgs values:
1541 const vector<double> & avgsValues1 = avgs1.at(blockId);
1542 const vector<double> & avgsValues2 = avgs2.at(blockId);
1543 // Get the diff:
1544 for( uint i = 0; i < velocityCellsPerBlock; ++i ) {
1545 double val1=avgsValues1[i]>threshold?avgsValues1[i]:threshold;
1546 double val2=avgsValues2[i]>threshold?avgsValues2[i]:threshold;
1547 if(avgsValues1[i]>threshold || avgsValues2[i]>threshold)
1548 numOfRelevantCells++;
1549
1550 avgsDiffs.push_back( abs(val1 - val2) );
1551 totalAbsAvgs += (abs(val1) + abs(val2));
1552 totalAbsDiff += abs(val1 - val2);
1553 totalAbsLog10Diff += abs(log10(val1) - log10(val2));
1554
1555 }
1556 }
1557 // Compare the avgs values of nonidentical blocks:
1558 vector<double> zeroAvgs(velocityCellsPerBlock, 0);
1559 for( vector<uint32_t>::const_iterator it = nonIdenticalBlockIds.begin(); it != nonIdenticalBlockIds.end(); ++it ) {
1560 // Get the block id
1561 const uint32_t blockId = *it;
1562 // Get avgs values:
1563
1564 const vector<double>* avgsValues1;
1565 const vector<double>* avgsValues2;
1566
1567 unordered_map<uint32_t, vector<double> >::const_iterator it2 = avgs1.find( blockId );
1568 if( it2 == avgs1.end() ) {
1569 avgsValues1 = &zeroAvgs;
1570 } else {
1571 avgsValues1 = &(it2->second);
1572 }
1573
1574 it2 = avgs2.find( blockId );
1575 if( it2 == avgs2.end() ) {
1576 avgsValues2 = &zeroAvgs;
1577 } else {
1578 avgsValues2 = &(it2->second);
1579 }
1580 // Get the diff:
1581 for( uint i = 0; i < velocityCellsPerBlock; ++i ) {
1582 double val1=avgsValues1->operator[](i)>threshold?avgsValues1->operator[](i):threshold;
1583 double val2=avgsValues2->operator[](i)>threshold?avgsValues2->operator[](i):threshold;
1584 if( avgsValues1->operator[](i)>threshold || avgsValues2->operator[](i)>threshold)
1585 numOfRelevantCells++;
1586
1587 avgsDiffs.push_back( abs(val1 - val2) );
1588 totalAbsAvgs += (abs(val1) + abs(val2));
1589 totalAbsDiff += abs(val1 - val2);
1590 totalAbsLog10Diff += abs(log10(val1) - log10(val2));
1591 }
1592 }
1593 numOfIdenticalBlocks += identicalBlockIds.size();
1594 numOfNonIdenticalBlocks += nonIdenticalBlockIds.size();
1595 }
1596 // Get the max and min diff, and the sum of the diff
1597 double maxDiff = 0;
1598 double minDiff = numeric_limits<Real>::max();
1599 double sumDiff = 0;
1600 for( vector<double>::const_iterator it = avgsDiffs.begin(); it != avgsDiffs.end(); ++it ) {
1601 sumDiff += *it;
1602 if( maxDiff < *it ) {
1603 maxDiff = *it;
1604 }
1605 if( minDiff > *it ) {
1606 minDiff = *it;
1607 }
1608 }
1609
1610 Real time1 {0.0};
1611 Real time2 {0.0};
1612 vlsvReader1.readParameter("time", time1);
1613 vlsvReader2.readParameter("time", time2);
1614
1615 const double relativeSumDiff = sumDiff / totalAbsAvgs;
1616 cout << "File names: " << fileName1 << " & " << fileName2 << endl <<
1617 setprecision(3) <<
1618 "NonIdenticalBlocks: " << numOfNonIdenticalBlocks << endl <<
1619 "IdenticalBlocks: " << numOfIdenticalBlocks << endl <<
1620 "Absolute_Error: " << totalAbsDiff << endl <<
1621 "Mean-Absolute-Error: " << totalAbsDiff / numOfRelevantCells << endl <<
1622 "Max-Absolute-Error: " << maxDiff << endl <<
1623 "Absolute-log-Error: " << totalAbsLog10Diff << endl <<
1624 "Mean-Absolute-log-Error: " << totalAbsLog10Diff / numOfRelevantCells << endl <<
1625 "Delta-t: " << time2 - time1 << endl;
1626
1627 return true;
1628}
