181void calculateGradPeTerm(
182 FsGrid< std::array<Real, fsgrids::egradpe::N_EGRADPE>, FS_STENCIL_WIDTH> & EGradPeGrid,
183 FsGrid< std::array<Real, fsgrids::moments::N_MOMENTS>, FS_STENCIL_WIDTH> & momentsGrid,
184 FsGrid< std::array<Real, fsgrids::dmoments::N_DMOMENTS>, FS_STENCIL_WIDTH> & dMomentsGrid,
185 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid,
186 cint i,
187 cint j,
188 cint k,
189 SysBoundary& sysBoundaries
190) {
191 #ifdef DEBUG_FSOLVER
192 if (technicalGrid.get(i,j,k) == NULL) {
193 cerr << "NULL pointer in " << __FILE__ << ":" << __LINE__ << endl;
194 exit(1);
195 }
196 #endif
197
198 cuint cellSysBoundaryFlag = technicalGrid.get(i,j,k)->sysBoundaryFlag;
199
200 if (cellSysBoundaryFlag == sysboundarytype::DO_NOT_COMPUTE || cellSysBoundaryFlag == sysboundarytype::OUTER_BOUNDARY_PADDING) return;
201
202 cuint cellSysBoundaryLayer = technicalGrid.get(i,j,k)->sysBoundaryLayer;
203
204 if ((cellSysBoundaryFlag != sysboundarytype::NOT_SYSBOUNDARY) && (cellSysBoundaryLayer != 1)) {
205 sysBoundaries.getSysBoundary(cellSysBoundaryFlag)->fieldSolverBoundaryCondGradPeElectricField(EGradPeGrid,i,j,k,0);
206 sysBoundaries.getSysBoundary(cellSysBoundaryFlag)->fieldSolverBoundaryCondGradPeElectricField(EGradPeGrid,i,j,k,1);
207 sysBoundaries.getSysBoundary(cellSysBoundaryFlag)->fieldSolverBoundaryCondGradPeElectricField(EGradPeGrid,i,j,k,2);
208 } else {
209 calculateEdgeGradPeTermXComponents(EGradPeGrid,momentsGrid,dMomentsGrid,i,j,k);
210 calculateEdgeGradPeTermYComponents(EGradPeGrid,momentsGrid,dMomentsGrid,i,j,k);
211 calculateEdgeGradPeTermZComponents(EGradPeGrid,momentsGrid,dMomentsGrid,i,j,k);
212 }
