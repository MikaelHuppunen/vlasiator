139 RP::get("ionosphere.centerY", this->center[1]);

153 RP::get("Magnetosphere.refine_L4nosexmin", this->refine_L4nosexmin);

207 // Some fail-safety checks
208 if(sP.taperInnerRadius < 0 || sP.taperOuterRadius < 0) {
209 if(myRank == MASTER_RANK) {
210 cerr << "Error: " << pop << "_Magnetosphere.taperInnerRadius and tapeOuterRadius should be >= 0! Aborting." << endl;

213 }

216 cerr << "Error: " << pop << "_Magnetosphere.taperInnerRadius should be <= taperOuterRadius! Aborting." << endl;
217 }
218 abort();
219 }

221 if(myRank == MASTER_RANK) {
222 cerr << "Error: " << pop << "_Magnetosphere.taperOuterRadius is non-zero yet smaller than ionosphere.radius / copysphere.radius! Aborting." << endl;
223 }
224 abort();
225 }
226 if(sP.taperInnerRadius == 0 && sP.taperOuterRadius > 0) {
227 if(myRank == MASTER_RANK) {
228 cerr << "Warning: " << pop << "_Magnetosphere.taperInnerRadius is zero (default), now setting this to the same value as ionosphere.radius / copysphere.radius, that is " << this->ionosphereRadius << ". Set/change " << pop << "_Magnetosphere.taperInnerRadius if this is not the expected behavior." << endl;
229 }
230 sP.taperInnerRadius = this->ionosphereRadius;
231 }
232 if(sP.ionosphereT == 0) {
233 if(myRank == MASTER_RANK) {
234 if (sysBoundaryContainer.existSysBoundary("Copysphere")) {
235 cerr << "Warning: " << pop << "_copysphere.T is zero (default), now setting to the same value as " << pop << "_Magnetosphere.T, that is " << sP.T << ". Set/change " << pop << "_copysphere.T if this is not the expected behavior." << endl;
236 } else if (sysBoundaryContainer.existSysBoundary("Ionosphere")) {
237 cerr << "Warning: " << pop << "_ionosphere.T is zero (default), now setting to the same value as " << pop << "_Magnetosphere.T, that is " << sP.T << ". Set/change " << pop << "_ionosphere.T if this is not the expected behavior." << endl;
238 }
239 }
240 sP.ionosphereT = sP.T;
241 }
242 if(sP.ionosphereRho == 0) {
243 if(myRank == MASTER_RANK) {
244 if (sysBoundaryContainer.existSysBoundary("Copysphere")) {
245 cerr << "Warning: " << pop << "_copysphere.rho is zero (default), now setting to the same value as " << pop << "_Magnetosphere.rho, that is " << sP.rho << ". Set/change " << pop << "_copysphere.rho if this is not the expected behavior." << endl;
246 } else if (sysBoundaryContainer.existSysBoundary("Ionosphere")) {
247 cerr << "Warning: " << pop << "_ionosphere.rho is zero (default), now setting to the same value as " << pop << "_Magnetosphere.rho, that is " << sP.rho << ". Set/change " << pop << "_ionosphere.rho if this is not the expected behavior." << endl;
248 }
249 }
250 sP.ionosphereRho = sP.rho;
251 }
252
253 speciesParams.push_back(sP);
254 }

264 Real Magnetosphere::geometryRadius(Real x, Real y, Real z) const {
265 Real radius;
266 switch(this->ionosphereGeometry) {
267 case 0:
268 // infinity-norm, result is a diamond/square with diagonals aligned on the axes in 2D
269 radius = fabs(x-center[0]) + fabs(y-center[1]) + fabs(z-center[2]);
270 break;
271 case 1:
272 // 1-norm, result is is a grid-aligned square in 2D
273 radius = max(max(fabs(x-center[0]), fabs(y-center[1])), fabs(z-center[2]));
274 break;
275 case 2:
276 // 2-norm (Cartesian), result is a circle in 2D
277 radius = sqrt((x-center[0])*(x-center[0]) + (y-center[1])*(y-center[1]) + (z-center[2])*(z-center[2]));
278 break;
279 case 3:
280 // cylinder aligned with y-axis, use with polar plane/line dipole
281 radius = sqrt((x-center[0])*(x-center[0]) + (z-center[2])*(z-center[2]));
282 break;
283 default:
284 std::cerr << __FILE__ << ":" << __LINE__ << ":" << "ionosphere.geometry has to be 0, 1, 2 or 3." << std::endl;
285 abort();
286 }
287 return radius;
288 }

290 /* set 0-centered dipole */
291 void Magnetosphere::setProjectBField(
292 FsGrid< std::array<Real, fsgrids::bfield::N_BFIELD>, FS_STENCIL_WIDTH> & perBGrid,
293 FsGrid< std::array<Real, fsgrids::bgbfield::N_BGB>, FS_STENCIL_WIDTH> & BgBGrid,
294 FsGrid< fsgrids::technical, FS_STENCIL_WIDTH> & technicalGrid
295 ) {
296 Dipole bgFieldDipole;
297 LineDipole bgFieldLineDipole;
298 VectorDipole bgVectorDipole;
299
300 phiprof::Timer switchDipoleTypeTimer {"switch-dipoleType"};
301 // The hardcoded constants of dipole and line dipole moments are obtained
302 // from Daldorff et al (2014), see
303 // https://github.com/fmihpc/vlasiator/issues/20 for a derivation of the
304 // values used here.
305 switch(this->dipoleType) {
306 case 0:
307 bgFieldDipole.initialize(8e15 *this->dipoleScalingFactor, 0.0, 0.0, 0.0, 0.0 );//set dipole moment
308 setBackgroundField(bgFieldDipole, BgBGrid);
309 SBC::ionosphereGrid.setDipoleField(bgFieldDipole);
310 break;
311 case 1:
312 bgFieldLineDipole.initialize(126.2e6 *this->dipoleScalingFactor, 0.0, 0.0, 0.0 );//set dipole moment
313 setBackgroundField(bgFieldLineDipole, BgBGrid);
314 SBC::ionosphereGrid.setDipoleField(bgFieldLineDipole);
315 break;
316 case 2:
317 bgFieldLineDipole.initialize(126.2e6 *this->dipoleScalingFactor, 0.0, 0.0, 0.0 );//set dipole moment
318 setBackgroundField(bgFieldLineDipole, BgBGrid);
319 //Append mirror dipole
320 bgFieldLineDipole.initialize(126.2e6 *this->dipoleScalingFactor, this->dipoleMirrorLocationX, 0.0, 0.0 );
321 setBackgroundField(bgFieldLineDipole, BgBGrid, true);
322 SBC::ionosphereGrid.setDipoleField(bgFieldLineDipole);
323 break;
324 case 3:
325 bgFieldDipole.initialize(8e15 *this->dipoleScalingFactor, 0.0, 0.0, 0.0, 0.0 );//set dipole moment
326 setBackgroundField(bgFieldDipole, BgBGrid);
327 SBC::ionosphereGrid.setDipoleField(bgFieldDipole);
328 //Append mirror dipole
329 bgFieldDipole.initialize(8e15 *this->dipoleScalingFactor, this->dipoleMirrorLocationX, 0.0, 0.0, 0.0 );//mirror
330 setBackgroundField(bgFieldDipole, BgBGrid, true);
331 break;
332 case 4: // Vector potential dipole, vanishes or optionally scales to static inflow value after a given x-coordinate
333 // What we in fact do is we place the regular dipole in the background field, and the
334 // corrective terms in the perturbed field. This maintains the BGB as curl-free.
335 bgFieldDipole.initialize(8e15 *this->dipoleScalingFactor, 0.0, 0.0, 0.0, 0.0 ); //set dipole moment
336 setBackgroundField(bgFieldDipole, BgBGrid);
337 SBC::ionosphereGrid.setDipoleField(bgFieldDipole);
338 // Now we calculate the difference required to scale the dipole to zero as we approach the inflow,
339 // and store it inside the BgBGrid object for use by e.g. boundary conditions.
340 bgFieldDipole.initialize(-8e15 *this->dipoleScalingFactor, 0.0, 0.0, 0.0, 0.0 );
341 setPerturbedField(bgFieldDipole, BgBGrid, fsgrids::bgbfield::BGBXVDCORR);
342 bgVectorDipole.initialize(8e15 *this->dipoleScalingFactor, 0.0, 0.0, 0.0, this->dipoleTiltPhi*M_PI/180., this->dipoleTiltTheta*M_PI/180., this->dipoleXFull, this->dipoleXZero, this->dipoleInflowB[0], this->dipoleInflowB[1], this->dipoleInflowB[2]);
343 setPerturbedField(bgVectorDipole, BgBGrid, fsgrids::bgbfield::BGBXVDCORR, true);
344 if (P::isRestart == false) {
345 // If we are starting a new simulation, we also copy this data into perB.
346 const auto localSize = BgBGrid.getLocalSize().data();
347 #pragma omp parallel for collapse(2)
348 for (int z = 0; z < localSize[2]; ++z) {
349 for (int y = 0; y < localSize[1]; ++y) {
350 for (int x = 0; x < localSize[0]; ++x) {
351 std::array<Real, fsgrids::bgbfield::N_BGB>* BGBcell = BgBGrid.get(x, y, z);
352 std::array<Real, fsgrids::bfield::N_BFIELD>* PERBcell = perBGrid.get(x, y, z);
353 PERBcell->at(fsgrids::bfield::PERBX) = BGBcell->at(fsgrids::bgbfield::BGBXVDCORR);
354 PERBcell->at(fsgrids::bfield::PERBY) = BGBcell->at(fsgrids::bgbfield::BGBYVDCORR);
355 PERBcell->at(fsgrids::bfield::PERBZ) = BGBcell->at(fsgrids::bgbfield::BGBZVDCORR);
356 }
357 }
358 }
359 }
360 break;
361 default:
362 setBackgroundFieldToZero(BgBGrid);
363 }
364 switchDipoleTypeTimer.stop();
365
366 const auto localSize = BgBGrid.getLocalSize().data();
367
368 phiprof::Timer zeroingTimer {"zeroing-out"};
369
370#pragma omp parallel
371 {
372 bool doZeroOut;
373 //Force field to zero in the perpendicular direction for 2D (1D) simulations. Otherwise we have unphysical components.
374 doZeroOut = P::xcells_ini ==1 && this->zeroOutComponents[0]==1;
375
376 if(doZeroOut) {
377#pragma omp for collapse(2)
378 for (FsGridTools::FsIndex_t z = 0; z < localSize[2]; ++z) {
379 for (FsGridTools::FsIndex_t y = 0; y < localSize[1]; ++y) {
380 for (FsGridTools::FsIndex_t x = 0; x < localSize[0]; ++x) {
381 std::array<Real, fsgrids::bgbfield::N_BGB>* cell = BgBGrid.get(x, y, z);
382 cell->at(fsgrids::bgbfield::BGBX)=0;
383 cell->at(fsgrids::bgbfield::BGBXVOL)=0.0;
384 cell->at(fsgrids::bgbfield::dBGBydx)=0.0;
385 cell->at(fsgrids::bgbfield::dBGBzdx)=0.0;
386 cell->at(fsgrids::bgbfield::dBGBxdy)=0.0;
387 cell->at(fsgrids::bgbfield::dBGBxdz)=0.0;
388 cell->at(fsgrids::bgbfield::dBGBYVOLdx)=0.0;
389 cell->at(fsgrids::bgbfield::dBGBZVOLdx)=0.0;
390 cell->at(fsgrids::bgbfield::dBGBXVOLdy)=0.0;
391 cell->at(fsgrids::bgbfield::dBGBXVOLdz)=0.0;
392 }
393 }
394 }
395 }
396
397 doZeroOut = P::ycells_ini ==1 && this->zeroOutComponents[1]==1;
398 if(doZeroOut) {
399 /*2D simulation in x and z. Set By and derivatives along Y, and derivatives of By to zero*/
400#pragma omp for collapse(2)
401 for (FsGridTools::FsIndex_t z = 0; z < localSize[2]; ++z) {
402 for (FsGridTools::FsIndex_t y = 0; y < localSize[1]; ++y) {
403 for (FsGridTools::FsIndex_t x = 0; x < localSize[0]; ++x) {
404 std::array<Real, fsgrids::bgbfield::N_BGB>* cell = BgBGrid.get(x, y, z);
405 cell->at(fsgrids::bgbfield::BGBY)=0.0;
406 cell->at(fsgrids::bgbfield::BGBYVOL)=0.0;
407 cell->at(fsgrids::bgbfield::dBGBxdy)=0.0;
408 cell->at(fsgrids::bgbfield::dBGBzdy)=0.0;
409 cell->at(fsgrids::bgbfield::dBGBydx)=0.0;
410 cell->at(fsgrids::bgbfield::dBGBydz)=0.0;
411 cell->at(fsgrids::bgbfield::dBGBXVOLdy)=0.0;
412 cell->at(fsgrids::bgbfield::dBGBZVOLdy)=0.0;
413 cell->at(fsgrids::bgbfield::dBGBYVOLdx)=0.0;
414 cell->at(fsgrids::bgbfield::dBGBYVOLdz)=0.0;
415 }
416 }
417 }
418 }
419
420 doZeroOut = P::zcells_ini ==1 && this->zeroOutComponents[2]==1;
421 if(doZeroOut) {

596 vector<std::array<Real, 3> > Magnetosphere::getV0(
597 creal x,
598 creal y,
599 creal z,
600 const uint popID
601 ) const {
602 const MagnetosphereSpeciesParameters& sP = this->speciesParams[popID];
603
604 vector<std::array<Real, 3> > centerPoints;
605 std::array<Real, 3> V0 {{sP.V0[0], sP.V0[1], sP.V0[2]}};
606 std::array<Real, 3> ionosphereV0 = {{sP.ionosphereV0[0], sP.ionosphereV0[1], sP.ionosphereV0[2]}};
607
608 Real radius = this->geometryRadius(x,y,z);
609 if(radius < sP.taperOuterRadius) {
610 // sine tapering
611 Real q=0.5*(1.0-sin(M_PI*(radius-sP.taperInnerRadius)/(sP.taperOuterRadius-sP.taperInnerRadius)+0.5*M_PI));
612
613 for(uint i=0; i<3; i++) {
614 V0[i]=q*(V0[i]-ionosphereV0[i])+ionosphereV0[i];
615 if(radius <= sP.taperInnerRadius) {
616 V0[i] = ionosphereV0[i];
617 }
618 }
619 }
620
621 centerPoints.push_back(V0);
622 return centerPoints;
