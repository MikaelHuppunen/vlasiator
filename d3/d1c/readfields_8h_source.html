106/* Read the "raw" field data in file order */
107template <class Reader>
108std::vector<double> readFieldData(Reader& r, std::string& name, unsigned int numcomponents) {
109
110 uint64_t arraySize=0;
111 uint64_t vectorSize=0;
112 uint64_t byteSize=0;
113 vlsv::datatype::type dataType;
114 std::list<std::pair<std::string,std::string> > attribs;
115 attribs.push_back(std::pair<std::string,std::string>("name",name));
116 if( r.getArrayInfo("VARIABLE",attribs, arraySize,vectorSize,dataType,byteSize) == false ) {
117 std::cerr << "getArrayInfo returned false when trying to read VARIABLE \""
118 << name << "\"." << std::endl;
119 exit(1);
120 }
121
122 if(byteSize == 8) {
123 /* Allocate memory for the data */
124 std::vector<double> buffer(arraySize*vectorSize);
125
126 if( r.readArray("VARIABLE",attribs,0,arraySize,(char*) buffer.data()) == false) {
127 std::cerr << "readArray failed when trying to read VARIABLE \"" << name << "\"." << std::endl;
128 exit(1);
129 }
130
131 return buffer;
132 } else if(byteSize == 4) {
133 /* Allocate memory for the data */
134 std::vector<double> buffer;
135 std::vector<float> fbuffer(arraySize*vectorSize);
136
137 if( r.readArray("VARIABLE",attribs,0,arraySize,(char*) fbuffer.data()) == false) {
138 std::cerr << "readArray faied when trying to read VARIABLE \"" << name << "\"." << std::endl;
139 exit(1);
140 }
141
142 for(float value : fbuffer) {
143 buffer.push_back((double)value);
144 }
145
146 return buffer;
147 } else {
148 std::cerr << "Datatype of VARIABLE \"" << name << "\" entries is not double." << std::endl;
149 exit(1);
150 }
151
152 return {0};
153}

155/* Read the "raw" FsGrid data in file order */
156template <class Reader>
157std::vector<double> readFsGridData(Reader& r, std::string& name, unsigned int numcomponents) {
158
159 uint64_t arraySize;
160 uint64_t vectorSize;
161 vlsv::datatype::type dataType;
162 uint64_t byteSize;
163 std::list<std::pair<std::string,std::string> > attribs;
164
165 attribs.push_back(std::make_pair("name",name));
166 attribs.push_back(std::make_pair("mesh","fsgrid"));
167
168 if (r.getArrayInfo("VARIABLE",attribs,arraySize,vectorSize,dataType,byteSize) == false) {
169 std::cerr << "getArrayInfo returned false when trying to read VARIABLE \""
170 << name << "\"." << std::endl;
171 exit(1);
172 }
173
174 if(dataType != vlsv::datatype::type::FLOAT || byteSize != 8 || vectorSize != numcomponents) {
175 std::cerr << "Datatype of VARIABLE \"" << name << "\" entries is not double." << std::endl;
176 exit(1);
177 }
178
179 int numWritingRanks=0;
180 if(r.readParameter("numWritingRanks",numWritingRanks) == false) {
181 std::cerr << "FSGrid writing rank number not found";
182 exit(1);
183 }
184
185 // Are we restarting from the same number of tasks, or a different number?
186 std::array<uint, 3> size;
187 r.readParameter("xcells_ini",size[0]);
188 r.readParameter("ycells_ini",size[1]);
189 r.readParameter("zcells_ini",size[2]);
190
191
192 // Determine our tasks storage size
193 size_t storageSize = size[0]*size[1]*size[2];
194 std::vector<Real> buffer(storageSize*numcomponents);
195 std::vector<Real> readBuffer(storageSize*numcomponents);
196
197 if(r.readArray("VARIABLE",attribs,0,arraySize,(char*) readBuffer.data()) == false) {
198 std::cerr << "readArray faied when trying to read VARIABLE \"" << name << "\"." << std::endl;
199 exit(1);
200 }
201
202
203 // More difficult case: different number of tasks.
204 // In this case, our own fsgrid domain overlaps (potentially many) domains in the file.
205 // We read the whole source rank into a temporary buffer, and transfer the overlapping
206 // part.
207 //
208 // +------------+----------------+
209 // | | |
210 // | . . . . . . . . . . . . |
211 // | .<----->|<----------->. |
212 // | .<----->|<----------->. |
213 // | .<----->|<----------->. |
214 // +----+-------+-------------+--|
215 // | .<----->|<----------->. |
216 // | .<----->|<----------->. |
217 // | .<----->|<----------->. |
218 // | . . . . . . . . . . . . |
219 // | | |
220 // +------------+----------------+
221
222 std::array<int,3> fileDecomposition;
223 FsGridTools::computeDomainDecomposition(size, numWritingRanks, fileDecomposition);
224 //computeFsGridDecomposition(size, numWritingRanks, fileDecomposition);
225
226
227 // Iterate through tasks and find their overlap with our domain.
228 size_t fileOffset = 0;
229 for(int task = 0; task < numWritingRanks; task++) {
230 std::array<int,3> overlapStart,overlapEnd,overlapSize;
231
232 overlapStart[0] = FsGridTools::calcLocalStart(size[0], fileDecomposition[0], task/fileDecomposition[2]/fileDecomposition[1]);
233 overlapStart[1] = FsGridTools::calcLocalStart(size[1], fileDecomposition[1], (task/fileDecomposition[2])%fileDecomposition[1]);
234 overlapStart[2] = FsGridTools::calcLocalStart(size[2], fileDecomposition[2], task%fileDecomposition[2]);
235
236 overlapSize[0] = FsGridTools::calcLocalSize(size[0], fileDecomposition[0], task/fileDecomposition[2]/fileDecomposition[1]);
237 overlapSize[1] = FsGridTools::calcLocalSize(size[1], fileDecomposition[1], (task/fileDecomposition[2])%fileDecomposition[1]);
238 overlapSize[2] = FsGridTools::calcLocalSize(size[2], fileDecomposition[2], task%fileDecomposition[2]);
239
240 overlapEnd[0] = overlapStart[0]+overlapSize[0];
241 overlapEnd[1] = overlapStart[1]+overlapSize[1];
242 overlapEnd[2] = overlapStart[2]+overlapSize[2];
243
244 // r.startMultiread("VARIABLE", attribs);
245 // // Read every source rank that we have an overlap with.
246 // if(r.addMultireadUnit((char*)readBuffer.data(), overlapSize[0]*overlapSize[1]*overlapSize[2])==false) {
247 // std::cerr << "ERROR: Failed to read fsgrid variable " << name << std::endl;
248 // exit(1);
249 // }
250 // r.endMultiread(fileOffset);
251
252 // Copy continuous stripes in x direction.
253 for(int z=overlapStart[2]; z<overlapEnd[2]; z++) {
254 for(int y=overlapStart[1]; y<overlapEnd[1]; y++) {
255 for(int x=overlapStart[0]; x<overlapEnd[0]; x++) {
256 int index = (z - overlapStart[2]) * overlapSize[0]*overlapSize[1]
257 + (y - overlapStart[1]) * overlapSize[0]
258 + (x - overlapStart[0]);
259
260 std::memcpy(&buffer[(size[0]*size[1]*z + size[0]*y + x)*numcomponents], &readBuffer[(fileOffset + index)*numcomponents], numcomponents*sizeof(Real));
261 }
262 }
263 }
264 fileOffset += overlapSize[0] * overlapSize[1] * overlapSize[2];
265 }
266 return buffer;
267}

269/* Read the next logical input file. Depending on sign of dt,
270 * this may be a numerically larger or smaller file.
271 * Return value: true if a new file was read, otherwise false.
272 * TODO: might need some DRY
273 */
274template <class Reader>
275bool readNextTimestep(const std::string& filename_pattern, double t, int step, Field& E0, Field& E1,
276 Field& B0, Field& B1, Field& V, bool doV, int& input_file_counter) {
277
278 char filename_buffer[256];
279 bool retval = false;
280
281 while(t < E0.time || t>= E1.time) {
282 input_file_counter += step;
283
284 E0=E1;
285 B0=B1;
286 snprintf(filename_buffer,256,filename_pattern.c_str(),input_file_counter);
287
288 /* Open next file */
289 Reader r;
290 r.open(filename_buffer);
291 double t;
292 if(!r.readParameter("time",t)) {
293 if(!r.readParameter("t",t)) {
294 std::cerr << "Time parameter in file " << filename_buffer << " is neither 't' nor 'time'. Bad file format?"
295 << std::endl;
296 exit(1);
297 }
298 }
299
300 E1.time = t;
301 B1.time = t;
302
303 uint64_t cells[3];
304 r.readParameter("xcells_ini",cells[0]);
305 r.readParameter("ycells_ini",cells[1]);
306 r.readParameter("zcells_ini",cells[2]);
307
308 /* Read CellIDs and Field data */
309 std::vector<uint64_t> cellIds = readCellIds(r);
310 std::string name(B_field_name);
311 std::vector<double> Bbuffer;
312 std::vector<double> Ebuffer;
313 if (B_field_name == "fg_b" || B_field_name == "fg_b_background") {
314 Bbuffer = readFsGridData(r,name,3u);
315 if (B_field_name == "fg_b_background") {
316 name = "fg_b_perturbed";
317 std::vector<double> perturbedBbuffer = readFsGridData(r,name,3u);
318 for (unsigned int i = 0; i < Bbuffer.size(); ++i) {
319 Bbuffer[i] += perturbedBbuffer[i];
320 }
321 }
322 name = E_field_name;
323 Ebuffer = readFsGridData(r,name,3u);
324 for (unsigned int i = 0; i < cellIds.size(); ++i) {
325 cellIds[i] = i+1;
326 }
327 } else {
328 Bbuffer = readFieldData(r,name,3u);
329 if (B_field_name == "vg_b_background_vol") {
330 name = "vg_b_perturbed_vol";
331 std::vector<double> perturbedBbuffer = readFieldData(r,name,3u);
332 for (unsigned int i = 0; i < Bbuffer.size(); ++i) {
333 Bbuffer[i] += perturbedBbuffer[i];
334 }
335 }
336 name = E_field_name;
337 Ebuffer = readFieldData(r,name,3u);
338 }
339 std::vector<double> Vbuffer;
340 if(doV) {
341 name = ParticleParameters::V_field_name;
342 std::vector<double> rho_v_buffer = readFieldData(r,name,3u);
343 if(ParticleParameters::divide_rhov_by_rho) {
344 name = ParticleParameters::rho_field_name;
345 std::vector<double> rho_buffer = readFieldData(r,name,1u);
346 for(unsigned int i=0; i<rho_buffer.size(); i++) {
347 Vbuffer.push_back(rho_v_buffer[3*i] / rho_buffer[i]);
348 Vbuffer.push_back(rho_v_buffer[3*i+1] / rho_buffer[i]);
349 Vbuffer.push_back(rho_v_buffer[3*i+2] / rho_buffer[i]);
350 }
351 }
352 }
353
354 /* Assign them, without sanity checking */
355 /* TODO: Is this actually a good idea? */
356 for(uint i=0; i< cellIds.size(); i++) {
357 uint64_t c = cellIds[i];
358 int64_t x = c % cells[0];
359 int64_t y = (c /cells[0]) % cells[1];
360 int64_t z = c /(cells[0]*cells[1]);
361
362 double* Etgt = E1.getCellRef(x,y,z);
363 double* Btgt = B1.getCellRef(x,y,z);
364 Etgt[0] = Ebuffer[3*i];
365 Etgt[1] = Ebuffer[3*i+1];
366 Etgt[2] = Ebuffer[3*i+2];
367 Btgt[0] = Bbuffer[3*i];
368 Btgt[1] = Bbuffer[3*i+1];
369 Btgt[2] = Bbuffer[3*i+2];
370
371 if(doV) {
372 double* Vtgt = V.getCellRef(x,y,z);
373 Vtgt[0] = Vbuffer[3*i];
374 Vtgt[1] = Vbuffer[3*i+1];
375 Vtgt[2] = Vbuffer[3*i+2];
376 }
377 }
378
379 r.close();
380 retval = true;
381 }
382
383 return retval;
384}

397/* Read E- and B-Fields as well as velocity field from a vlsv file */
398template <class Reader>
399void readfields(const char* filename, Field& E, Field& B, Field& V, bool doV=true) {
400 Reader r;
401
402#ifdef DEBUG
403 std::cerr << "Opening " << filename << "...";
404#endif
405 r.open(filename);
406#ifdef DEBUG
407 std::cerr <<"ok." << std::endl;
408#endif
409
410 /* Check whethere we got volume-centered fields */
411 detect_field_names<Reader>(r);
412
413 /* Read the MESH, yielding the CellIDs */
414 std::vector<uint64_t> cellIds = readCellIds(r);
415
416 /* Also read the raw field data */
417 std::vector<double> Bbuffer;
418 std::vector<double> Ebuffer;
419 std::string name = B_field_name;
420 if (B_field_name == "fg_b" || B_field_name == "fg_b_background") {
421 Bbuffer = readFsGridData(r,name,3u);
422 if (B_field_name == "fg_b_background") {
423 name = "fg_b_perturbed";
424 std::vector<double> perturbedBbuffer = readFsGridData(r,name,3u);
425 for (unsigned int i = 0; i < Bbuffer.size(); ++i) {
426 Bbuffer[i] += perturbedBbuffer[i];
427 }
428 }
429 name = E_field_name;
430 Ebuffer = readFsGridData(r,name,3u);
431 for (unsigned int i = 0; i < cellIds.size(); ++i) {
432 cellIds[i] = i+1;
433 }
434 } else {
435 Bbuffer = readFieldData(r,name,3u);
436 if (B_field_name == "vg_b_background_vol") {
437 name = "vg_b_perturbed_vol";
438 std::vector<double> perturbedBbuffer = readFieldData(r,name,3u);
439 for (unsigned int i = 0; i < Bbuffer.size(); ++i) {
440 Bbuffer[i] += perturbedBbuffer[i];
441 }
442 }
443 name = E_field_name;
444 Ebuffer = readFieldData(r,name,3u);
445 }
446 std::vector<double> rho_v_buffer,rho_buffer;
447 if(doV) {
448 name = ParticleParameters::V_field_name;
449 rho_v_buffer = readFieldData(r,name,3u);
450 if(ParticleParameters::divide_rhov_by_rho) {
451 name = ParticleParameters::rho_field_name;
452 rho_buffer = readFieldData(r,name,1u);
453 }
454 }
455
456 /* Coordinate Boundaries */
457 double min[3], max[3], time;
458 uint64_t cells[3];
459 r.readParameter("xmin",min[0]);
460 r.readParameter("ymin",min[1]);
461 r.readParameter("zmin",min[2]);
462 r.readParameter("xmax",max[0]);
463 r.readParameter("ymax",max[1]);
464 r.readParameter("zmax",max[2]);
465 r.readParameter("xcells_ini",cells[0]);
466 r.readParameter("ycells_ini",cells[1]);
467 r.readParameter("zcells_ini",cells[2]);
468 if(!r.readParameter("t",time)) {
469 r.readParameter("time",time);
470 }
471
472 //std::cerr << "Grid is " << cells[0] << " x " << cells[1] << " x " << cells[2] << " Cells, " << std::endl
473 // << " with dx = " << ((max[0]-min[0])/cells[0]) << ", dy = " << ((max[1]-min[1])/cells[1])
474 // << ", dz = " << ((max[2]-min[2])/cells[2]) << "." << std::endl;
475
476 /* Allocate space for the actual field structures */
477 E.data.resize(4*cells[0]*cells[1]*cells[2]);
478 B.data.resize(4*cells[0]*cells[1]*cells[2]);
479 if(doV) {
480 V.data.resize(4*cells[0]*cells[1]*cells[2]);
481 }
482
483 /* Sanity-check stored data sizes */
484 if(3*cellIds.size() != Bbuffer.size()) {
485 std::cerr << "3 * cellIDs.size (" << cellIds.size() << ") != Bbuffer.size (" << Bbuffer.size() << ")!"
486 << std::endl;
487 exit(1);
488 }
489 if(3*cellIds.size() != Ebuffer.size()) {
490 std::cerr << "3 * cellIDs.size (" << cellIds.size() << ") != Ebuffer.size (" << Ebuffer.size() << ")!"
491 << std::endl;
492 exit(1);
493 }
494 if(doV) {
495 if(3*cellIds.size() != rho_v_buffer.size()) {
496 std::cerr << "3 * cellIDs.size (" << cellIds.size() << ") != rho_v_buffer.size (" << Ebuffer.size() << ")!"
497 << std::endl;
498 exit(1);
499 }
500 if(ParticleParameters::divide_rhov_by_rho && cellIds.size() != rho_buffer.size()) {
501 std::cerr << "cellIDs.size (" << cellIds.size() << ") != rho_buffer.size (" << Ebuffer.size() << ")!"
502 << std::endl;
503 exit(1);
504 }
505 }
506
507 // Make sure the target fields have boundary data.
508 if(E.dimension[0] == nullptr || E.dimension[1] == nullptr || E.dimension[2] == nullptr) {
509 std::cerr << "Warning: Field boundary pointers uninitialized!" << std::endl;
510 E.dimension[0] = B.dimension[0] = V.dimension[0] = createBoundary<OpenBoundary>(0);
511 E.dimension[1] = B.dimension[1] = V.dimension[1] = createBoundary<OpenBoundary>(1);
512 E.dimension[2] = B.dimension[2] = V.dimension[2] = createBoundary<OpenBoundary>(2);
513 }
514 /* Set field sizes */
515 for(int i=0; i<3;i++) {
516 /* Volume-centered values -> shift by half a cell in all directions*/
517 E.dx[i] = B.dx[i] = V.dx[i] = (max[i]-min[i])/cells[i];
518 double shift = E.dx[i]/2;
519 E.dimension[i]->min = B.dimension[i]->min = V.dimension[i]->min = min[i]+shift;
520 E.dimension[i]->max = B.dimension[i]->max = V.dimension[i]->max = max[i]+shift;
521 E.dimension[i]->cells = B.dimension[i]->cells = V.dimension[i]->cells = cells[i];
522 }
523 E.time = B.time = V.time = time;
524
525 /* So, now we've got the cellIDs, the mesh size and the field values,
526 * we can sort them into place */
527 for(uint i=0; i< cellIds.size(); i++) {
528 uint64_t c = cellIds[i]-1;
529 int64_t x = c % cells[0];
530 int64_t y = (c /cells[0]) % cells[1];
531 int64_t z = c /(cells[0]*cells[1]);
532
533 double* Etgt = E.getCellRef(x,y,z);
534 double* Btgt = B.getCellRef(x,y,z);
535 Etgt[0] = Ebuffer[3*i];
536 Etgt[1] = Ebuffer[3*i+1];
537 Etgt[2] = Ebuffer[3*i+2];
538 Btgt[0] = Bbuffer[3*i];
539 Btgt[1] = Bbuffer[3*i+1];
540 Btgt[2] = Bbuffer[3*i+2];
541
542 if(doV) {
543 double* Vtgt = V.getCellRef(x,y,z);
544 if(ParticleParameters::divide_rhov_by_rho) {
545 Vtgt[0] = rho_v_buffer[3*i] / rho_buffer[i];
546 Vtgt[1] = rho_v_buffer[3*i+1] / rho_buffer[i];
547 Vtgt[2] = rho_v_buffer[3*i+2] / rho_buffer[i];
548 } else {
549 Vtgt[0] = rho_v_buffer[3*i];
550 Vtgt[1] = rho_v_buffer[3*i+1];
551 Vtgt[2] = rho_v_buffer[3*i+2];
552 }
553 }
554 }
555
556 r.close();
557}
